## 1. 线程与线程池

### 1.1 Thread 基础

```csharp
// 创建线程
Thread thread = new Thread(() =>
{
    Debug.Log("Worker thread");
});
thread.Start();

// 前台线程 vs 后台线程
thread.IsBackground = true;  // 后台线程，进程退出时自动终止

// 线程优先级
thread.Priority = ThreadPriority.AboveNormal;
```

### 1.2 ThreadPool

```csharp
// 使用线程池
ThreadPool.QueueUserWorkItem(state =>
{
    Debug.Log($"Thread pool work: {Thread.CurrentThread.ManagedThreadId}");
});

// 等待线程池工作完成
var resetEvent = new ManualResetEvent(false);
ThreadPool.QueueUserWorkItem(state =>
{
    // Do work
    resetEvent.Set();
});
resetEvent.WaitOne();
```

### 1.3 线程池配置

```csharp
// 查看线程池设置
int workerThreads, completionPortThreads;
ThreadPool.GetMinThreads(out workerThreads, out completionPortThreads);
ThreadPool.GetMaxThreads(out workerThreads, out completionPortThreads);

// 设置线程池大小
ThreadPool.SetMinThreads(8, 8);
ThreadPool.SetMaxThreads(1024, 1024);
```

## 2. 同步原语

### 2.1 lock 语句

```csharp
private readonly object _lock = new object();
private int _counter;

public void Increment()
{
    lock (_lock)
    {
        _counter++;
    }
}

// 编译器生成的代码
public void IncrementExpanded()
{
    Monitor.Enter(_lock);
    try
    {
        _counter++;
    }
    finally
    {
        Monitor.Exit(_lock);
    }
}
```

### 2.2 Monitor

```csharp
// 超时锁
if (Monitor.TryEnter(_lock, TimeSpan.FromSeconds(1)))
{
    try
    {
        // 临界区
    }
    finally
    {
        Monitor.Exit(_lock);
    }
}
else
{
    // 获取锁超时
}

// 等待和脉冲（生产者-消费者）
lock (_lock)
{
    while (!condition)
        Monitor.Wait(_lock);  // 释放锁并等待

    // 条件满足，执行操作
    Monitor.PulseAll(_lock);  // 通知等待的线程
}
```

### 2.3 ReaderWriterLockSlim

```csharp
private readonly ReaderWriterLockSlim _rwLock = new();

// 读锁（允许多个读者）
public int ReadData()
{
    _rwLock.EnterReadLock();
    try
    {
        return _data;
    }
    finally
    {
        _rwLock.ExitReadLock();
    }
}

// 写锁（独占）
public void WriteData(int value)
{
    _rwLock.EnterWriteLock();
    try
    {
        _data = value;
    }
    finally
    {
        _rwLock.ExitWriteLock();
    }
}

// 可升级读锁（读后可能写）
public void UpdateData(int value)
{
    _rwLock.EnterUpgradeableReadLock();
    try
    {
        if (_data != value)
        {
            _rwLock.EnterWriteLock();
            try
            {
                _data = value;
            }
            finally
            {
                _rwLock.ExitWriteLock();
            }
        }
    }
    finally
    {
        _rwLock.ExitUpgradeableReadLock();
    }
}
```

### 2.4 Semaphore/SemaphoreSlim

```csharp
// 限制并发数量
private readonly SemaphoreSlim _semaphore = new(3, 3);  // 最多3个并发

public async Task ProcessAsync()
{
    await _semaphore.WaitAsync();
    try
    {
        await DoWorkAsync();
    }
    finally
    {
        _semaphore.Release();
    }
}
```

### 2.5 Mutex

```csharp
// 跨进程互斥
using var mutex = new Mutex(false, "Global\\MyAppMutex");

if (!mutex.WaitOne(TimeSpan.FromSeconds(5)))
{
    Debug.Log("另一个实例正在运行");
    return;
}

try
{
    // 单实例代码
}
finally
{
    mutex.ReleaseMutex();
}
```

## 3. 无锁编程

### 3.1 Interlocked

```csharp
private int _counter;

// 原子操作
Interlocked.Increment(ref _counter);       // ++
Interlocked.Decrement(ref _counter);       // --
Interlocked.Add(ref _counter, 10);         // += 10
Interlocked.Exchange(ref _counter, 0);     // = 0
Interlocked.CompareExchange(ref _counter, 1, 0);  // if == 0 then = 1

// 原子读取
int value = Interlocked.CompareExchange(ref _counter, 0, 0);

// 64位原子操作（32位系统）
private long _longValue;
Interlocked.Read(ref _longValue);
Interlocked.Increment(ref _longValue);
```

### 3.2 Volatile

```csharp
private volatile bool _running = true;

public void Stop()
{
    _running = false;  // 立即可见
}

public void Worker()
{
    while (_running)  // 每次都从内存读取
    {
        // 工作
    }
}
```

### 3.3 Memory Barrier

```csharp
private int _a;
private int _b;
private int _c;

public void Write()
{
    _a = 1;
    _b = 2;
    Thread.MemoryBarrier();  // 确保写入顺序
    _c = 3;
}

public void Read()
{
    int c = _c;
    Thread.MemoryBarrier();  // 确保读取顺序
    int b = _b;
    int a = _a;
}
```

### 3.4 SpinLock

```csharp
private SpinLock _spinLock = new();

public void Process()
{
    bool lockTaken = false;
    try
    {
        _spinLock.Enter(ref lockTaken);
        // 临界区
    }
    finally
    {
        if (lockTaken)
            _spinLock.Exit();
    }
}
```

## 4. 并发集合

### 4.1 ConcurrentDictionary

```csharp
var dict = new ConcurrentDictionary<string, int>();

// 添加或更新
dict.AddOrUpdate("key", 1, (k, v) => v + 1);

// 获取或添加
int value = dict.GetOrAdd("key", k => ComputeValue(k));

// 尝试获取
if (dict.TryGetValue("key", out var val))
{
    // 找到
}

// 尝试移除
if (dict.TryRemove("key", out var removed))
{
    // 移除成功
}
```

### 4.2 ConcurrentQueue

```csharp
var queue = new ConcurrentQueue<int>();

// 入队
queue.Enqueue(1);
queue.Enqueue(2);

// 出队
if (queue.TryDequeue(out var item))
{
    Debug.Log(item);
}

// 查看队首
if (queue.TryPeek(out var peeked))
{
    Debug.Log(peeked);
}
```

### 4.3 ConcurrentBag

```csharp
var bag = new ConcurrentBag<int>();

// 添加
bag.Add(1);
bag.Add(2);

// 取出
if (bag.TryTake(out var item))
{
    Debug.Log(item);
}
```

### 4.4 BlockingCollection

```csharp
// 生产者-消费者
var collection = new BlockingCollection<int>(boundedCapacity: 100);

// 生产者
Task.Run(async () =>
{
    for (int i = 0; i < 1000; i++)
    {
        collection.Add(i);  // 满时阻塞
        await Task.Delay(10);
    }
    collection.CompleteAdding();
});

// 消费者
foreach (var item in collection.GetConsumingEnumerable())
{
    Process(item);  // 空时阻塞
}
```

## 5. TPL 数据流

### 5.1 ActionBlock

```csharp
using System.Threading.Tasks.Dataflow;

var actionBlock = new ActionBlock<int>(async x =>
{
    await Task.Delay(100);
    Debug.Log(x);
}, new ExecutionDataflowBlockOptions
{
    MaxDegreeOfParallelism = 4  // 最大并行度
});

// 发送数据
for (int i = 0; i < 100; i++)
{
    actionBlock.Post(i);
}

actionBlock.Complete();
await actionBlock.Completion;
```

### 5.2 TransformBlock

```csharp
var transformBlock = new TransformBlock<int, string>(x =>
{
    return x.ToString("X");  // 转十六进制
});

var actionBlock = new ActionBlock<string>(s =>
{
    Debug.Log(s);
});

// 链接
transformBlock.LinkTo(actionBlock, new DataflowLinkOptions { PropagateCompletion = true });

transformBlock.Post(255);  // 输出: FF
transformBlock.Complete();
await actionBlock.Completion;
```

## 本课小结

### 同步原语对比

| 类型 | 场景 | 性能 |
|------|------|------|
| lock | 通用互斥 | 中 |
| Monitor | 需要超时/等待 | 中 |
| ReaderWriterLockSlim | 读多写少 | 中 |
| SemaphoreSlim | 限制并发数 | 中 |
| SpinLock | 极短临界区 | 快（CPU密集） |
| Interlocked | 简单原子操作 | 最快 |

### 最佳实践

1. **优先使用高层抽象** - Task、Parallel、PLINQ
2. **最小化锁范围** - 只锁必要的代码
3. **避免嵌套锁** - 死锁风险
4. **使用并发集合** - 减少手动同步
5. **考虑无锁设计** - Interlocked、Volatile
