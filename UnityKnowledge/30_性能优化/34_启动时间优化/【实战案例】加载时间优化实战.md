## 3. 解决方案

### 3.1 迁移到 Addressables

**问题**：Resources.Load 同步加载阻塞主线程

**方案**：使用 Addressables 异步加载

#### 步骤1：资源迁移

```csharp
/// <summary>
/// 资源加载服务 - Addressables封装
/// </summary>
public class AssetLoader : MonoBehaviour
{
    private Dictionary<string, AsyncOperationHandle> loadedHandles = new Dictionary<string, AsyncOperationHandle>();

    /// <summary>
    /// 异步加载单个资源
    /// </summary>
    public async Task<T> LoadAssetAsync<T>(string address) where T : Object
    {
        // 检查缓存
        if (loadedHandles.TryGetValue(address, out var existingHandle))
        {
            if (existingHandle.IsDone)
            {
                return existingHandle.Result as T;
            }
            await existingHandle.Task;
            return existingHandle.Result as T;
        }

        // 异步加载
        var handle = Addressables.LoadAssetAsync<T>(address);
        loadedHandles[address] = handle;

        await handle.Task;

        if (handle.Status == AsyncOperationStatus.Succeeded)
        {
            return handle.Result;
        }

        Debug.LogError($"Failed to load asset: {address}");
        return null;
    }

    /// <summary>
    /// 批量异步加载
    /// </summary>
    public async Task<List<T>> LoadAssetsAsync<T>(IList<string> addresses) where T : Object
    {
        var results = new List<T>();
        var tasks = new List<Task<T>>();

        foreach (var address in addresses)
        {
            tasks.Add(LoadAssetAsync<T>(address));
        }

        var loadedAssets = await Task.WhenAll(tasks);
        results.AddRange(loadedAssets.Where(a => a != null));

        return results;
    }

    /// <summary>
    /// 释放资源
    /// </summary>
    public void ReleaseAsset(string address)
    {
        if (loadedHandles.TryGetValue(address, out var handle))
        {
            Addressables.Release(handle);
            loadedHandles.Remove(address);
        }
    }

    /// <summary>
    /// 释放所有资源
    /// </summary>
    public void ReleaseAll()
    {
        foreach (var handle in loadedHandles.Values)
        {
            Addressables.Release(handle);
        }
        loadedHandles.Clear();
    }
}
```

#### 步骤2：重构UI资源加载

```csharp
/// <summary>
/// UI管理器 - 异步加载版本
/// </summary>
public class UIManager : MonoBehaviour
{
    [SerializeField] private AssetLoader assetLoader;

    private Dictionary<string, GameObject> cachedPanels = new Dictionary<string, GameObject>();

    /// <summary>
    /// 异步打开面板
    /// </summary>
    public async Task<T> OpenPanelAsync<T>(string panelAddress) where T : Component
    {
        // 检查缓存
        if (cachedPanels.TryGetValue(panelAddress, out var cachedPanel))
        {
            cachedPanel.SetActive(true);
            return cachedPanel.GetComponent<T>();
        }

        // 异步加载
        var prefab = await assetLoader.LoadAssetAsync<GameObject>(panelAddress);
        if (prefab == null) return null;

        // 实例化
        var panel = Instantiate(prefab, transform);
        cachedPanels[panelAddress] = panel;

        return panel.GetComponent<T>();
    }

    /// <summary>
    /// 关闭面板（回收到缓存）
    /// </summary>
    public void ClosePanel(string panelAddress)
    {
        if (cachedPanels.TryGetValue(panelAddress, out var panel))
        {
            panel.SetActive(false);
        }
    }
}
```

### 3.2 异步场景加载

**问题**：SceneManager.LoadScene 同步阻塞

**方案**：使用 LoadSceneAsync + 进度显示

```csharp
/// <summary>
/// 场景管理器 - 异步加载版本
/// </summary>
public class SceneLoader : MonoBehaviour
{
    [Header("Loading UI")]
    [SerializeField] private GameObject loadingScreen;
    [SerializeField] private Slider progressBar;
    [SerializeField] private Text progressText;

    private AsyncOperation currentOperation;

    /// <summary>
    /// 异步加载场景（带进度显示）
    /// </summary>
    public async Task LoadSceneAsync(string sceneName)
    {
        // 显示加载界面
        loadingScreen.SetActive(true);
        UpdateProgress(0f);

        // 开始异步加载
        currentOperation = SceneManager.LoadSceneAsync(sceneName);
        currentOperation.allowSceneActivation = false;

        // 更新进度
        while (!currentOperation.isDone)
        {
            // AsyncOperation.progress 在 allowSceneActivation=false 时最大到0.9
            float progress = Mathf.Clamp01(currentOperation.progress / 0.9f);
            UpdateProgress(progress);

            // 进度达到90%时，准备激活场景
            if (currentOperation.progress >= 0.9f)
            {
                UpdateProgress(1f);
                await Task.Delay(100); // 短暂延迟让用户看到100%
                currentOperation.allowSceneActivation = true;
            }

            await Task.Yield();
        }

        // 隐藏加载界面
        loadingScreen.SetActive(false);
    }

    /// <summary>
    /// 带预加载的场景切换
    /// </summary>
    public async Task LoadSceneWithPreload(string sceneName, List<string> preloadAssets)
    {
        loadingScreen.SetActive(true);
        UpdateProgress(0f);

        // 阶段1: 加载场景 (0% - 50%)
        var sceneTask = LoadSceneInternal(sceneName);
        var progressTask = MonitorSceneProgress();

        await sceneTask;

        // 阶段2: 预加载资源 (50% - 100%)
        UpdateProgress(0.5f);
        await PreloadAssets(preloadAssets, (p) => UpdateProgress(0.5f + p * 0.5f));

        UpdateProgress(1f);
        await Task.Delay(100);

        loadingScreen.SetActive(false);
    }

    private void UpdateProgress(float progress)
    {
        if (progressBar != null)
        {
            progressBar.value = progress;
        }
        if (progressText != null)
        {
            progressText.text = $"加载中... {(int)(progress * 100)}%";
        }
    }
}
```

### 3.3 分帧初始化

**问题**：所有Manager在Awake/Start同步初始化导致卡顿

**方案**：使用协程分帧初始化

```csharp
/// <summary>
/// 游戏启动器 - 分帧初始化
/// </summary>
public class GameLauncher : MonoBehaviour
{
    [Header("Initialization Order")]
    [SerializeField] private List<MonoBehaviour> initOrder;

    [Header("Loading UI")]
    [SerializeField] private Slider initProgress;
    [SerializeField] private Text initStatus;

    private float initStartTime;

    private async void Start()
    {
        initStartTime = Time.realtimeSinceStartup;

        // 阶段1: 核心系统初始化
        await InitializePhase("核心系统", new List<Func<Task>>
        {
            () => InitializeManager<AudioManager>(),
            () => InitializeManager<EventManager>(),
            () => InitializeManager<SaveManager>(),
        });

        // 阶段2: UI初始化
        await InitializePhase("UI系统", new List<Func<Task>>
        {
            () => InitializeManager<UIManager>(),
            () => PreloadUIAssets(),
        });

        // 阶段3: 游戏系统初始化
        await InitializePhase("游戏系统", new List<Func<Task>>
        {
            () => InitializeManager<GameManager>(),
            () => InitializeManager<CharacterManager>(),
            () => LoadLevelConfig(),
        });

        // 阶段4: 可延后初始化
        StartCoroutine(DelayedInit());
    }

    private async Task InitializePhase(string phaseName, List<Func<Task>> tasks)
    {
        UpdateStatus($"正在初始化{phaseName}...");

        for (int i = 0; i < tasks.Count; i++)
        {
            await tasks[i]();
            UpdateProgress((float)(i + 1) / tasks.Count);

            // 每个任务后等待一帧，避免卡顿
            await Task.Yield();
        }
    }

    private IEnumerator DelayedInit()
    {
        // 等待3秒后初始化非关键系统
        yield return new WaitForSeconds(3f);

        // 后台初始化分析SDK
        Task.Run(() => AnalyticsService.Initialize());

        // 后台初始化广告SDK
        Task.Run(() => AdService.Initialize());
    }

    private void UpdateStatus(string status)
    {
        if (initStatus != null)
        {
            initStatus.text = status;
        }
    }

    private void UpdateProgress(float progress)
    {
        if (initProgress != null)
        {
            initProgress.value = progress;
        }
    }

    private async Task InitializeManager<T>() where T : MonoBehaviour
    {
        var manager = FindObjectOfType<T>();
        if (manager is IAsyncInitializable asyncInit)
        {
            await asyncInit.InitializeAsync();
        }
    }
}

/// <summary>
/// 异步初始化接口
/// </summary>
public interface IAsyncInitializable
{
    Task InitializeAsync();
}
```

### 3.4 配置预加载

**问题**：JSON配置在主线程解析阻塞

**方案**：后台线程预加载 + 缓存

```csharp
/// <summary>
/// 配置管理器 - 异步加载版本
/// </summary>
public class ConfigManager : MonoBehaviour, IAsyncInitializable
{
    private Dictionary<string, object> configCache = new Dictionary<string, object>();

    public async Task InitializeAsync()
    {
        // 在后台线程加载所有配置
        var configFiles = new Dictionary<string, Type>
        {
            { "Config/levels", typeof(LevelConfigData) },
            { "Config/characters", typeof(CharacterConfigData) },
            { "Config/items", typeof(ItemConfigData) },
            { "Config/iap", typeof(IAPConfigData) },
        };

        // 并行加载
        var tasks = configFiles.Select(kvp => LoadConfigAsync(kvp.Key, kvp.Value));
        await Task.WhenAll(tasks);
    }

    private async Task LoadConfigAsync(string address, Type configType)
    {
        var jsonAsset = await Addressables.LoadAssetAsync<TextAsset>(address).Task;

        if (jsonAsset != null)
        {
            // 在后台线程解析JSON
            var config = await Task.Run(() =>
            {
                return JsonUtility.FromJson(jsonAsset.text, configType);
            });

            configCache[address] = config;
        }
    }

    /// <summary>
    /// 获取配置（同步，必须已预加载）
    /// </summary>
    public T GetConfig<T>(string address) where T : class
    {
        if (configCache.TryGetValue(address, out var config))
        {
            return config as T;
        }
        Debug.LogError($"Config not loaded: {address}");
        return null;
    }
}
```

### 3.5 智能预加载策略

**问题**：全部资源一次性加载太慢

**方案**：按优先级分层预加载

```csharp
/// <summary>
/// 智能预加载管理器
/// </summary>
public class SmartPreloader : MonoBehaviour
{
    [System.Serializable]
    public class PreloadGroup
    {
        public string groupName;
        public List<string> assetAddresses;
        public PreloadPriority priority;
        public bool showInLoading;
    }

    public enum PreloadPriority
    {
        Critical,   // 启动必须，立即加载
        High,       // 首屏需要，启动后加载
        Normal,     // 游戏中需要，后台加载
        Low         // 可选内容，空闲加载
    }

    [Header("Preload Groups")]
    [SerializeField] private List<PreloadGroup> preloadGroups;

    private Dictionary<string, float> loadProgress = new Dictionary<string, float>();

    /// <summary>
    /// 开始预加载
    /// </summary>
    public async Task StartPreload(Action<float> onProgress)
    {
        // 按优先级分组
        var criticalGroup = preloadGroups.Where(g => g.priority == PreloadPriority.Critical);
        var highGroup = preloadGroups.Where(g => g.priority == PreloadPriority.High);
        var normalGroup = preloadGroups.Where(g => g.priority == PreloadPriority.Normal);
        var lowGroup = preloadGroups.Where(g => g.priority == PreloadPriority.Low);

        int totalGroups = preloadGroups.Count;
        int completedGroups = 0;

        // 1. 加载关键资源（阻塞）
        foreach (var group in criticalGroup)
        {
            await LoadGroup(group);
            completedGroups++;
            onProgress?.Invoke((float)completedGroups / totalGroups);
        }

        // 2. 启动游戏（关键资源加载完成即可）
        // ... 允许玩家进入游戏 ...

        // 3. 后台加载高优先级资源
        _ = Task.Run(async () =>
        {
            foreach (var group in highGroup)
            {
                await LoadGroup(group);
                completedGroups++;
                MainThreadDispatcher.ExecuteOnMainThread(() =>
                    onProgress?.Invoke((float)completedGroups / totalGroups));
            }

            // 4. 空闲时加载普通资源
            foreach (var group in normalGroup)
            {
                await LoadGroup(group);
                completedGroups++;
            }

            // 5. 低优先级资源
            foreach (var group in lowGroup)
            {
                await LoadGroup(group);
                completedGroups++;
            }
        });
    }

    private async Task LoadGroup(PreloadGroup group)
    {
        var tasks = group.assetAddresses.Select(addr =>
            Addressables.LoadAssetAsync<Object>(addr).Task);

        await Task.WhenAll(tasks);
    }
}
```
