## 一、加载策略对比

### 1.1 三种加载方式

```
┌─────────────────────────────────────────────────────────────┐
│                    资源加载方式对比                          │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  1. 同步加载（Resources.Load）                               │
│     ├─ 优点：简单、直接                                       │
│     ├─ 缺点：阻塞主线程、卡顿                                │
│     └─ 适用：小资源（<1MB）                                   │
│                                                             │
│  2. 异步加载（Resources.LoadAsync）                          │
│     ├─ 优点：不阻塞主线程、可显示进度                         │
│     ├─ 缺点：需要协程/异步处理                               │
│     └─ 适用：大资源（>1MB）                                   │
│                                                             │
│  3. Addressables加载                                         │
│     ├─ 优点：异步、可管理、支持热更新                         │
│     ├─ 缺点：学习成本高、配置复杂                             │
│     └─ 适用：大型项目、需要热更新                             │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

## 二、异步加载实现

### 2.1 基础异步加载

```csharp
using System.Collections;
using UnityEngine;

/// <summary>
/// 资源加载器 - 基础版
/// </summary>
public class ResourceLoader : MonoBehaviour
{
    /// <summary>
    /// 异步加载单个资源
    /// </summary>
    public IEnumerator LoadAssetAsync<T>(string path, System.Action<T> onComplete) where T : Object
    {
        var loadOperation = Resources.LoadAsync<T>(path);

        while (!loadOperation.isDone)
        {
            // 可在这里更新进度条
            Debug.Log($"加载进度: {loadOperation.progress * 100}%");
            yield return null;
        }

        if (loadOperation.asset != null)
        {
            onComplete?.Invoke(loadOperation.asset as T);
        }
        else
        {
            Debug.LogError($"加载失败: {path}");
        }
    }

    /// <summary>
    /// 异步加载多个资源
    /// </summary>
    public IEnumerator LoadAssetsAsync<T>(string[] paths, System.Action<T[]> onComplete) where T : Object
    {
        var assets = new T[paths.Length];
        var loadOperations = new ResourceRequest[paths.Length];

        // 同时开始所有加载
        for (int i = 0; i < paths.Length; i++)
        {
            loadOperations[i] = Resources.LoadAsync<T>(paths[i]);
        }

        // 等待所有加载完成
        bool allDone = false;
        while (!allDone)
        {
            allDone = true;
            float totalProgress = 0f;

            for (int i = 0; i < loadOperations.Length; i++)
            {
                if (!loadOperations[i].isDone)
                {
                    allDone = false;
                }
                totalProgress += loadOperations[i].progress;
            }

            float avgProgress = totalProgress / loadOperations.Length;
            Debug.Log($"总加载进度: {avgProgress * 100}%");

            yield return null;
        }

        // 收集结果
        for (int i = 0; i < paths.Length; i++)
        {
            assets[i] = loadOperations[i].asset as T;
        }

        onComplete?.Invoke(assets);
    }
}
```

### 2.2 使用示例

```csharp
public class GameManager : MonoBehaviour
{
    private ResourceLoader loader;

    IEnumerator Start()
    {
        loader = GetComponent<ResourceLoader>();

        // 加载单个资源
        yield return StartCoroutine(loader.LoadAssetAsync<Texture2D>("Textures/Background",
            texture => {
                Debug.Log($"加载完成: {texture.name}");
            }));

        // 加载多个资源
        string[] paths = {
            "Textures/Icon1",
            "Textures/Icon2",
            "Textures/Icon3"
        };

        yield return StartCoroutine(loader.LoadAssetsAsync<Texture2D>(paths,
            textures => {
                Debug.Log($"加载完成: {textures.Length}个纹理");
            }));
    }
}
```

## 三、分阶段加载策略

### 3.2 分阶段加载实现

```csharp
using System;
using System.Collections;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 资源优先级
/// </summary>
public enum AssetPriority
{
    P0_Immediate,  // 立即加载
    P1_High,       // 优先加载
    P2_Normal,     // 延迟加载
    P3_Low         // 按需加载
}

/// <summary>
/// 分阶段资源加载器
/// </summary>
public class PhasedAssetLoader : MonoBehaviour
{
    [System.Serializable]
    public class AssetGroup
    {
        public AssetPriority priority;
        public string[] paths;
    }

    [Header("资源配置")]
    [SerializeField] private AssetGroup[] assetGroups;

    public event Action<float> OnProgressChanged;
    public event Action OnLoadComplete;

    /// <summary>
    /// 分阶段加载
    /// </summary>
    public IEnumerator LoadAllPhased()
    {
        float totalProgress = 0f;
        float groupWeight = 1f / assetGroups.Length;

        foreach (var group in assetGroups)
        {
            yield return StartCoroutine(LoadGroup(group, progress =>
            {
                float weightedProgress = progress * groupWeight;
                OnProgressChanged?.Invoke(totalProgress + weightedProgress);
            }));

            totalProgress += groupWeight;
        }

        OnLoadComplete?.Invoke();
    }

    private IEnumerator LoadGroup(AssetGroup group, Action<float> onProgress)
    {
        int total = group.paths.Length;
        int loaded = 0;

        foreach (string path in group.paths)
        {
            var loadOp = Resources.LoadAsync<GameObject>(path);

            while (!loadOp.isDone)
            {
                float groupProgress = (float)loaded / total + loadOp.progress / total;
                onProgress?.Invoke(groupProgress);
                yield return null;
            }

            loaded++;
            onProgress?.Invoke((float)loaded / total);
        }
    }
}
```

### 3.3 使用示例

```csharp
public class GameInitializer : MonoBehaviour
{
    [SerializeField] private PhasedAssetLoader assetLoader;
    [SerializeField] private LoadingScreen loadingScreen;

    IEnumerator Start()
    {
        // 订阅进度事件
        assetLoader.OnProgressChanged += progress =>
        {
            loadingScreen.UpdateProgress(progress);
        };

        assetLoader.OnLoadComplete += () =>
        {
            loadingScreen.Hide();
            EnterGame();
        };

        // 开始分阶段加载
        yield return StartCoroutine(assetLoader.LoadAllPhased());
    }

    void EnterGame()
    {
        Debug.Log("游戏初始化完成");
        // 启动游戏逻辑
    }
}
```

## 四、进度反馈实现

### 4.1 优雅的加载界面

```csharp
using UnityEngine;
using UnityEngine.UI;

public class LoadingScreen : MonoBehaviour
{
    [Header("UI元素")]
    [SerializeField] private Slider progressBar;
    [SerializeField] private Text progressText;
    [SerializeField] private Text tipText;
    [SerializeField] private Image backgroundImage;

    [Header("提示文本")]
    [SerializeField] private string[] tips;

    [Header("动画")]
    [SerializeField] private float barSmoothSpeed = 5f;

    private float displayedProgress;
    private float targetProgress;

    private void OnEnable()
    {
        // 随机显示提示
        if (tips != null && tips.Length > 0)
        {
            tipText.text = tips[UnityEngine.Random.Range(0, tips.Length)];
        }
    }

    private void Update()
    {
        // 平滑进度条
        displayedProgress = Mathf.Lerp(displayedProgress, targetProgress, Time.deltaTime * barSmoothSpeed);

        if (progressBar != null)
        {
            progressBar.value = displayedProgress;
        }

        if (progressText != null)
        {
            progressText.text = $"{(int)(displayedProgress * 100)}%";
        }
    }

    /// <summary>
    /// 更新进度
    /// </summary>
    public void UpdateProgress(float progress)
    {
        targetProgress = Mathf.Clamp01(progress);
    }

    /// <summary>
    /// 隐藏加载界面
    /// </summary>
    public void Hide()
    {
        // 播放淡出动画
        gameObject.SetActive(false);
    }
}
```

### 4.2 真实进度 vs 假进度

```csharp
// ❌ 假进度：不反映真实加载状态
void FakeProgress()
{
    float progress = 0f;
    while (progress < 1f)
    {
        progress += Time.deltaTime * 0.5f;  // 假进度
        progressBar.value = progress;
    }
}

// ✅ 真实进度：反映实际加载状态
void RealProgress(ResourceRequest loadOp)
{
    while (!loadOp.isDone)
    {
        progressBar.value = loadOp.progress;  // 真实进度
    }
}
```

**建议**：
- 首次加载使用真实进度
- 后续加载可以混合真实和假进度（避免进度条卡顿）

## 五、后台预加载

### 5.1 场景闲置时预加载

```csharp
/// <summary>
/// 后台预加载管理器
/// </summary>
public class BackgroundPreloader : MonoBehaviour
{
    private Queue<string> preloadQueue = new Queue<string>();
    private bool isPreloading = false;

    /// <summary>
    /// 添加预加载任务
    /// </summary>
    public void AddPreloadTask(string path)
    {
        preloadQueue.Enqueue(path);
    }

    /// <summary>
    /// 在合适时机执行预加载
    /// </summary>
    public void TryStartPreload()
    {
        if (!isPreloading && preloadQueue.Count > 0)
        {
            StartCoroutine(PreloadNext());
        }
    }

    private IEnumerator PreloadNext()
    {
        isPreloading = true;

        while (preloadQueue.Count > 0)
        {
            string path = preloadQueue.Dequeue();
            var loadOp = Resources.LoadAsync<GameObject>(path);

            while (!loadOp.isDone)
            {
                // 如果游戏进入忙碌状态，暂停预加载
                if (IsGameBusy())
                {
                    yield return new WaitUntil(() => !IsGameBusy());
                }

                yield return null;
            }
        }

        isPreloading = false;
    }

    private bool IsGameBusy()
    {
        // 判断游戏是否忙碌（例如：战斗中、动画播放中）
        return false;  // 根据实际情况实现
    }
}
```

## 六、Addressables加载

### 6.1 为什么使用Addressables

```
传统Resources加载的问题：
├─ 所有资源打包进游戏
├─ 无法热更新
├─ 内存占用高
└─ 加载时间长

Addressables的优势：
├─ 资源可远程下载
├─ 支持热更新
├─ 内存管理更灵活
├─ 异步加载
└─ 支持依赖管理
```

---

### 6.2 Addressables基础使用

```csharp
using System.Collections;
using UnityEngine;
using UnityEngine.Addressables;
using UnityEngine.ResourceManagement.AsyncOperations;

public class AddressablesLoader : MonoBehaviour
{
    /// <summary>
    /// 加载单个资源
    /// </summary>
    public IEnumerator LoadAssetAsync(string address, System.Action<GameObject> onComplete)
    {
        AsyncOperationHandle<GameObject> handle = Addressables.LoadAssetAsync<GameObject>(address);

        while (!handle.IsDone)
        {
            Debug.Log($"加载进度: {handle.PercentComplete * 100}%");
            yield return null;
        }

        if (handle.Status == AsyncOperationStatus.Succeeded)
        {
            onComplete?.Invoke(handle.Result);
        }
        else
        {
            Debug.LogError($"加载失败: {address}");
        }

        // 使用完后释放
        // Addressables.Release(handle);
    }

    /// <summary>
    /// 加载场景
    /// </summary>
    public IEnumerator LoadSceneAsync(string address)
    {
        var handle = Addressables.LoadSceneAsync(address);

        while (!handle.IsDone)
        {
            Debug.Log($"场景加载进度: {handle.PercentComplete * 100}%");
            yield return null;
        }
    }
}
```

## 七、最佳实践总结

### ✅ DO

1. **使用异步加载**
   - 大于1MB的资源必须异步加载
   - 显示加载进度

2. **分阶段加载**
   - 按优先级分批加载
   - 立即加载：核心UI和逻辑
   - 延迟加载：非关键资源

3. **显示真实进度**
   - 使用真实加载进度
   - 平滑进度条动画

4. **后台预加载**
   - 在空闲时预加载
   - 不要影响当前体验

5. **资源管理**
   - 及时释放不需要的资源
   - 使用对象池复用

### ❌ DON'T

1. **不要在主线程同步加载大资源**
   ```csharp
   // ❌ 错误
   var bigTexture = Resources.Load<Texture2D>("BigTexture");  // 50MB
   ```

2. **不要一次性加载所有资源**
   ```csharp
   // ❌ 错误
   for (int i = 0; i < 1000; i++)
   {
       Resources.Load("Icon" + i);  // 一次性加载1000个资源
   }
   ```

3. **不要忽略加载错误**
   ```csharp
   // ✅ 正确
   if (asset == null)
   {
       Debug.LogError("加载失败");
       return;
   }
   ```

4. **不要忘记释放资源**
   ```csharp
   // ✅ 正确
   Resources.UnloadAssetUnusedAssets();
   ```
