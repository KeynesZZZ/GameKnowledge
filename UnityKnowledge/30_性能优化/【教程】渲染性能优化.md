### 1.2 批处理类型对比

```
┌─────────────────────────────────────────────────────────────┐
│                    批处理类型对比                            │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  静态批处理 (Static Batching)                               │
│  ├── 适用：永不移动的物体                                   │
│  ├── 条件：标记为Static，使用相同材质                       │
│  ├── 优点：合并后不受顶点数限制                             │

│                                                             │
│  动态批处理 (Dynamic Batching)                              │
│  ├── 适用：小型移动物体                                     │
│  ├── 条件：≤900顶点，相同材质，统一缩放                     │
│  ├── 优点：无需预先设置                                     │
│  ├── 缺点：每帧CPU开销合并顶点                              │
│  └── 设置：Player Settings > Dynamic Batching               │
│                                                             │

│  ├── 适用：大量相同网格不同属性                             │
│  ├── 条件：Shader支持，材质开启Instancing                   │
```

### 1.3 UI DrawCall优化

```csharp
using UnityEngine;
using UnityEngine.UI;

/// <summary>
/// UI DrawCall优化器
/// </summary>
public class UIDrawCallOptimizer : MonoBehaviour
{
    /*
    ========== UI合批规则 ==========

    1. 相同材质/图集
    2. 相同层级（Canvas下）
    3. 没有被其他元素打断
    4. 正确的渲染顺序

    ========== 打断合批的常见原因 ==========

    1. 不同图集的Image交叉排列
    2. Text组件穿插在Image之间
    3. 使用了Mask组件
    4. 不同材质的UI元素
    5. 激活/禁用组件改变层级
    */

    /// <summary>
    /// 分析Canvas的DrawCall情况
    /// </summary>
    public static void AnalyzeCanvas(Canvas canvas)
    {
        var graphics = canvas.GetComponentsInChildren<Graphic>(true);

        System.Collections.Generic.Dictionary<string, int> atlasCount =
            new System.Collections.Generic.Dictionary<string, int>();

        foreach (var graphic in graphics)
        {
            if (graphic is Image image && image.sprite != null)
            {
                string atlasName = image.sprite.texture?.name ?? "None";
                if (!atlasCount.ContainsKey(atlasName))
                    atlasCount[atlasName] = 0;
                atlasCount[atlasName]++;
            }
            else if (graphic is Text text)
            {
                string fontName = text.font?.name ?? "None";
                if (!atlasCount.ContainsKey($"Font:{fontName}"))
                    atlasCount[$"Font:{fontName}"] = 0;
                atlasCount[$"Font:{fontName}"]++;
            }
        }

        Debug.Log($"===== Canvas Analysis: {canvas.name} =====");
        Debug.Log($"Total Graphics: {graphics.Length}");
        Debug.Log($"Potential Atlases/Fonts: {atlasCount.Count}");

        foreach (var kvp in atlasCount)
        {
            Debug.Log($"  {kvp.Key}: {kvp.Value} elements");
        }
    }

    /// <summary>
    /// 重新排序UI元素以优化合批
    /// </summary>
    public static void ReorderForBatching(RectTransform parent)
    {
        var children = new System.Collections.Generic.List<RectTransform>();
        foreach (Transform child in parent)
        {
            children.Add(child as RectTransform);
        }

        // 按材质/图集分组排序
        children.Sort((a, b) =>
        {
            var graphicA = a.GetComponent<Graphic>();
            var graphicB = b.GetComponent<Graphic>();

            string keyA = GetSortingKey(graphicA);
            string keyB = GetSortingKey(graphicB);

            return string.Compare(keyA, keyB, System.StringComparison.Ordinal);
        });

        // 重新设置兄弟索引
        for (int i = 0; i < children.Count; i++)
        {
            children[i].SetSiblingIndex(i);
        }
    }

    private static string GetSortingKey(Graphic graphic)
    {
        if (graphic is Image image && image.sprite != null)
        {
            return image.sprite.texture?.name ?? "Image_None";
        }
        if (graphic is Text text)
        {
            return $"Text_{text.font?.name ?? "None"}";
        }
        return graphic.GetType().Name;
    }
}
```

### 1.4 图集管理

```csharp
using UnityEngine;
using UnityEngine.U2D;
using UnityEngine.UI;
using System.Collections.Generic;

/// <summary>
/// 图集管理器
/// </summary>
public class AtlasManager : MonoBehaviour
{
    private static AtlasManager instance;
    public static AtlasManager Instance => instance;

    [Header("Atlases")]
    [SerializeField] private SpriteAtlas[] atlases;

    private Dictionary<string, SpriteAtlas> atlasDict = new Dictionary<string, SpriteAtlas>();
    private Dictionary<string, Sprite> spriteCache = new Dictionary<string, Sprite>();

    private void Awake()
    {
        instance = this;

        foreach (var atlas in atlases)
        {
            if (atlas != null)
            {
                atlasDict[atlas.name] = atlas;
            }
        }
    }

    /// <summary>
    /// 从图集获取Sprite
    /// </summary>
    public Sprite GetSprite(string atlasName, string spriteName)
    {
        string cacheKey = $"{atlasName}_{spriteName}";

        if (spriteCache.TryGetValue(cacheKey, out var cached))
            return cached;

        if (atlasDict.TryGetValue(atlasName, out var atlas))
        {
            var sprite = atlas.GetSprite(spriteName);
            if (sprite != null)
            {
                spriteCache[cacheKey] = sprite;
                return sprite;
            }
        }

        Debug.LogWarning($"Sprite not found: {atlasName}/{spriteName}");
        return null;
    }

    /// <summary>
    /// 预加载图集
    /// </summary>
    public void PreloadAtlas(string atlasName)
    {
        if (atlasDict.TryGetValue(atlasName, out var atlas))
        {
            // 强制加载所有Sprite
            Sprite[] sprites = new Sprite[atlas.spriteCount];
            atlas.GetSprites(sprites);

            foreach (var sprite in sprites)
            {
                string cacheKey = $"{atlasName}_{sprite.name}";
                if (!spriteCache.ContainsKey(cacheKey))
                {
                    spriteCache[cacheKey] = sprite;
                }
            }
        }
    }

    /// <summary>
    /// 清理Sprite缓存
    /// </summary>
    public void ClearCache()
    {
        spriteCache.Clear();
    }
}

/// <summary>
/// 动态图集生成器
/// </summary>
public class DynamicAtlasGenerator
{
    /*
    ========== 图集打包最佳实践 ==========

    1. 按功能模块分组
       - UI_Common: 通用UI元素
       - UI_Game: 游戏内UI
       - Icons: 图标
       - Items: 物品图片

    2. 控制图集大小
       - 推荐: 2048x2048
       - 最大: 4096x4096（部分老设备不支持）

    3. 打包设置
       - Format: ASTC 4x4 (移动端)
       - Compression: 高质量
       - Include in Build: 是

    4. 避免常见问题
       - 不要把大图和小图混在一起
       - 避免空隙过大
       - 使用Tight Packing
    */
}
```

### 2.2 Shader变体管理

```csharp
using UnityEngine;
using UnityEngine.Rendering;

/// <summary>
/// Shader变体管理器
/// </summary>
public class ShaderVariantManager : MonoBehaviour
{
    /*
    ========== Shader变体问题 ==========

    每个关键字组合都会生成一个变体：
    _NORMALMAP * _ALPHATEST * _EMISSION = 8个变体

    变体过多会导致：
    1. 构建时间增加
    2. 内存占用增加
    3. 加载时间增加

    ========== 优化策略 ==========

    1. 限制关键字数量
       - 每个Shader最多 < 64个变体
       - 使用 shader_feature 代替 multi_compile（不用的变体不打包）

    2. 使用Shader Variant Collection
       - 预热常用变体
       - 只打包实际使用的变体

    3. 按需加载
       - 分离复杂Shader到独立AssetBundle
    */

    [Header("Variant Collections")]
    [SerializeField] private ShaderVariantCollection[] collections;

    /// <summary>
    /// 预热Shader变体
    /// </summary>
    public void WarmupShaders()
    {
        foreach (var collection in collections)
        {
            if (collection != null)
            {
                collection.WarmUp();
                Debug.Log($"Warmed up: {collection.name}");
            }
        }
    }

    /// <summary>
    /// 获取Shader变体数量
    /// </summary>
    public static int GetVariantCount(Shader shader)
    {
        int count = 0;
        var keywordSpace = new UnityEngine.Rendering.LocalKeywordSpace(shader);
        // 简化：实际需要遍历所有Pass和Keyword组合
        return count;
    }
}
```

```hlsl
        _Emission ("Emission", Color) = (0,0,0,1)
        _RimColor ("Rim Color", Color) = (1,1,1,1)
        _RimPower ("Rim Power", Range(1,10)) = 3
    }

    // 多个Pass、复杂光照...
}

// ========== 优化后：移动端简化Shader ==========
Shader "Custom/Mobile/Gem"
{
    Properties
    {
        _MainTex ("Texture", 2D) = "white" {}
        _Color ("Color", Color) = (1,1,1,1)
        [HDR] _Emission ("Emission", Color) = (0,0,0,1)
    }

    SubShader
    {
        Tags { "RenderType"="Opaque" "RenderPipeline"="UniversalPipeline" }
        LOD 100

        Pass
        {
            HLSLPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #pragma multi_compile_instancing

            #include "Packages/com.unity.render-pipelines.universal/ShaderLibrary/Core.hlsl"

            struct Attributes
            {
                float4 positionOS : POSITION;
                float2 uv : TEXCOORD0;
                UNITY_VERTEX_INPUT_INSTANCE_ID
            };

            struct Varyings
            {
                float4 positionCS : SV_POSITION;
                float2 uv : TEXCOORD0;
                UNITY_VERTEX_INPUT_INSTANCE_ID
            };

            TEXTURE2D(_MainTex);
            SAMPLER(sampler_MainTex);

            CBUFFER_START(UnityPerMaterial)
                float4 _MainTex_ST;
                float4 _Color;
                float4 _Emission;
            CBUFFER_END

            Varyings vert(Attributes input)
            {
                Varyings output;
                UNITY_SETUP_INSTANCE_ID(input);
                UNITY_TRANSFER_INSTANCE_ID(input, output);

                output.positionCS = TransformObjectToHClip(input.positionOS.xyz);
                output.uv = TRANSFORM_TEX(input.uv, _MainTex);
                return output;
            }

            half4 frag(Varyings input) : SV_Target
            {
                UNITY_SETUP_INSTANCE_ID(input);

                half4 texColor = SAMPLE_TEXTURE2D(_MainTex, sampler_MainTex, input.uv);
                half3 color = texColor.rgb * _Color.rgb + _Emission.rgb;

                return half4(color, texColor.a * _Color.a);
            }
            HLSLPROGRAM
        }
    }

    // 简化版LOD
    SubShader
    {
        Tags { "RenderType"="Opaque" }
        LOD 50

        Pass
        {
            // 超低配置版本
        }
    }
}
```

### 3.3 粒子系统Overdraw优化

```csharp
using UnityEngine;

/// <summary>
/// 粒子Overdraw优化
/// </summary>
public class ParticleOverdrawOptimizer : MonoBehaviour
{
    /*
    ========== 粒子Overdraw问题 ==========

    粒子通常是Overdraw的主要来源：
    - 大量重叠的粒子
    - 过大的粒子尺寸
    - 高发射率

    ========== 优化策略 ==========

    1. 减少粒子数量
    2. 缩小粒子尺寸
    3. 使用加法混合（减少视觉影响）
    4. 使用GPU粒子
    5. LOD系统
    */

    [Header("Optimization Settings")]
    [SerializeField] private int maxParticlesMobile = 100;
    [SerializeField] private int maxParticlesDesktop = 500;

    /// <summary>
    /// 优化粒子系统设置
    /// </summary>
    public void OptimizeParticleSystem(ParticleSystem ps, bool isMobile)
    {
        var main = ps.main;
        var emission = ps.emission;

        // 限制最大粒子数
        int maxParticles = isMobile ? maxParticlesMobile : maxParticlesDesktop;
        main.maxParticles = Mathf.Min(main.maxParticles, maxParticles);

        // 移动端减少发射率
        if (isMobile)
        {
            var rateOverTime = emission.rateOverTime;
            rateOverTime.constant *= 0.5f;
            emission.rateOverTime = rateOverTime;
        }

        // 优化渲染设置
        var renderer = ps.GetComponent<ParticleSystemRenderer>();
        if (renderer != null)
        {
            // 使用更简单的Shader
            // renderer.material = optimizedMaterial;
        }
    }

    /// <summary>
    /// 粒子LOD系统
    /// </summary>
    public class ParticleLOD : MonoBehaviour
    {
        [System.Serializable]
        public class LODLevel
        {
            public float distance;
            public int maxParticles;
            public float emissionRate;
        }

        [SerializeField] private LODLevel[] lodLevels;
        [SerializeField] private ParticleSystem targetParticle;

        private Transform cameraTransform;

        private void Start()
        {
            cameraTransform = Camera.main.transform;
        }

        private void Update()
        {
            float distance = Vector3.Distance(transform.position, cameraTransform.position);

            // 找到合适的LOD级别
            LODLevel currentLOD = lodLevels[lodLevels.Length - 1];
            foreach (var lod in lodLevels)
            {
                if (distance < lod.distance)
                {
                    currentLOD = lod;
                    break;
                }
            }

            // 应用LOD设置
            var main = targetParticle.main;
            main.maxParticles = currentLOD.maxParticles;

            var emission = targetParticle.emission;
            var rate = emission.rateOverTime;
            rate.constant = currentLOD.emissionRate;
            emission.rateOverTime = rate;
        }
    }
}
```

## 4. LOD系统

### 4.1 LOD Group配置

```csharp
using UnityEngine;

/// <summary>
/// LOD系统管理器
/// </summary>
public class LODManager : MonoBehaviour
{
    /*
    ========== LOD (Level of Detail) ==========

    根据距离使用不同精度的模型：

    LOD 0 (近距离): 高精度模型
    LOD 1 (中距离): 中精度模型
    LOD 2 (远距离): 低精度模型
    LOD 3 (超远):   billboard或禁用

    ========== 设置建议 ==========

    移动端：
    - LOD 0: 0-10m
    - LOD 1: 10-25m
    - LOD 2: 25-50m
    - Cull: 50m+

    PC：
    - LOD 0: 0-20m
    - LOD 1: 20-50m
    - LOD 2: 50-100m
    - Cull: 100m+
    */

    [Header("LOD Settings")]
    [SerializeField] private bool useLODForMobile = true;
    [SerializeField] private float lodBiasMobile = 0.7f;
    [SerializeField] private float lodBiasDesktop = 1.0f;

    private void Start()
    {
        ApplyLODSettings();
    }

    private void ApplyLODSettings()
    {
        bool isMobile = Application.isMobilePlatform;

        // 设置LOD Bias
        QualitySettings.lodBias = isMobile ? lodBiasMobile : lodBiasDesktop;

        // 设置最大LOD级别
        QualitySettings.maximumLODLevel = 0;
    }

    /// <summary>
    /// 为GameObject自动创建LOD Group
    /// </summary>
    public static LODGroup CreateLODGroup(GameObject obj, Mesh[] lodMeshes, float[] lodDistances)
    {
        var lodGroup = obj.GetComponent<LODGroup>();
        if (lodGroup == null)
            lodGroup = obj.AddComponent<LODGroup>();

        var lods = new LOD[lodMeshes.Length];
        var renderer = obj.GetComponent<MeshRenderer>();

        for (int i = 0; i < lodMeshes.Length; i++)
        {
            // 创建子对象包含不同LOD网格
            var lodObj = new GameObject($"LOD_{i}");
            lodObj.transform.SetParent(obj.transform);
            lodObj.AddComponent<MeshFilter>().sharedMesh = lodMeshes[i];
            var lodRenderer = lodObj.AddComponent<MeshRenderer>();

            // 复制材质
            lodRenderer.sharedMaterials = renderer.sharedMaterials;

            // 设置LOD
            lods[i] = new LOD(lodDistances[i], new Renderer[] { lodRenderer });
        }

        lodGroup.SetLODs(lods);
        lodGroup.RecalculateBounds();

        return lodGroup;
    }
}
```

## 5. 三消游戏渲染优化实战

### 5.1 棋盘渲染优化

```csharp
using UnityEngine;
using UnityEngine.Rendering;

/// <summary>
/// 三消棋盘渲染优化器
/// </summary>
public class BoardRenderOptimizer : MonoBehaviour
{
    [Header("References")]
    [SerializeField] private Mesh gemMesh;
    [SerializeField] private Material gemMaterial;

    [Header("Settings")]
    [SerializeField] private int boardSize = 8;
    [SerializeField] private bool useGPUInstancing = true;

    private Matrix4x4[] matrices;
    private MaterialPropertyBlock propertyBlock;
    private Vector4[] colors;
    private float[] highlightStrength;

    private const int BATCH_SIZE = 1023;

    private void Start()
    {
        InitializeInstancing();
    }

    private void InitializeInstancing()
    {
        int totalGems = boardSize * boardSize;
        matrices = new Matrix4x4[totalGems];
        colors = new Vector4[totalGems];
        highlightStrength = new float[totalGems];
        propertyBlock = new MaterialPropertyBlock();

        // 初始化矩阵和颜色
        int index = 0;
        for (int x = 0; x < boardSize; x++)
        {
            for (int y = 0; y < boardSize; y++)
            {
                matrices[index] = Matrix4x4.TRS(
                    new Vector3(x, y, 0),
                    Quaternion.identity,
                    Vector3.one * 0.9f
                );
                colors[index] = new Vector4(1, 1, 1, 1);
                highlightStrength[index] = 0f;
                index++;
            }
        }
    }

    private void Update()
    {
        if (useGPUInstancing)
        {
            RenderWithInstancing();
        }
    }

    private void RenderWithInstancing()
    {
        // 更新属性块
        propertyBlock.SetVectorArray("_Color", colors);
        propertyBlock.SetFloatArray("_Highlight", highlightStrength);

        // 分批渲染
        int totalGems = boardSize * boardSize;
        int batchCount = (totalGems + BATCH_SIZE - 1) / BATCH_SIZE;

        for (int batch = 0; batch < batchCount; batch++)
        {
            int startIndex = batch * BATCH_SIZE;
            int count = Mathf.Min(BATCH_SIZE, totalGems - startIndex);

            // 提取当前批次的矩阵
            Matrix4x4[] batchMatrices = new Matrix4x4[count];
            System.Array.Copy(matrices, startIndex, batchMatrices, 0, count);

            Graphics.DrawMeshInstanced(
                gemMesh,
                0,
                gemMaterial,
                batchMatrices,
                count,
                propertyBlock,
                ShadowCastingMode.Off,
                false,
                gameObject.layer
            );
        }
    }

    /// <summary>
    /// 更新单个棋子的颜色
    /// </summary>
    public void SetGemColor(int x, int y, Color color)
    {
        int index = y * boardSize + x;
        if (index >= 0 && index < colors.Length)
        {
            colors[index] = new Vector4(color.r, color.g, color.b, color.a);
        }
    }

    /// <summary>
    /// 设置棋子高亮
    /// </summary>
    public void SetGemHighlight(int x, int y, float strength)
    {
        int index = y * boardSize + x;
        if (index >= 0 && index < highlightStrength.Length)
        {
            highlightStrength[index] = strength;
        }
    }
}
```

### 5.2 特效渲染优化

```csharp
using UnityEngine;
using System.Collections.Generic;

/// <summary>
/// 特效渲染优化器
/// </summary>
public class EffectRenderOptimizer : MonoBehaviour
{
    [Header("Pool Settings")]
    [SerializeField] private int poolSize = 50;
    [SerializeField] private ParticleSystem matchEffectPrefab;

    private Queue<ParticleSystem> effectPool = new Queue<ParticleSystem>();
    private List<ParticleSystem> activeEffects = new List<ParticleSystem>();

    private void Start()
    {
        InitializePool();
    }

    private void InitializePool()
    {
        for (int i = 0; i < poolSize; i++)
        {
            var effect = Instantiate(matchEffectPrefab, transform);
            effect.gameObject.SetActive(false);
            effectPool.Enqueue(effect);
        }
    }

    /// <summary>
    /// 播放消除特效
    /// </summary>
    public void PlayMatchEffect(Vector3 position)
    {
        ParticleSystem effect;

        if (effectPool.Count > 0)
        {
            effect = effectPool.Dequeue();
        }
        else
        {
            // 池为空，复用最早的特效
            effect = activeEffects[0];
            activeEffects.RemoveAt(0);
        }

        effect.transform.position = position;
        effect.gameObject.SetActive(true);
        effect.Play();

        activeEffects.Add(effect);

        // 自动回收
        StartCoroutine(ReturnToPool(effect, effect.main.duration));
    }

    private System.Collections.IEnumerator ReturnToPool(ParticleSystem effect, float delay)
    {
        yield return new WaitForSeconds(delay);

        effect.gameObject.SetActive(false);
        activeEffects.Remove(effect);
        effectPool.Enqueue(effect);
    }

    /// <summary>
    /// 批量特效优化 - 合并相同类型特效
    /// </summary>
    public void PlayBatchEffects(List<Vector3> positions)
    {
        // 对于大量相同特效，考虑使用GPU粒子
        // 或合并为一个Particle System使用SubEmitters

        foreach (var pos in positions)
        {
            PlayMatchEffect(pos);
        }
    }
}
```
