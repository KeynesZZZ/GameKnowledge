## 1. 发现问题

### 1.1 问题现象

在休闲游戏项目中，发现以下问题：
- 游戏运行30分钟后内存从150MB增长到480MB
- 返回主菜单后内存不释放
- 低内存设备上频繁触发系统杀进程
- Memory Profiler显示内存持续增长趋势

## 3. 解决方案

### 3.1 解决Texture2D泄漏

**方案**：使用 Addressables + 显式卸载

```csharp
/// <summary>
/// 资源管理器 - 统一管理资源加载和卸载
/// </summary>
public class ResourceManager : MonoBehaviour
{
    private Dictionary<string, AsyncOperationHandle> loadedAssets = new Dictionary<string, AsyncOperationHandle>();

    /// <summary>
    /// 异步加载资源
    /// </summary>
    public async Task<T> LoadAssetAsync<T>(string address) where T : Object
    {
        if (loadedAssets.TryGetValue(address, out var existingHandle))
        {
            return existingHandle.Result as T;
        }

        var handle = Addressables.LoadAssetAsync<T>(address);
        await handle.Task;

        if (handle.Status == AsyncOperationStatus.Succeeded)
        {
            loadedAssets[address] = handle;
            return handle.Result;
        }

        Debug.LogError($"Failed to load asset: {address}");
        return null;
    }

    /// <summary>
    /// 卸载指定资源
    /// </summary>
    public void UnloadAsset(string address)
    {
        if (loadedAssets.TryGetValue(address, out var handle))
        {
            Addressables.Release(handle);
            loadedAssets.Remove(address);
        }
    }

    /// <summary>
    /// 卸载所有资源（场景切换时调用）
    /// </summary>
    public void UnloadAll()
    {
        foreach (var handle in loadedAssets.Values)
        {
            Addressables.Release(handle);
        }
        loadedAssets.Clear();

        // 强制GC
        Resources.UnloadUnusedAssets();
        GC.Collect();
    }

    /// <summary>
    /// 场景切换时自动清理
    /// </summary>
    private void OnSceneUnloaded(Scene scene)
    {
        // 清理该场景加载的资源
        var keysToRemove = loadedAssets.Keys
            .Where(k => k.StartsWith($"Scenes/{scene.name}"))
            .ToList();

        foreach (var key in keysToRemove)
        {
            UnloadAsset(key);
        }
    }
}
```

**使用示例**：

```csharp
public class LevelManager : MonoBehaviour
{
    [SerializeField] private ResourceManager resourceManager;

    private string currentBackgroundKey;

    public async Task LoadLevel(int levelId)
    {
        // 卸载旧资源
        if (!string.IsNullOrEmpty(currentBackgroundKey))
        {
            resourceManager.UnloadAsset(currentBackgroundKey);
        }

        // 加载新资源
        currentBackgroundKey = $"Levels/Background_{levelId}";
        var texture = await resourceManager.LoadAssetAsync<Texture2D>(currentBackgroundKey);

        if (texture != null)
        {
            backgroundImage.sprite = Sprite.Create(texture,
                new Rect(0, 0, texture.width, texture.height),
                Vector2.zero);
        }
    }
}
```

### 3.2 解决事件泄漏

**方案**：使用安全的订阅模式

```csharp
/// <summary>
/// 安全的事件订阅基类
/// </summary>
public abstract class EventSubscriber : MonoBehaviour
{
    private List<Action> unsubscribeActions = new List<Action>();

    /// <summary>
    /// 安全订阅事件
    /// </summary>
    protected void SafeSubscribe<T>(EventChannel<T> channel, Action<T> handler)
    {
        channel.Subscribe(handler);
        unsubscribeActions.Add(() => channel.Unsubscribe(handler));
    }

    /// <summary>
    /// 自动取消所有订阅
    /// </summary>
    protected virtual void OnDestroy()
    {
        foreach (var unsubscribe in unsubscribeActions)
        {
            unsubscribe?.Invoke();
        }
        unsubscribeActions.Clear();
    }
}

/// <summary>
/// 使用示例
/// </summary>
public class GameUI : EventSubscriber
{
    [SerializeField] private EventChannel<int> scoreChannel;

    protected override void Awake()
    {
        base.Awake();
        SafeSubscribe(scoreChannel, UpdateScore);
    }

    private void UpdateScore(int score)
    {
        scoreText.text = score.ToString();
    }
}
```

**方案2：使用弱引用事件系统**

```csharp
/// <summary>
/// 弱引用事件系统 - 不会阻止订阅者被GC
/// </summary>
public class WeakEventChannel<T>
{
    private List<WeakReference<Action<T>>> subscribers = new List<WeakReference<Action<T>>>();

    public void Subscribe(Action<T> handler)
    {
        subscribers.Add(new WeakReference<Action<T>>(handler));
    }

    public void Publish(T data)
    {
        // 清理已释放的订阅者
        subscribers.RemoveAll(weakRef =>
        {
            if (!weakRef.TryGetTarget(out var handler))
            {
                return true; // 移除已GC的订阅者
            }
            handler.Invoke(data);
            return false;
        });
    }
}
```

### 3.3 解决协程泄漏

**方案**：显式管理协程生命周期

```csharp
/// <summary>
/// 协程管理器 - 自动追踪和清理协程
/// </summary>
public class CoroutineTracker : MonoBehaviour
{
    private Dictionary<string, Coroutine> activeCoroutines = new Dictionary<string, Coroutine>();

    /// <summary>
    /// 启动追踪协程
    /// </summary>
    public void StartTrackedCoroutine(string key, IEnumerator routine)
    {
        // 如果已存在同名协程，先停止
        StopTrackedCoroutine(key);

        activeCoroutines[key] = StartCoroutine(routine);
    }

    /// <summary>
    /// 停止指定协程
    /// </summary>
    public void StopTrackedCoroutine(string key)
    {
        if (activeCoroutines.TryGetValue(key, out var coroutine))
        {
            StopCoroutine(coroutine);
            activeCoroutines.Remove(key);
        }
    }

    /// <summary>
    /// 停止所有协程
    /// </summary>
    public void StopAllTrackedCoroutines()
    {
        foreach (var coroutine in activeCoroutines.Values)
        {
            StopCoroutine(coroutine);
        }
        activeCoroutines.Clear();
    }

    /// <summary>
    /// 对象销毁时自动清理
    /// </summary>
    private void OnDestroy()
    {
        StopAllTrackedCoroutines();
    }
}

/// <summary>
/// 使用示例
/// </summary>
public class EffectPlayer : MonoBehaviour
{
    private CoroutineTracker tracker;

    private void Awake()
    {
        tracker = gameObject.AddComponent<CoroutineTracker>();
    }

    public void PlayEffect()
    {
        tracker.StartTrackedCoroutine("EffectLoop", PlayEffectLoop());
    }

    public void StopEffect()
    {
        tracker.StopTrackedCoroutine("EffectLoop");
    }

    private IEnumerator PlayEffectLoop()
    {
        while (true)
        {
            // 播放特效
            yield return new WaitForSeconds(1f);
        }
    }
}
```

### 3.4 解决对象池泄漏

**方案**：场景感知的对象池

```csharp
/// <summary>
/// 场景感知对象池 - 自动清理场景切换时的引用
/// </summary>
public class SceneAwarePool<T> : IDisposable where T : Component
{
    private Stack<T> pool = new Stack<T>();
    private HashSet<T> activeObjects = new HashSet<T>();
    private T prefab;
    private Transform poolContainer;
    private Scene ownerScene;

    public SceneAwarePool(T prefab, int initialSize, Scene scene)
    {
        this.prefab = prefab;
        this.ownerScene = scene;

        // 创建容器
        var containerObj = new GameObject($"Pool_{typeof(T).Name}");
        SceneManager.MoveGameObjectToScene(containerObj, scene);
        poolContainer = containerObj.transform;

        // 预热
        for (int i = 0; i < initialSize; i++)
        {
            var obj = CreateNew();
            obj.gameObject.SetActive(false);
            pool.Push(obj);
        }

        // 注册场景卸载事件
        SceneManager.sceneUnloaded += OnSceneUnloaded;
    }

    public T Get()
    {
        T obj;

        if (pool.Count > 0)
        {
            obj = pool.Pop();
        }
        else
        {
            obj = CreateNew();
        }

        obj.gameObject.SetActive(true);
        activeObjects.Add(obj);
        return obj;
    }

    public void Return(T obj)
    {
        if (activeObjects.Remove(obj))
        {
            obj.gameObject.SetActive(false);
            obj.transform.SetParent(poolContainer);
            pool.Push(obj);
        }
    }

    private T CreateNew()
    {
        var obj = GameObject.Instantiate(prefab, poolContainer);
        SceneManager.MoveGameObjectToScene(obj.gameObject, ownerScene);
        return obj;
    }

    private void OnSceneUnloaded(Scene unloadedScene)
    {
        if (unloadedScene == ownerScene)
        {
            Dispose();
        }
    }

    public void Dispose()
    {
        SceneManager.sceneUnloaded -= OnSceneUnloaded;

        // 销毁所有对象
        foreach (var obj in activeObjects)
        {
            if (obj != null) GameObject.Destroy(obj.gameObject);
        }
        foreach (var obj in pool)
        {
            if (obj != null) GameObject.Destroy(obj.gameObject);
        }

        activeObjects.Clear();
        pool.Clear();

        if (poolContainer != null)
        {
            GameObject.Destroy(poolContainer.gameObject);
        }
    }
}
```


### 5.2 自动化检测工具

```csharp
/// <summary>
/// 内存泄漏检测工具（Editor Only）
/// </summary>
#if UNITY_EDITOR
public static class MemoryLeakDetector
{
    /// <summary>
    /// 检测潜在的事件泄漏
    /// </summary>
    [MenuItem("Tools/Memory/Check Event Leaks")]
    public static void CheckEventLeaks()
    {
        var allMonoBehaviours = GameObject.FindObjectsOfType<MonoBehaviour>();
        var potentialLeaks = new List<string>();

        foreach (var mb in allMonoBehaviours)
        {
            var type = mb.GetType();

            // 检查有OnEnable但没有OnDisable的类型
            var hasOnEnable = type.GetMethod("OnEnable", BindingFlags.NonPublic | BindingFlags.Instance) != null;
            var hasOnDisable = type.GetMethod("OnDisable", BindingFlags.NonPublic | BindingFlags.Instance) != null;

            if (hasOnEnable && !hasOnDisable)
            {
                potentialLeaks.Add($"{mb.gameObject.name}.{type.Name}: 有OnEnable但无OnDisable");
            }
        }

        if (potentialLeaks.Count > 0)
        {
            Debug.LogWarning($"发现 {potentialLeaks.Count} 个潜在事件泄漏：\n{string.Join("\n", potentialLeaks)}");
        }
        else
        {
            Debug.Log("未发现明显的事件泄漏风险");
        }
    }

    /// <summary>
    /// 检测静态引用
    /// </summary>
    [MenuItem("Tools/Memory/Check Static References")]
    public static void CheckStaticReferences()
    {
        var staticFields = new List<string>();

        foreach (var assembly in AppDomain.CurrentDomain.GetAssemblies())
        {
            try
            {
                foreach (var type in assembly.GetTypes())
                {
                    if (!type.IsClass) continue;

                    foreach (var field in type.GetFields(BindingFlags.Static | BindingFlags.NonPublic | BindingFlags.Public))
                    {
                        if (typeof(Component).IsAssignableFrom(field.FieldType) ||
                            typeof(GameObject).IsAssignableFrom(field.FieldType))
                        {
                            var value = field.GetValue(null);
                            if (value != null)
                            {
                                staticFields.Add($"{type.Name}.{field.Name} = {value}");
                            }
                        }
                    }
                }
            }
            catch { }
        }

        if (staticFields.Count > 0)
        {
            Debug.LogWarning($"发现 {staticFields.Count} 个静态引用：\n{string.Join("\n", staticFields)}");
        }
        else
        {
            Debug.Log("未发现静态引用问题");
        }
    }
}
#endif
```
