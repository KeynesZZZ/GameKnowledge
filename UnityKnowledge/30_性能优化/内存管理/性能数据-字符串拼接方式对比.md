## 测试1: 简单拼接

### 测试代码

```csharp
using System.Text;
using UnityEngine;

public class StringConcatenationBenchmark : MonoBehaviour
{
    private const int ITERATIONS = 10000;
    private string a = "Hello";
    private string b = "World";
    private string c = "Unity";
    private string d = "2021";

    private void RunBenchmark()
    {
        // 1. + 操作符
        var sw1 = System.Diagnostics.Stopwatch.StartNew();
        long mem1 = GC.GetTotalMemory(true);
        for (int i = 0; i < ITERATIONS; i++)
        {
            string result = a + b + c + d;
        }
        sw1.Stop();
        long alloc1 = GC.GetTotalMemory(false) - mem1;

        // 2. string.Concat
        var sw2 = System.Diagnostics.Stopwatch.StartNew();
        long mem2 = GC.GetTotalMemory(true);
        for (int i = 0; i < ITERATIONS; i++)
        {
            string result = string.Concat(a, b, c, d);
        }
        sw2.Stop();
        long alloc2 = GC.GetTotalMemory(false) - mem2;

        // 3. string.Format
        var sw3 = System.Diagnostics.Stopwatch.StartNew();
        long mem3 = GC.GetTotalMemory(true);
        for (int i = 0; i < ITERATIONS; i++)
        {
            string result = string.Format("{0}{1}{2}{3}", a, b, c, d);
        }
        sw3.Stop();
        long alloc3 = GC.GetTotalMemory(false) - mem3;

        // 4. 字符串插值
        var sw4 = System.Diagnostics.Stopwatch.StartNew();
        long mem4 = GC.GetTotalMemory(true);
        for (int i = 0; i < ITERATIONS; i++)
        {
            string result = $"{a}{b}{c}{d}";
        }
        sw4.Stop();
        long alloc4 = GC.GetTotalMemory(false) - mem4;

        // 5. StringBuilder
        var sw5 = System.Diagnostics.Stopwatch.StartNew();
        long mem5 = GC.GetTotalMemory(true);
        var sb = new StringBuilder();
        for (int i = 0; i < ITERATIONS; i++)
        {
            sb.Clear();
            sb.Append(a).Append(b).Append(c).Append(d);
            string result = sb.ToString();
        }
        sw5.Stop();
        long alloc5 = GC.GetTotalMemory(false) - mem5;

        Debug.Log($"+: {sw1.ElapsedMilliseconds}ms, {alloc1 / 1024}KB");
        Debug.Log($"Concat: {sw2.ElapsedMilliseconds}ms, {alloc2 / 1024}KB");
        Debug.Log($"Format: {sw3.ElapsedMilliseconds}ms, {alloc3 / 1024}KB");
        Debug.Log($"Interpolation: {sw4.ElapsedMilliseconds}ms, {alloc4 / 1024}KB");
        Debug.Log($"StringBuilder: {sw5.ElapsedMilliseconds}ms, {alloc5 / 1024}KB");
    }
}
```

### 测试结果

| 方式 | 执行时间 | 内存分配 | 性能评级 |
|------|----------|----------|----------|
| **+ 操作符** | 2ms | 1.2MB | ⭐⭐⭐ |
| **string.Concat** | 1ms | 1.2MB | ⭐⭐⭐⭐ |
| **string.Format** | 5ms | 1.8MB | ⭐⭐ |
| **字符串插值** | 2ms | 1.2MB | ⭐⭐⭐⭐ |
| **StringBuilder** | 3ms | 0.8MB | ⭐⭐⭐ |

### 结论

- **少量拼接**：使用 `+` 或字符串插值（编译优化）
- **大量拼接**：使用 StringBuilder
- **避免**：string.Format（装箱开销）

## 测试2: 循环拼接

### 测试代码

```csharp
private void LoopConcatenationBenchmark()
{
    const int COUNT = 1000;

    // 1. + 操作符在循环中 (最差)
    var sw1 = Stopwatch.StartNew();
    string result1 = "";
    for (int i = 0; i < COUNT; i++)
    {
        result1 += i.ToString();  // 每次创建新字符串！
    }
    sw1.Stop();

    // 2. StringBuilder
    var sw2 = Stopwatch.StartNew();
    var sb = new StringBuilder();
    for (int i = 0; i < COUNT; i++)
    {
        sb.Append(i.ToString());
    }
    string result2 = sb.ToString();
    sw2.Stop();

    // 3. StringBuilder 预设容量
    var sw3 = Stopwatch.StartNew();
    var sb3 = new StringBuilder(COUNT * 5);  // 预估容量
    for (int i = 0; i < COUNT; i++)
    {
        sb3.Append(i.ToString());
    }
    string result3 = sb3.ToString();
    sw3.Stop();

    Debug.Log($"+ in loop: {sw1.ElapsedMilliseconds}ms");
    Debug.Log($"StringBuilder: {sw2.ElapsedMilliseconds}ms");
    Debug.Log($"StringBuilder (pre-sized): {sw3.ElapsedMilliseconds}ms");
}
```

### 测试结果

| 方式 | 执行时间 | 内存分配 | 性能差异 |
|------|----------|----------|----------|
| **+ 在循环中** | 156ms | ~2.5MB | 基准 |
| **StringBuilder** | 2ms | ~20KB | **78x快** |
| **StringBuilder (预设容量)** | 1ms | ~15KB | **156x快** |

### 结论

**绝对不要在循环中使用 + 拼接字符串！**

## 测试3: 数字格式化

### 测试代码

```csharp
private void NumberFormattingBenchmark()
{
    const int ITERATIONS = 10000;
    int number = 1234567;

    // 1. ToString()
    var sw1 = Stopwatch.StartNew();
    for (int i = 0; i < ITERATIONS; i++)
    {
        string s = number.ToString();
    }
    sw1.Stop();

    // 2. ToString("N0")
    var sw2 = Stopwatch.StartNew();
    for (int i = 0; i < ITERATIONS; i++)
    {
        string s = number.ToString("N0");  // 1,234,567
    }
    sw2.Stop();

    // 3. 字符串插值
    var sw3 = Stopwatch.StartNew();
    for (int i = 0; i < ITERATIONS; i++)
    {
        string s = $"{number:N0}";
    }
    sw3.Stop();

    Debug.Log($"ToString(): {sw1.ElapsedMilliseconds}ms");
    Debug.Log($"ToString(N0): {sw2.ElapsedMilliseconds}ms");
    Debug.Log($"Interpolation: {sw3.ElapsedMilliseconds}ms");
}
```

### 测试结果

| 方式 | 执行时间 | 输出示例 |
|------|----------|----------|
| **ToString()** | 3ms | 1234567 |
| **ToString("N0")** | 8ms | 1,234,567 |
| **字符串插值格式化** | 9ms | 1,234,567 |

## 测试4: 避免装箱

### 测试代码

```csharp
private void BoxingBenchmark()
{
    const int ITERATIONS = 10000;

    // 1. 装箱
    var sw1 = Stopwatch.StartNew();
    long mem1 = GC.GetTotalMemory(true);
    for (int i = 0; i < ITERATIONS; i++)
    {
        string s = "Value: " + i;  // i被装箱！
    }
    sw1.Stop();
    long alloc1 = GC.GetTotalMemory(false) - mem1;

    // 2. 避免装箱
    var sw2 = Stopwatch.StartNew();
    long mem2 = GC.GetTotalMemory(true);
    for (int i = 0; i < ITERATIONS; i++)
    {
        string s = "Value: " + i.ToString();  // 无装箱
    }
    sw2.Stop();
    long alloc2 = GC.GetTotalMemory(false) - mem2;

    Debug.Log($"Boxing: {sw1.ElapsedMilliseconds}ms, {alloc1 / 1024}KB");
    Debug.Log($"No boxing: {sw2.ElapsedMilliseconds}ms, {alloc2 / 1024}KB");
}
```

### 测试结果

| 方式 | 执行时间 | 内存分配 |
|------|----------|----------|
| **字符串 + 整数** | 5ms | 320KB |
| **字符串 + ToString()** | 3ms | 160KB |

### 结论

在拼接前先调用 `ToString()` 避免装箱开销。

## 最佳实践总结

### 快速选择指南

| 场景 | 推荐方式 | 原因 |
|------|----------|------|
| 2-3个字符串拼接 | `+` 或 `$""` | 简洁，编译器优化 |
| 格式化输出 | `$"{value:N0}"` | 可读性好 |
| 循环拼接 | `StringBuilder` | 避免重复分配 |
| 大量拼接 | `StringBuilder` (预设容量) | 最高效 |
| 高频更新 | 缓存结果，脏标记 | 避免重复计算 |

### 代码示例

```csharp
// ✅ 推荐：简单拼接
string message = $"Score: {score}, Level: {level}";

// ✅ 推荐：循环拼接
var sb = new StringBuilder(items.Count * 20);
foreach (var item in items)
{
    sb.Append(item.Name).Append(": ").Append(item.Value).Append(", ");
}

// ✅ 推荐：脏标记更新
private int lastScore = -1;
private int currentScore;

void UpdateScoreDisplay()
{
    if (currentScore != lastScore)
    {
        scoreText.text = $"Score: {currentScore:N0}";
        lastScore = currentScore;
    }
}

// ❌ 避免：循环中使用 +
string result = "";
for (int i = 0; i < items.Count; i++)
{
    result += items[i].ToString();  // 性能灾难！
}
```
