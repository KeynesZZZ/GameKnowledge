## 二、存档槽位UI

```csharp
using System;
using UnityEngine;
using UnityEngine.UI;
using TMPro;

namespace CloudSave.UI
{
    /// <summary>
    /// 存档槽位UI组件
    /// </summary>
    public class SaveSlotUI : MonoBehaviour
    {
        [Header("UI引用")]
        [SerializeField] private Image screenshotImage;
        [SerializeField] private Image syncStatusIcon;
        [SerializeField] private TextMeshProUGUI slotNameText;
        [SerializeField] private TextMeshProUGUI levelText;
        [SerializeField] private TextMeshProUGUI playTimeText;
        [SerializeField] private TextMeshProUGUI lastSavedText;
        [SerializeField] private TextMeshProUGUI deviceText;
        [SerializeField] private Button loadButton;
        [SerializeField] private Button deleteButton;
        [SerializeField] private GameObject emptyOverlay;
        [SerializeField] private GameObject conflictIndicator;

        [Header("图标资源")]
        [SerializeField] private Sprite syncedIcon;
        [SerializeField] private Sprite pendingIcon;
        [SerializeField] private Sprite conflictIcon;
        [SerializeField] private Sprite offlineIcon;

        private SlotInfo slotInfo;
        private Action<int> onLoadClicked;
        private Action<int> onDeleteClicked;

        public int SlotIndex => slotInfo?.slotIndex ?? -1;

        private void Awake()
        {
            loadButton?.onClick.AddListener(OnLoadButtonClick);
            deleteButton?.onClick.AddListener(OnDeleteButtonClick);
        }

        /// <summary>
        /// 更新显示
        /// </summary>
        public void UpdateDisplay(SlotInfo info, Action<int> onLoad, Action<int> onDelete)
        {
            slotInfo = info;
            onLoadClicked = onLoad;
            onDeleteClicked = onDelete;

            if (info == null || !info.hasData)
            {
                ShowEmpty();
                return;
            }

            ShowData(info);
        }

        private void ShowEmpty()
        {
            emptyOverlay?.SetActive(true);
            conflictIndicator?.SetActive(false);

            if (slotNameText != null)
                slotNameText.text = "空槽位";

            if (levelText != null)
                levelText.text = "";

            if (playTimeText != null)
                playTimeText.text = "";

            if (lastSavedText != null)
                lastSavedText.text = "";

            if (deviceText != null)
                deviceText.text = "";

            if (loadButton != null)
                loadButton.GetComponentInChildren<TextMeshProUGUI>().text = "新建存档";

            if (deleteButton != null)
                deleteButton.gameObject.SetActive(false);
        }

        private void ShowData(SlotInfo info)
        {
            emptyOverlay?.SetActive(false);
            conflictIndicator?.SetActive(info.hasConflict);

            // 槽位名称
            if (slotNameText != null)
                slotNameText.text = $"存档 {info.slotIndex + 1}";

            // 游戏数据
            if (info.summary != null)
            {
                if (levelText != null)
                    levelText.text = $"等级: {info.summary.level}";

                if (playTimeText != null)
                    playTimeText.text = FormatPlayTime(info.summary.playTime);
            }

            // 时间
            if (lastSavedText != null)
            {
                var time = DateTimeOffset.FromUnixTimeMilliseconds(info.timestamp).LocalDateTime;
                lastSavedText.text = time.ToString("MM/dd HH:mm");
            }

            // 设备
            if (deviceText != null)
                deviceText.text = $"{info.deviceName} | {info.platformName}";

            // 同步状态
            UpdateSyncStatus(info.syncState);

            // 按钮
            if (loadButton != null)
                loadButton.GetComponentInChildren<TextMeshProUGUI>().text = "加载";

            if (deleteButton != null)
                deleteButton.gameObject.SetActive(true);
        }

        private void UpdateSyncStatus(SyncState state)
        {
            if (syncStatusIcon == null) return;

            syncStatusIcon.sprite = state switch
            {
                SyncState.Synced => syncedIcon,
                SyncState.Syncing => pendingIcon,
                SyncState.Conflict => conflictIcon,
                SyncState.PendingSync => offlineIcon,
                SyncState.LocalOnly => offlineIcon,
                _ => offlineIcon
            };
        }

        private void OnLoadButtonClick()
        {
            onLoadClicked?.Invoke(SlotIndex);
        }

        private void OnDeleteButtonClick()
        {
            onDeleteClicked?.Invoke(SlotIndex);
        }

        private string FormatPlayTime(long seconds)
        {
            if (seconds < 60)
                return $"{seconds}秒";
            else if (seconds < 3600)
                return $"{seconds / 60}分钟";
            else
                return $"{seconds / 3600}小时{(seconds % 3600) / 60}分";
        }
    }
}
```

## 三、存档列表面板

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;
using Cysharp.Threading.Tasks;
using CloudSave.Core;

namespace CloudSave.UI
{
    /// <summary>
    /// 存档列表面板
    /// </summary>
    public class SaveSlotListPanel : MonoBehaviour
    {
        [Header("UI引用")]
        [SerializeField] private Transform slotContainer;
        [SerializeField] private SaveSlotUI slotPrefab;
        [SerializeField] private Button syncAllButton;
        [SerializeField] private Button backButton;
        [SerializeField] private GameObject loadingOverlay;
        [SerializeField] private TextMeshProUGUI statusText;

        [Header("其他面板")]
        [SerializeField] private ConflictDialog conflictDialog;
        [SerializeField] private ConfirmDialog confirmDialog;

        private CloudSaveManager saveManager;
        private List<SaveSlotUI> slotUIs = new List<SaveSlotUI>();
        private Action<int> onSlotSelected;
        private Action onBack;

        private void Awake()
        {
            syncAllButton?.onClick.AddListener(OnSyncAllClick);
            backButton?.onClick.AddListener(OnBackClick);
        }

        /// <summary>
        /// 初始化面板
        /// </summary>
        public void Initialize(CloudSaveManager manager, Action<int> onSlotSelected, Action onBack)
        {
            this.saveManager = manager;
            this.onSlotSelected = onSlotSelected;
            this.onBack = onBack;

            // 创建槽位UI
            CreateSlotUIs();

            // 注册事件
            saveManager.OnSyncProgress += OnSyncProgress;
            saveManager.OnSyncComplete += OnSyncComplete;
            saveManager.OnConflict += OnConflict;

            // 初始加载
            RefreshAsync().Forget();
        }

        private void CreateSlotUIs()
        {
            // 清除旧的
            foreach (var slot in slotUIs)
            {
                if (slot != null)
                    Destroy(slot.gameObject);
            }
            slotUIs.Clear();

            // 创建新的
            for (int i = 0; i < saveManager.PlatformInfo?.maxSlots ?? 3; i++)
            {
                var slotUI = Instantiate(slotPrefab, slotContainer);
                slotUIs.Add(slotUI);
            }
        }

        /// <summary>
        /// 刷新列表
        /// </summary>
        public async UniTask RefreshAsync()
        {
            if (loadingOverlay != null)
                loadingOverlay.SetActive(true);

            try
            {
                var slots = await saveManager.GetSlotListAsync();

                for (int i = 0; i < slotUIs.Count; i++)
                {
                    var slotInfo = slots.Find(s => s.slotIndex == i);
                    slotUIs[i].UpdateDisplay(slotInfo, OnLoadSlot, OnDeleteSlot);
                }

                UpdateStatusText();
            }
            finally
            {
                if (loadingOverlay != null)
                    loadingOverlay.SetActive(false);
            }
        }

        private async void OnLoadSlot(int slotIndex)
        {
            onSlotSelected?.Invoke(slotIndex);
        }

        private async void OnDeleteSlot(int slotIndex)
        {
            if (confirmDialog != null)
            {
                bool confirmed = await confirmDialog.ShowAsync(
                    "确认删除",
                    $"确定要删除存档 {slotIndex + 1} 吗？\n此操作不可撤销。"
                );

                if (!confirmed) return;
            }

            bool success = await saveManager.DeleteAsync(slotIndex);

            if (success)
            {
                await RefreshAsync();
            }
        }

        private async void OnSyncAllClick()
        {
            await saveManager.SyncAsync();
            await RefreshAsync();
        }

        private void OnBackClick()
        {
            onBack?.Invoke();
        }

        private void OnSyncProgress(SyncProgress progress)
        {
            if (statusText != null)
            {
                statusText.text = $"{progress.statusText} ({progress.OverallProgress:P0})";
            }
        }

        private void OnSyncComplete(SyncResult result)
        {
            UpdateStatusText();

            if (result.success)
            {
                Debug.Log($"同步完成: 上传{result.uploadedCount}, 下载{result.downloadedCount}");
            }
            else
            {
                Debug.LogError($"同步失败: {result.errorMessage}");
            }
        }

        private async void OnConflict(ConflictEvent conflict)
        {
            if (conflictDialog != null)
            {
                var resolution = await conflictDialog.ShowAsync(conflict);
                // 冲突解决后刷新
                await RefreshAsync();
            }
        }

        private void UpdateStatusText()
        {
            if (statusText != null)
            {
                int pending = saveManager.PendingOperations;
                bool online = saveManager.IsOnline;

                if (!online)
                {
                    statusText.text = "离线模式";
                }
                else if (pending > 0)
                {
                    statusText.text = $"待同步: {pending}";
                }
                else
                {
                    statusText.text = "已同步";
                }
            }
        }

        private void OnDestroy()
        {
            if (saveManager != null)
            {
                saveManager.OnSyncProgress -= OnSyncProgress;
                saveManager.OnSyncComplete -= OnSyncComplete;
                saveManager.OnConflict -= OnConflict;
            }
        }
    }
}
```

## 四、冲突对话框

```csharp
using System;
using UnityEngine;
using UnityEngine.UI;
using TMPro;
using Cysharp.Threading.Tasks;
using CloudSave.Core;

namespace CloudSave.UI
{
    /// <summary>
    /// 冲突解决对话框
    /// </summary>
    public class ConflictDialog : MonoBehaviour
    {
        [Header("UI引用")]
        [SerializeField] private GameObject panel;
        [SerializeField] private TextMeshProUGUI titleText;

        [Header("本地信息")]
        [SerializeField] private TextMeshProUGUI localTimeText;
        [SerializeField] private TextMeshProUGUI localLevelText;
        [SerializeField] private TextMeshProUGUI localDeviceText;

        [Header("云端信息")]
        [SerializeField] private TextMeshProUGUI cloudTimeText;
        [SerializeField] private TextMeshProUGUI cloudLevelText;
        [SerializeField] private TextMeshProUGUI cloudDeviceText;

        [Header("按钮")]
        [SerializeField] private Button useLocalButton;
        [SerializeField] private Button useCloudButton;
        [SerializeField] private Button keepBothButton;

        private ConflictEvent currentConflict;
        private UniTaskCompletionSource<ConflictResolution> tcs;

        private void Awake()
        {
            useLocalButton?.onClick.AddListener(() => Resolve(ConflictResolution.UseLocal));
            useCloudButton?.onClick.AddListener(() => Resolve(ConflictResolution.UseCloud));
            keepBothButton?.onClick.AddListener(() => Resolve(ConflictResolution.KeepBoth));

            panel?.SetActive(false);
        }

        /// <summary>
        /// 显示冲突对话框
        /// </summary>
        public async UniTask<ConflictResolution> ShowAsync(ConflictEvent conflict)
        {
            currentConflict = conflict;
            tcs = new UniTaskCompletionSource<ConflictResolution>();

            // 更新显示
            UpdateDisplay(conflict);

            // 显示面板
            panel?.SetActive(true);

            // 等待用户选择
            return await tcs.Task;
        }

        private void UpdateDisplay(ConflictEvent conflict)
        {
            // 本地信息
            if (localTimeText != null)
                localTimeText.text = conflict.LocalTime.ToString("MM/dd HH:mm");

            if (localLevelText != null && conflict.localData?.summary != null)
                localLevelText.text = $"等级: {conflict.localData.summary.level}";

            if (localDeviceText != null)
                localDeviceText.text = $"{conflict.localData?.deviceId} | {conflict.localData?.platform}";

            // 云端信息
            if (cloudTimeText != null)
                cloudTimeText.text = conflict.CloudTime.ToString("MM/dd HH:mm");

            if (cloudLevelText != null && conflict.cloudData?.summary != null)
                cloudLevelText.text = $"等级: {conflict.cloudData.summary.level}";

            if (cloudDeviceText != null)
                cloudDeviceText.text = $"{conflict.cloudData?.deviceId} | {conflict.cloudData?.platform}";

            // 标题
            if (titleText != null)
                titleText.text = $"检测到存档冲突 - 槽位 {conflict.slotIndex + 1}";

            // 根据建议高亮按钮
            UpdateButtonHighlight(conflict.suggestedResolution);
        }

        private void UpdateButtonHighlight(ConflictResolution suggested)
        {
            // 重置所有按钮颜色
            ResetButtonHighlight(useLocalButton);
            ResetButtonHighlight(useCloudButton);
            ResetButtonHighlight(keepBothButton);

            // 高亮建议的按钮
            switch (suggested)
            {
                case ConflictResolution.UseLocal:
                    HighlightButton(useLocalButton);
                    break;
                case ConflictResolution.UseCloud:
                    HighlightButton(useCloudButton);
                    break;
            }
        }

        private void HighlightButton(Button button)
        {
            if (button != null)
            {
                var colors = button.colors;
                colors.normalColor = Color.green;
                button.colors = colors;
            }
        }

        private void ResetButtonHighlight(Button button)
        {
            if (button != null)
            {
                var colors = button.colors;
                colors.normalColor = Color.white;
                button.colors = colors;
            }
        }

        private void Resolve(ConflictResolution resolution)
        {
            panel?.SetActive(false);
            tcs?.TrySetResult(resolution);
        }
    }
}
```

## 五、确认对话框

```csharp
using UnityEngine;
using UnityEngine.UI;
using TMPro;
using Cysharp.Threading.Tasks;

namespace CloudSave.UI
{
    /// <summary>
    /// 确认对话框
    /// </summary>
    public class ConfirmDialog : MonoBehaviour
    {
        [SerializeField] private GameObject panel;
        [SerializeField] private TextMeshProUGUI titleText;
        [SerializeField] private TextMeshProUGUI messageText;
        [SerializeField] private Button confirmButton;
        [SerializeField] private Button cancelButton;

        private UniTaskCompletionSource<bool> tcs;

        private void Awake()
        {
            confirmButton?.onClick.AddListener(OnConfirm);
            cancelButton?.onClick.AddListener(OnCancel);
            panel?.SetActive(false);
        }

        public async UniTask<bool> ShowAsync(string title, string message)
        {
            tcs = new UniTaskCompletionSource<bool>();

            if (titleText != null)
                titleText.text = title;

            if (messageText != null)
                messageText.text = message;

            panel?.SetActive(true);

            return await tcs.Task;
        }

        private void OnConfirm()
        {
            panel?.SetActive(false);
            tcs?.TrySetResult(true);
        }

        private void OnCancel()
        {
            panel?.SetActive(false);
            tcs?.TrySetResult(false);
        }
    }
}
```

## 六、游戏集成示例

```csharp
using UnityEngine;
using UnityEngine.SceneManagement;
using Cysharp.Threading.Tasks;
using CloudSave.Core;
using CloudSave.UI;

namespace CloudSave.Examples
{
    /// <summary>
    /// 游戏管理器 - 完整集成示例
    /// </summary>
    public class GameManager : MonoBehaviour
    {
        public static GameManager Instance { get; private set; }

        [Header("UI")]
        [SerializeField] private SaveSlotListPanel saveSlotPanel;
        [SerializeField] private GameObject mainMenuUI;
        [SerializeField] private GameObject gameUI;

        [Header("配置")]
        [SerializeField] private string encryptionKey = "your-secret-key";

        private CloudSaveManager saveManager;
        private GameSaveData currentGameData;
        private int currentSlot = -1;

        #region 生命周期

        private void Awake()
        {
            if (Instance != null && Instance != this)
            {
                Destroy(gameObject);
                return;
            }

            Instance = this;
            DontDestroyOnLoad(gameObject);
        }

        private async void Start()
        {
            await InitializeCloudSaveAsync();
            ShowMainMenu();
        }

        private async UniTask InitializeCloudSaveAsync()
        {
            saveManager = await new CloudSaveBuilder()
                .WithMaxSlots(3)
                .WithAutoSave(300f)
                .WithEncryption(encryptionKey)
                .WithDebugMode(Debug.isDebugBuild)
                .BuildAsync();

            Debug.Log("云存档系统初始化完成");

            // 监听自动保存
            saveManager.OnSaveComplete += OnAutoSaveComplete;
        }

        #endregion

        #region 游戏流程

        private void ShowMainMenu()
        {
            mainMenuUI?.SetActive(true);
            gameUI?.SetActive(false);
        }

        public void ShowSaveSlotPanel()
        {
            saveSlotPanel?.Initialize(
                saveManager,
                OnSlotSelected,
                OnSlotPanelClosed
            );

            saveSlotPanel.gameObject.SetActive(true);
        }

        private async void OnSlotSelected(int slotIndex)
        {
            currentSlot = slotIndex;

            // 加载存档
            var result = await saveManager.LoadAsync<GameSaveData>(slotIndex);

            if (result.success)
            {
                currentGameData = result.data;
            }
            else
            {
                // 新存档
                currentGameData = new GameSaveData
                {
                    level = 1,
                    coins = 0,
                    playTime = 0,
                    createTime = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds()
                };
            }

            // 进入游戏
            saveSlotPanel.gameObject.SetActive(false);
            StartGame();
        }

        private void OnSlotPanelClosed()
        {
            ShowMainMenu();
        }

        private void StartGame()
        {
            mainMenuUI?.SetActive(false);
            gameUI?.SetActive(true);

            // 应用存档数据
            ApplyGameData(currentGameData);

            Debug.Log($"游戏开始: slot={currentSlot}, level={currentGameData.level}");
        }

        #endregion

        #region 存档操作

        /// <summary>
        /// 手动保存
        /// </summary>
        public async UniTaskVoid ManualSave()
        {
            if (currentSlot < 0) return;

            // 收集当前数据
            CollectGameData();

            var result = await saveManager.SaveAsync(currentSlot, currentGameData);

            if (result.success)
            {
                Debug.Log($"手动保存成功: slot={currentSlot}");
            }
            else
            {
                Debug.LogError($"手动保存失败: {result.errorMessage}");
            }
        }

        /// <summary>
        /// 自动保存回调
        /// </summary>
        private void OnAutoSaveComplete(int slot, SaveResult result)
        {
            if (slot == currentSlot && result.success)
            {
                Debug.Log("自动保存完成");
            }
        }

        private void CollectGameData()
        {
            // 从游戏状态收集数据
            // currentGameData.level = Player.Instance.Level;
            // currentGameData.coins = Player.Instance.Coins;
            // currentGameData.playTime += Time.deltaTime;
            currentGameData.updateTime = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();
        }

        private void ApplyGameData(GameSaveData data)
        {
            // 应用存档数据到游戏
            // Player.Instance.SetLevel(data.level);
            // Player.Instance.SetCoins(data.coins);
            Debug.Log($"应用存档: level={data.level}, coins={data.coins}");
        }

        #endregion

        #region 场景切换

        private void OnSceneChanged(Scene previous, Scene next)
        {
            if (saveManager != null && currentSlot >= 0)
            {
                // 场景切换时自动保存
                CollectGameData();
                saveManager.SaveAsync(currentSlot, currentGameData).Forget();
            }
        }

        #endregion

        #region 应用退出

        private void OnApplicationPause(bool pauseStatus)
        {
            if (pauseStatus && currentSlot >= 0)
            {
                CollectGameData();
                saveManager.SaveAsync(currentSlot, currentGameData).Forget();
            }
        }

        private void OnApplicationQuit()
        {
            if (currentSlot >= 0)
            {
                CollectGameData();
                // 同步保存
                saveManager.SaveAsync(currentSlot, currentGameData).GetAwaiter().GetResult();
            }

            saveManager?.Dispose();
        }

        #endregion
    }

    #region 示例数据结构

    [Serializable]
    public class GameSaveData
    {
        public int level;
        public int coins;
        public long playTime;
        public long createTime;
        public long updateTime;
        public List<string> unlockedLevels;
        public Dictionary<string, int> items;
    }

    #endregion
}
```
