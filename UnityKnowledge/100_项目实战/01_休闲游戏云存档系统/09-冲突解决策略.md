## 二、解决策略

```csharp
namespace CloudSave.Core
{
    /// <summary>
    /// 冲突解决策略
    /// </summary>
    public enum ConflictResolution
    {
        /// <summary>使用云端数据</summary>
        UseCloud,

        /// <summary>使用本地数据</summary>
        UseLocal,

        /// <summary>使用较新的数据（按时间戳）</summary>
        UseNewer,

        /// <summary>使用较旧的数据（按时间戳）</summary>
        UseOlder,

        /// <summary>需要用户选择</summary>
        AskUser,

        /// <summary>尝试智能合并</summary>
        Merge,

        /// <summary>保留两份（创建备份）</summary>
        KeepBoth
    }

    /// <summary>
    /// 冲突解决配置
    /// </summary>
    public class ConflictConfig
    {
        /// <summary>默认解决策略</summary>
        public ConflictResolution defaultResolution = ConflictResolution.UseNewer;

        /// <summary>时间戳容差（毫秒），小于此值认为同时</summary>
        public long timestampTolerance = 5000;

        /// <summary>是否自动解决冲突</summary>
        public bool autoResolve = true;

        /// <summary>是否创建备份</summary>
        public bool createBackup = true;

        /// <summary>合并策略（游戏特定）</summary>
        public IMergeStrategy mergeStrategy;
    }
}
```

## 三、冲突检测器

```csharp
using System;
using UnityEngine;

namespace CloudSave.Core
{
    /// <summary>
    /// 冲突检测器
    /// </summary>
    public class ConflictDetector
    {
        private readonly ConflictConfig config;

        public ConflictDetector(ConflictConfig config = null)
        {
            this.config = config ?? new ConflictConfig();
        }

        /// <summary>
        /// 检测是否存在冲突
        /// </summary>
        public ConflictEvent Detect(int slot, CloudSaveData localData, CloudSaveData cloudData)
        {
            if (localData == null || cloudData == null)
            {
                return null;  // 没有冲突
            }

            // 检查校验和
            if (localData.checksum == cloudData.checksum)
            {
                return null;  // 数据相同，无冲突
            }

            // 检查时间戳
            long timeDiff = Math.Abs(localData.timestamp - cloudData.timestamp);

            if (timeDiff <= config.timestampTolerance)
            {
                // 时间戳相近，可能是并发写入
                return CreateConflictEvent(slot, localData, cloudData, ConflictType.ConcurrentWrite);
            }

            // 检查是否基于同一版本
            if (localData.gameDataVersion == cloudData.gameDataVersion)
            {
                // 同一版本的不同修改，冲突
                return CreateConflictEvent(slot, localData, cloudData, ConflictType.VersionConflict);
            }

            // 时间戳不同，可能是一方更新过
            return CreateConflictEvent(slot, localData, cloudData, ConflictType.TimestampConflict);
        }

        /// <summary>
        /// 获取建议的解决策略
        /// </summary>
        public ConflictResolution GetSuggestedResolution(ConflictEvent conflict)
        {
            switch (conflict.type)
            {
                case ConflictType.ConcurrentWrite:
                    // 并发写入，建议用户选择
                    return ConflictResolution.AskUser;

                case ConflictType.VersionConflict:
                    // 版本冲突，使用较新的
                    return ConflictResolution.UseNewer;

                case ConflictType.TimestampConflict:
                    // 时间戳冲突，比较时间
                    return conflict.LocalTime > conflict.CloudTime
                        ? ConflictResolution.UseLocal
                        : ConflictResolution.UseCloud;

                default:
                    return config.defaultResolution;
            }
        }

        private ConflictEvent CreateConflictEvent(int slot, CloudSaveData local, CloudSaveData cloud, ConflictType type)
        {
            var conflict = new ConflictEvent
            {
                slotIndex = slot,
                localData = local,
                cloudData = cloud,
                type = type,
                suggestedResolution = ConflictResolution.UseNewer
            };

            conflict.suggestedResolution = GetSuggestedResolution(conflict);
            return conflict;
        }
    }

    /// <summary>
    /// 冲突类型
    /// </summary>
    public enum ConflictType
    {
        /// <summary>时间戳冲突</summary>
        TimestampConflict,

        /// <summary>版本冲突</summary>
        VersionConflict,

        /// <summary>并发写入</summary>
        ConcurrentWrite,

        /// <summary>数据损坏</summary>
        DataCorruption
    }
}
```

## 四、冲突解决器

```csharp
using System;
using UnityEngine;
using Cysharp.Threading.Tasks;

namespace CloudSave.Core
{
    /// <summary>
    /// 冲突解决器
    /// </summary>
    public class ConflictResolver
    {
        private readonly ConflictConfig config;
        private readonly ICloudProvider cloudProvider;
        private readonly ICacheProvider cacheProvider;

        public event Action<ConflictEvent> OnConflictDetected;
        public event Action<ConflictEvent, ConflictResolution, CloudSaveData> OnConflictResolved;

        public ConflictResolver(
            ConflictConfig config,
            ICloudProvider cloudProvider,
            ICacheProvider cacheProvider)
        {
            this.config = config;
            this.cloudProvider = cloudProvider;
            this.cacheProvider = cacheProvider;
        }

        /// <summary>
        /// 自动解决冲突
        /// </summary>
        public async UniTask<CloudSaveData> ResolveAsync(ConflictEvent conflict)
        {
            var resolution = conflict.suggestedResolution;

            // 如果需要用户选择且没有自动解决
            if (resolution == ConflictResolution.AskUser && !config.autoResolve)
            {
                OnConflictDetected?.Invoke(conflict);
                return null;  // 等待用户选择
            }

            // 自动解决
            return await ApplyResolutionAsync(conflict, resolution);
        }

        /// <summary>
        /// 应用解决策略
        /// </summary>
        public async UniTask<CloudSaveData> ApplyResolutionAsync(ConflictEvent conflict, ConflictResolution resolution)
        {
            CloudSaveData resolvedData = null;

            switch (resolution)
            {
                case ConflictResolution.UseCloud:
                    resolvedData = conflict.cloudData;
                    break;

                case ConflictResolution.UseLocal:
                    resolvedData = conflict.localData;
                    break;

                case ConflictResolution.UseNewer:
                    resolvedData = conflict.LocalTime > conflict.CloudTime
                        ? conflict.localData
                        : conflict.cloudData;
                    break;

                case ConflictResolution.UseOlder:
                    resolvedData = conflict.LocalTime < conflict.CloudTime
                        ? conflict.localData
                        : conflict.cloudData;
                    break;

                case ConflictResolution.Merge:
                    resolvedData = await MergeAsync(conflict);
                    break;

                case ConflictResolution.KeepBoth:
                    resolvedData = conflict.localData;  // 使用本地
                    await CreateBackupAsync(conflict);   // 备份云端
                    break;
            }

            if (resolvedData != null)
            {
                // 更新版本号
                resolvedData.gameDataVersion++;

                // 同步到本地和云端
                cacheProvider.SaveToLocal(conflict.slotIndex, resolvedData);
                await cloudProvider.SaveAsync(conflict.slotIndex, resolvedData);

                OnConflictResolved?.Invoke(conflict, resolution, resolvedData);
            }

            return resolvedData;
        }

        /// <summary>
        /// 合并数据
        /// </summary>
        private async UniTask<CloudSaveData> MergeAsync(ConflictEvent conflict)
        {
            if (config.mergeStrategy != null)
            {
                return await config.mergeStrategy.MergeAsync(
                    conflict.localData,
                    conflict.cloudData
                );
            }

            // 默认合并策略：使用较新的作为基础
            var baseData = conflict.LocalTime > conflict.CloudTime
                ? conflict.localData
                : conflict.cloudData;

            Debug.LogWarning($"使用默认合并策略（选择较新数据）: slot={conflict.slotIndex}");

            return baseData;
        }

        /// <summary>
        /// 创建备份
        /// </summary>
        private async UniTask CreateBackupAsync(ConflictEvent conflict)
        {
            var backupData = conflict.cloudData;
            backupData.slotIndex = conflict.slotIndex + 100;  // 备份槽位

            await cacheProvider.SaveToLocal(backupData.slotIndex, backupData);

            Debug.Log($"已创建备份: original_slot={conflict.slotIndex}, backup_slot={backupData.slotIndex}");
        }
    }
}
```

## 五、合并策略接口

```csharp
using Cysharp.Threading.Tasks;

namespace CloudSave.Core
{
    /// <summary>
    /// 数据合并策略接口
    /// 游戏可根据自身数据结构实现
    /// </summary>
    public interface IMergeStrategy
    {
        /// <summary>
        /// 合并两个存档数据
        /// </summary>
        UniTask<CloudSaveData> MergeAsync(CloudSaveData local, CloudSaveData cloud);
    }

    /// <summary>
    /// 三消游戏合并策略示例
    /// </summary>
    public class Match3MergeStrategy : IMergeStrategy
    {
        public async UniTask<CloudSaveData> MergeAsync(CloudSaveData local, CloudSaveData cloud)
        {
            // 解析游戏数据
            var localGame = local.ParseGameData<Match3GameData>();
            var cloudGame = cloud.ParseGameData<Match3GameData>();

            // 合并策略：
            // - 金币：取最大值
            // - 关卡：取最大值
            // - 道具：合并（去重）
            // - 成就：合并

            var merged = new Match3GameData
            {
                // 基础数据取最大
                coins = Math.Max(localGame.coins, cloudGame.coins),
                level = Math.Max(localGame.level, cloudGame.level),
                exp = Math.Max(localGame.exp, cloudGame.exp),

                // 游戏时间累加
                playTime = localGame.playTime + cloudGame.playTime,

                // 道具合并
                items = MergeItems(localGame.items, cloudGame.items),

                // 成就合并
                achievements = MergeAchievements(localGame.achievements, cloudGame.achievements),

                // 创建时间取最早
                createTime = Math.Min(localGame.createTime, cloudGame.createTime),

                // 更新时间取当前
                updateTime = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds()
            };

            // 创建新的CloudSaveData
            var result = CloudSaveData.Create(local.slotIndex, merged, local.deviceId, local.platform);
            result.userId = local.userId;
            result.gameDataVersion = Math.Max(local.gameDataVersion, cloud.gameDataVersion) + 1;

            await UniTask.CompletedTask;
            return result;
        }

        private List<GameItem> MergeItems(List<GameItem> a, List<GameItem> b)
        {
            var result = new List<GameItem>(a);

            foreach (var item in b)
            {
                var existing = result.Find(i => i.id == item.id);
                if (existing != null)
                {
                    // 取数量最大
                    existing.count = Math.Max(existing.count, item.count);
                }
                else
                {
                    result.Add(item);
                }
            }

            return result;
        }

        private List<string> MergeAchievements(List<string> a, List<string> b)
        {
            var result = new HashSet<string>(a);
            foreach (var ach in b)
            {
                result.Add(ach);
            }
            return new List<string>(result);
        }
    }

    #region 示例数据结构

    [Serializable]
    public class Match3GameData
    {
        public int coins;
        public int level;
        public int exp;
        public long playTime;
        public long createTime;
        public long updateTime;
        public List<GameItem> items;
        public List<string> achievements;
    }

    [Serializable]
    public class GameItem
    {
        public string id;
        public int count;
    }

    #endregion
}
```

## 六、使用示例

```csharp
using UnityEngine;
using Cysharp.Threading.Tasks;
using CloudSave.Core;

namespace CloudSave.Examples
{
    /// <summary>
    /// 冲突解决使用示例
    /// </summary>
    public class ConflictExample : MonoBehaviour
    {
        private ConflictDetector detector;
        private ConflictResolver resolver;
        private ICloudProvider cloudProvider;
        private ICacheProvider cacheProvider;

        private void Start()
        {
            // 配置
            var config = new ConflictConfig
            {
                defaultResolution = ConflictResolution.UseNewer,
                autoResolve = true,
                createBackup = true,
                mergeStrategy = new Match3MergeStrategy()
            };

            detector = new ConflictDetector(config);
            resolver = new ConflictResolver(config, cloudProvider, cacheProvider);

            // 监听冲突
            resolver.OnConflictDetected += OnConflictDetected;
            resolver.OnConflictResolved += OnConflictResolved;
        }

        /// <summary>
        /// 同步时检测并解决冲突
        /// </summary>
        public async UniTask<CloudSaveData> SyncWithConflictCheck(int slot)
        {
            // 获取本地和云端数据
            var localData = cacheProvider.LoadFromLocal(slot);
            var cloudResult = await cloudProvider.LoadAsync(slot);

            if (!cloudResult.success)
            {
                return localData;  // 云端加载失败，使用本地
            }

            var cloudData = cloudResult.data;

            // 检测冲突
            var conflict = detector.Detect(slot, localData, cloudData);

            if (conflict != null)
            {
                Debug.Log($"检测到冲突: slot={slot}, type={conflict.type}");

                // 自动解决
                var resolved = await resolver.ResolveAsync(conflict);

                return resolved;
            }

            // 无冲突，使用较新的
            return localData.timestamp > cloudData.timestamp ? localData : cloudData;
        }

        private void OnConflictDetected(ConflictEvent conflict)
        {
            Debug.Log($"需要用户选择: slot={conflict.slotIndex}");

            // 显示UI让用户选择
            // UIManager.ShowConflictDialog(conflict);
        }

        private void OnConflictResolved(ConflictEvent conflict, ConflictResolution resolution, CloudSaveData data)
        {
            Debug.Log($"冲突已解决: slot={conflict.slotIndex}, resolution={resolution}");
        }

        /// <summary>
        /// 用户选择后应用
        /// </summary>
        public async UniTaskVoid UserSelectedResolution(ConflictEvent conflict, ConflictResolution resolution)
        {
            var result = await resolver.ApplyResolutionAsync(conflict, resolution);

            if (result != null)
            {
                Debug.Log("用户选择已应用");
            }
        }

        private void OnDestroy()
        {
            resolver.OnConflictDetected -= OnConflictDetected;
            resolver.OnConflictResolved -= OnConflictResolved;
        }
    }
}
```
