## 二、完整实现

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;
using Cysharp.Threading.Tasks;
using CloudSave.Platform;
using CloudSave.Storage;

namespace CloudSave.Core
{
    /// <summary>
    /// 云存档管理器 - 统一入口
    /// </summary>
    public class CloudSaveManager : ICloudSaveManager, IDisposable
    {
        #region 单例访问（可选）

        private static CloudSaveManager _instance;

        /// <summary>
        /// 全局实例（推荐使用依赖注入替代）
        /// </summary>
        public static CloudSaveManager Instance
        {
            get
            {
                if (_instance == null)
                {
                    Debug.LogWarning("CloudSaveManager未初始化，请先调用Initialize");
                }
                return _instance;
            }
        }

        /// <summary>
        /// 是否已初始化
        /// </summary>
        public static bool IsInitialized => _instance != null && _instance.isInitialized;

        #endregion

        #region 私有字段

        private readonly CloudSaveConfig config;
        private readonly CloudProviderFactory providerFactory;
        private readonly ICacheProvider cacheProvider;
        private readonly PlatformDetector platformDetector;
        private readonly NetworkMonitor networkMonitor;
        private readonly ConflictResolver conflictResolver;
        private readonly ConflictDetector conflictDetector;
        private readonly OfflineQueueManager offlineQueue;

        private ICloudProvider cloudProvider;
        private PlatformInfo platformInfo;
        private bool isInitialized;
        private bool isSyncing;
        private int autoSaveHandle = -1;

        #endregion

        #region 事件

        /// <summary>
        /// 同步进度事件
        /// </summary>
        public event Action<SyncProgress> OnSyncProgress;

        /// <summary>
        /// 冲突检测事件
        /// </summary>
        public event Action<ConflictEvent> OnConflict;

        /// <summary>
        /// 同步完成事件
        /// </summary>
        public event Action<SyncResult> OnSyncComplete;

        /// <summary>
        /// 网络状态变化事件
        /// </summary>
        public event Action<bool> OnNetworkChanged;

        /// <summary>
        /// 存档保存完成事件
        /// </summary>
        public event Action<int, SaveResult> OnSaveComplete;

        /// <summary>
        /// 存档加载完成事件
        /// </summary>
        public event Action<int, LoadResult> OnLoadComplete;

        #endregion

        #region 属性

        /// <summary>
        /// 当前平台信息
        /// </summary>
        public PlatformInfo PlatformInfo => platformInfo;

        /// <summary>
        /// 是否在线
        /// </summary>
        public bool IsOnline => networkMonitor?.IsOnline ?? false;

        /// <summary>
        /// 是否正在同步
        /// </summary>
        public bool IsSyncing => isSyncing;

        /// <summary>
        /// 待同步操作数
        /// </summary>
        public int PendingOperations => offlineQueue?.PendingCount ?? 0;

        #endregion

        #region 构造函数

        public CloudSaveManager(
            CloudSaveConfig config,
            CloudProviderFactory providerFactory,
            ICacheProvider cacheProvider)
        {
            this.config = config;
            this.providerFactory = providerFactory;
            this.cacheProvider = cacheProvider;

            // 创建平台检测器
            platformDetector = new PlatformDetector();

            // 创建网络监听器
            networkMonitor = new NetworkMonitor(platformDetector);
            networkMonitor.OnNetworkStateChanged += HandleNetworkStateChanged;

            // 创建冲突解决器
            var conflictConfig = new ConflictConfig
            {
                defaultResolution = ConflictResolution.UseNewer,
                autoResolve = true,
                createBackup = true
            };
            conflictDetector = new ConflictDetector(conflictConfig);

            // 创建离线队列
            offlineQueue = new OfflineQueueManager(config);
        }

        #endregion

        #region 初始化

        /// <summary>
        /// 初始化云存档系统
        /// </summary>
        public async UniTask<bool> InitializeAsync()
        {
            if (isInitialized)
            {
                return true;
            }

            try
            {
                // 1. 检测平台
                platformInfo = await platformDetector.InitializeAsync();
                Debug.Log($"平台检测完成: {platformInfo.platformName}");

                // 2. 创建对应的Provider
                cloudProvider = providerFactory.GetProvider(platformInfo.cloudService);

                // 3. 初始化Provider
                bool providerReady = await cloudProvider.InitializeAsync();

                if (!providerReady)
                {
                    Debug.LogWarning($"Provider初始化失败，将使用本地存储");
                }

                // 4. 初始化冲突解决器
                conflictResolver = new ConflictResolver(
                    new ConflictConfig { autoResolve = true },
                    cloudProvider,
                    cacheProvider
                );
                conflictResolver.OnConflictDetected += HandleConflictDetected;
                conflictResolver.OnConflictResolved += HandleConflictResolved;

                // 5. 启动网络监听
                networkMonitor.StartMonitoring();

                // 6. 启动自动保存
                if (config.enableAutoSave)
                {
                    StartAutoSave();
                }

                // 7. 设置单例
                _instance = this;

                isInitialized = true;
                Debug.Log("云存档系统初始化完成");

                // 8. 如果有离线操作，提示同步
                if (offlineQueue.HasPendingOperations)
                {
                    Debug.Log($"有 {offlineQueue.PendingCount} 个待同步操作");
                }

                return true;
            }
            catch (Exception e)
            {
                Debug.LogError($"云存档初始化失败: {e.Message}");
                return false;
            }
        }

        #endregion

        #region 保存

        /// <summary>
        /// 保存游戏数据
        /// </summary>
        public async UniTask<SaveResult> SaveAsync(int slot, object gameData)
        {
            if (!ValidateSlot(slot))
            {
                return SaveResult.Failed("INVALID_SLOT", $"无效的存档槽位: {slot}");
            }

            try
            {
                // 创建存档数据
                var saveData = CloudSaveData.Create(
                    slot,
                    gameData,
                    platformInfo.deviceId,
                    platformInfo.platformName
                );

                // 设置用户ID
                if (!string.IsNullOrEmpty(platformInfo.userId))
                {
                    saveData.userId = platformInfo.userId;
                }

                // 通过Provider保存
                var result = await cloudProvider.SaveAsync(slot, saveData);

                // 触发事件
                OnSaveComplete?.Invoke(slot, result);

                if (result.success && !result.syncedToCloud)
                {
                    // 本地保存成功但未同步，加入离线队列
                    offlineQueue.EnqueueSave(slot, saveData);
                }

                return result;
            }
            catch (Exception e)
            {
                Debug.LogError($"保存失败: {e.Message}");
                return SaveResult.Failed("SAVE_ERROR", e.Message);
            }
        }

        /// <summary>
        /// 保存当前槽位（快捷方法）
        /// </summary>
        public async UniTask<SaveResult> SaveCurrentAsync(object gameData)
        {
            int currentSlot = GetCurrentSlot();
            return await SaveAsync(currentSlot, gameData);
        }

        #endregion

        #region 加载

        /// <summary>
        /// 加载游戏数据
        /// </summary>
        public async UniTask<LoadResult> LoadAsync(int slot)
        {
            if (!ValidateSlot(slot))
            {
                return LoadResult.Failed("INVALID_SLOT", $"无效的存档槽位: {slot}");
            }

            try
            {
                var result = await cloudProvider.LoadAsync(slot);

                // 触发事件
                OnLoadComplete?.Invoke(slot, result);

                return result;
            }
            catch (Exception e)
            {
                Debug.LogError($"加载失败: {e.Message}");
                return LoadResult.Failed("LOAD_ERROR", e.Message);
            }
        }

        /// <summary>
        /// 加载并解析为指定类型
        /// </summary>
        public async UniTask<LoadResult<T>> LoadAsync<T>(int slot) where T : class
        {
            var result = await LoadAsync(slot);

            if (result.success && result.data != null)
            {
                try
                {
                    var gameData = result.data.ParseGameData<T>();
                    return LoadResult<T>.Succeeded(gameData, result.source, result.isCloudNewer);
                }
                catch (Exception e)
                {
                    return LoadResult<T>.Failed("PARSE_ERROR", $"数据解析失败: {e.Message}");
                }
            }

            return LoadResult<T>.Failed(result.errorCode, result.errorMessage);
        }

        /// <summary>
        /// 加载当前槽位
        /// </summary>
        public async UniTask<LoadResult> LoadCurrentAsync()
        {
            int currentSlot = GetCurrentSlot();
            return await LoadAsync(currentSlot);
        }

        #endregion

        #region 删除

        /// <summary>
        /// 删除存档
        /// </summary>
        public async UniTask<bool> DeleteAsync(int slot)
        {
            if (!ValidateSlot(slot))
            {
                return false;
            }

            try
            {
                bool success = await cloudProvider.DeleteAsync(slot);

                if (!success && !IsOnline)
                {
                    // 离线时加入队列
                    offlineQueue.EnqueueDelete(slot);
                    return true;
                }

                return success;
            }
            catch (Exception e)
            {
                Debug.LogError($"删除失败: {e.Message}");
                return false;
            }
        }

        #endregion

        #region 同步

        /// <summary>
        /// 手动触发同步
        /// </summary>
        public async UniTask<SyncResult> SyncAsync()
        {
            var result = new SyncResult
            {
                success = true,
                failedSlots = new List<int>()
            };

            if (isSyncing)
            {
                result.success = false;
                result.errorMessage = "正在同步中";
                return result;
            }

            isSyncing = true;

            try
            {
                // 1. 处理离线队列
                await ProcessOfflineQueueAsync(result);

                // 2. 同步所有槽位
                for (int i = 0; i < config.maxSlots; i++)
                {
                    try
                    {
                        await SyncSlotAsync(i, result);
                    }
                    catch (Exception e)
                    {
                        Debug.LogError($"同步槽位 {i} 失败: {e.Message}");
                        result.failedSlots.Add(i);
                    }
                }

                OnSyncComplete?.Invoke(result);
                return result;
            }
            finally
            {
                isSyncing = false;
            }
        }

        private async UniTask ProcessOfflineQueueAsync(SyncResult result)
        {
            while (offlineQueue.HasPendingOperations)
            {
                var operation = offlineQueue.Dequeue();

                try
                {
                    if (operation.type == OfflineOperationType.Save)
                    {
                        var saveResult = await cloudProvider.SaveAsync(operation.slotIndex, operation.saveData);
                        if (saveResult.success)
                        {
                            result.uploadedCount++;
                        }
                        else
                        {
                            offlineQueue.MarkFailed(operation, true);
                        }
                    }
                    else if (operation.type == OfflineOperationType.Delete)
                    {
                        var deleteResult = await cloudProvider.DeleteAsync(operation.slotIndex);
                        if (deleteResult)
                        {
                            result.uploadedCount++;
                        }
                        else
                        {
                            offlineQueue.MarkFailed(operation, true);
                        }
                    }
                }
                catch (Exception e)
                {
                    Debug.LogError($"处理离线操作失败: {e.Message}");
                    offlineQueue.MarkFailed(operation, true);
                }

                // 报告进度
                OnSyncProgress?.Invoke(new SyncProgress
                {
                    operation = "离线同步",
                    currentSlot = 0,
                    totalSlots = 1,
                    progress = 1f,
                    statusText = "处理离线操作"
                });
            }
        }

        private async UniTask SyncSlotAsync(int slot, SyncResult result)
        {
            // 报告进度
            OnSyncProgress?.Invoke(new SyncProgress
            {
                operation = "同步",
                currentSlot = slot,
                totalSlots = config.maxSlots,
                progress = 0f,
                statusText = $"同步槽位 {slot}"
            });

            // 获取本地和云端数据
            var localData = cacheProvider.LoadFromLocal(slot);
            var cloudResult = await cloudProvider.LoadAsync(slot);

            if (cloudResult.success && localData != null)
            {
                // 检测冲突
                var conflict = conflictDetector.Detect(slot, localData, cloudResult.data);

                if (conflict != null)
                {
                    result.conflictCount++;

                    // 自动解决冲突
                    await conflictResolver.ResolveAsync(conflict);
                }
                else
                {
                    // 无冲突，比较时间戳
                    if (cloudResult.data.timestamp > localData.timestamp)
                    {
                        // 云端更新，下载
                        cacheProvider.SaveToLocal(slot, cloudResult.data);
                        result.downloadedCount++;
                    }
                }
            }
            else if (cloudResult.success && localData == null)
            {
                // 只有云端有数据，下载
                cacheProvider.SaveToLocal(slot, cloudResult.data);
                result.downloadedCount++;
            }
            else if (localData != null && !cloudResult.success)
            {
                // 只有本地有数据，上传
                await cloudProvider.SaveAsync(slot, localData);
                result.uploadedCount++;
            }

            // 报告完成
            OnSyncProgress?.Invoke(new SyncProgress
            {
                operation = "同步",
                currentSlot = slot,
                totalSlots = config.maxSlots,
                progress = 1f,
                statusText = $"槽位 {slot} 同步完成"
            });
        }

        #endregion

        #region 槽位管理

        /// <summary>
        /// 获取所有槽位信息
        /// </summary>
        public async UniTask<List<SlotInfo>> GetSlotListAsync()
        {
            return await cloudProvider.GetSlotListAsync();
        }

        /// <summary>
        /// 设置当前槽位
        /// </summary>
        public void SetCurrentSlot(int slot)
        {
            if (ValidateSlot(slot))
            {
                PlayerPrefs.SetInt("CloudSave_CurrentSlot", slot);
            }
        }

        /// <summary>
        /// 获取当前槽位
        /// </summary>
        public int GetCurrentSlot()
        {
            return PlayerPrefs.GetInt("CloudSave_CurrentSlot", 0);
        }

        #endregion

        #region 自动保存

        private void StartAutoSave()
        {
            if (autoSaveHandle != -1)
            {
                StopAutoSave();
            }

            autoSaveHandle = StartAutoSaveCoroutine().GetAwaiter();
        }

        private async UniTaskVoid StartAutoSaveCoroutine()
        {
            while (true)
            {
                await UniTask.Delay(TimeSpan.FromSeconds(config.autoSaveInterval));

                if (isInitialized)
                {
                    // 触发自动保存事件
                    Debug.Log("自动保存触发");
                    // 由游戏层调用SaveCurrentAsync
                }
            }
        }

        private void StopAutoSave()
        {
            if (autoSaveHandle != -1)
            {
                // 取消协程
                autoSaveHandle = -1;
            }
        }

        #endregion

        #region 事件处理

        private void HandleNetworkStateChanged(bool isOnline)
        {
            Debug.Log($"网络状态变化: {(isOnline ? "在线" : "离线")}");

            OnNetworkChanged?.Invoke(isOnline);

            if (isOnline && offlineQueue.HasPendingOperations)
            {
                // 网络恢复，自动同步离线队列
                SyncAsync().Forget();
            }
        }

        private void HandleConflictDetected(ConflictEvent conflict)
        {
            Debug.Log($"检测到冲突: slot={conflict.slotIndex}");
            OnConflict?.Invoke(conflict);
        }

        private void HandleConflictResolved(ConflictEvent conflict, ConflictResolution resolution, CloudSaveData data)
        {
            Debug.Log($"冲突已解决: slot={conflict.slotIndex}, resolution={resolution}");
        }

        #endregion

        #region 辅助方法

        private bool ValidateSlot(int slot)
        {
            return slot >= 0 && slot < config.maxSlots;
        }

        #endregion

        #region IDisposable

        public void Dispose()
        {
            StopAutoSave();

            networkMonitor?.Dispose();
            networkMonitor.OnNetworkStateChanged -= HandleNetworkStateChanged;

            conflictResolver.OnConflictDetected -= HandleConflictDetected;
            conflictResolver.OnConflictResolved -= HandleConflictResolved;

            if (_instance == this)
            {
                _instance = null;
            }

            isInitialized = false;
        }

        #endregion
    }

    #region 泛型加载结果

    /// <summary>
    /// 泛型加载结果
    /// </summary>
    public class LoadResult<T> where T : class
    {
        public bool success;
        public string errorCode;
        public string errorMessage;
        public T data;
        public DataSource source;
        public bool isCloudNewer;

        public static LoadResult<T> Succeeded(T data, DataSource source, bool cloudNewer = false)
        {
            return new LoadResult<T>
            {
                success = true,
                data = data,
                source = source,
                isCloudNewer = cloudNewer
            };
        }

        public static LoadResult<T> Failed(string code, string message)
        {
            return new LoadResult<T>
            {
                success = false,
                errorCode = code,
                errorMessage = message
            };
        }
    }

    #endregion
}
```

## 三、Builder模式

```csharp
using CloudSave.Platform;
using CloudSave.Storage;

namespace CloudSave.Core
{
    /// <summary>
    /// CloudSaveManager构建器
    /// </summary>
    public class CloudSaveBuilder
    {
        private CloudSaveConfig config = new CloudSaveConfig();
        private PlatformInfo platformInfo;

        public CloudSaveBuilder WithMaxSlots(int maxSlots)
        {
            config.maxSlots = maxSlots;
            return this;
        }

        public CloudSaveBuilder WithAutoSave(float interval = 300f)
        {
            config.enableAutoSave = true;
            config.autoSaveInterval = interval;
            return this;
        }

        public CloudSaveBuilder WithEncryption(string key)
        {
            config.enableEncryption = true;
            config.EncryptionKey = key;
            return this;
        }

        public CloudSaveBuilder WithAutoSync(bool enabled = true)
        {
            config.enableAutoSync = enabled;
            return this;
        }

        public CloudSaveBuilder WithDebugMode(bool enabled = true)
        {
            config.enableDebugLog = enabled;
            return this;
        }

        public CloudSaveBuilder WithPlatform(PlatformInfo info)
        {
            this.platformInfo = info;
            return this;
        }

        public async UniTask<CloudSaveManager> BuildAsync()
        {
            // 创建缓存提供者
            var cacheProvider = new LocalCacheProvider(config);

            // 检测平台
            if (platformInfo == null)
            {
                var detector = new PlatformDetector();
                platformInfo = await detector.InitializeAsync();
            }

            // 创建工厂
            var factory = new CloudProviderFactory(config, cacheProvider, platformInfo);

            // 创建管理器
            var manager = new CloudSaveManager(config, factory, cacheProvider);

            // 初始化
            await manager.InitializeAsync();

            return manager;
        }
    }
}
```

## 四、使用示例

```csharp
using UnityEngine;
using Cysharp.Threading.Tasks;
using CloudSave.Core;

namespace CloudSave.Examples
{
    /// <summary>
    /// 云存档管理器使用示例
    /// </summary>
    public class CloudSaveManagerExample : MonoBehaviour
    {
        private CloudSaveManager saveManager;

        private async void Start()
        {
            // 方式1: 使用Builder
            saveManager = await new CloudSaveBuilder()
                .WithMaxSlots(3)
                .WithAutoSave(300f)
                .WithEncryption("my-secret-key")
                .WithDebugMode(true)
                .BuildAsync();

            // 方式2: 直接创建
            // var config = new CloudSaveConfig { maxSlots = 3 };
            // saveManager = await CloudSaveManager.Initialize(config);

            // 注册事件
            saveManager.OnSyncComplete += OnSyncComplete;
            saveManager.OnConflict += OnConflict;
            saveManager.OnNetworkChanged += OnNetworkChanged;
        }

        #region 基本操作

        /// <summary>
        /// 保存游戏
        /// </summary>
        public async UniTaskVoid SaveGame()
        {
            var gameData = new GameSaveData
            {
                level = 10,
                coins = 1000,
                playTime = 3600
            };

            var result = await saveManager.SaveAsync(0, gameData);

            if (result.success)
            {
                Debug.Log($"保存成功，已同步到云端: {result.syncedToCloud}");
            }
            else
            {
                Debug.LogError($"保存失败: {result.errorMessage}");
            }
        }

        /// <summary>
        /// 加载游戏
        /// </summary>
        public async UniTaskVoid LoadGame()
        {
            var result = await saveManager.LoadAsync<GameSaveData>(0);

            if (result.success)
            {
                Debug.Log($"加载成功: level={result.data.level}, coins={result.data.coins}");
            }
            else
            {
                Debug.LogError($"加载失败: {result.errorMessage}");
            }
        }

        /// <summary>
        /// 手动同步
        /// </summary>
        public async UniTaskVoid ManualSync()
        {
            var result = await saveManager.SyncAsync();

            Debug.Log($"同步完成: 上传{result.uploadedCount}, 下载{result.downloadedCount}, 冲突{result.conflictCount}");
        }

        #endregion

        #region 事件处理

        private void OnSyncComplete(SyncResult result)
        {
            Debug.Log($"同步完成: {result.success}");
        }

        private void OnConflict(ConflictEvent conflict)
        {
            Debug.Log($"检测到冲突: slot={conflict.slotIndex}");
            // 显示冲突解决UI
        }

        private void OnNetworkChanged(bool isOnline)
        {
            Debug.Log($"网络状态: {(isOnline ? "在线" : "离线")}");
        }

        #endregion

        private void OnDestroy()
        {
            if (saveManager != null)
            {
                saveManager.OnSyncComplete -= OnSyncComplete;
                saveManager.OnConflict -= OnConflict;
                saveManager.OnNetworkChanged -= OnNetworkChanged;
                saveManager.Dispose();
            }
        }
    }
}
```
