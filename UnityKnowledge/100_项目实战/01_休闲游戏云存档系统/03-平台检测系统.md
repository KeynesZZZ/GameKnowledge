## 二、平台类型定义

```csharp
namespace CloudSave.Platform
{
    /// <summary>
    /// 运行平台类型
    /// </summary>
    public enum PlatformType
    {
        /// <summary>未知平台</summary>
        Unknown,

        // ===== 原生平台 =====
        /// <summary>iOS原生</summary>
        IOS,

        /// <summary>Android原生</summary>
        Android,

        /// <summary>PC/Mac独立版</summary>
        Standalone,

        // ===== 小游戏平台 =====
        /// <summary>微信小游戏</summary>
        WeChatMiniGame,

        /// <summary>抖音小游戏</summary>
        DouyinMiniGame,

        /// <summary>华为快游戏</summary>
        HuaweiQuickGame,

        /// <summary>通用WebGL</summary>
        WebGL
    }

    /// <summary>
    /// 云服务类型
    /// </summary>
    public enum CloudServiceType
    {
        /// <summary>Firebase (海外)</summary>
        Firebase,

        /// <summary>PlayFab (海外)</summary>
        PlayFab,

        /// <summary>微信云开发</summary>
        WeChatCloud,

        /// <summary>抖音云开发</summary>
        DouyinCloud,

        /// <summary>华为AGC</summary>
        HuaweiAGC,

        /// <summary>REST API</summary>
        REST,

        /// <summary>仅本地</summary>
        LocalOnly
    }

    /// <summary>
    /// 平台信息
    /// </summary>
    public class PlatformInfo
    {
        public PlatformType platform;
        public CloudServiceType cloudService;
        public string platformName;
        public string sdkVersion;
        public bool isNetworkAvailable;
        public bool isLoggedIn;
        public string userId;
        public string deviceId;

        /// <summary>
        /// 是否支持云存档
        /// </summary>
        public bool SupportsCloudSave => cloudService != CloudServiceType.LocalOnly;

        /// <summary>
        /// 是否是小游戏平台
        /// </summary>
        public bool IsMiniGame => platform == PlatformType.WeChatMiniGame ||
                                  platform == PlatformType.DouyinMiniGame ||
                                  platform == PlatformType.HuaweiQuickGame;
    }
}
```

## 三、平台检测器

```csharp
using System;
using UnityEngine;
using Cysharp.Threading.Tasks;

namespace CloudSave.Platform
{
    /// <summary>
    /// 平台检测器 - 检测运行平台并返回对应信息
    /// </summary>
    public class PlatformDetector
    {
        private PlatformInfo cachedInfo;
        private bool isInitialized;

        /// <summary>
        /// 当前平台信息
        /// </summary>
        public PlatformInfo CurrentPlatform => cachedInfo;

        /// <summary>
        /// 初始化平台检测
        /// </summary>
        public async UniTask<PlatformInfo> InitializeAsync()
        {
            if (isInitialized)
            {
                return cachedInfo;
            }

            cachedInfo = new PlatformInfo
            {
                deviceId = GenerateDeviceId(),
                isNetworkAvailable = CheckNetworkAvailable()
            };

            // 检测平台
            if (Application.platform == RuntimePlatform.WebGLPlayer)
            {
                await DetectWebGLPlatformAsync(cachedInfo);
            }
            else
            {
                DetectNativePlatform(cachedInfo);
            }

            isInitialized = true;
            return cachedInfo;
        }

        /// <summary>
        /// 检测原生平台
        /// </summary>
        private void DetectNativePlatform(PlatformInfo info)
        {
            switch (Application.platform)
            {
                case RuntimePlatform.IPhonePlayer:
                    info.platform = PlatformType.IOS;
                    info.cloudService = CloudServiceType.Firebase;
                    info.platformName = "iOS";
                    break;

                case RuntimePlatform.Android:
                    info.platform = PlatformType.Android;
                    info.cloudService = CloudServiceType.Firebase;
                    info.platformName = "Android";
                    break;

                case RuntimePlatform.OSXPlayer:
                case RuntimePlatform.WindowsPlayer:
                case RuntimePlatform.LinuxPlayer:
                    info.platform = PlatformType.Standalone;
                    info.cloudService = CloudServiceType.LocalOnly;
                    info.platformName = "Standalone";
                    break;

                default:
                    info.platform = PlatformType.Unknown;
                    info.cloudService = CloudServiceType.LocalOnly;
                    info.platformName = "Unknown";
                    break;
            }

#if DEBUG
            // 编辑器模式模拟
            if (Application.isEditor)
            {
                info.platformName = "Editor";
            }
#endif
        }

        /// <summary>
        /// 检测WebGL平台（通过JS交互）
        /// </summary>
        private async UniTask DetectWebGLPlatformAsync(PlatformInfo info)
        {
            info.platform = PlatformType.WebGL;
            info.platformName = "WebGL";

#if UNITY_WEBGL && !UNITY_EDITOR
            try
            {
                // 检测微信小游戏
                bool isWeChat = await CheckJsGlobalObjectAsync("wx");
                if (isWeChat)
                {
                    info.platform = PlatformType.WeChatMiniGame;
                    info.cloudService = CloudServiceType.WeChatCloud;
                    info.platformName = "微信小游戏";
                    info.sdkVersion = await GetJsSdkVersionAsync("wx");
                    return;
                }

                // 检测抖音小游戏
                bool isDouyin = await CheckJsGlobalObjectAsync("tt");
                if (isDouyin)
                {
                    info.platform = PlatformType.DouyinMiniGame;
                    info.cloudService = CloudServiceType.DouyinCloud;
                    info.platformName = "抖音小游戏";
                    info.sdkVersion = await GetJsSdkVersionAsync("tt");
                    return;
                }

                // 检测华为快游戏
                bool isHuawei = await CheckJsGlobalObjectAsync("hbs");
                if (isHuawei)
                {
                    info.platform = PlatformType.HuaweiQuickGame;
                    info.cloudService = CloudServiceType.HuaweiAGC;
                    info.platformName = "华为快游戏";
                    return;
                }

                // 通用WebGL
                info.cloudService = CloudServiceType.REST;
            }
            catch (Exception e)
            {
                Debug.LogWarning($"平台检测失败: {e.Message}");
                info.cloudService = CloudServiceType.LocalOnly;
            }
#else
            // 编辑器模拟
            info.cloudService = CloudServiceType.LocalOnly;
            await UniTask.CompletedTask;
#endif
        }

        /// <summary>
        /// 检查网络可用性
        /// </summary>
        public bool CheckNetworkAvailable()
        {
#if UNITY_WEBGL && !UNITY_EDITOR
            return CheckNetworkJS();
#else
            return Application.internetReachability != NetworkReachability.NotReachable;
#endif
        }

        /// <summary>
        /// 刷新平台信息
        /// </summary>
        public async UniTask RefreshAsync()
        {
            if (cachedInfo != null)
            {
                cachedInfo.isNetworkAvailable = CheckNetworkAvailable();
            }
            await UniTask.CompletedTask;
        }

        /// <summary>
        /// 生成设备唯一标识
        /// </summary>
        private string GenerateDeviceId()
        {
            string deviceId = PlayerPrefs.GetString("CloudSave_DeviceId", "");

            if (string.IsNullOrEmpty(deviceId))
            {
                deviceId = Guid.NewGuid().ToString("N");
                PlayerPrefs.SetString("CloudSave_DeviceId", deviceId);
                PlayerPrefs.Save();
            }

            return deviceId;
        }

#if UNITY_WEBGL && !UNITY_EDITOR
        #region JS交互方法

        [System.Runtime.InteropServices.DllImport("__Internal")]
        private static extern bool CheckJsGlobalObject(string objName);

        [System.Runtime.InteropServices.DllImport("__Internal")]
        private static extern string GetJsSdkVersion(string objName);

        [System.Runtime.InteropServices.DllImport("__Internal")]
        private static extern bool CheckNetworkJS();

        private async UniTask<bool> CheckJsGlobalObjectAsync(string objName)
        {
            await UniTask.SwitchToMainThread();
            return CheckJsGlobalObject(objName);
        }

        private async UniTask<string> GetJsSdkVersionAsync(string objName)
        {
            await UniTask.SwitchToMainThread();
            return GetJsSdkVersion(objName);
        }

        #endregion
#else
        private async UniTask<bool> CheckJsGlobalObjectAsync(string objName)
        {
            await UniTask.CompletedTask;
            return false;
        }

        private async UniTask<string> GetJsSdkVersionAsync(string objName)
        {
            await UniTask.CompletedTask;
            return "";
        }
#endif
    }
}
```

## 四、WebGL JS交互库

需要在Unity项目中创建 `.jslib` 文件放在 `Assets/Plugins/WebGL/` 目录下。

### 4.1 PlatformDetector.jslib

```javascript
// PlatformDetector.jslib
// 放置路径: Assets/Plugins/WebGL/PlatformDetector.jslib

var PlatformDetector = {
    // 检查JS全局对象是否存在
    CheckJsGlobalObject: function(objName) {
        var name = UTF8ToString(objName);
        var obj = window[name];
        return obj !== undefined && obj !== null;
    },

    // 获取SDK版本
    GetJsSdkVersion: function(objName) {
        var name = UTF8ToString(objName);
        var obj = window[name];

        if (obj && obj.getSystemInfoSync) {
            try {
                var info = obj.getSystemInfoSync();
                var version = info.SDKVersion || info.version || "unknown";
                return allocateUTF8(version);
            } catch (e) {
                return allocateUTF8("unknown");
            }
        }

        return allocateUTF8("unknown");
    },

    // 检查网络状态
    CheckNetworkJS: function() {
        return navigator.onLine;
    },

    // 获取用户信息（微信）
    GetWeChatUserInfo: function() {
        if (window.wx && window.wx.getUserInfo) {
            try {
                var userInfo = window.wx.getUserInfoSync();
                if (userInfo && userInfo.userInfo) {
                    return allocateUTF8(JSON.stringify({
                        nickName: userInfo.userInfo.nickName,
                        avatarUrl: userInfo.userInfo.avatarUrl,
                        gender: userInfo.userInfo.gender
                    }));
                }
            } catch (e) {
                // 用户未授权
            }
        }
        return allocateUTF8("");
    },

    // 检查微信登录状态
    CheckWeChatLogin: function() {
        if (window.wx && window.wx.checkSession) {
            window.wx.checkSession({
                success: function() {
                    // session有效
                },
                fail: function() {
                    // session失效，需要重新登录
                }
            });
        }
        return false;
    }
};

mergeInto(LibraryManager.library, PlatformDetector);
```

## 五、网络状态监听

```csharp
using System;
using UnityEngine;
using Cysharp.Threading.Tasks;

namespace CloudSave.Platform
{
    /// <summary>
    /// 网络状态监听器
    /// </summary>
    public class NetworkMonitor : IDisposable
    {
        private PlatformDetector detector;
        private bool isMonitoring;
        private float checkInterval = 5f;
        private bool lastNetworkState;

        /// <summary>
        /// 网络状态变化事件
        /// </summary>
        public event Action<bool> OnNetworkStateChanged;

        /// <summary>
        /// 当前是否在线
        /// </summary>
        public bool IsOnline { get; private set; }

        public NetworkMonitor(PlatformDetector detector)
        {
            this.detector = detector;
            IsOnline = detector.CheckNetworkAvailable();
            lastNetworkState = IsOnline;
        }

        /// <summary>
        /// 开始监听
        /// </summary>
        public void StartMonitoring()
        {
            if (isMonitoring) return;
            isMonitoring = true;

            MonitorAsync().Forget();
        }

        /// <summary>
        /// 停止监听
        /// </summary>
        public void StopMonitoring()
        {
            isMonitoring = false;
        }

        private async UniTaskVoid MonitorAsync()
        {
            while (isMonitoring)
            {
                await UniTask.Delay(TimeSpan.FromSeconds(checkInterval));

                bool currentState = detector.CheckNetworkAvailable();

                if (currentState != lastNetworkState)
                {
                    lastNetworkState = currentState;
                    IsOnline = currentState;
                    OnNetworkStateChanged?.Invoke(currentState);

                    Debug.Log($"网络状态变化: {(currentState ? "在线" : "离线")}");
                }
            }
        }

#if UNITY_WEBGL && !UNITY_EDITOR
        /// <summary>
        /// 注册JS网络事件监听
        /// </summary>
        public void RegisterWebGLEvents()
        {
            // 需要配合JS监听online/offline事件
            // 这里通过JS调用C#方法实现
        }
#endif

        public void Dispose()
        {
            StopMonitoring();
            OnNetworkStateChanged = null;
        }
    }
}
```

## 六、Provider工厂

```csharp
using UnityEngine;

namespace CloudSave.Platform
{
    /// <summary>
    /// 云服务提供者工厂
    /// </summary>
    public static class CloudProviderFactory
    {
        /// <summary>
        /// 创建云服务提供者
        /// </summary>
        public static ICloudProvider CreateProvider(PlatformInfo platformInfo)
        {
            switch (platformInfo.cloudService)
            {
#if UNITY_WEBGL && !UNITY_EDITOR
                case CloudServiceType.WeChatCloud:
                    return new WeChatProvider();

                case CloudServiceType.DouyinCloud:
                    return new DouyinProvider();

                case CloudServiceType.HuaweiAGC:
                    return new HuaweiProvider();
#endif

#if !UNITY_WEBGL || UNITY_EDITOR
                case CloudServiceType.Firebase:
                    return new FirebaseProvider();

                case CloudServiceType.PlayFab:
                    return new PlayFabProvider();
#endif

                case CloudServiceType.REST:
                    return new RESTProvider();

                case CloudServiceType.LocalOnly:
                default:
                    return new LocalProvider();
            }
        }

        /// <summary>
        /// 获取推荐的云服务类型
        /// </summary>
        public static CloudServiceType GetRecommendedService(PlatformType platform)
        {
            return platform switch
            {
                PlatformType.IOS => CloudServiceType.Firebase,
                PlatformType.Android => CloudServiceType.Firebase,
                PlatformType.WeChatMiniGame => CloudServiceType.WeChatCloud,
                PlatformType.DouyinMiniGame => CloudServiceType.DouyinCloud,
                PlatformType.HuaweiQuickGame => CloudServiceType.HuaweiAGC,
                PlatformType.WebGL => CloudServiceType.REST,
                _ => CloudServiceType.LocalOnly
            };
        }
    }
}
```

## 七、使用示例

```csharp
using UnityEngine;
using Cysharp.Threading.Tasks;

namespace CloudSave.Examples
{
    /// <summary>
    /// 平台检测使用示例
    /// </summary>
    public class PlatformDetectorExample : MonoBehaviour
    {
        private PlatformDetector detector;
        private NetworkMonitor networkMonitor;

        private async void Start()
        {
            // 创建检测器
            detector = new PlatformDetector();

            // 初始化并获取平台信息
            var platformInfo = await detector.InitializeAsync();

            Debug.Log($"平台: {platformInfo.platformName}");
            Debug.Log($"云服务: {platformInfo.cloudService}");
            Debug.Log($"支持云存档: {platformInfo.SupportsCloudSave}");
            Debug.Log($"是小游戏: {platformInfo.IsMiniGame}");

            // 创建对应的Provider
            var provider = CloudProviderFactory.CreateProvider(platformInfo);

            // 启动网络监听
            networkMonitor = new NetworkMonitor(detector);
            networkMonitor.OnNetworkStateChanged += OnNetworkChanged;
            networkMonitor.StartMonitoring();
        }

        private void OnNetworkChanged(bool isOnline)
        {
            Debug.Log($"网络状态: {(isOnline ? "已连接" : "已断开")}");

            if (isOnline)
            {
                // 网络恢复，触发离线队列同步
                SyncOfflineData().Forget();
            }
        }

        private async UniTaskVoid SyncOfflineData()
        {
            // 同步离线数据
            Debug.Log("开始同步离线数据...");
            await UniTask.Delay(1000);
            Debug.Log("离线数据同步完成");
        }

        private void OnDestroy()
        {
            networkMonitor?.Dispose();
        }
    }
}
```
