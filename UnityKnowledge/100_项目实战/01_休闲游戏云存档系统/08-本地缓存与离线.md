## 三、平台存储实现

### 3.1 原生平台存储（iOS/Android/Standalone）

```csharp
using System;
using System.IO;
using System.Text;
using UnityEngine;

namespace CloudSave.Storage
{
    /// <summary>
    /// 原生平台存储适配器
    /// 使用PlayerPrefs + 文件存储
    /// </summary>
    public class NativeStorageAdapter : IStorageAdapter
    {
        private readonly string saveDirectory;
        private const string KEY_PREFIX = "CloudSave_";

        public NativeStorageAdapter()
        {
            saveDirectory = Path.Combine(Application.persistentDataPath, "CloudSave");

            if (!Directory.Exists(saveDirectory))
            {
                Directory.CreateDirectory(saveDirectory);
            }
        }

        /// <summary>
        /// 保存数据
        /// 小数据用PlayerPrefs，大数据用文件
        /// </summary>
        public bool Save(string key, string value)
        {
            if (string.IsNullOrEmpty(key))
            {
                return false;
            }

            string fullKey = KEY_PREFIX + key;

            try
            {
                // 小于1KB用PlayerPrefs
                if (value == null || value.Length < 1024)
                {
                    PlayerPrefs.SetString(fullKey, value ?? "");
                    PlayerPrefs.Save();
                }
                else
                {
                    // 大数据用文件
                    string filePath = GetFilePath(key);
                    File.WriteAllText(filePath, value, Encoding.UTF8);

                    // PlayerPrefs记录文件存在标记
                    PlayerPrefs.SetInt(fullKey + "_file", 1);
                    PlayerPrefs.Save();
                }

                return true;
            }
            catch (Exception e)
            {
                Debug.LogError($"保存数据失败: {key}, {e.Message}");
                return false;
            }
        }

        /// <summary>
        /// 加载数据
        /// </summary>
        public string Load(string key)
        {
            if (string.IsNullOrEmpty(key))
            {
                return null;
            }

            string fullKey = KEY_PREFIX + key;

            try
            {
                // 检查是否使用文件存储
                if (PlayerPrefs.GetInt(fullKey + "_file", 0) == 1)
                {
                    string filePath = GetFilePath(key);
                    if (File.Exists(filePath))
                    {
                        return File.ReadAllText(filePath, Encoding.UTF8);
                    }
                }
                else
                {
                    return PlayerPrefs.GetString(fullKey, null);
                }
            }
            catch (Exception e)
            {
                Debug.LogError($"加载数据失败: {key}, {e.Message}");
            }

            return null;
        }

        /// <summary>
        /// 删除数据
        /// </summary>
        public bool Delete(string key)
        {
            if (string.IsNullOrEmpty(key))
            {
                return false;
            }

            string fullKey = KEY_PREFIX + key;

            try
            {
                // 删除PlayerPrefs
                PlayerPrefs.DeleteKey(fullKey);
                PlayerPrefs.DeleteKey(fullKey + "_file");

                // 删除文件
                string filePath = GetFilePath(key);
                if (File.Exists(filePath))
                {
                    File.Delete(filePath);
                }

                PlayerPrefs.Save();
                return true;
            }
            catch (Exception e)
            {
                Debug.LogError($"删除数据失败: {key}, {e.Message}");
                return false;
            }
        }

        /// <summary>
        /// 检查键是否存在
        /// </summary>
        public bool Exists(string key)
        {
            if (string.IsNullOrEmpty(key))
            {
                return false;
            }

            string fullKey = KEY_PREFIX + key;

            if (PlayerPrefs.HasKey(fullKey))
            {
                return true;
            }

            string filePath = GetFilePath(key);
            return File.Exists(filePath);
        }

        /// <summary>
        /// 获取所有键
        /// </summary>
        public string[] GetAllKeys()
        {
            // 注意：PlayerPrefs没有直接获取所有键的API
            // 这里需要维护一个键列表
            var keysJson = PlayerPrefs.GetString(KEY_PREFIX + "_keylist", "[]");
            return ParseKeyList(keysJson);
        }

        /// <summary>
        /// 清空所有数据
        /// </summary>
        public void Clear()
        {
            var keys = GetAllKeys();
            foreach (var key in keys)
            {
                Delete(key);
            }

            // 清空目录
            if (Directory.Exists(saveDirectory))
            {
                Directory.Delete(saveDirectory, true);
                Directory.CreateDirectory(saveDirectory);
            }
        }

        private string GetFilePath(string key)
        {
            // 使用安全的文件名
            string safeKey = Convert.ToBase64String(Encoding.UTF8.GetBytes(key));
            return Path.Combine(saveDirectory, safeKey + ".dat");
        }

        private string[] ParseKeyList(string json)
        {
            try
            {
                return JsonUtility.FromJson<KeyList>(json).keys.ToArray();
            }
            catch
            {
                return Array.Empty<string>();
            }
        }

        [Serializable]
        private class KeyList
        {
            public List<string> keys = new List<string>();
        }
    }
}
```

### 3.2 WebGL通用存储

```csharp
using System;
using System.Runtime.InteropServices;
using UnityEngine;

namespace CloudSave.Storage
{
    /// <summary>
    /// WebGL通用存储适配器
    /// 使用IndexedDB通过JS交互
    /// </summary>
    public class WebGLStorageAdapter : IStorageAdapter
    {
        private const string KEY_PREFIX = "CloudSave_";

        public WebGLStorageAdapter()
        {
#if UNITY_WEBGL && !UNITY_EDITOR
            InitIndexedDB();
#endif
        }

        public bool Save(string key, string value)
        {
            if (string.IsNullOrEmpty(key)) return false;

            string fullKey = KEY_PREFIX + key;

#if UNITY_WEBGL && !UNITY_EDITOR
            return SaveToIndexedDB(fullKey, value);
#else
            return false;
#endif
        }

        public string Load(string key)
        {
            if (string.IsNullOrEmpty(key)) return null;

            string fullKey = KEY_PREFIX + key;

#if UNITY_WEBGL && !UNITY_EDITOR
            return LoadFromIndexedDB(fullKey);
#else
            return null;
#endif
        }

        public bool Delete(string key)
        {
            if (string.IsNullOrEmpty(key)) return false;

            string fullKey = KEY_PREFIX + key;

#if UNITY_WEBGL && !UNITY_EDITOR
            return DeleteFromIndexedDB(fullKey);
#else
            return false;
#endif
        }

        public bool Exists(string key)
        {
            if (string.IsNullOrEmpty(key)) return false;

            string fullKey = KEY_PREFIX + key;

#if UNITY_WEBGL && !UNITY_EDITOR
            return ExistsInIndexedDB(fullKey);
#else
            return false;
#endif
        }

        public string[] GetAllKeys()
        {
#if UNITY_WEBGL && !UNITY_EDITOR
            return GetAllKeysFromIndexedDB(KEY_PREFIX);
#else
            return Array.Empty<string>();
#endif
        }

        {
#if UNITY_WEBGL && !UNITY_EDITOR

#endif

        [DllImport("__Internal")]

        [DllImport("__Internal")]
```

### 3.3 IndexedDB JS实现

```javascript
// Assets/Plugins/WebGL/IndexedDBStorage.jslib

var IndexedDBStorage = {
    db: null,
    dbName: 'CloudSaveDB',
    storeName: 'saves',

    InitIndexedDB: function() {
        var request = indexedDB.open(this.dbName, 1);

        request.onerror = function(event) {
            console.error('IndexedDB初始化失败:', event.target.error);
        };

        request.onupgradeneeded = function(event) {
            var db = event.target.result;
            if (!db.objectStoreNames.contains(this.storeName)) {
                db.createObjectStore(this.storeName, { keyPath: 'key' });
            }
        }.bind(this);

        request.onsuccess = function(event) {
            this.db = event.target.result;
            console.log('IndexedDB初始化成功');
        }.bind(this);
    },

    SaveToIndexedDB: function(key, value) {
        var k = UTF8ToString(key);
        var v = UTF8ToString(value);

        if (!this.db) {
            console.error('IndexedDB未初始化');
            return false;
        }

        try {
            var transaction = this.db.transaction([this.storeName], 'readwrite');
            var store = transaction.objectStore(this.storeName);
            var request = store.put({ key: k, value: v });

            request.onsuccess = function() {
                return true;
            };

            request.onerror = function(event) {
                console.error('保存失败:', event.target.error);
            };

            return true;
        } catch (e) {
            console.error('保存异常:', e);
            return false;
        }
    },

    LoadFromIndexedDB: function(key) {
        var k = UTF8ToString(key);

        if (!this.db) {
            return allocateUTF8("");
        }

        try {
            var transaction = this.db.transaction([this.storeName], 'readonly');
            var store = transaction.objectStore(this.storeName);
            var request = store.get(k);

            request.onsuccess = function(event) {
                if (event.target.result) {
                    return allocateUTF8(event.target.result.value);
                }
                return allocateUTF8("");
            };

            request.onerror = function() {
                return allocateUTF8("");
            };

            // 同步返回（简化处理）
            return allocateUTF8("");
        } catch (e) {
            return allocateUTF8("");
        }
    },

    DeleteFromIndexedDB: function(key) {
        var k = UTF8ToString(key);

        if (!this.db) return false;

        try {
            var transaction = this.db.transaction([this.storeName], 'readwrite');
            var store = transaction.objectStore(this.storeName);
            store.delete(k);
            return true;
        } catch (e) {
            return false;
        }
    },

    ExistsInIndexedDB: function(key) {
        var k = UTF8ToString(key);

        if (!this.db) return false;

        try {
            var transaction = this.db.transaction([this.storeName], 'readonly');
            var store = transaction.objectStore(this.storeName);
            var request = store.get(k);

            return request.result !== undefined;
        } catch (e) {
            return false;
        }
    },

    ClearIndexedDB: function(prefix) {
        if (!this.db) return;

        try {
            var transaction = this.db.transaction([this.storeName], 'readwrite');
            var store = transaction.objectStore(this.storeName);
            store.clear();
        } catch (e) {
            console.error('清空失败:', e);
        }
    }
};

mergeInto(LibraryManager.library, IndexedDBStorage);
```

## 四、本地缓存提供者

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;
using Cysharp.Threading.Tasks;
using CloudSave.Storage;
using CloudSave.Platform;

namespace CloudSave.Core
{
    /// <summary>
    /// 本地缓存提供者实现
    /// </summary>
    public class LocalCacheProvider : ICacheProvider
    {
        private readonly CloudSaveConfig config;
        private readonly IStorageAdapter storageAdapter;
        private readonly IEncryptor encryptor;

        private const string SLOT_KEY_PREFIX = "slot_";
        private const string META_KEY = "metadata";
        private const string OFFLINE_QUEUE_KEY = "offline_queue";

        public LocalCacheProvider(CloudSaveConfig config, PlatformType platform = PlatformType.Unknown)
        {
            this.config = config;

            // 根据平台选择存储适配器
            storageAdapter = CreateStorageAdapter(platform);

            // 初始化加密器
            if (config.enableEncryption)
            {
                encryptor = new AesEncryptor(config.EncryptionKey);
            }
        }

        #region ICacheProvider实现

        public bool SaveToLocal(int slot, CloudSaveData data)
        {
            if (data == null) return false;

            try
            {
                string key = SLOT_KEY_PREFIX + slot;
                string json = JsonUtility.ToJson(data);

                // 加密
                if (encryptor != null)
                {
                    json = encryptor.Encrypt(json);
                }

                bool success = storageAdapter.Save(key, json);

                if (success)
                {
                    UpdateMetadata(slot, data);
                }

                return success;
            }
            catch (Exception e)
            {
                Debug.LogError($"本地保存失败: slot={slot}, {e.Message}");
                return false;
            }
        }

        public CloudSaveData LoadFromLocal(int slot)
        {
            try
            {
                string key = SLOT_KEY_PREFIX + slot;
                string json = storageAdapter.Load(key);

                if (string.IsNullOrEmpty(json))
                {
                    return null;
                }

                // 解密
                if (encryptor != null)
                {
                    json = encryptor.Decrypt(json);
                }

                return JsonUtility.FromJson<CloudSaveData>(json);
            }
            catch (Exception e)
            {
                Debug.LogError($"本地加载失败: slot={slot}, {e.Message}");
                return null;
            }
        }

        public bool DeleteFromLocal(int slot)
        {
            try
            {
                string key = SLOT_KEY_PREFIX + slot;
                bool success = storageAdapter.Delete(key);

                if (success)
                {
                    RemoveFromMetadata(slot);
                }

                return success;
            }
            catch (Exception e)
            {
                Debug.LogError($"本地删除失败: slot={slot}, {e.Message}");
                return false;
            }
        }

        public List<SlotInfo> GetLocalSlotList()
        {
            var result = new List<SlotInfo>();

            try
            {
                var metadata = LoadMetadata();

                for (int i = 0; i < config.maxSlots; i++)
                {
                    if (metadata.TryGetValue(i, out var info))
                    {
                        result.Add(info);
                    }
                    else
                    {
                        result.Add(SlotInfo.Empty(i));
                    }
                }
            }
            catch (Exception e)
            {
                Debug.LogError($"获取本地槽位列表失败: {e.Message}");
            }

            return result;
        }

        public void ClearAllCache()
        {
            try
            {
                storageAdapter.Clear();
                Debug.Log("本地缓存已清空");
            }
            catch (Exception e)
            {
                Debug.LogError($"清空缓存失败: {e.Message}");
            }
        }

        #endregion

        #region 元数据管理

        private Dictionary<int, SlotInfo> LoadMetadata()
        {
            var result = new Dictionary<int, SlotInfo>();

            try
            {
                string json = storageAdapter.Load(META_KEY);

                if (!string.IsNullOrEmpty(json))
                {
                    var metadata = JsonUtility.FromJson<SlotMetadata>(json);

                    foreach (var slot in metadata.slots)
                    {
                        result[slot.slotIndex] = slot;
                    }
                }
            }
            catch (Exception e)
            {
                Debug.LogWarning($"加载元数据失败: {e.Message}");
            }

            return result;
        }

        private void UpdateMetadata(int slot, CloudSaveData data)
        {
            try
            {
                var metadata = LoadMetadata();

                metadata[slot] = new SlotInfo
                {
                    slotIndex = slot,
                    hasData = true,
                    timestamp = data.timestamp,
                    deviceName = data.deviceId,
                    platformName = data.platform,
                    summary = data.summary,
                    syncState = SyncState.LocalOnly
                };

                var saveData = new SlotMetadata { slots = new List<SlotInfo>() };
                foreach (var kvp in metadata)
                {
                    saveData.slots.Add(kvp.Value);
                }

                string json = JsonUtility.ToJson(saveData);
                storageAdapter.Save(META_KEY, json);
            }
            catch (Exception e)
            {
                Debug.LogWarning($"更新元数据失败: {e.Message}");
            }
        }

        private void RemoveFromMetadata(int slot)
        {
            try
            {
                var metadata = LoadMetadata();
                metadata.Remove(slot);

                var saveData = new SlotMetadata { slots = new List<SlotInfo>() };
                foreach (var kvp in metadata)
                {
                    saveData.slots.Add(kvp.Value);
                }

                string json = JsonUtility.ToJson(saveData);
                storageAdapter.Save(META_KEY, json);
            }
            catch (Exception e)
            {
                Debug.LogWarning($"移除元数据失败: {e.Message}");
            }
        }

        [Serializable]
        private class SlotMetadata
        {
            public List<SlotInfo> slots;
        }

        #endregion

        #region 存储适配器工厂

        private IStorageAdapter CreateStorageAdapter(PlatformType platform)
        {
#if UNITY_WEBGL && !UNITY_EDITOR
            return new WebGLStorageAdapter();
#else
            return new NativeStorageAdapter();
#endif
        }

        #endregion
    }
}
```

## 五、离线队列

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;
using Cysharp.Threading.Tasks;
using CloudSave.Storage;

namespace CloudSave.Core
{
    /// <summary>
    /// 离线操作队列
    /// 网络恢复后自动同步
    /// </summary>
    public class OfflineQueueManager
    {
        private readonly IStorageAdapter storageAdapter;
        private readonly IEncryptor encryptor;

        private const string QUEUE_KEY = "offline_queue";
        private OfflineQueue queue;

        public event Action<OfflineOperation> OnOperationEnqueued;
        public event Action<OfflineOperation> OnOperationDequeued;
        public event Action<int> OnQueueChanged;

        public bool HasPendingOperations => queue?.HasPendingOperations ?? false;
        public int PendingCount => queue?.PendingCount ?? 0;

        public OfflineQueueManager(CloudSaveConfig config)
        {
            // 使用简单的PlayerPrefs存储队列
            storageAdapter = new NativeStorageAdapter();

            if (config.enableEncryption)
            {
                encryptor = new AesEncryptor(config.EncryptionKey);
            }

            LoadQueue();
        }

        /// <summary>
        /// 添加保存操作到队列
        /// </summary>
        public void EnqueueSave(int slot, CloudSaveData data)
        {
            var operation = OfflineOperation.CreateSave(slot, data);
            queue.Enqueue(operation);
            SaveQueue();

            OnOperationEnqueued?.Invoke(operation);
            OnQueueChanged?.Invoke(PendingCount);

            Debug.Log($"离线保存操作已入队: slot={slot}, queueSize={PendingCount}");
        }

        /// <summary>
        /// 添加删除操作到队列
        /// </summary>
        public void EnqueueDelete(int slot)
        {
            var operation = OfflineOperation.CreateDelete(slot);
            queue.Enqueue(operation);
            SaveQueue();

            OnOperationEnqueued?.Invoke(operation);
            OnQueueChanged?.Invoke(PendingCount);

            Debug.Log($"离线删除操作已入队: slot={slot}, queueSize={PendingCount}");
        }

        /// <summary>
        /// 取出下一个操作
        /// </summary>
        public OfflineOperation Dequeue()
        {
            var operation = queue.Dequeue();
            if (operation != null)
            {
                SaveQueue();
                OnOperationDequeued?.Invoke(operation);
                OnQueueChanged?.Invoke(PendingCount);
            }
            return operation;
        }

        /// <summary>
        /// 查看队首操作（不移除）
        /// </summary>
        public OfflineOperation Peek()
        {
            return queue.Peek();
        }

        /// <summary>
        /// 标记操作失败（重新入队或放弃）
        /// </summary>
        public void MarkFailed(OfflineOperation operation, bool retry = true)
        {
            operation.retryCount++;
            operation.lastAttemptTime = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();

            if (retry && operation.retryCount < 5)  // 最多重试5次
            {
                queue.Enqueue(operation);
                SaveQueue();
            }
            else
            {
                Debug.LogWarning($"操作已达到最大重试次数，放弃: {operation.operationId}");
            }
        }

        /// <summary>
        /// 清空队列
        /// </summary>
        public void Clear()
        {
            queue = new OfflineQueue();
            SaveQueue();
            OnQueueChanged?.Invoke(0);
        }

        private void LoadQueue()
        {
            try
            {
                string json = storageAdapter.Load(QUEUE_KEY);

                if (!string.IsNullOrEmpty(json))
                {
                    if (encryptor != null)
                    {
                        json = encryptor.Decrypt(json);
                    }

                    queue = JsonUtility.FromJson<OfflineQueue>(json);
                }
                else
                {
                    queue = new OfflineQueue();
                }
            }
            catch (Exception e)
            {
                Debug.LogWarning($"加载离线队列失败: {e.Message}");
                queue = new OfflineQueue();
            }
        }

        private void SaveQueue()
        {
            try
            {
                string json = JsonUtility.ToJson(queue);

                if (encryptor != null)
                {
                    json = encryptor.Encrypt(json);
                }

                storageAdapter.Save(QUEUE_KEY, json);
            }
            catch (Exception e)
            {
```

## 六、加密工具

```csharp
using System;
using System.IO;
using System.Security.Cryptography;
using System.Text;
using UnityEngine;

namespace CloudSave.Core
{
    /// <summary>
    /// 加密器接口
    /// </summary>
    public interface IEncryptor
    {
        string Encrypt(string plainText);
        string Decrypt(string cipherText);
    }

    /// <summary>
    /// AES加密器
    /// </summary>
    public class AesEncryptor : IEncryptor
    {
        private readonly byte[] key;
        private readonly byte[] iv;

        public AesEncryptor(string secretKey)
        {
            // 从密钥生成固定长度的key和iv
            using (var sha256 = SHA256.Create())
            {
                var hash = sha256.ComputeHash(Encoding.UTF8.GetBytes(secretKey));
                key = new byte[16];
                Array.Copy(hash, key, 16);
                iv = new byte[16];
                Array.Copy(hash, 16, iv, 0, 16);
            }
        }

        public string Encrypt(string plainText)
        {
            if (string.IsNullOrEmpty(plainText))
            {
                return plainText;
            }

            try
            {
                using (var aes = Aes.Create())
                {
                    aes.Key = key;
                    aes.IV = iv;
                    aes.Mode = CipherMode.CBC;
                    aes.Padding = PaddingMode.PKCS7;

                    using (var encryptor = aes.CreateEncryptor())
                    using (var msEncrypt = new MemoryStream())
                    using (var csEncrypt = new CryptoStream(msEncrypt, encryptor, CryptoStreamMode.Write))
                    {
                        byte[] plainBytes = Encoding.UTF8.GetBytes(plainText);
                        csEncrypt.Write(plainBytes, 0, plainBytes.Length);
                        csEncrypt.FlushFinalBlock();

                        return Convert.ToBase64String(msEncrypt.ToArray());
                    }
                }
            }
            catch (Exception e)
            {
                Debug.LogError($"加密失败: {e.Message}");
                return plainText;
            }
        }

        public string Decrypt(string cipherText)
        {
            if (string.IsNullOrEmpty(cipherText))
            {
                return cipherText;
            }

            try
            {
                byte[] cipherBytes = Convert.FromBase64String(cipherText);

                using (var aes = Aes.Create())
                {
                    aes.Key = key;
                    aes.IV = iv;
                    aes.Mode = CipherMode.CBC;
                    aes.Padding = PaddingMode.PKCS7;

                    using (var decryptor = aes.CreateDecryptor())
                    using (var msDecrypt = new MemoryStream(cipherBytes))
                    using (var csDecrypt = new CryptoStream(msDecrypt, decryptor, CryptoStreamMode.Read))
                    using (var srDecrypt = new StreamReader(csDecrypt))
                    {
                        return srDecrypt.ReadToEnd();
                    }
                }
            }
            catch (Exception e)
            {
                Debug.LogError($"解密失败: {e.Message}");
                return cipherText;
            }
        }
    }
}
```

```csharp
using UnityEngine;
using Cysharp.Threading.Tasks;
using CloudSave.Core;
using CloudSave.Storage;

namespace CloudSave.Examples
{
    /// <summary>
    /// 本地缓存使用示例
    /// </summary>
    public class LocalCacheExample : MonoBehaviour
    {
        private LocalCacheProvider cacheProvider;
        private OfflineQueueManager offlineQueue;
        private CloudSaveConfig config;

        private void Start()
        {
            config = new CloudSaveConfig
            {
                maxSlots = 3,
                enableEncryption = true,
                EncryptionKey = "my-secret-key-12345"
            };

            // 创建本地缓存提供者
            cacheProvider = new LocalCacheProvider(config);

            // 创建离线队列管理器
            offlineQueue = new OfflineQueueManager(config);

            // 监听队列变化
            offlineQueue.OnQueueChanged += OnQueueChanged;
        }

        private void OnQueueChanged(int count)
        {
            Debug.Log($"离线队列大小: {count}");
        }

        /// <summary>
        /// 保存到本地
        /// </summary>
        public void SaveLocal(int slot)
        {
            var gameData = new GameSaveData
            {
                level = 10,
                coins = 1000
            };

            var saveData = CloudSaveData.Create(slot, gameData, "device001", "Android");

            bool success = cacheProvider.SaveToLocal(slot, saveData);

            if (success)
            {
                Debug.Log("本地保存成功");
            }
        }

        /// <summary>
        /// 从本地加载
        /// </summary>
        public void LoadLocal(int slot)
        {
            var data = cacheProvider.LoadFromLocal(slot);

            if (data != null)
            {
                var gameData = data.ParseGameData<GameSaveData>();
                Debug.Log($"加载成功: level={gameData.level}");
            }
        }

        /// <summary>
        /// 添加到离线队列
        /// </summary>
        public void AddToOfflineQueue(int slot)
        {
            var gameData = new GameSaveData { level = 20 };
            var saveData = CloudSaveData.Create(slot, gameData, "device001", "Android");

            offlineQueue.EnqueueSave(slot, saveData);
        }

        private void OnDestroy()
        {
            offlineQueue.OnQueueChanged -= OnQueueChanged;
        }
    }
}
```
