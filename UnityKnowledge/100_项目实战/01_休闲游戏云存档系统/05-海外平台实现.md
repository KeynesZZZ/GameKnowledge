## 二、Firebase实现

### 2.1 SDK安装

```markdown
# Firebase SDK安装步骤

1. 下载 Firebase Unity SDK
   https://firebase.google.com/download/unity

2. 导入必要包：
   - FirebaseAuth.unitypackage
   - FirebaseFirestore.unitypackage

3. 配置 GoogleService-Info.plist (iOS)
   或 google-services.json (Android)

4. 在Firebase控制台创建项目并添加应用
```

### 2.2 FirebaseProvider实现

```csharp
using System;
using System.Collections.Generic;
using System.Threading;
using UnityEngine;
using Cysharp.Threading.Tasks;
using Firebase;
using Firebase.Auth;
using Firebase.Extensions;
using Firebase.Firestore;

namespace CloudSave.Providers
{
    /// <summary>
    /// Firebase云存档提供者
    /// </summary>
    public class FirebaseProvider : BaseCloudProvider
    {
        public override string ProviderName => "Firebase";

        private FirebaseFirestore firestore;
        private FirebaseAuth auth;
        private FirebaseUser currentUser;
        private bool isFirebaseInitialized;

        public FirebaseProvider(ICacheProvider cacheProvider, CloudSaveConfig config)
            : base(cacheProvider, config)
        {
        }

        /// <summary>
        /// 初始化Firebase
        /// </summary>
        public override async UniTask<bool> InitializeAsync()
        {
            if (isFirebaseInitialized)
            {
                return true;
            }

            try
            {
                // 检查依赖
                var dependencyStatus = await FirebaseApp.CheckAndFixDependenciesAsync();

                if (dependencyStatus == DependencyStatus.Available)
                {
                    // 初始化Firebase
                    FirebaseApp app = FirebaseApp.DefaultInstance;
                    firestore = FirebaseFirestore.DefaultInstance;
                    auth = FirebaseAuth.DefaultInstance;

                    // 检查登录状态
                    if (auth.CurrentUser != null)
                    {
                        currentUser = auth.CurrentUser;
                    }

                    isFirebaseInitialized = true;
                    Debug.Log("Firebase初始化成功");
                    return true;
                }
                else
                {
                    Debug.LogError($"Firebase依赖不可用: {dependencyStatus}");
                    return false;
                }
            }
            catch (Exception e)
            {
                Debug.LogError($"Firebase初始化失败: {e.Message}");
                return false;
            }
        }

        /// <summary>
        /// 匿名登录（首次使用）
        /// </summary>
        public async UniTask<bool> SignInAnonymouslyAsync()
        {
            try
            {
                var result = await auth.SignInAnonymouslyAsync();
                currentUser = result.User;
                Debug.Log($"匿名登录成功: {currentUser.UserId}");
                return true;
            }
            catch (Exception e)
            {
                Debug.LogError($"匿名登录失败: {e.Message}");
                return false;
            }
        }

        /// <summary>
        /// 检查服务是否可用
        /// </summary>
        public override async UniTask<bool> IsAvailableAsync()
        {
            if (!isFirebaseInitialized)
            {
                await InitializeAsync();
            }

            return isFirebaseInitialized && currentUser != null;
        }

        /// <summary>
        /// 保存到Firestore
        /// </summary>
        protected override async UniTask<SaveResult> SaveToCloudAsync(int slot, CloudSaveData data)
        {
            try
            {
                string docPath = GetDocumentPath(slot);

                // 转换为字典
                var saveDict = ConvertToDict(data);

                // 保存到Firestore
                await firestore.Document(docPath).SetAsync(saveDict, SetOptions.MergeAll);

                Debug.Log($"Firebase保存成功: slot={slot}");
                return SaveResult.Succeeded(data, true);
            }
            catch (Exception e)
            {
                Debug.LogError($"Firebase保存失败: {e.Message}");
                return SaveResult.Failed("FIREBASE_SAVE_ERROR", e.Message);
            }
        }

        /// <summary>
        /// 从Firestore加载
        /// </summary>
        protected override async UniTask<LoadResult> LoadFromCloudAsync(int slot)
        {
            try
            {
                string docPath = GetDocumentPath(slot);

                var snapshot = await firestore.Document(docPath).GetSnapshotAsync();

                if (snapshot.Exists)
                {
                    var data = ConvertFromDict(snapshot.ToDictionary());
                    Debug.Log($"Firebase加载成功: slot={slot}");
                    return LoadResult.Succeeded(data, DataSource.Cloud);
                }
                else
                {
                    return LoadResult.Failed("NO_DATA", $"存档槽位 {slot} 没有云端数据");
                }
            }
            catch (Exception e)
            {
                Debug.LogError($"Firebase加载失败: {e.Message}");
                return LoadResult.Failed("FIREBASE_LOAD_ERROR", e.Message);
            }
        }

        /// <summary>
        /// 从Firestore删除
        /// </summary>
        protected override async UniTask<bool> DeleteFromCloudAsync(int slot)
        {
            try
            {
                string docPath = GetDocumentPath(slot);
                await firestore.Document(docPath).DeleteAsync();
                Debug.Log($"Firebase删除成功: slot={slot}");
                return true;
            }
            catch (Exception e)
            {
                Debug.LogError($"Firebase删除失败: {e.Message}");
                return false;
            }
        }

        /// <summary>
        /// 获取云端槽位列表
        /// </summary>
        protected override async UniTask<List<SlotInfo>> GetCloudSlotListAsync()
        {
            var result = new List<SlotInfo>();

            try
            {
                string collectionPath = GetUserCollectionPath();
                var snapshot = await firestore.Collection(collectionPath).GetSnapshotAsync();

                foreach (var doc in snapshot.Documents)
                {
                    if (doc.Exists)
                    {
                        var data = ConvertFromDict(doc.ToDictionary());
                        var slotInfo = SlotInfo.FromSaveData(data);
                        result.Add(slotInfo);
                    }
                }
            }
            catch (Exception e)
            {
                Debug.LogError($"获取云端槽位失败: {e.Message}");
            }

            return result;
        }

        #region 辅助方法

        private string GetDocumentPath(int slot)
        {
            return $"{GetUserCollectionPath()}/slot_{slot}";
        }

        private string GetUserCollectionPath()
        {
            string userId = currentUser?.UserId ?? "anonymous";
            return $"users/{userId}/saves";
        }

        private Dictionary<string, object> ConvertToDict(CloudSaveData data)
        {
            return new Dictionary<string, object>
            {
                { "formatVersion", data.formatVersion },
                { "gameDataVersion", data.gameDataVersion },
                { "timestamp", data.timestamp },
                { "deviceId", data.deviceId ?? "" },
                { "platform", data.platform ?? "" },
                { "userId", data.userId ?? "" },
                { "slotIndex", data.slotIndex },
                { "gameData", data.gameData ?? "" },
                { "checksum", data.checksum ?? "" },
                { "dataSize", data.dataSize },
                { "summary_playerName", data.summary?.playerName ?? "" },
                { "summary_level", data.summary?.level ?? 0 },
                { "summary_playTime", data.summary?.playTime ?? 0 },
                { "summary_coins", data.summary?.coins ?? 0 },
                { "summary_currentScene", data.summary?.currentScene ?? "" }
            };
        }

        private CloudSaveData ConvertFromDict(Dictionary<string, object> dict)
        {
            var data = new CloudSaveData
            {
                formatVersion = GetValue<int>(dict, "formatVersion"),
                gameDataVersion = GetValue<int>(dict, "gameDataVersion"),
                timestamp = GetValue<long>(dict, "timestamp"),
                deviceId = GetValue<string>(dict, "deviceId"),
                platform = GetValue<string>(dict, "platform"),
                userId = GetValue<string>(dict, "userId"),
                slotIndex = GetValue<int>(dict, "slotIndex"),
                gameData = GetValue<string>(dict, "gameData"),
                checksum = GetValue<string>(dict, "checksum"),
                dataSize = GetValue<int>(dict, "dataSize"),
                summary = new SaveSummary
                {
                    playerName = GetValue<string>(dict, "summary_playerName"),
                    level = GetValue<int>(dict, "summary_level"),
                    playTime = GetValue<long>(dict, "summary_playTime"),
                    coins = GetValue<int>(dict, "summary_coins"),
                    currentScene = GetValue<string>(dict, "summary_currentScene")
                }
            };

            return data;
        }

        private T GetValue<T>(Dictionary<string, object> dict, string key)
        {
            if (dict.TryGetValue(key, out var value))
            {
                if (value is T tValue)
                {
                    return tValue;
                }
                try
                {
                    return (T)Convert.ChangeType(value, typeof(T));
                }
                catch
                {
                    return default;
                }
            }
            return default;
        }

        #endregion
    }
}
```

### 3.2 PlayFabProvider实现

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;
using Cysharp.Threading.Tasks;
using PlayFab;
using PlayFab.ClientModels;

namespace CloudSave.Providers
{
    /// <summary>
    /// PlayFab云存档提供者
    /// </summary>
    public class PlayFabProvider : BaseCloudProvider
    {
        public override string ProviderName => "PlayFab";

        private string titleId;
        private string playFabId;
        private bool isLoggedIn;

        public PlayFabProvider(ICacheProvider cacheProvider, CloudSaveConfig config)
            : base(cacheProvider, config)
        {
            titleId = PlayFabSettings.TitleId;
        }

        /// <summary>
        /// 初始化
        /// </summary>
        public override async UniTask<bool> InitializeAsync()
        {
            if (isInitialized)
            {
                return true;
            }

            try
            {
                // 检查Title ID
                if (string.IsNullOrEmpty(titleId))
                {
                    Debug.LogError("PlayFab Title ID未配置");
                    return false;
                }

                isInitialized = true;
                return true;
            }
            catch (Exception e)
            {
                Debug.LogError($"PlayFab初始化失败: {e.Message}");
                return false;
            }
        }

        /// <summary>
        /// 匿名登录（设备ID）
        /// </summary>
        public async UniTask<bool> LoginWithDeviceAsync(string deviceId)
        {
            try
            {
                var request = new LoginWithCustomIDRequest
                {
                    CustomId = deviceId,
                    CreateAccount = true
                };

                var result = await PlayFabClientAPI.LoginWithCustomIDAsync(request);

                if (result.Error != null)
                {
                    Debug.LogError($"PlayFab登录失败: {result.Error.ErrorMessage}");
                    return false;
                }

                playFabId = result.Result.PlayFabId;
                isLoggedIn = true;
                Debug.Log($"PlayFab登录成功: {playFabId}");
                return true;
            }
            catch (Exception e)
            {
                Debug.LogError($"PlayFab登录异常: {e.Message}");
                return false;
            }
        }

        /// <summary>
        /// 检查服务是否可用
        /// </summary>
        public override async UniTask<bool> IsAvailableAsync()
        {
            if (!isInitialized)
            {
                await InitializeAsync();
            }

            return isInitialized && isLoggedIn;
        }

        /// <summary>
        /// 保存到PlayFab
        /// </summary>
        protected override async UniTask<SaveResult> SaveToCloudAsync(int slot, CloudSaveData data)
        {
            try
            {
                // 使用UserData存储
                var json = JsonUtility.ToJson(data);

                var request = new UpdateUserDataRequest
                {
                    Data = new Dictionary<string, string>
                    {
                        { $"save_slot_{slot}", json }
                    }
                };

                var result = await PlayFabClientAPI.UpdateUserDataAsync(request);

                if (result.Error != null)
                {
                    return SaveResult.Failed("PLAYFAB_SAVE_ERROR", result.Error.ErrorMessage);
                }

                Debug.Log($"PlayFab保存成功: slot={slot}");
                return SaveResult.Succeeded(data, true);
            }
            catch (Exception e)
            {
                Debug.LogError($"PlayFab保存失败: {e.Message}");
                return SaveResult.Failed("PLAYFAB_ERROR", e.Message);
            }
        }

        /// <summary>
        /// 从PlayFab加载
        /// </summary>
        protected override async UniTask<LoadResult> LoadFromCloudAsync(int slot)
        {
            try
            {
                var request = new GetUserDataRequest
                {
                    Keys = new List<string> { $"save_slot_{slot}" }
                };

                var result = await PlayFabClientAPI.GetUserDataAsync(request);

                if (result.Error != null)
                {
                    return LoadResult.Failed("PLAYFAB_LOAD_ERROR", result.Error.ErrorMessage);
                }

                if (result.Result.Data.TryGetValue($"save_slot_{slot}", out var record))
                {
                    var data = JsonUtility.FromJson<CloudSaveData>(record.Value);
                    Debug.Log($"PlayFab加载成功: slot={slot}");
                    return LoadResult.Succeeded(data, DataSource.Cloud);
                }

                return LoadResult.Failed("NO_DATA", $"存档槽位 {slot} 没有云端数据");
            }
            catch (Exception e)
            {
                Debug.LogError($"PlayFab加载失败: {e.Message}");
                return LoadResult.Failed("PLAYFAB_ERROR", e.Message);
            }
        }

        /// <summary>
        /// 从PlayFab删除
        /// </summary>
        protected override async UniTask<bool> DeleteFromCloudAsync(int slot)
        {
            try
            {
                var request = new UpdateUserDataRequest
                {
                    Data = new Dictionary<string, string>
                    {
                        { $"save_slot_{slot}", null }  // null表示删除
                    },
                    DataToRemove = new List<string> { $"save_slot_{slot}" }
                };

                var result = await PlayFabClientAPI.UpdateUserDataAsync(request);

                if (result.Error != null)
                {
                    Debug.LogError($"PlayFab删除失败: {result.Error.ErrorMessage}");
                    return false;
                }

                Debug.Log($"PlayFab删除成功: slot={slot}");
                return true;
            }
            catch (Exception e)
            {
                Debug.LogError($"PlayFab删除异常: {e.Message}");
                return false;
            }
        }

        /// <summary>
        /// 获取云端槽位列表
        /// </summary>
        protected override async UniTask<List<SlotInfo>> GetCloudSlotListAsync()
        {
            var result = new List<SlotInfo>();

            try
            {
                var request = new GetUserDataRequest();
                var response = await PlayFabClientAPI.GetUserDataAsync(request);

                if (response.Error != null)
                {
                    return result;
                }

                foreach (var kvp in response.Result.Data)
                {
                    if (kvp.Key.StartsWith("save_slot_"))
                    {
                        try
                        {
                            var data = JsonUtility.FromJson<CloudSaveData>(kvp.Value.Value);
                            var slotInfo = SlotInfo.FromSaveData(data);
                            result.Add(slotInfo);
                        }
                        catch (Exception e)
                        {
                            Debug.LogWarning($"解析存档失败: {kvp.Key}, {e.Message}");
                        }
                    }
                }
            }
            catch (Exception e)
            {
                Debug.LogError($"获取云端槽位失败: {e.Message}");
            }

            return result;
        }
    }
}
```

## 四、使用示例

```csharp
using UnityEngine;
using Cysharp.Threading.Tasks;
using CloudSave.Core;
using CloudSave.Providers;

namespace CloudSave.Examples
{
    /// <summary>
    /// 海外平台使用示例
    /// </summary>
    public class OverseasPlatformExample : MonoBehaviour
    {
        private FirebaseProvider firebaseProvider;
        private LocalCacheProvider cacheProvider;
        private CloudSaveConfig config;

        private async void Start()
        {
            config = new CloudSaveConfig
            {
                maxSlots = 3,
                enableAutoSave = true,
                autoSaveInterval = 300f
            };

            cacheProvider = new LocalCacheProvider(config);

            // 使用Firebase
            firebaseProvider = new FirebaseProvider(cacheProvider, config);

            // 初始化
            bool initSuccess = await firebaseProvider.InitializeAsync();

            if (initSuccess)
            {
                // 匿名登录
                await firebaseProvider.SignInAnonymouslyAsync();

                Debug.Log("Firebase准备就绪");
            }
        }

        /// <summary>
        /// 保存游戏
        /// </summary>
        public async UniTaskVoid SaveGame(int slot)
        {
            var gameData = new GameSaveData
            {
                level = 10,
                coins = 1000,
                playTime = 3600
            };

            var saveData = CloudSaveData.Create(slot, gameData, "device001", "iOS");
            saveData.summary = new SaveSummary
            {
                playerName = "玩家1",
                level = gameData.level,
                coins = gameData.coins,
                playTime = gameData.playTime
            };

            var result = await firebaseProvider.SaveAsync(slot, saveData);

            if (result.success)
            {
                Debug.Log($"保存成功，同步到云端: {result.syncedToCloud}");
            }
            else
            {
                Debug.LogError($"保存失败: {result.errorMessage}");
            }
        }

        /// <summary>
        /// 加载游戏
        /// </summary>
        public async UniTaskVoid LoadGame(int slot)
        {
            var result = await firebaseProvider.LoadAsync(slot);

            if (result.success)
            {
                var gameData = result.data.ParseGameData<GameSaveData>();
                Debug.Log($"加载成功: level={gameData.level}, coins={gameData.coins}");
                Debug.Log($"数据来源: {result.source}");
            }
            else
            {
                Debug.LogError($"加载失败: {result.errorMessage}");
            }
        }
    }

    /// <summary>
    /// 测试用游戏数据
    /// </summary>
    [Serializable]
    public class GameSaveData
    {
        public int level;
        public int coins;
        public long playTime;
    }
}
```
