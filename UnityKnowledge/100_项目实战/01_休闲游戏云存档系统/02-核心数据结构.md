# 02-核心数据结构

## 二、核心数据结构

### 2.1 CloudSaveData

```csharp
using System;
using System.Collections.Generic;

namespace CloudSave.Core
{
    /// <summary>
    /// 云存档数据 - 存储和传输的基本单位
    /// </summary>
    [Serializable]
    public class CloudSaveData
    {
        /// <summary>
        /// 数据格式版本（用于迁移）
        /// </summary>
        public int formatVersion = 1;

        /// <summary>
        /// 游戏数据版本（用于兼容性）
        /// </summary>
        public int gameDataVersion;

        /// <summary>
        /// 存档时间戳（毫秒）
        /// </summary>
        public long timestamp;

        /// <summary>
        /// 设备唯一标识
        /// </summary>
        public string deviceId;

        /// <summary>
        /// 平台标识
        /// </summary>
        public string platform;

        /// <summary>
        /// 用户ID（可能为空）
        /// </summary>
        public string userId;

        /// <summary>
        /// 存档槽索引
        /// </summary>
        public int slotIndex;

        /// <summary>
        /// 游戏数据（JSON字符串）
        /// </summary>
        public string gameData;

        /// <summary>
        /// 数据校验和
        /// </summary>
        public string checksum;

        /// <summary>
        /// 数据大小（字节）
        /// </summary>
        public int dataSize;

        /// <summary>
        /// 游戏内摘要（用于UI显示）
        /// </summary>
        public SaveSummary summary;

        /// <summary>
        /// 创建用于传输的实例
        /// </summary>
        public static CloudSaveData Create<T>(int slot, T data, string deviceId, string platform)
        {
            var gameDataJson = JsonUtility.ToJson(data);

            return new CloudSaveData
            {
                formatVersion = 1,
                timestamp = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds(),
                deviceId = deviceId,
                platform = platform,
                slotIndex = slot,
                gameData = gameDataJson,
                dataSize = gameDataJson.Length,
                checksum = ComputeChecksum(gameDataJson)
            };
        }

        /// <summary>
        /// 解析游戏数据
        /// </summary>
        public T ParseGameData<T>()
        {
            return JsonUtility.FromJson<T>(gameData);
        }

        /// <summary>
        /// 验证数据完整性
        /// </summary>
        public bool Validate()
        {
            if (string.IsNullOrEmpty(gameData)) return false;
            if (timestamp <= 0) return false;

            var expectedChecksum = ComputeChecksum(gameData);
            return checksum == expectedChecksum;
        }

        private static string ComputeChecksum(string data)
        {
            using (var sha256 = System.Security.Cryptography.SHA256.Create())
            {
                var bytes = System.Text.Encoding.UTF8.GetBytes(data);
                var hash = sha256.ComputeHash(bytes);
                return Convert.ToBase64String(hash);
            }
        }
    }

    /// <summary>
    /// 存档摘要（用于UI显示）
    /// </summary>
    [Serializable]
    public class SaveSummary
    {
        public string playerName;
        public int level;
        public long playTime;       // 秒
        public int coins;
        public string currentScene;
        public string customSummary; // 自定义摘要
    }
}
```

### 2.2 SlotInfo

```csharp
namespace CloudSave.Core
{
    /// <summary>
    /// 存档槽位信息（用于列表显示）
    /// </summary>
    [Serializable]
    public class SlotInfo
    {
        /// <summary>
        /// 槽位索引（0-2）
        /// </summary>
        public int slotIndex;

        /// <summary>
        /// 是否有数据
        /// </summary>
        public bool hasData;

        /// <summary>
        /// 最后修改时间
        /// </summary>
        public long timestamp;

        /// <summary>
        /// 设备名称
        /// </summary>
        public string deviceName;

        /// <summary>
        /// 平台名称
        /// </summary>
        public string platformName;

        /// <summary>
        /// 游戏摘要
        /// </summary>
        public SaveSummary summary;

        /// <summary>
        /// 同步状态
        /// </summary>
        public SyncState syncState;

        /// <summary>
        /// 是否有冲突
        /// </summary>
        public bool hasConflict;

        /// <summary>
        /// 创建空槽位
        /// </summary>
        public static SlotInfo Empty(int index)
        {
            return new SlotInfo
            {
                slotIndex = index,
                hasData = false,
                syncState = SyncState.LocalOnly
            };
        }

        /// <summary>
        /// 从CloudSaveData创建
        /// </summary>
        public static SlotInfo FromSaveData(CloudSaveData data)
        {
            return new SlotInfo
            {
                slotIndex = data.slotIndex,
                hasData = true,
                timestamp = data.timestamp,
                deviceName = GetDeviceDisplayName(data.deviceId),
                platformName = GetPlatformDisplayName(data.platform),
                summary = data.summary
            };
        }

        private static string GetDeviceDisplayName(string deviceId)
        {
            if (string.IsNullOrEmpty(deviceId)) return "未知设备";
            if (deviceId.Length <= 8) return deviceId;
            return deviceId.Substring(0, 4) + "****" + deviceId.Substring(deviceId.Length - 4);
        }

        private static string GetPlatformDisplayName(string platform)
        {
            return platform switch
            {
                "WeChat" => "微信",
                "Douyin" => "抖音",
                "iOS" => "iOS",
                "Android" => "Android",
                "WebGL" => "网页",
                _ => platform ?? "未知"
            };
        }
    }

    /// <summary>
    /// 同步状态
    /// </summary>
    public enum SyncState
    {
        /// <summary>仅本地</summary>
        LocalOnly,

        /// <summary>同步中</summary>
        Syncing,

        /// <summary>已同步</summary>
        Synced,

        /// <summary>同步失败</summary>
        SyncFailed,

        /// <summary>有冲突</summary>
        Conflict,

        /// <summary>待同步（离线）</summary>
        PendingSync
    }
}
```

## 三、结果类型

### 3.1 操作结果

```csharp
namespace CloudSave.Core
{
    /// <summary>
    /// 保存操作结果
    /// </summary>
    public class SaveResult
    {
        public bool success;
        public string errorCode;
        public string errorMessage;

        /// <summary>是否同步到云端</summary>
        public bool syncedToCloud;

        /// <summary>本地是否保存成功</summary>
        public bool savedLocally;

        /// <summary>存档数据</summary>
        public CloudSaveData savedData;

        public static SaveResult Succeeded(CloudSaveData data, bool synced)
        {
            return new SaveResult
            {
                success = true,
                savedData = data,
                syncedToCloud = synced,
                savedLocally = true
            };
        }

        public static SaveResult Failed(string code, string message)
        {
            return new SaveResult
            {
                success = false,
                errorCode = code,
                errorMessage = message
            };
        }
    }

    /// <summary>
    /// 加载操作结果
    /// </summary>
    public class LoadResult
    {
        public bool success;
        public string errorCode;
        public string errorMessage;

        /// <summary>加载的存档数据</summary>
        public CloudSaveData data;

        /// <summary>数据来源</summary>
        public DataSource source;

        /// <summary>是否是云端更新的数据</summary>
        public bool isCloudNewer;

        public static LoadResult Succeeded(CloudSaveData data, DataSource source, bool cloudNewer = false)
        {
            return new LoadResult
            {
                success = true,
                data = data,
                source = source,
                isCloudNewer = cloudNewer
            };
        }

        public static LoadResult Failed(string code, string message)
        {
            return new LoadResult
            {
                success = false,
                errorCode = code,
                errorMessage = message
            };
        }
    }

    /// <summary>
    /// 数据来源
    /// </summary>
    public enum DataSource
    {
        /// <summary>本地缓存</summary>
        LocalCache,

        /// <summary>云端下载</summary>
        Cloud,

        /// <summary>云端但使用缓存（网络不可用）</summary>
        CloudCached
    }

    /// <summary>
    /// 同步操作结果
    /// </summary>
    public class SyncResult
    {
        public bool success;
        public string errorMessage;

        /// <summary>上传数量</summary>
        public int uploadedCount;

        /// <summary>下载数量</summary>
        public int downloadedCount;

        /// <summary>冲突数量</summary>
        public int conflictCount;

        /// <summary>失败的槽位</summary>
        public List<int> failedSlots;
    }
}
```

## 四、事件类型

### 4.1 同步事件

```csharp
namespace CloudSave.Core
{
    /// <summary>
    /// 同步进度事件
    /// </summary>
    public class SyncProgress
    {
        /// <summary>当前操作（上传/下载）</summary>
        public string operation;

        /// <summary>当前槽位</summary>
        public int currentSlot;

        /// <summary>总槽位数</summary>
        public int totalSlots;

        /// <summary>进度 0-1</summary>
        public float progress;

        /// <summary>状态描述</summary>
        public string statusText;

        public float OverallProgress => (currentSlot + progress) / totalSlots;
    }

    /// <summary>
    /// 冲突事件
    /// </summary>
    public class ConflictEvent
    {
        /// <summary>冲突的槽位</summary>
        public int slotIndex;

        /// <summary>本地数据</summary>
        public CloudSaveData localData;

        /// <summary>云端数据</summary>
        public CloudSaveData cloudData;

        /// <summary>本地修改时间</summary>
        public DateTime LocalTime => DateTimeOffset.FromUnixTimeMilliseconds(localData.timestamp).LocalDateTime;

        /// <summary>云端修改时间</summary>
        public DateTime CloudTime => DateTimeOffset.FromUnixTimeMilliseconds(cloudData.timestamp).LocalDateTime;

        /// <summary>建议的解决策略</summary>
        public ConflictResolution suggestedResolution;
    }

    /// <summary>
    /// 冲突解决策略
    /// </summary>
    public enum ConflictResolution
    {
        /// <summary>使用云端数据</summary>
        UseCloud,

        /// <summary>使用本地数据</summary>
        UseLocal,

        /// <summary>需要用户选择</summary>
        AskUser,

        /// <summary>尝试合并</summary>
        Merge
    }
}
```

## 六、离线队列数据

### 6.1 离线操作

```csharp
namespace CloudSave.Core
{
    /// <summary>
    /// 离线操作类型
    /// </summary>
    public enum OfflineOperationType
    {
        Save,
        Delete
    }

    /// <summary>
    /// 离线操作记录
    /// </summary>
    [Serializable]
    public class OfflineOperation
    {
        /// <summary>操作ID</summary>
        public string operationId;

        /// <summary>操作类型</summary>
        public OfflineOperationType type;

        /// <summary>目标槽位</summary>
        public int slotIndex;

        /// <summary>存档数据（仅Save操作）</summary>
        public CloudSaveData saveData;

        /// <summary>创建时间</summary>
        public long createdTime;

        /// <summary>重试次数</summary>
        public int retryCount;

        /// <summary>最后尝试时间</summary>
        public long lastAttemptTime;

        public static OfflineOperation CreateSave(int slot, CloudSaveData data)
        {
            return new OfflineOperation
            {
                operationId = Guid.NewGuid().ToString(),
                type = OfflineOperationType.Save,
                slotIndex = slot,
                saveData = data,
                createdTime = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds(),
                retryCount = 0
            };
        }

        public static OfflineOperation CreateDelete(int slot)
        {
            return new OfflineOperation
            {
                operationId = Guid.NewGuid().ToString(),
                type = OfflineOperationType.Delete,
                slotIndex = slot,
                createdTime = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds(),
                retryCount = 0
            };
        }
    }

    /// <summary>
    /// 离线队列
    /// </summary>
    [Serializable]
    public class OfflineQueue
    {
        public List<OfflineOperation> operations = new List<OfflineOperation>();

        /// <summary>
        /// 最大队列长度
        /// </summary>
        public const int MaxQueueSize = 100;

        public void Enqueue(OfflineOperation operation)
        {
            // 移除同槽位的旧操作
            operations.RemoveAll(o => o.slotIndex == operation.slotIndex);

            operations.Add(operation);

            // 限制队列大小
            while (operations.Count > MaxQueueSize)
            {
                operations.RemoveAt(0);
            }
        }

        public OfflineOperation Dequeue()
        {
            if (operations.Count == 0) return null;

            var op = operations[0];
            operations.RemoveAt(0);
            return op;
        }

        public OfflineOperation Peek()
        {
            return operations.Count > 0 ? operations[0] : null;
        }

        public bool HasPendingOperations => operations.Count > 0;

        public int PendingCount => operations.Count;
    }
}
```
