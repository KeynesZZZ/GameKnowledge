### 2.2 saveData云函数

```javascript
// cloudfunctions/saveData/index.js
const cloud = require('wx-server-sdk')

cloud.init({
  env: cloud.DYNAMIC_CURRENT_ENV
})

const db = cloud.database
const _ = db.command

exports.main = async (event, context) => {
  const wxContext = cloud.getWXContext()
  const openid = wxContext.OPENID

  const { slot, data } = event

  try {
    // 验证参数
    if (slot === undefined || slot < 0 || slot > 2) {
      return {
        success: false,
        errorCode: 'INVALID_SLOT',
        errorMessage: '无效的存档槽位'
      }
    }

    if (!data) {
      return {
        success: false,
        errorCode: 'NULL_DATA',
        errorMessage: '存档数据为空'
      }
    }

    // 添加元数据
    const saveData = {
      ...data,
      openid: openid,
      updatedAt: db.serverDate()
    }

    // 查询是否已存在
    const collection = db.collection('saves')
    const existResult = await collection
      .where({
        openid: openid,
        slot: slot
      })
      .get()

    if (existResult.data.length > 0) {
      // 更新
      await collection
        .doc(existResult.data[0]._id)
        .update({
          data: saveData
        })
    } else {
      // 新建
      saveData.createdAt = db.serverDate()
      await collection.add({
        data: saveData
      })
    }

    return {
      success: true,
      timestamp: Date.now()
    }
  } catch (err) {
    console.error('保存失败:', err)
    return {
      success: false,
      errorCode: 'SAVE_ERROR',
      errorMessage: err.message
    }
  }
}
```

### 2.4 getSlotList云函数

```javascript
// cloudfunctions/getSlotList/index.js
const cloud = require('wx-server-sdk')

cloud.init({
  env: cloud.DYNAMIC_CURRENT_ENV
})

const db = cloud.database

exports.main = async (event, context) => {
  const wxContext = cloud.getWXContext()
  const openid = wxContext.OPENID

  try {
    const result = await db.collection('saves')
      .where({
        openid: openid
      })
      .orderBy('slot', 'asc')
      .get()

    const slots = result.data.map(item => ({
      slot: item.slot,
      timestamp: item.timestamp,
      deviceId: item.deviceId,
      platform: item.platform,
      summary: item.summary || {},
      hasData: true
    }))

    // 填充空槽位
    const allSlots = []
    for (let i = 0; i < 3; i++) {
      const found = slots.find(s => s.slot === i)
      if (found) {
        allSlots.push(found)
      } else {
        allSlots.push({
          slot: i,
          hasData: false
        })
      }
    }

    return {
      success: true,
      slots: allSlots
    }
  } catch (err) {
    console.error('获取槽位列表失败:', err)
    return {
      success: false,
      errorCode: 'LIST_ERROR',
      errorMessage: err.message
    }
  }
}
```

```javascript
// Assets/Plugins/WebGL/WeChatCloud.jslib

var WeChatCloud = {
    // 初始化云开发
    InitCloud: function(envId) {
        var env = UTF8ToString(envId);
        if (typeof wx !== 'undefined' && wx.cloud) {
            wx.cloud.init({
                env: env,
                traceUser: true
            });
            return true;
        }
        return false;
    },

    // 检查是否在微信环境
    IsWeChatEnvironment: function() {
        return typeof wx !== 'undefined';
    },

    // 获取用户信息
    GetUserInfo: function(callback) {
        var cb = callback;
        if (typeof wx !== 'undefined') {
            wx.getUserInfo({
                success: function(res) {
                    var json = JSON.stringify(res.userInfo);
                    var ptr = allocateUTF8(json);
                    {{{ makeDynCall('vi', 'cb') }}}(ptr);
                    _free(ptr);
                },
                fail: function() {
                    {{{ makeDynCall('vi', 'cb') }}}(0);
                }
            });
        } else {
            {{{ makeDynCall('vi', 'cb') }}}(0);
        }
    },

    // 调用云函数
    CallCloudFunction: function(name, data, successCallback, failCallback) {
        var funcName = UTF8ToString(name);
        var jsonData = UTF8ToString(data);
        var successCb = successCallback;
        var failCb = failCallback;

        if (typeof wx !== 'undefined' && wx.cloud) {
            wx.cloud.callFunction({
                name: funcName,
                data: JSON.parse(jsonData),
                success: function(res) {
                    var json = JSON.stringify(res.result);
                    var ptr = allocateUTF8(json);
                    {{{ makeDynCall('vi', 'successCb') }}}(ptr);
                    _free(ptr);
                },
                fail: function(err) {
                    var json = JSON.stringify(err);
                    var ptr = allocateUTF8(json);
                    {{{ makeDynCall('vi', 'failCb') }}}(ptr);
                    _free(ptr);
                }
            });
        } else {
            var errPtr = allocateUTF8('{"errCode": -1, "errMsg": "Not in WeChat environment"}');
            {{{ makeDynCall('vi', 'failCb') }}}(errPtr);
            _free(errPtr);
        }
    },

    // 获取网络状态
    GetNetworkType: function(callback) {
        var cb = callback;
        if (typeof wx !== 'undefined') {
            wx.getNetworkType({
                success: function(res) {
                    var ptr = allocateUTF8(res.networkType);
                    {{{ makeDynCall('vi', 'cb') }}}(ptr);
                    _free(ptr);
                }
            });
        }
    },

    // 监听网络变化
    OnNetworkStatusChange: function(callback) {
        var cb = callback;
        if (typeof wx !== 'undefined') {
            wx.onNetworkStatusChange(function(res) {
                var json = JSON.stringify({
                    isConnected: res.isConnected,
                    networkType: res.networkType
                });
                var ptr = allocateUTF8(json);
                {{{ makeDynCall('vi', 'cb') }}}(ptr);
                _free(ptr);
            });
        }
    },

    // 显示提示
    ShowToast: function(title, icon) {
        var t = UTF8ToString(title);
        var i = UTF8ToString(icon);
        if (typeof wx !== 'undefined') {
            wx.showToast({
                title: t,
                icon: i,
                duration: 2000
            });
        }
    },

    // 显示加载
    ShowLoading: function(title) {
        var t = UTF8ToString(title);
        if (typeof wx !== 'undefined') {
            wx.showLoading({
                title: t,
                mask: true
            });
        }
    },

    // 隐藏加载
    HideLoading: function() {
        if (typeof wx !== 'undefined') {
            wx.hideLoading();
        }
    }
};
```

### 3.2 WeChatProvider实现

```csharp
using System;
using System.Collections.Generic;
using System.Runtime.InteropServices;
using System.Text;
using UnityEngine;
using Cysharp.Threading.Tasks;

namespace CloudSave.Providers
{
    /// <summary>
    /// 微信小游戏云存档提供者
    /// </summary>
    public class WeChatProvider : BaseCloudProvider
    {
        public override string ProviderName => "WeChat";

        private string cloudEnvId;
        private bool isCloudInitialized;

        public WeChatProvider(ICacheProvider cacheProvider, CloudSaveConfig config)
            : base(cacheProvider, config)
        {
            // 云开发环境ID，需要在微信云开发控制台获取
            cloudEnvId = "your-cloud-env-id";
        }

        /// <summary>
        /// 设置云开发环境ID
        /// </summary>
        public void SetCloudEnvId(string envId)
        {
            cloudEnvId = envId;
        }

        /// <summary>
        /// 初始化
        /// </summary>
        public override async UniTask<bool> InitializeAsync()
        {
            if (isInitialized)
            {
                return true;
            }

            try
            {
#if UNITY_WEBGL && !UNITY_EDITOR
                // 检查微信环境
                if (!IsWeChatEnvironment())
                {
                    Debug.LogWarning("不在微信小游戏环境中");
                    return false;
                }

                // 初始化云开发
                if (!InitCloud(cloudEnvId))
                {
                    Debug.LogWarning("云开发初始化失败");
                    return false;
                }

                isCloudInitialized = true;
#endif

                isInitialized = true;
                await UniTask.CompletedTask;
                return true;
            }
            catch (Exception e)
            {
                Debug.LogError($"微信Provider初始化失败: {e.Message}");
                return false;
            }
        }

        /// <summary>
        /// 检查服务是否可用
        /// </summary>
        public override async UniTask<bool> IsAvailableAsync()
        {
            if (!isInitialized)
            {
                await InitializeAsync();
            }

#if UNITY_WEBGL && !UNITY_EDITOR
            return isCloudInitialized;
#else
            return false;
#endif
        }

        /// <summary>
        /// 保存到云端
        /// </summary>
        protected override async UniTask<SaveResult> SaveToCloudAsync(int slot, CloudSaveData data)
        {
#if UNITY_WEBGL && !UNITY_EDITOR
            try
            {
                var requestData = new
                {
                    slot = slot,
                    data = new
                    {
                        formatVersion = data.formatVersion,
                        gameDataVersion = data.gameDataVersion,
                        timestamp = data.timestamp,
                        deviceId = data.deviceId,
                        platform = data.platform,
                        userId = data.userId,
                        slotIndex = data.slotIndex,
                        gameData = data.gameData,
                        checksum = data.checksum,
                        dataSize = data.dataSize,
                        summary = data.summary
                    }
                };

                string json = JsonUtility.ToJson(requestData);
                var result = await CallCloudFunctionAsync("saveData", json);

                if (result.success)
                {
                    Debug.Log($"微信云保存成功: slot={slot}");
                    return SaveResult.Succeeded(data, true);
                }
                else
                {
                    return SaveResult.Failed(result.errorCode, result.errorMessage);
                }
            }
            catch (Exception e)
            {
                Debug.LogError($"微信云保存失败: {e.Message}");
                return SaveResult.Failed("WECHAT_SAVE_ERROR", e.Message);
            }
#else
            await UniTask.CompletedTask;
            return SaveResult.Failed("NOT_WEBGL", "非WebGL环境");
#endif
        }

        /// <summary>
        /// 从云端加载
        /// </summary>
        protected override async UniTask<LoadResult> LoadFromCloudAsync(int slot)
        {
#if UNITY_WEBGL && !UNITY_EDITOR
            try
            {
                var requestData = new { slot = slot };
                string json = JsonUtility.ToJson(requestData);

                var result = await CallCloudFunctionAsync("loadData", json);

                if (result.success)
                {
                    var cloudData = ParseCloudData(result.data);
                    Debug.Log($"微信云加载成功: slot={slot}");
                    return LoadResult.Succeeded(cloudData, DataSource.Cloud);
                }
                else
                {
                    return LoadResult.Failed(result.errorCode, result.errorMessage);
                }
            }
            catch (Exception e)
            {
                Debug.LogError($"微信云加载失败: {e.Message}");
                return LoadResult.Failed("WECHAT_LOAD_ERROR", e.Message);
            }
#else
            await UniTask.CompletedTask;
            return LoadResult.Failed("NOT_WEBGL", "非WebGL环境");
#endif
        }

        /// <summary>
        /// 从云端删除
        /// </summary>
        protected override async UniTask<bool> DeleteFromCloudAsync(int slot)
        {
#if UNITY_WEBGL && !UNITY_EDITOR
            try
            {
                var requestData = new { slot = slot };
                string json = JsonUtility.ToJson(requestData);

                var result = await CallCloudFunctionAsync("deleteData", json);

                if (result.success)
                {
                    Debug.Log($"微信云删除成功: slot={slot}");
                    return true;
                }
                else
                {
                    Debug.LogError($"微信云删除失败: {result.errorMessage}");
                    return false;
                }
            }
            catch (Exception e)
            {
                Debug.LogError($"微信云删除异常: {e.Message}");
                return false;
            }
#else
            await UniTask.CompletedTask;
            return false;
#endif
        }

        /// <summary>
        /// 获取云端槽位列表
        /// </summary>
        protected override async UniTask<List<SlotInfo>> GetCloudSlotListAsync()
        {
            var result = new List<SlotInfo>();

#if UNITY_WEBGL && !UNITY_EDITOR
            try
            {
                var response = await CallCloudFunctionAsync("getSlotList", "{}");

                if (response.success && response.data != null)
                {
                    var slotList = ParseSlotList(response.data);
                    return slotList;
                }
            }
            catch (Exception e)
            {
                Debug.LogError($"获取云端槽位列表失败: {e.Message}");
            }
#else
            await UniTask.CompletedTask;
#endif

            return result;
        }

        #region JS交互

#if UNITY_WEBGL && !UNITY_EDITOR
        [DllImport("__Internal")]
        private static extern bool IsWeChatEnvironment();

        [DllImport("__Internal")]
        private static extern bool InitCloud(string envId);

        [DllImport("__Internal")]
        private static extern void CallCloudFunction(
            string name,
            string data,
            Action<IntPtr> successCallback,
            Action<IntPtr> failCallback
        );
#endif

        private class CloudFunctionResult
        {
            public bool success;
            public string errorCode;
            public string errorMessage;
            public Dictionary<string, object> data;
        }

        private async UniTask<CloudFunctionResult> CallCloudFunctionAsync(string name, string data)
        {
            var tcs = new UniTaskCompletionSource<CloudFunctionResult>();

#if UNITY_WEBGL && !UNITY_EDITOR
            CallCloudFunction(
                name,
                data,
                (resultPtr) =>
                {
                    string json = Marshal.PtrToStringAnsi(resultPtr);
                    var result = ParseFunctionResult(json);
                    tcs.TrySetResult(result);
                },
                (errorPtr) =>
                {
                    string json = Marshal.PtrToStringAnsi(errorPtr);
                    var result = new CloudFunctionResult
                    {
                        success = false,
                        errorMessage = json
                    };
                    tcs.TrySetResult(result);
                }
            );
#else
            tcs.TrySetResult(new CloudFunctionResult
            {
                success = false,
                errorMessage = "非WebGL环境"
            });
#endif

            return await tcs.Task;
        }

        #endregion

        #region 数据解析

        private CloudFunctionResult ParseFunctionResult(string json)
        {
            try
            {
                var dict = MiniJson.Deserialize(json) as Dictionary<string, object>;
                var result = new CloudFunctionResult
                {
                    success = GetBool(dict, "success"),
                    errorCode = GetString(dict, "errorCode"),
                    errorMessage = GetString(dict, "errorMessage")
                };

                if (dict.TryGetValue("data", out var data))
                {
                    result.data = data as Dictionary<string, object>;
                }

                return result;
            }
            catch (Exception e)
            {
                return new CloudFunctionResult
                {
                    success = false,
                    errorMessage = $"解析结果失败: {e.Message}"
                };
            }
        }

        private CloudSaveData ParseCloudData(object dataObj)
        {
            var dict = dataObj as Dictionary<string, object>;

            return new CloudSaveData
            {
                formatVersion = GetInt(dict, "formatVersion"),
                gameDataVersion = GetInt(dict, "gameDataVersion"),
                timestamp = GetLong(dict, "timestamp"),
                deviceId = GetString(dict, "deviceId"),
                platform = GetString(dict, "platform"),
                userId = GetString(dict, "userId"),
                slotIndex = GetInt(dict, "slotIndex"),
                gameData = GetString(dict, "gameData"),
                checksum = GetString(dict, "checksum"),
                dataSize = GetInt(dict, "dataSize"),
                summary = ParseSummary(dict)
            };
        }

        private SaveSummary ParseSummary(Dictionary<string, object> dict)
        {
            if (!dict.TryGetValue("summary", out var summaryObj))
            {
                return null;
            }

            var summaryDict = summaryObj as Dictionary<string, object>;

            return new SaveSummary
            {
                playerName = GetString(summaryDict, "playerName"),
                level = GetInt(summaryDict, "level"),
                playTime = GetLong(summaryDict, "playTime"),
                coins = GetInt(summaryDict, "coins"),
                currentScene = GetString(summaryDict, "currentScene")
            };
        }

        private List<SlotInfo> ParseSlotList(object dataObj)
        {
            var result = new List<SlotInfo>();
            var dict = dataObj as Dictionary<string, object>;

            if (dict.TryGetValue("slots", out var slotsObj) && slotsObj is List<object> slots)
            {
                foreach (var slotObj in slots)
                {
                    var slotDict = slotObj as Dictionary<string, object>;
                    var slotInfo = new SlotInfo
                    {
                        slotIndex = GetInt(slotDict, "slot"),
                        hasData = GetBool(slotDict, "hasData"),
                        timestamp = GetLong(slotDict, "timestamp"),
                        deviceName = GetString(slotDict, "deviceId"),
                        platformName = GetString(slotDict, "platform"),
                        syncState = SyncState.Synced
                    };

                    if (slotDict.TryGetValue("summary", out var summaryObj))
                    {
                        slotInfo.summary = ParseSummary(slotDict);
                    }

                    result.Add(slotInfo);
                }
            }

            return result;
        }

        #endregion

        #region 辅助方法

        private bool GetBool(Dictionary<string, object> dict, string key)
        {
            if (dict.TryGetValue(key, out var value))
            {
                return value is bool b && b;
            }
            return false;
        }

        private int GetInt(Dictionary<string, object> dict, string key)
        {
            if (dict.TryGetValue(key, out var value))
            {
                return Convert.ToInt32(value);
            }
            return 0;
        }

        private long GetLong(Dictionary<string, object> dict, string key)
        {
            if (dict.TryGetValue(key, out var value))
            {
                return Convert.ToInt64(value);
            }
            return 0;
        }

        private string GetString(Dictionary<string, object> dict, string key)
        {
            if (dict.TryGetValue(key, out var value))
            {
                return value?.ToString() ?? "";
            }
            return "";
        }

        #endregion
    }
}
```

```csharp
using UnityEngine;

namespace CloudSave.Examples
{
    /// <summary>
    /// 微信云存档使用示例

    {

            }
        }

        /// <summary>
        /// 保存游戏
        /// </summary>
        public async UniTaskVoid SaveGame(int slot)
        {
            var gameData = new GameSaveData
            {

                coins = 1000,
                playTime = 3600
            };

            var saveData = CloudSaveData.Create(slot, gameData, "wechat_device", "WeChat");
            saveData.userId = "wechat_user_123";
            saveData.summary = new SaveSummary
            {
                playerName = "微信玩家",
                level = gameData.level,
                coins = gameData.coins,
                playTime = gameData.playTime
            };

            var result = await wechatProvider.SaveAsync(slot, saveData);

            if (result.success)
            {
                Debug.Log($"保存成功");
            }
        }

        /// <summary>
        /// 加载游戏
        /// </summary>
        public async UniTaskVoid LoadGame(int slot)
        {
            var result = await wechatProvider.LoadAsync(slot);

            if (result.success)
            {
                var gameData = result.data.ParseGameData<GameSaveData>();
                Debug.Log($"加载成功: level={gameData.level}");
            }
        }
    }
}
```
