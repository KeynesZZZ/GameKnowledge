## 二、场景管理器实现

### 2.1 基础实现

```csharp
using System;
using System.Collections;
using UnityEngine;
using UnityEngine.SceneManagement;

/// <summary>
/// 场景管理器
/// </summary>
public class SceneLoader : MonoBehaviour
{
    public static SceneLoader Instance { get; private set; }

    [Header("配置")]
    [SerializeField] private float minimumLoadingTime = 1f;
    [SerializeField] private string loadingSceneName = "Loading";

    [Header("过渡动画")]
    [SerializeField] private Animator transitionAnimator;
    [SerializeField] private float transitionDuration = 0.5f;

    private bool isLoading;
    private AsyncOperation currentOperation;

    public event Action<float> OnProgressChanged;
    public event Action<string> OnSceneLoaded;
    public event Action<string> OnSceneUnloaded;

    private void Awake()
    {
        if (Instance != null && Instance != this)
        {
            Destroy(gameObject);
            return;
        }
        Instance = this;
        DontDestroyOnLoad(gameObject);
    }

    /// <summary>
    /// 同步加载场景
    /// </summary>
    public void LoadScene(string sceneName)
    {
        if (isLoading)
        {
            Debug.LogWarning("正在加载中，请等待");
            return;
        }

        StartCoroutine(LoadSceneCoroutine(sceneName));
    }

    /// <summary>
    /// 异步加载场景（带加载界面）
    /// </summary>
    public void LoadSceneAsync(string sceneName, bool showLoadingScreen = true)
    {
        if (isLoading)
        {
            Debug.LogWarning("正在加载中，请等待");
            return;
        }

        StartCoroutine(LoadSceneAsyncCoroutine(sceneName, showLoadingScreen));
    }

    /// <summary>
    /// 加载场景（Additive模式）
    /// </summary>
    public void LoadSceneAdditive(string sceneName)
    {
        StartCoroutine(LoadSceneAdditiveCoroutine(sceneName));
    }

    /// <summary>
    /// 卸载场景
    /// </summary>
    public void UnloadScene(string sceneName)
    {
        StartCoroutine(UnloadSceneCoroutine(sceneName));
    }

    private IEnumerator LoadSceneCoroutine(string sceneName)
    {
        isLoading = true;

        // 播放过渡动画（淡出）
        if (transitionAnimator != null)
        {
            transitionAnimator.SetTrigger("FadeOut");
            yield return new WaitForSeconds(transitionDuration);
        }

        // 卸载当前场景（保留Loading场景）
        var currentScene = SceneManager.GetActiveScene();
        if (currentScene.name != loadingSceneName)
        {
            yield return SceneManager.UnloadSceneAsync(currentScene);
        }

        // 加载新场景
        SceneManager.LoadScene(sceneName, LoadSceneMode.Single);

        // 播放过渡动画（淡入）
        if (transitionAnimator != null)
        {
            transitionAnimator.SetTrigger("FadeIn");
            yield return new WaitForSeconds(transitionDuration);
        }

        isLoading = false;
        OnSceneLoaded?.Invoke(sceneName);
    }

    private IEnumerator LoadSceneAsyncCoroutine(string sceneName, bool showLoadingScreen)
    {
        isLoading = true;
        float startTime = Time.time;

        // 播放过渡动画（淡出）
        if (transitionAnimator != null)
        {
            transitionAnimator.SetTrigger("FadeOut");
            yield return new WaitForSeconds(transitionDuration);
        }

        // 加载Loading场景
        if (showLoadingScreen && !string.IsNullOrEmpty(loadingSceneName))
        {
            yield return SceneManager.LoadSceneAsync(loadingSceneName, LoadSceneMode.Single);
        }

        // 开始异步加载目标场景
        currentOperation = SceneManager.LoadSceneAsync(sceneName, LoadSceneMode.Additive);
        currentOperation.allowSceneActivation = false;

        // 等待加载完成
        while (!currentOperation.isDone)
        {
            float progress = currentOperation.progress;

            // Unity的progress最大值是0.9，需要转换
            if (progress >= 0.9f)
            {
                progress = 1f;
            }

            OnProgressChanged?.Invoke(progress);

            // 检查是否达到最小加载时间
            if (progress >= 0.9f && (Time.time - startTime) >= minimumLoadingTime)
            {
                currentOperation.allowSceneActivation = true;
            }

            yield return null;
        }

        // 卸载Loading场景
        if (showLoadingScreen && !string.IsNullOrEmpty(loadingSceneName))
        {
            var loadingScene = SceneManager.GetSceneByName(loadingSceneName);
            if (loadingScene.isLoaded)
            {
                yield return SceneManager.UnloadSceneAsync(loadingScene);
            }
        }

        // 设置新场景为活动场景
        var newScene = SceneManager.GetSceneByName(sceneName);
        SceneManager.SetActiveScene(newScene);

        // 播放过渡动画（淡入）
        if (transitionAnimator != null)
        {
            transitionAnimator.SetTrigger("FadeIn");
            yield return new WaitForSeconds(transitionDuration);
        }

        isLoading = false;
        currentOperation = null;
        OnSceneLoaded?.Invoke(sceneName);
    }

    private IEnumerator LoadSceneAdditiveCoroutine(string sceneName)
    {
        var operation = SceneManager.LoadSceneAsync(sceneName, LoadSceneMode.Additive);
        yield return operation;
        OnSceneLoaded?.Invoke(sceneName);
    }

    private IEnumerator UnloadSceneCoroutine(string sceneName)
    {
        var operation = SceneManager.UnloadSceneAsync(sceneName);
        yield return operation;
        OnSceneUnloaded?.Invoke(sceneName);
    }

    /// <summary>
    /// 获取当前加载进度
    /// </summary>
    public float CurrentProgress => currentOperation?.progress ?? 0f;
}
```

## 三、加载界面实现

### 3.1 加载界面控制器

```csharp
using UnityEngine;
using UnityEngine.UI;
using TMPro;

/// <summary>
/// 加载界面控制器
/// </summary>
public class LoadingScreen : MonoBehaviour
{
    [Header("UI引用")]
    [SerializeField] private Slider progressBar;
    [SerializeField] private TMP_Text progressText;
    [SerializeField] private TMP_Text loadingTip;
    [SerializeField] private Image backgroundImage;

    [Header("提示文本")]
    [SerializeField] private string[] tips;

    [Header("动画")]
    [SerializeField] private float barSmoothSpeed = 5f;

    private float displayedProgress;
    private float targetProgress;

    private void OnEnable()
    {
        SceneLoader.Instance.OnProgressChanged += OnProgressChanged;

        // 随机显示提示
        if (tips != null && tips.Length > 0)
        {
            loadingTip.text = tips[UnityEngine.Random.Range(0, tips.Length)];
        }
    }

    private void OnDisable()
    {
        if (SceneLoader.Instance != null)
        {
            SceneLoader.Instance.OnProgressChanged -= OnProgressChanged;
        }
    }

    private void Update()
    {
        // 平滑进度条
        displayedProgress = Mathf.Lerp(displayedProgress, targetProgress, Time.deltaTime * barSmoothSpeed);

        if (progressBar != null)
        {
            progressBar.value = displayedProgress;
        }

        if (progressText != null)
        {
            progressText.text = $"{(int)(displayedProgress * 100)}%";
        }
    }

    private void OnProgressChanged(float progress)
    {
        targetProgress = progress;
    }
}
```

### 3.2 过渡动画（淡入淡出）

```csharp
using UnityEngine;
using UnityEngine.UI;

/// <summary>
/// 场景过渡动画
/// </summary>
public class SceneTransition : MonoBehaviour
{
    [Header("配置")]
    [SerializeField] private Image fadeImage;
    [SerializeField] private float fadeDuration = 0.5f;

    private Coroutine currentFade;

    private void Awake()
    {
        // 初始状态：完全透明
        if (fadeImage != null)
        {
            fadeImage.color = new Color(0, 0, 0, 0);
        }
    }

    public void FadeOut(System.Action onComplete = null)
    {
        if (currentFade != null)
        {
            StopCoroutine(currentFade);
        }
        currentFade = StartCoroutine(FadeCoroutine(0f, 1f, onComplete));
    }

    public void FadeIn(System.Action onComplete = null)
    {
        if (currentFade != null)
        {
            StopCoroutine(currentFade);
        }
        currentFade = StartCoroutine(FadeCoroutine(1f, 0f, onComplete));
    }

    private System.Collections.IEnumerator FadeCoroutine(float from, float to, System.Action onComplete)
    {
        float elapsed = 0f;

        while (elapsed < fadeDuration)
        {
            elapsed += Time.deltaTime;
            float t = elapsed / fadeDuration;
            float alpha = Mathf.Lerp(from, to, t);

            if (fadeImage != null)
            {
                fadeImage.color = new Color(0, 0, 0, alpha);
            }

            yield return null;
        }

        if (fadeImage != null)
        {
            fadeImage.color = new Color(0, 0, 0, to);
        }

        onComplete?.Invoke();
        currentFade = null;
    }
}
```

## 四、场景数据传递

### 4.1 场景数据容器

```csharp
using System.Collections.Generic;

/// <summary>
/// 场景数据传递容器
/// </summary>
public static class SceneData
{
    private static Dictionary<string, object> data = new Dictionary<string, object>();

    /// <summary>
    /// 设置数据
    /// </summary>
    public static void Set<T>(string key, T value)
    {
        data[key] = value;
    }

    /// <summary>
    /// 获取数据
    /// </summary>
    public static T Get<T>(string key, T defaultValue = default)
    {
        if (data.TryGetValue(key, out var value) && value is T typedValue)
        {
            return typedValue;
        }
        return defaultValue;
    }

    /// <summary>
    /// 获取并移除数据（一次性）
    /// </summary>
    public static T GetAndRemove<T>(string key, T defaultValue = default)
    {
        var value = Get<T>(key, defaultValue);
        data.Remove(key);
        return value;
    }

    /// <summary>
    /// 清空数据
    /// </summary>
    public static void Clear()
    {
        data.Clear();
    }
}
```

## 五、多场景管理

### 5.1 场景分组

```csharp
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.SceneManagement;

/// <summary>
/// 场景组管理器
/// </summary>
public class SceneGroupManager : MonoBehaviour
{
    [System.Serializable]
    public class SceneGroup
    {
        public string groupName;
        public string[] sceneNames;
    }

    [Header("场景组配置")]
    [SerializeField] private List<SceneGroup> sceneGroups;

    private HashSet<string> loadedScenes = new HashSet<string>();

    /// <summary>
    /// 加载场景组
    /// </summary>
    public void LoadGroup(string groupName)
    {
        var group = sceneGroups.Find(g => g.groupName == groupName);
        if (group == null)
        {
            Debug.LogError($"场景组不存在: {groupName}");
            return;
        }

        StartCoroutine(LoadGroupCoroutine(group));
    }

    private System.Collections.IEnumerator LoadGroupCoroutine(SceneGroup group)
    {
        foreach (var sceneName in group.sceneNames)
        {
            if (!loadedScenes.Contains(sceneName))
            {
                yield return SceneManager.LoadSceneAsync(sceneName, LoadSceneMode.Additive);
                loadedScenes.Add(sceneName);
                Debug.Log($"加载场景: {sceneName}");
            }
        }
    }

    /// <summary>
    /// 卸载场景组
    /// </summary>
    public void UnloadGroup(string groupName)
    {
        var group = sceneGroups.Find(g => g.groupName == groupName);
        if (group == null) return;

        StartCoroutine(UnloadGroupCoroutine(group));
    }

    private System.Collections.IEnumerator UnloadGroupCoroutine(SceneGroup group)
    {
        foreach (var sceneName in group.sceneNames)
        {
            if (loadedScenes.Contains(sceneName))
            {
                yield return SceneManager.UnloadSceneAsync(sceneName);
                loadedScenes.Remove(sceneName);
                Debug.Log($"卸载场景: {sceneName}");
            }
        }
    }

    /// <summary>
    /// 卸载所有附加场景
    /// </summary>
    public void UnloadAllAdditiveScenes()
    {
        var activeScene = SceneManager.GetActiveScene();

        for (int i = SceneManager.sceneCount - 1; i >= 0; i--)
        {
            var scene = SceneManager.GetSceneAt(i);
            if (scene != activeScene && scene.isLoaded)
            {
                SceneManager.UnloadSceneAsync(scene);
            }
        }

        loadedScenes.Clear();
    }
}
```
