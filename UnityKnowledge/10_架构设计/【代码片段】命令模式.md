## 一、基础实现

### 1.1 命令接口

```csharp
/// <summary>
/// 命令接口
/// </summary>
public interface ICommand
{
    /// <summary>
    /// 执行命令
    /// </summary>
    void Execute();

    /// <summary>
    /// 撤销命令
    /// </summary>
    void Undo();

    /// <summary>
    /// 命令描述（用于调试）
    /// </summary>
    string Description { get; }
}
```

### 1.2 命令管理器（撤销/重做）

```csharp
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 命令管理器 - 支持撤销/重做
/// </summary>
public class CommandManager : MonoBehaviour
{
    [Header("配置")]
    [SerializeField] private int maxHistorySize = 100;

    private readonly Stack<ICommand> undoStack = new Stack<ICommand>();
    private readonly Stack<ICommand> redoStack = new Stack<ICommand>();

    /// <summary>
    /// 执行命令
    /// </summary>
    public void ExecuteCommand(ICommand command)
    {
        command.Execute();

        undoStack.Push(command);
        redoStack.Clear();  // 新操作清空重做栈

        // 限制历史大小
        if (undoStack.Count > maxHistorySize)
        {
            var temp = new Stack<ICommand>();
            for (int i = 0; i < maxHistorySize - 1; i++)
            {
                temp.Push(undoStack.Pop());
            }
            undoStack.Clear();
            while (temp.Count > 0)
            {
                undoStack.Push(temp.Pop());
            }
        }

        Debug.Log($"执行命令: {command.Description}");
    }

    /// <summary>
    /// 撤销上一个命令
    /// </summary>
    public bool Undo()
    {
        if (undoStack.Count == 0) return false;

        var command = undoStack.Pop();
        command.Undo();
        redoStack.Push(command);

        Debug.Log($"撤销命令: {command.Description}");
        return true;
    }

    /// <summary>
    /// 重做下一个命令
    /// </summary>
    public bool Redo()
    {
        if (redoStack.Count == 0) return false;

        var command = redoStack.Pop();
        command.Execute();
        undoStack.Push(command);

        Debug.Log($"重做命令: {command.Description}");
        return true;
    }

    /// <summary>
    /// 清空历史
    /// </summary>
    public void ClearHistory()
    {
        undoStack.Clear();
        redoStack.Clear();
    }

    /// <summary>
    /// 是否可以撤销
    /// </summary>
    public bool CanUndo => undoStack.Count > 0;

    /// <summary>
    /// 是否可以重做
    /// </summary>
    public bool CanRedo => redoStack.Count > 0;
}
```

## 二、常用命令实现

### 2.1 移动命令

```csharp
using UnityEngine;

/// <summary>
/// 移动命令
/// </summary>
public class MoveCommand : ICommand
{
    private readonly Transform target;
    private readonly Vector3 newPosition;
    private readonly Vector3 previousPosition;

    public string Description => $"移动 {target.name} 到 {newPosition}";

    public MoveCommand(Transform target, Vector3 newPosition)
    {
        this.target = target;
        this.newPosition = newPosition;
        this.previousPosition = target.position;
    }

    public void Execute()
    {
        target.position = newPosition;
    }

    public void Undo()
    {
        target.position = previousPosition;
    }
}
```

### 2.2 属性修改命令

```csharp
using UnityEngine;

/// <summary>
/// 属性修改命令
/// </summary>
public class SetPropertyCommand<T> : ICommand
{
    private readonly object target;
    private readonly System.Action<T> setter;
    private readonly T newValue;
    private readonly T oldValue;

    public string Description => $"修改属性为 {newValue}";

    public SetPropertyCommand(System.Func<T> getter, System.Action<T> setter, T newValue)
    {
        this.setter = setter;
        this.newValue = newValue;
        this.oldValue = getter();
    }

    public void Execute()
    {
        setter(newValue);
    }

    public void Undo()
    {
        setter(oldValue);
    }
}

// 使用示例
var healthCommand = new SetPropertyCommand<float>(
    () => player.Health,
    (v) => player.SetHealth(v),
    50f
);
commandManager.ExecuteCommand(healthCommand);
```

### 2.3 创建/删除对象命令

```csharp
using UnityEngine;

/// <summary>
/// 创建对象命令
/// </summary>
public class CreateObjectCommand : ICommand
{
    private readonly GameObject prefab;
    private readonly Vector3 position;
    private readonly Quaternion rotation;
    private GameObject createdObject;

    public string Description => $"创建 {prefab.name}";

    public CreateObjectCommand(GameObject prefab, Vector3 position, Quaternion rotation)
    {
        this.prefab = prefab;
        this.position = position;
        this.rotation = rotation;
    }

    public void Execute()
    {
        createdObject = Object.Instantiate(prefab, position, rotation);
    }

    public void Undo()
    {
        if (createdObject != null)
        {
            Object.Destroy(createdObject);
        }
    }

    /// <summary>
    /// 获取创建的对象
    /// </summary>
    public GameObject CreatedObject => createdObject;
}

/// <summary>
/// 删除对象命令
/// </summary>
public class DeleteObjectCommand : ICommand
{
    private readonly GameObject target;
    private readonly Vector3 position;
    private readonly Quaternion rotation;
    private GameObject prefabReference;

    public string Description => $"删除 {target.name}";

    public DeleteObjectCommand(GameObject target)
    {
        this.target = target;
        this.position = target.transform.position;
        this.rotation = target.transform.rotation;
    }

    public void Execute()
    {
        // 保存预制体引用用于重建
        prefabReference = target;  // 实际应保存预制体引用
        target.SetActive(false);
    }

    public void Undo()
    {
        target.transform.position = position;
        target.transform.rotation = rotation;
        target.SetActive(true);
    }
}
```

## 三、技能系统实现

### 3.1 技能命令基类

```csharp
using UnityEngine;

/// <summary>
/// 技能命令基类
/// </summary>
public abstract class SkillCommand : ICommand
{
    protected readonly Character caster;
    protected float cooldown;
    protected float lastCastTime;

    public abstract string Description { get; }

    protected SkillCommand(Character caster)
    {
        this.caster = caster;
    }

    public void Execute()
    {
        if (!CanCast())
        {
            Debug.Log($"{Description} 冷却中");
            return;
        }

        OnExecute();
        lastCastTime = Time.time;
    }

    public void Undo()
    {
        OnUndo();
    }

    /// <summary>
    /// 是否可以施放
    /// </summary>
    public bool CanCast()
    {
        return Time.time >= lastCastTime + cooldown;
    }

    /// <summary>
    /// 剩余冷却时间
    /// </summary>
    public float RemainingCooldown => Mathf.Max(0, lastCastTime + cooldown - Time.time);

    /// <summary>
    /// 冷却进度 (0-1)
    /// </summary>
    public float CooldownProgress => CanCast() ? 1f : 1f - (RemainingCooldown / cooldown);

    /// <summary>
    /// 实际执行逻辑
    /// </summary>
    protected abstract void OnExecute();

    /// <summary>
    /// 撤销逻辑
    /// </summary>
    protected abstract void OnUndo();
}
```

### 3.2 具体技能实现

```csharp
using UnityEngine;

/// <summary>
/// 火球术技能
/// </summary>
public class FireballSkill : SkillCommand
{
    private readonly Vector3 targetPosition;
    private GameObject fireballInstance;

    public override string Description => "火球术";

    public FireballSkill(Character caster, Vector3 target) : base(caster)
    {
        cooldown = 2f;
        targetPosition = target;
    }

    protected override void OnExecute()
    {
        // 创建火球
        fireballInstance = Object.Instantiate(
            caster.FireballPrefab,
            caster.CastPoint.position,
            Quaternion.LookRotation(targetPosition - caster.CastPoint.position)
        );

        // 设置伤害
        var fireball = fireballInstance.GetComponent<Fireball>();
        fireball.damage = 50;
        fireball.owner = caster;

        // 播放动画和音效
        caster.PlayAnimation("Cast");
        caster.PlaySound("Fireball");
    }

    protected override void OnUndo()
    {
        if (fireballInstance != null)
        {
            Object.Destroy(fireballInstance);
        }
    }
}

/// <summary>
/// 治疗技能
/// </summary>
public class HealSkill : SkillCommand
{
    private readonly float healAmount;
    private float previousHealth;

    public override string Description => $"治疗 {healAmount}";

    public HealSkill(Character caster, float amount) : base(caster)
    {
        cooldown = 5f;
        healAmount = amount;
    }

    protected override void OnExecute()
    {
        previousHealth = caster.Health;
        caster.Heal(healAmount);

        // 播放效果
        caster.PlayEffect("Heal");
    }

    protected override void OnUndo()
    {
        caster.SetHealth(previousHealth);
    }
}

/// <summary>
/// 闪现技能
/// </summary>
public class BlinkSkill : SkillCommand
{
    private readonly float distance;
    private Vector3 previousPosition;

    public override string Description => $"闪现 {distance}米";

    public BlinkSkill(Character caster, float distance) : base(caster)
    {
        cooldown = 8f;
        this.distance = distance;
    }

    protected override void OnExecute()
    {
        previousPosition = caster.transform.position;

        // 计算目标位置
        Vector3 targetPos = caster.transform.position + caster.transform.forward * distance;

        // 检查碰撞
        if (Physics.Raycast(caster.transform.position, caster.transform.forward, out var hit, distance))
        {
            targetPos = hit.point - caster.transform.forward * 1f;
        }

        caster.transform.position = targetPos;

        // 播放效果
        caster.PlayEffect("Blink");
    }

    protected override void OnUndo()
    {
        caster.transform.position = previousPosition;
    }
}
```

### 3.3 组合技能（连招）

```csharp
using System.Collections.Generic;

/// <summary>
/// 组合命令 - 多个命令一起执行
/// </summary>
public class ComboCommand : ICommand
{
    private readonly List<ICommand> commands;
    public string Description => $"连招 ({commands.Count}个技能)";

    public ComboCommand(params ICommand[] commands)
    {
        this.commands = new List<ICommand>(commands);
    }

    public void Execute()
    {
        foreach (var cmd in commands)
        {
            cmd.Execute();
        }
    }

    public void Undo()
    {
        // 逆序撤销
        for (int i = commands.Count - 1; i >= 0; i--)
        {
            commands[i].Undo();
        }
    }

    public void AddCommand(ICommand command)
    {
        commands.Add(command);
    }
}

// 使用示例
var combo = new ComboCommand(
    new BlinkSkill(player, 5f),
    new FireballSkill(player, enemy.position),
    new HealSkill(player, 30f)
);
commandManager.ExecuteCommand(combo);
```

## 四、技能队列系统

```csharp
using System.Collections.Generic;
using System.Collections;
using UnityEngine;

/// <summary>
/// 技能队列管理器
/// </summary>
public class SkillQueue : MonoBehaviour
{
    [Header("配置")]
    [SerializeField] private float skillInterval = 0.5f;

    private readonly Queue<SkillCommand> skillQueue = new Queue<SkillCommand>();
    private bool isExecuting;
    private Coroutine executeCoroutine;

    /// <summary>
    /// 添加技能到队列
    /// </summary>
    public void EnqueueSkill(SkillCommand skill)
    {
        skillQueue.Enqueue(skill);

        if (!isExecuting)
        {
            executeCoroutine = StartCoroutine(ExecuteQueue());
        }
    }

    /// <summary>
    /// 清空队列
    /// </summary>
    public void ClearQueue()
    {
        skillQueue.Clear();
        if (executeCoroutine != null)
        {
            StopCoroutine(executeCoroutine);
        }
        isExecuting = false;
    }

    private IEnumerator ExecuteQueue()
    {
        isExecuting = true;

        while (skillQueue.Count > 0)
        {
            var skill = skillQueue.Dequeue();

            if (skill.CanCast())
            {
                skill.Execute();
                yield return new WaitForSeconds(skillInterval);
            }
            else
            {
                // 技能冷却中，可以选择等待或跳过
                Debug.Log($"技能 {skill.Description} 冷却中，跳过");
            }
        }

        isExecuting = false;
    }

    /// <summary>
    /// 队列中的技能数量
    /// </summary>
    public int QueueCount => skillQueue.Count;

    /// <summary>
    /// 是否正在执行
    /// </summary>
    public bool IsExecuting => isExecuting;
}
```

## 五、输入系统绑定

```csharp
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 输入命令绑定器 - 可重映射的输入系统
/// </summary>
public class InputCommandBinder : MonoBehaviour
{
    [System.Serializable]
    public class KeyBinding
    {
        public KeyCode key;
        public string commandId;
    }

    [Header("按键绑定")]
    [SerializeField] private List<KeyBinding> bindings = new List<KeyBinding>();

    private Dictionary<KeyCode, ICommand> keyCommands = new Dictionary<KeyCode, ICommand>();
    private Dictionary<string, System.Func<ICommand>> commandFactories = new Dictionary<string, System.Func<ICommand>>();

    private CommandManager commandManager;
    private Character player;

    private void Awake()
    {
        commandManager = GetComponent<CommandManager>();
        player = GetComponent<Character>();

        // 注册命令工厂
        RegisterCommandFactory("Fireball", () => new FireballSkill(player, GetMousePosition()));
        RegisterCommandFactory("Heal", () => new HealSkill(player, 30f));
        RegisterCommandFactory("Blink", () => new BlinkSkill(player, 5f));
    }

    private void Start()
    {
        // 应用按键绑定
        foreach (var binding in bindings)
        {
            if (commandFactories.TryGetValue(binding.commandId, out var factory))
            {
                keyCommands[binding.key] = factory();
            }
        }
    }

    private void Update()
    {
        foreach (var kvp in keyCommands)
        {
            if (Input.GetKeyDown(kvp.Key))
            {
                // 每次按键创建新命令实例
                if (commandFactories.TryGetValue(GetCommandId(kvp.Key), out var factory))
                {
                    var command = factory();
                    commandManager.ExecuteCommand(command);
                }
            }
        }

        // 撤销/重做快捷键
        if (Input.GetKey(KeyCode.LeftControl) && Input.GetKeyDown(KeyCode.Z))
        {
            commandManager.Undo();
        }
        if (Input.GetKey(KeyCode.LeftControl) && Input.GetKeyDown(KeyCode.Y))
        {
            commandManager.Redo();
        }
    }

    /// <summary>
    /// 注册命令工厂
    /// </summary>
    public void RegisterCommandFactory(string id, System.Func<ICommand> factory)
    {
        commandFactories[id] = factory;
    }

    /// <summary>
    /// 重新绑定按键
    /// </summary>
    public void RebindKey(KeyCode oldKey, KeyCode newKey)
    {
        if (keyCommands.TryGetValue(oldKey, out var command))
        {
            keyCommands.Remove(oldKey);
            keyCommands[newKey] = command;
        }
    }

    private string GetCommandId(KeyCode key)
    {
        foreach (var binding in bindings)
        {
            if (binding.key == key)
                return binding.commandId;
        }
        return null;
    }

    private Vector3 GetMousePosition()
    {
        var ray = Camera.main.ScreenPointToRay(Input.mousePosition);
        if (Physics.Raycast(ray, out var hit))
        {
            return hit.point;
        }
        return player.transform.position + player.transform.forward * 10f;
    }
}
```
