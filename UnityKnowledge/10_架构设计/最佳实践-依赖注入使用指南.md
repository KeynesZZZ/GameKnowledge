# 最佳实践 - 依赖注入使用指南

> VContainer在Unity中的配置与使用 `#最佳实践` `#依赖注入` `#VContainer`

## 一、为什么使用依赖注入

### 1.1 单例的问题

```csharp
// ❌ 单例问题：隐式依赖、难以测试
public class Player : MonoBehaviour
{
    private void TakeDamage(int damage)
    {
        // 隐式依赖AudioManager
        AudioManager.Instance.PlaySound("Hurt");

        // 隐式依赖SaveManager
        SaveManager.Instance.Save();
    }
}
```

### 1.2 依赖注入的优势

```csharp
// ✅ 依赖注入：显式依赖、易于测试
public class Player : MonoBehaviour
{
    private readonly IAudioManager audioManager;
    private readonly ISaveManager saveManager;

    // 构造函数明确声明依赖
    public Player(IAudioManager audio, ISaveManager save)
    {
        audioManager = audio;
        saveManager = save;
    }

    private void TakeDamage(int damage)
    {
        audioManager.PlaySound("Hurt");
        saveManager.Save();
    }
}
```

## 二、VContainer配置

### 2.1 安装VContainer

```
// 通过Package Manager安装
// 或 manifest.json添加
"com.vcontainer": "https://github.com/hadashiA/VContainer.git"
```

### 2.2 创建生命周期配置

```csharp
using UnityEngine;
using VContainer;
using VContainer.Unity;

/// <summary>
/// 游戏生命周期配置
/// </summary>
public class GameLifetimeScope : LifetimeScope
{
    [Header("配置")]
    [SerializeField] private GameConfig gameConfig;

    protected override void Configure(IContainerBuilder builder)
    {
        // 注册单例服务
        builder.Register<IGameManager, GameManager>(Lifetime.Singleton);
        builder.Register<IAudioManager, AudioManager>(Lifetime.Singleton);
        builder.Register<ISaveManager, SaveManager>(Lifetime.Singleton);

        // 注册配置
        builder.RegisterInstance(gameConfig);

        // 注册MonoBehaviour（使用FindComponent）
        builder.RegisterComponent<UIManager>(FindComponent<UIManager>());
    }
}
```

### 2.3 场景生命周期

```csharp
using VContainer;
using VContainer.Unity;

/// <summary>
/// 场景级别的生命周期配置

public class SceneLifetimeScope : LifetimeScope

        builder.Register<IEnemySpawner, EnemySpawner>(Lifetime.Scoped);
        builder.Register<ILevelController, LevelController>(Lifetime.Scoped);
```

## 三、注册方式详解

### 3.1 注册类型

```csharp
protected override void Configure(IContainerBuilder builder)
{
    // 1. 单例 - 整个游戏只有一个实例
    builder.Register<IGameManager, GameManager>(Lifetime.Singleton);

    // 2. 瞬态 - 每次请求都创建新实例
    builder.Register<IEnemy, Enemy>(Lifetime.Transient);

    // 3. 作用域 - 每个Scope一个实例
    builder.Register<IPlayerData, PlayerData>(Lifetime.Scoped);

    // 4. 注册已存在的实例
    builder.RegisterInstance(gameConfig);

    // 5. 注册组件（场景中的MonoBehaviour）
    builder.RegisterComponent<UIManager>(FindComponent<UIManager>());

    // 6. 使用工厂注册
    builder.RegisterFactory<IEnemy, EnemyFactory>(Lifetime.Scoped);
}
```

### 3.2 注册接口与实现

```csharp
// 接口定义
public interface IGameManager
{
    GameState State { get; }
    void StartGame();
    void EndGame();
}

// 实现
public class GameManager : IGameManager
{
    private readonly IAudioManager audioManager;
    private readonly ISaveManager saveManager;

    // 通过构造函数注入依赖
    public GameManager(IAudioManager audio, ISaveManager save)
    {
        audioManager = audio;
        saveManager = save;
    }

    public GameState State { get; private set; }

    public void StartGame()
    {
        State = GameState.Playing;
        audioManager.PlayBGM("Game");
    }

    public void EndGame()
    {
        State = GameState.GameOver;
        saveManager.Save();
    }
}
```

## 四、注入方式

### 4.1 构造函数注入（推荐）

```csharp
public class PlayerController : MonoBehaviour
{
    private readonly IInputManager inputManager;
    private readonly IGameManager gameManager;

    // 构造函数注入
    [Inject]
    public PlayerController(IInputManager input, IGameManager game)
    {
        inputManager = input;
        gameManager = game;
    }
}
```

### 4.2 属性注入

```csharp
public class PlayerController : MonoBehaviour
{
    // 属性注入
    [Inject]
    public IInputManager InputManager { get; private set; }

    [Inject]
    public IGameManager GameManager { get; private set; }
}
```

### 4.3 方法注入

```csharp
public class PlayerController : MonoBehaviour
{
    private IInputManager inputManager;
    private IGameManager gameManager;

    // 方法注入
    [Inject]
    public void Initialize(IInputManager input, IGameManager game)
    {
        inputManager = input;
        gameManager = game;
    }
}
```

### 5.2 完整代码

```csharp
// ============ GameLifetimeScope.cs ============
using VContainer;
using VContainer.Unity;

public class GameLifetimeScope : LifetimeScope
{
    protected override void Configure(IContainerBuilder builder)
    {
        // 核心服务
        builder.Register<IGameManager, GameManager>(Lifetime.Singleton);
        builder.Register<IAudioManager, AudioManager>(Lifetime.Singleton);
        builder.Register<ISaveManager, SaveManager>(Lifetime.Singleton);

        // 事件系统
        builder.Register<IEventBus, EventBus>(Lifetime.Singleton);
    }
}

// ============ IGameManager.cs ============
public interface IGameManager
{
    GameState State { get; }
    void StartGame();
    void PauseGame();
    void EndGame();
    event Action<GameState> OnStateChanged;
}

// ============ GameManager.cs ============
public class GameManager : IGameManager
{
    private readonly IEventBus eventBus;
    private readonly IAudioManager audioManager;

    public GameState State { get; private set; }
    public event Action<GameState> OnStateChanged;

    public GameManager(IEventBus eventBus, IAudioManager audio)
    {
        this.eventBus = eventBus;
        audioManager = audio;
    }

    public void StartGame()
    {
        State = GameState.Playing;
        audioManager.PlayBGM("Game");
        OnStateChanged?.Invoke(State);
        eventBus.Publish(new GameStartedEvent());
    }

    public void PauseGame()
    {
        State = GameState.Paused;
        OnStateChanged?.Invoke(State);
    }

    public void EndGame()
    {
        State = GameState.GameOver;
        audioManager.StopBGM();
        OnStateChanged?.Invoke(State);
        eventBus.Publish(new GameEndedEvent());
    }
}

// ============ PlayerController.cs ============
using UnityEngine;
using VContainer;

public class PlayerController : MonoBehaviour
{
    [Inject] private readonly IInputManager inputManager;
    [Inject] private readonly IGameManager gameManager;
    [Inject] private readonly IEventBus eventBus;

    private void Update()
    {
        if (gameManager.State != GameState.Playing) return;

        var input = inputManager.GetMovementInput();
        Move(input);
    }

    private void Move(Vector2 direction)
    {
        transform.position += new Vector3(direction.x, 0, direction.y) * 5f * Time.deltaTime;
    }
}
```

## 六、测试示例

### 6.1 单元测试

```csharp
using NUnit.Framework;
using NSubstitute;

[TestFixture]
public class GameManagerTests
{
    private IEventBus eventBus;
    private IAudioManager audioManager;
    private GameManager gameManager;

    [SetUp]
    public void Setup()
    {
        // 创建Mock对象
        eventBus = Substitute.For<IEventBus>();
        audioManager = Substitute.For<IAudioManager>();

        // 注入Mock依赖
        gameManager = new GameManager(eventBus, audioManager);
    }

    [Test]
    public void StartGame_ShouldChangeState()
    {
        gameManager.StartGame();
        Assert.AreEqual(GameState.Playing, gameManager.State);
    }

    [Test]
    public void StartGame_ShouldPlayBGM()
    {
        gameManager.StartGame();
        audioManager.Received().PlayBGM("Game");
    }

    [Test]
    public void StartGame_ShouldPublishEvent()
    {
        gameManager.StartGame();
        eventBus.Received().Publish(Arg.Any<GameStartedEvent>());
    }
}
```

## 七、常见问题

### 7.1 循环依赖

```csharp
// ❌ 问题：A依赖B，B依赖A
public class A
{
    public A(B b) { }
}

public class B
{
    public B(A a) { }
}

// ✅ 解决：重构或使用事件
public class A
{
    public A(IB b) { }
}

public class B
{
    // 不直接依赖A，使用事件通信
    private readonly IEventBus eventBus;
    public B(IEventBus eventBus)
    {
        this.eventBus = eventBus;
    }
}
```

### 7.2 MonoBehaviour注入

```csharp
// ❌ 问题：MonoBehaviour不能使用构造函数注入
public class Player : MonoBehaviour
{
    private readonly IGameManager gameManager;

    public Player(IGameManager game) // 不工作！
    {
        gameManager = game;
    }
}

// ✅ 解决：使用属性或方法注入
public class Player : MonoBehaviour
{
    [Inject]
    public IGameManager GameManager { get; private set; }
}
```
