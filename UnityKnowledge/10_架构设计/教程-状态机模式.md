# 状态机模式

> 第5课 | 脚本与架构模块

## 1. 什么是有限状态机(FSM)？

**定义**：状态机是一种行为模型，由有限数量的状态、转换和动作组成。

```
┌─────────────────────────────────────────────────────┐
│                   状态机示意图                        │
│                                                     │
│     ┌──────────┐    开始游戏    ┌──────────┐       │
│     │   Menu   │───────────────→│ Playing  │       │
│     │  (菜单)   │←──────────────│  (游戏中) │       │
│     └──────────┘    返回菜单    └────┬─────┘       │
│          ↑                            │             │
│          │ 重新开始                   │ 游戏结束    │
│          │                            ↓             │
│          │                      ┌──────────┐       │
│          └──────────────────────│ GameOver │       │
│                                 │ (结束)    │       │
│                                 └──────────┘       │
└─────────────────────────────────────────────────────┘
```

**核心要素**：
- **状态(State)**：对象在某一时刻的行为模式
- **转换(Transition)**：从一个状态到另一个状态的条件
- **事件(Event)**：触发状态转换的输入
- **动作(Action)**：进入/退出/处于某状态时的行为

## 2. 简单状态机实现

### 基于switch的简单实现

```csharp
/// <summary>
/// 游戏状态枚举
/// </summary>
public enum GameState
{
    Menu,
    Playing,
    Paused,
    GameOver
}

/// <summary>
/// 简单状态机（基于switch）
/// </summary>
public class SimpleGameStateMachine : MonoBehaviour
{
    private GameState currentState;

    public GameState CurrentState => currentState;

    private void Start()
    {
        ChangeState(GameState.Menu);
    }

    public void ChangeState(GameState newState)
    {
        if (currentState == newState) return;

        // 退出当前状态
        ExitState(currentState);

        // 更新状态
        var oldState = currentState;
        currentState = newState;

        // 进入新状态
        EnterState(newState);

        Debug.Log($"State changed: {oldState} → {newState}");
    }

    private void EnterState(GameState state)
    {
        switch (state)
        {
            case GameState.Menu:
                Time.timeScale = 0;
                ShowMenu();
                break;
            case GameState.Playing:
                Time.timeScale = 1;
                HideAllPanels();
                break;
            case GameState.Paused:
                Time.timeScale = 0;
                ShowPausePanel();
                break;
            case GameState.GameOver:
                Time.timeScale = 0;
                ShowGameOverPanel();
                break;
        }
    }

    private void ExitState(GameState state)
    {
        switch (state)
        {
            case GameState.Playing:
                SaveGameProgress();
                break;
        }
    }

    // 处理输入
    private void Update()
    {
        switch (currentState)
        {
            case GameState.Menu:
                UpdateMenu();
                break;
            case GameState.Playing:
                UpdatePlaying();
                break;
            case GameState.Paused:
                UpdatePaused();
                break;
            case GameState.GameOver:
                UpdateGameOver();
                break;
        }
    }

    private void UpdatePlaying()
    {
        if (Input.GetKeyDown(KeyCode.Escape))
        {
            ChangeState(GameState.Paused);
        }
    }

    private void UpdatePaused()
    {
        if (Input.GetKeyDown(KeyCode.Escape))
        {
            ChangeState(GameState.Playing);
        }
    }

    // ... 其他方法
}
```

**缺点**：
- 状态多了代码会变得冗长
- 不易扩展
- 状态逻辑混在一起

```csharp
/// <summary>
/// 状态接口
/// </summary>
public interface IState
{
    void Enter();
    void Exit();
    void Update();
    void FixedUpdate();
}

/// <summary>
/// 状态机基类
/// </summary>
public class StateMachine
{
    private IState currentState;

    public IState CurrentState => currentState;

    public void ChangeState(IState newState)
    {
        if (currentState == newState) return;

        currentState?.Exit();
        currentState = newState;
        currentState?.Enter();
    }

    public void Update()
    {
        currentState?.Update();
    }

    public void FixedUpdate()
    {
        currentState?.FixedUpdate();
    }
}

    [SerializeField] private GameObject pausePanel;
    [SerializeField] private GameObject gameOverPanel;

    private void Awake()
    {
        // 创建状态机
        StateMachine = new StateMachine();

        // 创建所有状态
        MenuState = new MenuState(this);
        PlayingState = new PlayingState(this);
        PausedState = new PausedState(this);
        GameOverState = new GameOverState(this);
    }

    private void Start()
    {
        // 进入初始状态
        StateMachine.ChangeState(MenuState);
    }

    private void Update()
    {
        StateMachine.Update();
    }

    private void FixedUpdate()
    {
        StateMachine.FixedUpdate();
    }

    // 公共方法供状态调用
    public void ShowMenu() => menuPanel.SetActive(true);
    public void HideMenu() => menuPanel.SetActive(false);
    public void ShowPausePanel() => pausePanel.SetActive(true);
    public void HidePausePanel() => pausePanel.SetActive(false);
    public void ShowGameOverPanel() => gameOverPanel.SetActive(true);
    public void HideGameOverPanel() => gameOverPanel.SetActive(false);
    public void HideAllPanels()
    {
        menuPanel.SetActive(false);
        pausePanel.SetActive(false);

    public void ResumeGame() => StateMachine.ChangeState(PlayingState);
    public void EndGame() => StateMachine.ChangeState(GameOverState);
    public void ReturnToMenu() => StateMachine.ChangeState(MenuState);
}
```

## 4. 通用泛型状态机

```csharp
/// <summary>
/// 泛型状态接口
/// </summary>
public interface IState<TState> where TState : System.Enum
{
    TState StateType { get; }
    void Enter();
    void Exit();
    void Update();
}

/// <summary>
/// 泛型状态机
/// </summary>
public class StateMachine<TState> where TState : System.Enum
{
    private readonly Dictionary<TState, IState<TState>> states;
    private IState<TState> currentState;

    public TState CurrentStateType => currentState != null ? currentState.StateType : default;
    public IState<TState> CurrentState => currentState;

    public event Action<TState, TState> OnStateChanged;

    public StateMachine()
    {
        states = new Dictionary<TState, IState<TState>>();
    }

    public void RegisterState(IState<TState> state)
    {
        states[state.StateType] = state;
    }

    public bool ChangeState(TState newStateType)
    {
        if (!states.TryGetValue(newStateType, out var newState))
        {
            Debug.LogError($"State not found: {newStateType}");
            return false;
        }

        if (currentState != null && currentState.StateType.Equals(newStateType))
        {
            return false;
        }

        var oldStateType = currentState != null ? currentState.StateType : default;

        currentState?.Exit();
        currentState = newState;
        currentState.Enter();

        OnStateChanged?.Invoke(oldStateType, newStateType);

        return true;
    }

    public void Update()
    {
        currentState?.Update();
    }
}

/// <summary>
/// 状态基类（简化实现）
/// </summary>
public abstract class StateBase<TState> : IState<TState> where TState : System.Enum
{
    public abstract TState StateType { get; }

    protected readonly StateMachine<TState> stateMachine;

    protected StateBase(StateMachine<TState> stateMachine)
    {
        this.stateMachine = stateMachine;
    }

    public virtual void Enter() { }
    public virtual void Exit() { }
    public virtual void Update() { }
}

// 使用示例
public enum CombatState
{
    Idle,
    Attack,
    Skill,
    Hurt,
    Dead
}

public class CombatStateMachine : MonoBehaviour
{
    private StateMachine<CombatState> stateMachine;

    private void Awake()
    {
        stateMachine = new StateMachine<CombatState>();
        stateMachine.RegisterState(new IdleCombatState(stateMachine, this));
        stateMachine.RegisterState(new AttackCombatState(stateMachine, this));
        stateMachine.RegisterState(new SkillCombatState(stateMachine, this));
        stateMachine.RegisterState(new HurtCombatState(stateMachine, this));
        stateMachine.RegisterState(new DeadCombatState(stateMachine, this));

        stateMachine.OnStateChanged += OnStateChanged;
    }

    private void Start()
    {
        stateMachine.ChangeState(CombatState.Idle);
    }

    private void Update()
    {
        stateMachine.Update();
    }

    private void OnStateChanged(CombatState oldState, CombatState newState)
    {
        Debug.Log($"Combat state: {oldState} → {newState}");
    }
}
```

## 5. 层次状态机(HFSM)

**问题**：状态多了之后，状态机会变得复杂

**解决方案**：层次状态机 - 状态可以包含子状态

```
┌─────────────────────────────────────────────┐
│                 Game                        │
│  ┌───────────────────────────────────────┐  │
│  │            Combat (战斗)               │  │
│  │  ┌─────────┐  ┌─────────┐            │  │
│  │  │ Attack  │  │ Defend  │            │  │
│  │  └─────────┘  └─────────┘            │  │
│  └───────────────────────────────────────┘  │
│  ┌───────────────────────────────────────┐  │
│  │            Explore (探索)              │  │
│  │  ┌─────────┐  ┌─────────┐            │  │
│  │  │  Walk   │  │  Idle   │            │  │
│  │  └─────────┘  └─────────┘            │  │
│  └───────────────────────────────────────┘  │
└─────────────────────────────────────────────┘
```

```csharp
/// <summary>
/// 层次状态接口
/// </summary>
public interface IHierarchicalState
{
    void Enter();
    void Exit();
    void Update();
    IHierarchicalState Parent { get; set; }
}

/// <summary>
/// 层次状态机
/// </summary>
public class HierarchicalStateMachine
{
    private IHierarchicalState currentState;
    private readonly Dictionary<string, IHierarchicalState> states = new();

    public void AddState(string name, IHierarchicalState state)
    {
        states[name] = state;
    }

    public void ChangeState(string stateName)
    {
        if (!states.TryGetValue(stateName, out var newState))
        {
            Debug.LogError($"State not found: {stateName}");
            return;
        }

        // 找到公共祖先
        var commonAncestor = FindCommonAncestor(currentState, newState);

        // 从当前状态退出到公共祖先
        ExitToAncestor(currentState, commonAncestor);

        // 从公共祖先进入新状态
        EnterFromAncestor(newState, commonAncestor);

        currentState = newState;
    }

    private IHierarchicalState FindCommonAncestor(IHierarchicalState a, IHierarchicalState b)
    {
        var ancestorsA = new HashSet<IHierarchicalState>();
        var current = a;
        while (current != null)
        {
            ancestorsA.Add(current);
            current = current.Parent;
        }

        current = b;
        while (current != null)
        {
            if (ancestorsA.Contains(current))
                return current;
            current = current.Parent;
        }

        return null;
    }

    private void ExitToAncestor(IHierarchicalState state, IHierarchicalState ancestor)
    {
        var current = state;
        while (current != null && current != ancestor)
        {
            current.Exit();
            current = current.Parent;
        }
    }

    private void EnterFromAncestor(IHierarchicalState state, IHierarchicalState ancestor)
    {
        var path = new Stack<IHierarchicalState>();
        var current = state;
        while (current != null && current != ancestor)
        {
            path.Push(current);
            current = current.Parent;
        }

        while (path.Count > 0)
        {
            path.Pop().Enter();
        }
    }

    public void Update()
    {
        currentState?.Update();
    }
}
```

## 6. 三消游戏实战：游戏状态机

```csharp
/// <summary>
/// 三消游戏状态枚举
/// </summary>
public enum Match3GameState
{
    None,
    Loading,      // 加载中
    Ready,        // 准备开始（显示棋盘）
    PlayerInput,  // 等待玩家输入
    Swapping,     // 交换动画中
    Matching,     // 消除检测中
    Removing,     // 消除动画中
    Filling,      // 填充新棋子中
    Combat,       // 战斗中
    Victory,      // 胜利
    Defeat        // 失败
}

/// <summary>
/// 三消游戏状态机
/// </summary>
public class Match3GameFSM : MonoBehaviour
{
    private StateMachine<Match3GameState> stateMachine;

    [Header("Components")]
    [SerializeField] private Board board;
    [SerializeField] private MatchDetector matchDetector;
    [SerializeField] private CombatSystem combatSystem;

    public Match3GameState CurrentState => stateMachine.CurrentStateType;

    public event Action<Match3GameState, Match3GameState> OnStateChanged;

    private void Awake()
    {
        InitializeStateMachine();
    }

    private void InitializeStateMachine()
    {
        stateMachine = new StateMachine<Match3GameState>();

        // 注册所有状态
        stateMachine.RegisterState(new LoadingState(this));
        stateMachine.RegisterState(new ReadyState(this));
        stateMachine.RegisterState(new PlayerInputState(this));
        stateMachine.RegisterState(new SwappingState(this));
        stateMachine.RegisterState(new MatchingState(this));
        stateMachine.RegisterState(new RemovingState(this));
        stateMachine.RegisterState(new FillingState(this));
        stateMachine.RegisterState(new CombatState(this));
        stateMachine.RegisterState(new VictoryState(this));
        stateMachine.RegisterState(new DefeatState(this));

        stateMachine.OnStateChanged += (oldState, newState) =>
        {
            Debug.Log($"[Match3FSM] {oldState} → {newState}");
            OnStateChanged?.Invoke(oldState, newState);
        };
    }

    private void Start()
    {
        stateMachine.ChangeState(Match3GameState.Loading);
    }

    private void Update()
    {
        stateMachine.Update();
    }

    // 状态转换方法
    public void ChangeState(Match3GameState newState) => stateMachine.ChangeState(newState);

    // ============ 状态实现 ============

    private abstract class Match3StateBase : StateBase<Match3GameState>
    {
        protected readonly Match3GameFSM fsm;
        protected readonly Board board;
        protected readonly MatchDetector matchDetector;
        protected readonly CombatSystem combatSystem;

        protected Match3StateBase(Match3GameFSM fsm) : base(fsm.stateMachine)
        {
            this.fsm = fsm;
            this.board = fsm.board;
            this.matchDetector = fsm.matchDetector;
            this.combatSystem = fsm.combatSystem;
        }
    }

    private class LoadingState : Match3StateBase
    {
        public override Match3GameState StateType => Match3GameState.Loading;

        public LoadingState(Match3GameFSM fsm) : base(fsm) { }

        public override void Enter()
        {
            // 加载资源、初始化棋盘
            fsm.StartCoroutine(LoadGame());
        }

        private IEnumerator LoadGame()
        {
            yield return null; // 模拟加载

            board.Initialize(7, 7);
            board.FillBoard();

            stateMachine.ChangeState(Match3GameState.Ready);
        }
    }

    private class ReadyState : Match3StateBase
    {
        public override Match3GameState StateType => Match3GameState.Ready;

        public ReadyState(Match3GameFSM fsm) : base(fsm) { }

        public override void Enter()
        {
            // 显示开始UI
            Debug.Log("Game Ready! Tap to start.");

            // 自动进入输入状态
            fsm.StartCoroutine(DelayedTransition());
        }

        private IEnumerator DelayedTransition()
        {
            yield return new WaitForSeconds(0.5f);
            stateMachine.ChangeState(Match3GameState.PlayerInput);
        }
    }

    private class PlayerInputState : Match3StateBase
    {
        public override Match3GameState StateType => Match3GameState.PlayerInput;

        private Vector2Int? selectedGem;

        public PlayerInputState(Match3GameFSM fsm) : base(fsm) { }

        public override void Enter()
        {
            selectedGem = null;
            Debug.Log("Waiting for player input...");
        }

        public override void Exit()
        {
            selectedGem = null;
        }

        public override void Update()
        {
            if (Input.GetMouseButtonDown(0))
            {
                var pos = GetGridPosition(Input.mousePosition);
                if (pos.HasValue)
                {
                    HandleGemSelection(pos.Value);
                }
            }
        }

        private void HandleGemSelection(Vector2Int pos)
        {
            if (!selectedGem.HasValue)
            {
                selectedGem = pos;
                board.HighlightGem(pos);
            }
            else
            {
                // 检查是否相邻
                if (IsAdjacent(selectedGem.Value, pos))
                {
                    // 开始交换
                    fsm.StartSwap(selectedGem.Value, pos);
                    stateMachine.ChangeState(Match3GameState.Swapping);
                }
                else
                {
                    // 选择新的棋子
                    board.ClearHighlight();
                    selectedGem = pos;
                    board.HighlightGem(pos);
                }
            }
        }

        private bool IsAdjacent(Vector2Int a, Vector2Int b)
        {
            var dx = Mathf.Abs(a.x - b.x);
            var dy = Mathf.Abs(a.y - b.y);
            return (dx == 1 && dy == 0) || (dx == 0 && dy == 1);
        }

        private Vector2Int? GetGridPosition(Vector3 mousePosition)
        {
            // 转换鼠标位置到棋盘坐标
            // 实现省略...
            return null;
        }
    }

    private class SwappingState : Match3StateBase
    {
        public override Match3GameState StateType => Match3GameState.Swapping;

        public SwappingState(Match3GameFSM fsm) : base(fsm) { }

        public override void Enter()
        {
            fsm.StartCoroutine(SwapAnimation());
        }

        private IEnumerator SwapAnimation()
        {
            // 播放交换动画
            yield return new WaitForSeconds(0.3f);

            // 检查是否有消除
            stateMachine.ChangeState(Match3GameState.Matching);
        }
    }

    private class MatchingState : Match3StateBase
    {
        public override Match3GameState StateType => Match3GameState.Matching;

        public MatchingState(Match3GameFSM fsm) : base(fsm) { }

        public override void Enter()
        {
            var matches = matchDetector.DetectMatches();

            if (matches.Count > 0)
            {
                // 有消除，进入消除状态
                fsm.StoreMatches(matches);
                stateMachine.ChangeState(Match3GameState.Removing);
            }
            else
            {
                // 没有消除，交换回来，返回输入状态
                fsm.ReverseSwap();
                stateMachine.ChangeState(Match3GameState.PlayerInput);
            }
        }
    }

    private class RemovingState : Match3StateBase
    {
        public override Match3GameState StateType => Match3GameState.Removing;

        public RemovingState(Match3GameFSM fsm) : base(fsm) { }

        public override void Enter()
        {
            fsm.StartCoroutine(RemoveAnimation());
        }

        private IEnumerator RemoveAnimation()
        {
            var matches = fsm.GetStoredMatches();

            // 播放消除动画
            foreach (var match in matches)
            {
                board.RemoveGems(match.Positions);
                yield return new WaitForSeconds(0.1f);
            }

            // 触发战斗
            fsm.ProcessMatchCombat(matches);

            stateMachine.ChangeState(Match3GameState.Combat);
        }
    }

    private class FillingState : Match3StateBase
    {
        public override Match3GameState StateType => Match3GameState.Filling;

        public FillingState(Match3GameFSM fsm) : base(fsm) { }

        public override void Enter()
        {
            fsm.StartCoroutine(FillBoard());
        }

        private IEnumerator FillBoard()
        {
            // 下落动画
            yield return board.DropGemsCoroutine();

            // 填充新棋子
            yield return board.FillBoardCoroutine();

            // 检查是否有新的消除
            stateMachine.ChangeState(Match3GameState.Matching);
        }
    }

    private class CombatState : Match3StateBase
    {
        public override Match3GameState StateType => Match3GameState.Combat;

        public CombatState(Match3GameFSM fsm) : base(fsm) { }

        public override void Enter()
        {
            fsm.StartCoroutine(ProcessCombat());
        }

        private IEnumerator ProcessCombat()
        {
            yield return combatSystem.ProcessCombatCoroutine();

            // 检查胜负
            if (combatSystem.IsVictory)
            {
                stateMachine.ChangeState(Match3GameState.Victory);
            }
            else if (combatSystem.IsDefeat)
            {
                stateMachine.ChangeState(Match3GameState.Defeat);
            }
            else
            {
                // 继续填充棋盘
                stateMachine.ChangeState(Match3GameState.Filling);
            }
        }
    }

    private class VictoryState : Match3StateBase
    {
        public override Match3GameState StateType => Match3GameState.Victory;

        public VictoryState(Match3GameFSM fsm) : base(fsm) { }

        public override void Enter()
        {
            Debug.Log("Victory!");
            // 显示胜利UI、奖励等
        }

        public override void Update()
        {
            if (Input.anyKeyDown)
            {
                // 返回主菜单或下一关
            }
        }
    }

    private class DefeatState : Match3StateBase
    {
        public override Match3GameState StateType => Match3GameState.Defeat;

        public DefeatState(Match3GameFSM fsm) : base(fsm) { }

        public override void Enter()
        {
            Debug.Log("Defeat...");
            // 显示失败UI
        }
    }

    // ============ 辅助方法 ============

    private List<Match> storedMatches;

    public void StartSwap(Vector2Int from, Vector2Int to)
    {
        board.SwapGems(from, to);
    }

    public void ReverseSwap()
    {
        board.ReverseLastSwap();
    }

    public void StoreMatches(List<Match> matches)
    {
        storedMatches = matches;
    }

    public List<Match> GetStoredMatches() => storedMatches;

    public void ProcessMatchCombat(List<Match> matches)
    {
        combatSystem.ProcessMatches(matches);
    }
}
```

## 本课小结

### 核心知识点

| 知识点 | 核心要点 |
|--------|----------|
| FSM概念 | 有限状态、转换、事件、动作 |
| 简单实现 | switch-case，适合少量状态 |
| 状态模式 | 每个状态独立类，易扩展 |
| 泛型状态机 | 类型安全，可复用 |
| 层次状态机 | 状态嵌套，适合复杂场景 |

### 状态机选择指南

| 场景 | 推荐方案 |
|------|----------|
| 3-5个简单状态 | switch-case |
| 5-10个状态 | 状态模式 |
| 需要复用 | 泛型状态机 |
| 复杂嵌套状态 | 层次状态机(HFSM) |
| AI行为 | 行为树 |
