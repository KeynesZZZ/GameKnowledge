# 【代码片段】事件总线

> 核心价值：通过事件驱动实现模块间解耦，替代直接引用和单例调用

## 快速参考

```csharp
// 定义事件
public struct ScoreChangedEvent { public int Score; }

// 订阅
EventBus.Subscribe<ScoreChangedEvent>(OnScoreChanged);

// 发布
EventBus.Publish(new ScoreChangedEvent { Score = 100 });

// 取消订阅
```

## 完整实现

### 泛型事件总线

```csharp
using System;
using System.Collections.Generic;

/// <summary>
/// 全局事件总线 - 类型安全的事件系统
/// </summary>
public static class EventBus
{
    private static readonly Dictionary<Type, List<Delegate>> handlers = new();
    private static readonly Dictionary<Type, List<Delegate>> pendingRemoves = new();
    private static bool isPublishing;

    /// <summary>
    /// 订阅事件
    /// </summary>
    public static void Subscribe<T>(Action<T> handler) where T : struct
    {
        var type = typeof(T);
        if (!handlers.ContainsKey(type))
        {
            handlers[type] = new List<Delegate>();
        }
        handlers[type].Add(handler);
    }

    /// <summary>
    /// 取消订阅
    /// </summary>
    public static void Unsubscribe<T>(Action<T> handler) where T : struct
    {
        var type = typeof(T);

        if (isPublishing)
        {
            // 发布期间延迟移除，避免修改集合
            if (!pendingRemoves.ContainsKey(type))
            {
                pendingRemoves[type] = new List<Delegate>();
            }
            pendingRemoves[type].Add(handler);
        }
        else if (handlers.TryGetValue(type, out var list))
        {
            list.Remove(handler);
        }
    }

    /// <summary>
    /// 发布事件
    /// </summary>
    public static void Publish<T>(T eventData) where T : struct
    {
        var type = typeof(T);
        if (!handlers.TryGetValue(type, out var list))
        {
            return;
        }

        isPublishing = true;

        // 从后向前遍历，支持在回调中移除
        for (int i = list.Count - 1; i >= 0; i--)
        {
            try
            {
                ((Action<T>)list[i]).Invoke(eventData);
            }
            catch (Exception e)
            {
                UnityEngine.Debug.LogException(e);
            }
        }

        isPublishing = false;

        // 处理延迟移除
        if (pendingRemoves.TryGetValue(type, out var removes) && removes.Count > 0)
        {
            foreach (var handler in removes)
            {
                list.Remove(handler);
            }
            removes.Clear();
        }
    }

    /// <summary>
    /// 清除所有订阅
    /// </summary>
    public static void Clear()
    {
        handlers.Clear();
        pendingRemoves.Clear();
    }

    /// <summary>
    /// 获取指定事件的订阅者数量
    /// </summary>
    public static int GetSubscriberCount<T>() where T : struct
    {
        return handlers.TryGetValue(typeof(T), out var list) ? list.Count : 0;
    }
}
```

### 带优先级的事件总线

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

/// <summary>
/// 支持优先级的事件总线
/// </summary>
public static class PriorityEventBus
{
    private static readonly Dictionary<Type, List<(Delegate handler, int priority)>> handlers = new();

    public static void Subscribe<T>(Action<T> handler, int priority = 0) where T : struct
    {
        var type = typeof(T);
        if (!handlers.ContainsKey(type))
        {
            handlers[type] = new List<(Delegate, int)>();
        }

        var list = handlers[type];
        list.Add((handler, priority));

        // 按优先级排序（高优先级先执行）
        list.Sort((a, b) => b.priority.CompareTo(a.priority));
    }

    public static void Unsubscribe<T>(Action<T> handler) where T : struct
    {
        var type = typeof(T);
        if (handlers.TryGetValue(type, out var list))
        {
            list.RemoveAll(x => x.handler.Equals(handler));
        }
    }

    public static void Publish<T>(T eventData) where T : struct
    {
        var type = typeof(T);
        if (!handlers.TryGetValue(type, out var list))
        {
            return;
        }

        foreach (var (handler, _) in list.ToList())
        {
            try
            {
                ((Action<T>)handler).Invoke(eventData);
            }
            catch (Exception e)
            {
                UnityEngine.Debug.LogException(e);
            }
        }
    }

    public static void Clear()
    {
        handlers.Clear();
    }
}
```

### IDisposable 订阅模式

```csharp
using System;
using System.Collections.Generic;

/// <summary>
/// 支持IDisposable的事件总线 - 可用using管理订阅
/// </summary>
public static class DisposableEventBus
{
    private static readonly Dictionary<Type, List<Delegate>> handlers = new();

    /// <summary>
    /// 订阅事件，返回IDisposable用于取消订阅
    /// </summary>
    public static IDisposable Subscribe<T>(Action<T> handler) where T : struct
    {
        var type = typeof(T);
        if (!handlers.ContainsKey(type))
        {
            handlers[type] = new List<Delegate>();
        }
        handlers[type].Add(handler);

        return new EventSubscription<T>(handler);
    }

    public static void Unsubscribe<T>(Action<T> handler) where T : struct
    {
        if (handlers.TryGetValue(typeof(T), out var list))
        {
            list.Remove(handler);
        }
    }

    public static void Publish<T>(T eventData) where T : struct
    {
        if (!handlers.TryGetValue(typeof(T), out var list))
        {
            return;
        }

        foreach (var handler in list.ToArray())
        {
            ((Action<T>)handler).Invoke(eventData);
        }
    }

    private class EventSubscription<T> : IDisposable where T : struct
    {
        private Action<T> handler;
        private bool disposed;

        public EventSubscription(Action<T> handler)
        {
            this.handler = handler;
        }

        public void Dispose()
        {
            if (!disposed)
            {
                Unsubscribe(handler);
                handler = null;
                disposed = true;
            }
        }
    }
}
```

### 封装便捷类

```csharp
/// <summary>
/// 自动管理订阅的基类
/// </summary>
public abstract class EventListener : MonoBehaviour
{
    private readonly List<IDisposable> subscriptions = new();

    protected void Subscribe<T>(Action<T> handler) where T : struct
    {
        subscriptions.Add(DisposableEventBus.Subscribe(handler));
    }

    protected virtual void OnDestroy()
    {
        foreach (var sub in subscriptions)
        {
            sub.Dispose();
        }
        subscriptions.Clear();
    }
}
```

## 最佳实践

### DO ✅

- 使用 struct 定义事件数据，避免GC
- 在 OnDisable/OnDestroy 中取消订阅
- 使用有意义的事件名称
- 事件数据包含必要的上下文信息

### DON'T ❌

- 不要在事件处理中抛出未捕获异常
- 不要创建循环事件（A→B→A）
- 不要在事件处理中执行耗时操作
- 不要忘记取消订阅导致内存泄漏
