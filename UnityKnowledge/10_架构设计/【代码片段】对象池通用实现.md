## 完整实现

### 泛型对象池

```csharp
using System;
using System.Collections.Generic;

/// <summary>
/// 通用对象池 - 减少实例化开销和GC分配
/// </summary>
/// <typeparam name="T">池化对象类型</typeparam>
public class ObjectPool<T> where T : class
{
    private readonly Stack<T> pool;
    private readonly Func<T> createFunc;
    private readonly Action<T> onGet;
    private readonly Action<T> onReturn;
    private readonly Action<T> onDestroy;
    private readonly int maxSize;

    /// <summary>
    /// 当前池中可用对象数量
    /// </summary>
    public int CountInactive => pool.Count;

    /// <summary>
    /// 创建对象池
    /// </summary>
    /// <param name="createFunc">创建新对象的工厂方法</param>
    /// <param name="onGet">获取对象时的回调（可选）</param>
    /// <param name="onReturn">归还对象时的回调（可选，用于重置状态）</param>
    /// <param name="onDestroy">销毁对象时的回调（可选）</param>
    /// <param name="maxSize">池最大容量，超过时新归还的对象将被丢弃</param>
    /// <param name="initialSize">初始预创建数量</param>
    public ObjectPool(
        Func<T> createFunc,
        Action<T> onGet = null,
        Action<T> onReturn = null,
        Action<T> onDestroy = null,
        int maxSize = 1000,
        int initialSize = 0)
    {
        this.createFunc = createFunc ?? throw new ArgumentNullException(nameof(createFunc));
        this.onGet = onGet;
        this.onReturn = onReturn;
        this.onDestroy = onDestroy;
        this.maxSize = maxSize;

        pool = new Stack<T>(Math.Min(initialSize, maxSize));

        // 预创建对象
        for (int i = 0; i < initialSize; i++)
        {
            pool.Push(createFunc());
        }
    }

    /// <summary>
    /// 从池中获取对象
    /// </summary>
    public T Get()
    {
        T item = pool.Count > 0 ? pool.Pop() : createFunc();
        onGet?.Invoke(item);
        return item;
    }

    /// <summary>
    /// 归还对象到池中
    /// </summary>
    public void Return(T item)
    {
        if (item == null) return;

        onReturn?.Invoke(item);

        if (pool.Count < maxSize)
        {
            pool.Push(item);
        }
        else
        {
            onDestroy?.Invoke(item);
        }
    }

    /// <summary>
    /// 清空池并销毁所有对象
    /// </summary>
    public void Clear()
    {
        if (onDestroy != null)
        {
            foreach (var item in pool)
            {
                onDestroy(item);
            }
        }
        pool.Clear();
    }
}
```

### GameObject 专用池

```csharp
using UnityEngine;
using System.Collections.Generic;

/// <summary>
/// GameObject 专用对象池 - 管理Unity游戏对象
/// </summary>
public class GameObjectPool
{
    private readonly Stack<GameObject> pool;
    private readonly GameObject prefab;
    private readonly Transform parent;
    private readonly int maxSize;

    public int CountInactive => pool.Count;

    /// <summary>
    /// 创建GameObject池
    /// </summary>
    /// <param name="prefab">预制体</param>
    /// <param name="parent">池对象的父物体（用于整理层级）</param>
    /// <param name="maxSize">最大容量</param>
    /// <param name="initialSize">初始预创建数量</param>
    public GameObjectPool(
        GameObject prefab,
        Transform parent = null,
        int maxSize = 100,
        int initialSize = 0)
    {
        this.prefab = prefab;
        this.parent = parent;
        this.maxSize = maxSize;
        pool = new Stack<GameObject>(maxSize);

        // 预创建对象
        for (int i = 0; i < initialSize; i++)
        {
            var go = CreateNew();
            go.SetActive(false);
            pool.Push(go);
        }
    }

    /// <summary>
    /// 获取GameObject
    /// </summary>
    public GameObject Get(Vector3 position = default, Quaternion rotation = default)
    {
        GameObject go;

        if (pool.Count > 0)
        {
            go = pool.Pop();
        }
        else
        {
            go = CreateNew();
        }

        go.transform.position = position;
        go.transform.rotation = rotation;
        go.SetActive(true);

        return go;
    }

    /// <summary>
    /// 获取GameObject并设置父物体
    /// </summary>
    public GameObject Get(Transform parent, bool worldPositionStays = false)
    {
        var go = Get();
        go.transform.SetParent(parent, worldPositionStays);
        return go;
    }

    /// <summary>
    /// 归还GameObject
    /// </summary>
    public void Return(GameObject go)
    {
        if (go == null) return;

        go.SetActive(false);

        if (parent != null)
        {
            go.transform.SetParent(parent);
        }

        if (pool.Count < maxSize)
        {
            pool.Push(go);
        }
        else
        {
            UnityEngine.Object.Destroy(go);
        }
    }

    /// <summary>
    /// 清空池
    /// </summary>
    public void Clear()
    {
        foreach (var go in pool)
        {
            if (go != null)
            {
                UnityEngine.Object.Destroy(go);
            }
        }
        pool.Clear();
    }

    private GameObject CreateNew()
    {
        var go = UnityEngine.Object.Instantiate(prefab, parent);
        go.name = prefab.name; // 移除 (Clone) 后缀
        return go;
    }
}
```

### 组件池（带IPoolable接口）

```csharp
using UnityEngine;
using System.Collections.Generic;

/// <summary>
/// 可池化对象接口
/// </summary>
public interface IPoolable
{
    void OnSpawn();
    void OnDespawn();
}

/// <summary>
/// 组件对象池 - 支持自动调用生命周期方法
/// </summary>
public class ComponentPool<T> where T : Component, IPoolable
{
    private readonly Stack<T> pool;
    private readonly T prefab;
    private readonly Transform parent;
    private readonly int maxSize;

    public int CountInactive => pool.Count;

    public ComponentPool(T prefab, Transform parent = null, int maxSize = 100, int initialSize = 0)
    {
        this.prefab = prefab;
        this.parent = parent;
        this.maxSize = maxSize;
        pool = new Stack<T>(maxSize);

        for (int i = 0; i < initialSize; i++)
        {
            var comp = CreateNew();
            comp.gameObject.SetActive(false);
            pool.Push(comp);
        }
    }

    public T Get()
    {
        T item;

        if (pool.Count > 0)
        {
            item = pool.Pop();
        }
        else
        {
            item = CreateNew();
        }

        item.gameObject.SetActive(true);
        item.OnSpawn();
        return item;
    }

    public void Return(T item)
    {
        if (item == null) return;

        item.OnDespawn();
        item.gameObject.SetActive(false);

        if (parent != null)
        {
            item.transform.SetParent(parent);
        }

        if (pool.Count < maxSize)
        {
            pool.Push(item);
        }
        else
        {
            Object.Destroy(item.gameObject);
        }
    }

    public void Clear()
    {
        foreach (var item in pool)
        {
            if (item != null)
            {
                Object.Destroy(item.gameObject);
            }
        }
        pool.Clear();
    }

    private T CreateNew()
    {
        var go = Object.Instantiate(prefab, parent);
        go.name = prefab.name;
        return go;
    }
}
```

## 使用示例

### 子弹池示例

```csharp
public class Bullet : MonoBehaviour, IPoolable
{
    private Rigidbody rb;

    private void Awake()
    {
        rb = GetComponent<Rigidbody>();
    }

    public void Launch(Vector3 direction, float speed)
    {
        rb.velocity = direction * speed;
    }

    void IPoolable.OnSpawn()
    {
        // 重置状态
        rb.velocity = Vector3.zero;
    }

    void IPoolable.OnDespawn()
    {
        // 清理状态
        rb.velocity = Vector3.zero;
    }

    private void OnTriggerEnter(Collider other)
    {
        // 归还到池
        BulletManager.Instance.ReturnBullet(this);
    }
}

// 管理类
public class BulletManager : MonoBehaviour
{
    public static BulletManager Instance { get; private set; }

    [SerializeField] private Bullet bulletPrefab;
    private ComponentPool<Bullet> bulletPool;

    private void Awake()
    {
        Instance = this;
        bulletPool = new ComponentPool<Bullet>(
            bulletPrefab,
            transform,
            maxSize: 200,
            initialSize: 50
        );
    }

    public Bullet SpawnBullet(Vector3 position, Vector3 direction, float speed)
    {
        var bullet = bulletPool.Get();
        bullet.transform.position = position;
        bullet.Launch(direction, speed);
        return bullet;
    }

    public void ReturnBullet(Bullet bullet)
    {
        bulletPool.Return(bullet);
    }
}
```

## 最佳实践

### DO ✅

- 预创建常用数量的对象（initialSize）
- 在OnDespawn中重置对象状态
- 为不同类型的对象使用独立的池
- 设置合理的maxSize防止内存泄漏

### DON'T ❌

- 不要归还null对象
- 不要在外部持有已归还对象的引用
- 不要在OnSpawn中执行耗时操作
- 不要忘记调用Return导致对象泄漏

## 注意事项

1. **线程安全**: 此实现非线程安全，Unity主线程使用
2. **内存泄漏**: 确保所有Get的对象最终都被Return
3. **状态重置**: 在onReturn/OnDespawn中完整重置状态
