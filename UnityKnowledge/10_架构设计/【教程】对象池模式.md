## 1. 为什么需要对象池？

### 问题场景

```csharp
// ❌ 没有对象池的情况
void SpawnBullet()
{
    // 每次都new，产生大量GC
    var bullet = Instantiate(bulletPrefab, position, rotation);

    // 子弹消失时销毁
    Destroy(bullet, 2f);  // 销毁也会产生GC
}

// 如果每秒发射10发子弹，1分钟就是600次实例化/销毁
// 这会导致：
// 1. 频繁的内存分配和回收
// 2. GC.Collect被触发，游戏卡顿
// 3. CPU性能浪费
```

### 对象池的核心思想

- **用完不销毁，放回池里**
- **需要时从池里取，不够再创建**
- **空间换时间**

```
┌─────────────────────────────────────┐
│           Object Pool               │
│  ┌───┐ ┌───┐ ┌───┐ ┌───┐ ┌───┐    │
│  │ ● │ │ ● │ │ ● │ │   │ │   │    │
│  └───┘ └───┘ └───┘ └───┘ └───┘    │
│   可用   可用   可用   空位   空位   │
└─────────────────────────────────────┘
         ↑ Get()        ↑ Return()
      取出使用         放回池中
```

## 2. 通用对象池实现

```csharp
/// <summary>
/// 通用对象池接口
/// </summary>
public interface IObjectPool<T>
{
    T Get();
    void Return(T obj);
    void Clear();
    int Count { get; }
    int ActiveCount { get; }
}

/// <summary>
/// 通用对象池实现
/// </summary>
public class ObjectPool<T> : IObjectPool<T> where T : class
{
    private readonly Stack<T> pool;
    private readonly Func<T> createFunc;
    private readonly Action<T> onGet;
    private readonly Action<T> onReturn;
    private readonly Action<T> onDestroy;
    private readonly int maxSize;
    private int activeCount;

    /// <summary>
    /// 对象池构造函数
    /// </summary>
    /// <param name="createFunc">创建新对象的函数</param>
    /// <param name="onGet">从池中取出时的回调</param>
    /// <param name="onReturn">放回池中时的回调</param>
    /// <param name="onDestroy">销毁时的回调</param>
    /// <param name="maxSize">池最大容量</param>
    /// <param name="initialSize">初始容量</param>
    public ObjectPool(
        Func<T> createFunc,
        Action<T> onGet = null,
        Action<T> onReturn = null,
        Action<T> onDestroy = null,
        int maxSize = 100,
        int initialSize = 0)
    {
        this.createFunc = createFunc ?? throw new ArgumentNullException(nameof(createFunc));
        this.onGet = onGet;
        this.onReturn = onReturn;
        this.onDestroy = onDestroy;
        this.maxSize = maxSize;
        this.pool = new Stack<T>(maxSize);
        this.activeCount = 0;

        // 预热：预先创建对象
        for (int i = 0; i < initialSize; i++)
        {
            pool.Push(createFunc());
        }
    }

    public T Get()
    {
        T obj;

        if (pool.Count > 0)
        {
            obj = pool.Pop();
        }
        else
        {
            obj = createFunc();
        }

        activeCount++;
        onGet?.Invoke(obj);
        return obj;
    }

    public void Return(T obj)
    {
        if (obj == null) return;

        activeCount--;

        if (pool.Count < maxSize)
        {
            onReturn?.Invoke(obj);
            pool.Push(obj);
        }
        else
        {
            // 池已满，销毁对象
            onDestroy?.Invoke(obj);
        }
    }

    public void Clear()
    {
        while (pool.Count > 0)
        {
            var obj = pool.Pop();
            onDestroy?.Invoke(obj);
        }
        activeCount = 0;
    }

    public int Count => pool.Count;
    public int ActiveCount => activeCount;
}
```

## 3. Unity GameObject专用对象池

```csharp
/// <summary>
/// Unity GameObject对象池
/// </summary>
public class GameObjectPool : IObjectPool<GameObject>
{
    private readonly ObjectPool<GameObject> innerPool;
    private readonly GameObject prefab;
    private readonly Transform poolParent;

    public GameObjectPool(
        GameObject prefab,
        Transform poolParent = null,
        int maxSize = 100,
        int initialSize = 0)
    {
        this.prefab = prefab;
        this.poolParent = poolParent;

        innerPool = new ObjectPool<GameObject>(
            createFunc: CreateObject,
            onGet: OnGetObject,
            onReturn: OnReturnObject,
            onDestroy: OnDestroyObject,
            maxSize: maxSize,
            initialSize: initialSize
        );
    }

    private GameObject CreateObject()
    {
        var obj = Object.Instantiate(prefab);
        obj.SetActive(false);
        return obj;
    }

    private void OnGetObject(GameObject obj)
    {
        obj.SetActive(true);
    }

    private void OnReturnObject(GameObject obj)
    {
        obj.SetActive(false);
        if (poolParent != null)
        {
            obj.transform.SetParent(poolParent);
        }
    }

    private void OnDestroyObject(GameObject obj)
    {
        Object.Destroy(obj);
    }

    public GameObject Get() => innerPool.Get();
    public GameObject Get(Vector3 position, Quaternion rotation)
    {
        var obj = Get();
        obj.transform.SetPositionAndRotation(position, rotation);
        return obj;
    }

    public void Return(GameObject obj) => innerPool.Return(obj);
    public void Clear() => innerPool.Clear();
    public int Count => innerPool.Count;
    public int ActiveCount => innerPool.ActiveCount;
}

// 使用示例
public class BulletSpawner : MonoBehaviour
{
    [SerializeField] private GameObject bulletPrefab;
    private GameObjectPool bulletPool;

    void Start()
    {
        bulletPool = new GameObjectPool(
            prefab: bulletPrefab,
            poolParent: transform,
            maxSize: 50,
            initialSize: 10
        );
    }

    void Fire(Vector3 position, Vector3 direction)
    {
        var bullet = bulletPool.Get(position, Quaternion.LookRotation(direction));
        var bulletScript = bullet.GetComponent<Bullet>();
        bulletScript.Initialize(direction, () => bulletPool.Return(bullet));
    }
}

// 子弹脚本
public class Bullet : MonoBehaviour
{
    private Action onHit;

    public void Initialize(Vector3 direction, Action onHitCallback)
    {
        this.onHit = onHitCallback;
        // 设置速度等...
    }

    void OnTriggerEnter(Collider other)
    {
        // 处理碰撞...
        onHit?.Invoke();  // 放回池中
    }
}
```

## 4. 对象池管理器（统一管理多个池）

```csharp
/// <summary>
/// 对象池管理器 - 管理多种类型的对象池
/// </summary>
public class PoolManager : MonoSingleton<PoolManager>
{
    private Dictionary<string, GameObjectPool> pools;
    private Transform poolContainer;

    protected override void Awake()
    {
        base.Awake();
        pools = new Dictionary<string, GameObjectPool>();
        poolContainer = new GameObject("[PoolContainer]").transform;
        poolContainer.SetParent(transform);
    }

    /// <summary>
    /// 创建或获取对象池
    /// </summary>
    public GameObjectPool GetOrCreatePool(
        string poolName,
        GameObject prefab,
        int maxSize = 100,
        int initialSize = 0)
    {
        if (pools.TryGetValue(poolName, out var pool))
        {
            return pool;
        }

        var poolParent = new GameObject($"[Pool_{poolName}]").transform;
        poolParent.SetParent(poolContainer);

        pool = new GameObjectPool(prefab, poolParent, maxSize, initialSize);
        pools[poolName] = pool;

        return pool;
    }

    /// <summary>
    /// 从指定池中获取对象
    /// </summary>
    public GameObject Get(string poolName)
    {
        if (pools.TryGetValue(poolName, out var pool))
        {
            return pool.Get();
        }
        Debug.LogError($"Pool not found: {poolName}");
        return null;
    }

    /// <summary>
    /// 将对象放回池中
    /// </summary>
    public void Return(string poolName, GameObject obj)
    {
        if (pools.TryGetValue(poolName, out var pool))
        {
            pool.Return(obj);
        }
        else
        {
            Debug.LogWarning($"Pool not found: {poolName}, destroying object");
            Destroy(obj);
        }
    }

    /// <summary>
    /// 清空所有池
    /// </summary>
    public void ClearAll()
    {
        foreach (var pool in pools.Values)
        {
            pool.Clear();
        }
        pools.Clear();
    }

    /// <summary>
    /// 预热指定池
    /// </summary>
    public void WarmUp(string poolName, int count)
    {
        if (pools.TryGetValue(poolName, out var pool))
        {
            var temp = new List<GameObject>(count);
            for (int i = 0; i < count; i++)
            {
                temp.Add(pool.Get());
            }
            foreach (var obj in temp)
            {
                pool.Return(obj);
            }
        }
    }
}

// 使用示例
public class GameInitializer : MonoBehaviour
{
    [SerializeField] private GameObject gemPrefab;
    [SerializeField] private GameObject effectPrefab;

    void Start()
    {
        // 初始化对象池
        PoolManager.Instance.GetOrCreatePool("Gem", gemPrefab, 100, 50);
        PoolManager.Instance.GetOrCreatePool("Effect", effectPrefab, 30, 10);
    }
}
```

## 5. 池化策略与内存管理

### 什么时候使用对象池？

| 场景 | 是否使用对象池 | 原因 |
|------|---------------|------|
| 子弹、特效 | ✅ 是 | 高频创建/销毁 |
| 敌人、NPC | ✅ 是 | 频繁生成/死亡 |
| UI弹窗 | ⚠️ 可选 | 看频率决定 |
| 过场动画物体 | ❌ 否 | 一次性使用 |
| 配置数据 | ❌ 否 | 不是GameObject |

### 池大小设置策略

```csharp
/// <summary>
/// 动态调整池大小
/// </summary>
public class DynamicPool : GameObjectPool
{
    private int peakUsage;
    private float lastAdjustTime;
    private const float ADJUST_INTERVAL = 10f; // 每10秒检查一次

    public DynamicPool(GameObject prefab, Transform poolParent, int maxSize, int initialSize)
        : base(prefab, poolParent, maxSize, initialSize) { }

    public new void Return(GameObject obj)
    {
        base.Return(obj);
        peakUsage = Mathf.Max(peakUsage, ActiveCount);

        // 定期调整
        if (Time.time - lastAdjustTime > ADJUST_INTERVAL)
        {
            AdjustPoolSize();
            lastAdjustTime = Time.time;
        }
    }

    private void AdjustPoolSize()
    {
        // 如果峰值使用量远小于当前池大小，可以释放一些对象
        // 如果峰值接近最大容量，考虑增加容量
        Debug.Log($"Pool stats - Peak: {peakUsage}, Current: {Count}, Active: {ActiveCount}");
        peakUsage = 0; // 重置峰值
    }
}
```

### 内存泄漏检查

```csharp
/// <summary>
/// 带泄漏检测的对象池
/// </summary>
public class LeakDetectingPool<T> : ObjectPool<T> where T : class
{
    private readonly HashSet<T> activeObjects = new();
    private readonly string poolName;

    public LeakDetectingPool(
        string poolName,
        Func<T> createFunc,
        Action<T> onGet = null,
        Action<T> onReturn = null,
        Action<T> onDestroy = null,
        int maxSize = 100,
        int initialSize = 0)
        : base(createFunc, onGet, onReturn, onDestroy, maxSize, initialSize)
    {
        this.poolName = poolName;
    }

    public new T Get()
    {
        var obj = base.Get();
        activeObjects.Add(obj);
        return obj;
    }

    public new void Return(T obj)
    {
        if (!activeObjects.Remove(obj))
        {
            Debug.LogWarning($"[{poolName}] Attempting to return object that wasn't from this pool or already returned");
        }
        base.Return(obj);
    }

    public void CheckLeaks()
    {
        if (activeObjects.Count > 0)
        {
            Debug.LogWarning($"[{poolName}] Potential leak: {activeObjects.Count} objects still active");
        }
    }
}
```

## 6. 三消游戏实战：棋子池化

```csharp
/// <summary>
/// 棋子类型
/// </summary>
public enum GemType
{
    Hero,
    Skill,
    Empty
}

/// <summary>
/// 棋子数据
/// </summary>
public class Gem : MonoBehaviour
{
    public GemType Type { get; private set; }
    public int Id { get; private set; }
    public Vector2Int Position { get; set; }

    private SpriteRenderer spriteRenderer;
    private Action<Gem> returnToPool;

    public void Initialize(GemType type, int id, Sprite sprite, Action<Gem> returnCallback)
    {
        Type = type;
        Id = id;
        returnToPool = returnCallback;

        if (spriteRenderer == null)
            spriteRenderer = GetComponent<SpriteRenderer>();
        spriteRenderer.sprite = sprite;
    }

    public void SetPosition(Vector2Int pos)
    {
        Position = pos;
        transform.localPosition = new Vector3(pos.x, pos.y, 0);
    }

    public void Return()
    {
        returnToPool?.Invoke(this);
    }
}

/// <summary>
/// 棋子对象池管理器
/// </summary>
public class GemPoolManager : MonoSingleton<GemPoolManager>
{
    [System.Serializable]
    public class GemConfig
    {
        public int id;
        public GemType type;
        public Sprite sprite;
    }

    [SerializeField] private GemConfig[] gemConfigs;
    [SerializeField] private GameObject gemPrefab;

    private Dictionary<int, GameObjectPool> gemPools;
    private Transform poolContainer;

    protected override void Awake()
    {
        base.Awake();
        InitializePools();
    }

    private void InitializePools()
    {
        gemPools = new Dictionary<int, GameObjectPool>();
        poolContainer = new GameObject("[GemPools]").transform;
        poolContainer.SetParent(transform);

        foreach (var config in gemConfigs)
        {
            // 为每种棋子创建预制体实例作为模板
            var template = Instantiate(gemPrefab);
            var sr = template.GetComponent<SpriteRenderer>();
            sr.sprite = config.sprite;
            template.SetActive(false);

            var poolParent = new GameObject($"[Pool_Gem_{config.id}]").transform;
            poolParent.SetParent(poolContainer);

            var pool = new GameObjectPool(
                prefab: template,
                poolParent: poolParent,
                maxSize: 100,
                initialSize: 20
            );

            gemPools[config.id] = pool;
            Destroy(template); // 销毁模板，池会自己创建实例
        }
    }

    public Gem GetGem(int gemId, Vector2Int position)
    {
        if (!gemPools.TryGetValue(gemId, out var pool))
        {
            Debug.LogError($"Gem pool not found: {gemId}");
            return null;
        }

        var go = pool.Get();
        go.transform.localPosition = new Vector3(position.x, position.y, 0);

        var gem = go.GetComponent<Gem>();
        var config = System.Array.Find(gemConfigs, c => c.id == gemId);
        gem.Initialize(config.type, config.id, config.sprite, (g) => ReturnGem(gemId, g));
        gem.SetPosition(position);

        return gem;
    }

    public void ReturnGem(int gemId, Gem gem)
    {
        if (gemPools.TryGetValue(gemId, out var pool))
        {
            pool.Return(gem.gameObject);
        }
    }

    public void ReturnAllGems()
    {
        // 找到场景中所有激活的Gem并放回池中
        var gems = FindObjectsOfType<Gem>();
        foreach (var gem in gems)
        {
            gem.Return();
        }
    }
}

/// <summary>
/// 棋盘使用示例
/// </summary>
public class Board : MonoBehaviour
{
    private Gem[,] grid;
    private int width = 7;
    private int height = 7;

    public void Initialize(int w, int h)
    {
        width = w;
        height = h;
        grid = new Gem[width, height];
        FillBoard();
    }

    public void FillBoard()
    {
        for (int x = 0; x < width; x++)
        {
            for (int y = 0; y < height; y++)
            {
                // 随机选择一个棋子类型
                int randomGemId = Random.Range(0, 5);
                var gem = GemPoolManager.Instance.GetGem(randomGemId, new Vector2Int(x, y));
                grid[x, y] = gem;
            }
        }
    }

    public void RemoveGem(int x, int y)
    {
        if (grid[x, y] != null)
        {
            grid[x, y].Return();
            grid[x, y] = null;
        }
    }

    public void ClearBoard()
    {
        for (int x = 0; x < width; x++)
        {
            for (int y = 0; y < height; y++)
            {
                if (grid[x, y] != null)
                {
                    grid[x, y].Return();
                    grid[x, y] = null;
                }
            }
        }
    }
}
```

## 本课小结

### 核心知识点

| 知识点 | 核心要点 |
|--------|----------|
| 对象池原理 | 用完不销毁，放回池中复用 |
| 通用实现 | 使用泛型 + Stack，支持回调 |
| GameObject池 | 处理Active状态、父子关系 |
| 池管理器 | 统一管理多个池，便于扩展 |
| 池化策略 | 高频创建/销毁的对象才需要池化 |
| 内存管理 | 动态调整大小、泄漏检测 |

### 性能对比

| 操作 | 无对象池 | 有对象池 | 提升 |
|------|----------|----------|------|
| 创建1000个对象 | ~15ms + GC | ~0.5ms | 30x |
| 销毁1000个对象 | ~10ms + GC | ~0.3ms | 33x |
| GC触发频率 | 频繁 | 极少 | 显著 |
