# 【最佳实践】组件化重构案例

> 核心价值：通过完整案例展示如何从继承架构重构为组件化架构

## 一、重构前代码（继承架构）

### 1.1 问题代码

```csharp
// ❌ 问题：深层继承，代码重复，难以扩展

/// <summary>
/// 角色基类
/// </summary>
public abstract class Character : MonoBehaviour
{
    public float health;
    public float maxHealth;
    public float moveSpeed;
    public float attackDamage;

    protected virtual void Move(Vector3 direction)
    {
        transform.position += direction * moveSpeed * Time.deltaTime;
    }

    protected virtual void TakeDamage(float damage)
    {
        health -= damage;
        if (health <= 0) Die();
    }

    protected virtual void Die()
    {
        Destroy(gameObject);
    }
}

/// <summary>
/// 战斗角色（中间层）
/// </summary>
public abstract class CombatCharacter : Character
{
    public float attackRange;
    public float attackCooldown;
    protected float lastAttackTime;

    public virtual void Attack(Character target)
    {
        if (Time.time - lastAttackTime < attackCooldown) return;
        target.TakeDamage(attackDamage);
        lastAttackTime = Time.time;
    }
}

/// <summary>
/// 玩家
/// </summary>
public class Player : CombatCharacter
{
    public float jumpForce;
    public int level;
    public float experience;

    private void Update()
    {
        // 移动
        float h = Input.GetAxis("Horizontal");
        float v = Input.GetAxis("Vertical");
        Move(new Vector3(h, 0, v));

        // 跳跃
        if (Input.GetKeyDown(KeyCode.Space))
        {
            Jump();
        }

        // 攻击
        if (Input.GetMouseButtonDown(0))
        {
            var enemy = FindEnemy();
            if (enemy != null && Vector3.Distance(transform.position, enemy.transform.position) < attackRange)
            {
                Attack(enemy);
            }
        }
    }

    private void Jump()
    {
        GetComponent<Rigidbody>().AddForce(Vector3.up * jumpForce, ForceMode.Impulse);
    }

    private Character FindEnemy()
    {
        var hits = Physics.OverlapSphere(transform.position, attackRange);
        foreach (var hit in hits)
        {
            var enemy = hit.GetComponent<Enemy>();
            if (enemy != null) return enemy;
        }
        return null;
    }

    protected override void Die()
    {
        // 玩家特殊死亡逻辑
        Debug.Log("Game Over");
    }
}

/// <summary>
/// 敌人
/// </summary>
public class Enemy : CombatCharacter
{
    public float detectionRange;
    public float chaseSpeed;
    private Transform player;

    private void Start()
    {
        player = GameObject.FindWithTag("Player").transform;
    }

    private void Update()
    {
        if (player == null) return;

        float distance = Vector3.Distance(transform.position, player.position);

        if (distance < detectionRange)
        {
            // 追击
            Vector3 direction = (player.position - transform.position).normalized;
            Move(direction * chaseSpeed / moveSpeed);

            // 攻击
            if (distance < attackRange)
            {
                var playerComponent = player.GetComponent<Player>();
                if (playerComponent != null)
                {
                    Attack(playerComponent);
                }
            }
        }
    }

    protected override void Die()
    {
        // 敌人特殊死亡逻辑
        Destroy(gameObject);
        // 掉落物品等
    }
}

/// <summary>
/// NPC（不需要战斗）
/// </summary>
public class NPC : Character
{
    public string dialogue;
    public float interactRange;

    // 问题：NPC不需要attackDamage，但继承了它
    // 问题：NPC不需要attackRange，但继承了它

    public void Interact()
    {
        // 显示对话
        Debug.Log(dialogue);
    }

    private void Update()
    {
        // 检测玩家靠近
        var player = GameObject.FindWithTag("Player");
        if (player != null && Vector3.Distance(transform.position, player.transform.position) < interactRange)
        {
            if (Input.GetKeyDown(KeyCode.E))
            {
                Interact();
            }
        }
    }
}
```

## 二、组件化重构

### 2.1 组件拆分

```csharp
// ✅ 重构后：职责分离的组件

/// <summary>
/// 移动组件
/// </summary>
public class MovementComponent : MonoBehaviour
{
    [SerializeField] private float moveSpeed = 5f;

    public void Move(Vector3 direction)
    {
        transform.position += direction * moveSpeed * Time.deltaTime;
    }

    public void MoveTowards(Vector3 target, float speed)
    {
        Vector3 direction = (target - transform.position).normalized;
        transform.position += direction * speed * Time.deltaTime;
    }
}

/// <summary>
/// 生命值组件
/// </summary>
public class HealthComponent : MonoBehaviour
{
    [SerializeField] private float maxHealth = 100f;

    public float CurrentHealth { get; private set; }
    public float MaxHealth => maxHealth;
    public bool IsDead => CurrentHealth <= 0;

    public event Action<float> OnDamageTaken;
    public event Action OnDeath;

    private void Awake()
    {
        CurrentHealth = maxHealth;
    }

    public void TakeDamage(float damage)
    {
        if (IsDead) return;

        CurrentHealth = Mathf.Max(0, CurrentHealth - damage);
        OnDamageTaken?.Invoke(damage);

        if (IsDead)
        {
            OnDeath?.Invoke();
        }
    }

    public void Heal(float amount)
    {
        CurrentHealth = Mathf.Min(maxHealth, CurrentHealth + amount);
    }
}

/// <summary>
/// 攻击组件
/// </summary>
public class AttackComponent : MonoBehaviour
{
    [SerializeField] private float damage = 10f;
    [SerializeField] private float range = 2f;
    [SerializeField] private float cooldown = 1f;

    private float lastAttackTime;

    public event Action<IDamageable> OnAttack;

    public bool CanAttack => Time.time - lastAttackTime >= cooldown;

    public bool TryAttack(IDamageable target)
    {
        if (!CanAttack) return false;

        float distance = Vector3.Distance(transform.position, target.Position);
        if (distance > range) return false;

        target.TakeDamage(damage);
        lastAttackTime = Time.time;
        OnAttack?.Invoke(target);
        return true;
    }
}

/// <summary>
/// 可受伤接口
/// </summary>
public interface IDamageable
{
    Vector3 Position { get; }
    void TakeDamage(float damage);
}

/// <summary>
/// 玩家输入组件
/// </summary>
public class PlayerInputComponent : MonoBehaviour
{
    public event Action<Vector3> OnMove;
    public event Action OnJump;
    public event Action OnAttack;
    public event Action OnInteract;

    private void Update()
    {
        // 移动输入
        Vector3 move = new Vector3(
            Input.GetAxis("Horizontal"),
            0,
            Input.GetAxis("Vertical")
        );
        if (move != Vector3.zero)
        {
            OnMove?.Invoke(move);
        }

        // 跳跃
        if (Input.GetKeyDown(KeyCode.Space))
        {
            OnJump?.Invoke();
        }

        // 攻击
        if (Input.GetMouseButtonDown(0))
        {
            OnAttack?.Invoke();
        }

        // 交互
        if (Input.GetKeyDown(KeyCode.E))
        {
            OnInteract?.Invoke();
        }
    }
}

/// <summary>
/// 跳跃组件
/// </summary>
[RequireComponent(typeof(Rigidbody))]
public class JumpComponent : MonoBehaviour
{
    [SerializeField] private float jumpForce = 5f;
    [SerializeField] private LayerMask groundLayer;

    private Rigidbody rb;
    private bool isGrounded;

    public bool IsGrounded => isGrounded;

    private void Awake()
    {
        rb = GetComponent<Rigidbody>();
    }

    public void Jump()
    {
        if (!isGrounded) return;
        rb.AddForce(Vector3.up * jumpForce, ForceMode.Impulse);
    }

    private void OnCollisionEnter(Collision collision)
    {
        if (((1 << collision.gameObject.layer) & groundLayer) != 0)
        {
            isGrounded = true;
        }
    }

    private void OnCollisionExit(Collision collision)
    {
        if (((1 << collision.gameObject.layer) & groundLayer) != 0)
        {
            isGrounded = false;
        }
    }
}

/// <summary>
/// AI行为组件
/// </summary>
public class AIComponent : MonoBehaviour
{
    [SerializeField] private float detectionRange = 10f;

    private Transform target;

    public event Action<Transform> OnTargetFound;
    public event Action OnTargetLost;

    public Transform Target => target;

    private void Update()
    {
        if (target == null)
        {
            FindTarget();
        }
        else
        {
            CheckTargetDistance();
        }
    }

    private void FindTarget()
    {
        var hits = Physics.OverlapSphere(transform.position, detectionRange);
        foreach (var hit in hits)
        {
            if (hit.CompareTag("Player"))
            {
                target = hit.transform;
                OnTargetFound?.Invoke(target);
                return;
            }
        }
    }

    private void CheckTargetDistance()
    {
        if (Vector3.Distance(transform.position, target.position) > detectionRange * 1.5f)
        {
            target = null;
            OnTargetLost?.Invoke();
        }
    }
}

/// <summary>
/// 交互组件
/// </summary>
public class InteractComponent : MonoBehaviour
{
    [SerializeField] private float interactRange = 2f;

    public event Action<IInteractable> OnInteract;

    public bool TryInteract()
    {
        var hits = Physics.OverlapSphere(transform.position, interactRange);
        foreach (var hit in hits)
        {
            var interactable = hit.GetComponent<IInteractable>();
            if (interactable != null)
            {
                interactable.Interact();
                OnInteract?.Invoke(interactable);
                return true;
            }
        }
        return false;
    }
}

/// <summary>
/// 可交互接口
/// </summary>
public interface IInteractable
{
    void Interact();
}
```

### 2.2 组装玩家

```csharp
/// <summary>
/// 玩家 - 只负责协调组件
/// </summary>
public class Player : MonoBehaviour, IDamageable
{
    private MovementComponent movement;
    private HealthComponent health;
    private AttackComponent attack;
    private PlayerInputComponent input;
    private JumpComponent jump;

    public Vector3 Position => transform.position;

    private void Awake()
    {
        // 获取组件引用
        movement = GetComponent<MovementComponent>();
        health = GetComponent<HealthComponent>();
        attack = GetComponent<AttackComponent>();
        input = GetComponent<PlayerInputComponent>();
        jump = GetComponent<JumpComponent>();

        // 订阅事件
        input.OnMove += OnMoveInput;
        input.OnJump += OnJumpInput;
        input.OnAttack += OnAttackInput;

        health.OnDeath += OnDeath;
    }

    private void OnDestroy()
    {
        // 取消订阅
        input.OnMove -= OnMoveInput;
        input.OnJump -= OnJumpInput;
        input.OnAttack -= OnAttackInput;

        health.OnDeath -= OnDeath;
    }

    private void OnMoveInput(Vector3 direction)
    {
        movement.Move(direction);
    }

    private void OnJumpInput()
    {
        jump.Jump();
    }

    private void OnAttackInput()
    {
        // 查找目标并攻击
        var hits = Physics.OverlapSphere(transform.position, 10f);
        foreach (var hit in hits)
        {
            var damageable = hit.GetComponent<IDamageable>();
            if (damageable != null && damageable != this)
            {
                attack.TryAttack(damageable);
                break;
            }
        }
    }

    private void OnDeath()
    {
        Debug.Log("Player Died");
        // 游戏结束逻辑
    }

    public void TakeDamage(float damage)
    {
        health.TakeDamage(damage);
    }
}
```

### 2.3 组装敌人

```csharp
/// <summary>
/// 敌人 - 只负责协调组件
/// </summary>
public class Enemy : MonoBehaviour, IDamageable
{
    private MovementComponent movement;
    private HealthComponent health;
    private AttackComponent attack;
    private AIComponent ai;

    [SerializeField] private float chaseSpeed = 3f;

    public Vector3 Position => transform.position;

    private void Awake()
    {
        movement = GetComponent<MovementComponent>();
        health = GetComponent<HealthComponent>();
        attack = GetComponent<AttackComponent>();
        ai = GetComponent<AIComponent>();

        // 订阅AI事件
        ai.OnTargetFound += OnTargetFound;
        ai.OnTargetLost += OnTargetLost;

        health.OnDeath += OnDeath;
    }

    private void OnDestroy()
    {
        ai.OnTargetFound -= OnTargetFound;
        ai.OnTargetLost -= OnTargetLost;
        health.OnDeath -= OnDeath;
    }

    private void OnTargetFound(Transform target)
    {
        // 开始追击
        StartCoroutine(ChaseTarget(target));
    }

    private void OnTargetLost()
    {
        StopAllCoroutines();
    }

    private System.Collections.IEnumerator ChaseTarget(Transform target)
    {
        while (target != null)
        {
            float distance = Vector3.Distance(transform.position, target.position);

            // 追击
            if (distance > attack.Range)
            {
                movement.MoveTowards(target.position, chaseSpeed);
            }
            // 攻击
            else
            {
                var damageable = target.GetComponent<IDamageable>();
                if (damageable != null)
                {
                    attack.TryAttack(damageable);
                }
            }

            yield return null;
        }
    }

    private void OnDeath()
    {
        Destroy(gameObject);
        // 掉落逻辑
    }

    public void TakeDamage(float damage)
    {
        health.TakeDamage(damage);
    }
}
```

## 三、重构前后对比

### 3.1 类图对比

```
重构前（继承）:
                    Character
                        │
            ┌─────────┴─────────┐
            │                   │
    CombatCharacter           NPC (不需要战斗但继承了)
            │
    ┌─────────┴─────────┐
    │                   │
  Player              Enemy


重构后（组件）:
┌─────────────────────────────────────────────────────────┐
│                   独立组件                              │
├─────────────────────────────────────────────────────────┤
│ MovementComponent   - 所有角色共用                     │
│ HealthComponent     - 所有角色共用                     │
│ AttackComponent     - Player/Enemy共用                 │
│ JumpComponent       - 仅Player使用                     │
│ AIComponent         - 仅Enemy使用                      │
│ InteractComponent   - Player/NPC共用                   │
│ PlayerInputComponent- 仅Player使用                     │
└─────────────────────────────────────────────────────────┘

Player = Movement + Health + Attack + Jump + Input + Interact

```

