## 三、序列化策略

### 3.1 序列化接口

```csharp
/// <summary>
/// 序列化器接口
/// </summary>
public interface ISerializer
{
    string Serialize<T>(T data);
    T Deserialize<T>(string data);
}
```

### 3.2 JSON序列化器（Unity原生）

```csharp
using UnityEngine;

/// <summary>
/// JSON序列化器 - 使用Unity的JsonUtility
/// </summary>
public class JsonNetSerializer : ISerializer
{
    public string Serialize<T>(T data)
    {
        return JsonUtility.ToJson(data, true);
    }

    public T Deserialize<T>(string data)
    {
        return JsonUtility.FromJson<T>(data);
    }
}
```

### 3.3 二进制序列化器（更安全）

```csharp
using System;
using System.IO;
using System.Runtime.Serialization.Formatters.Binary;

/// <summary>
/// 二进制序列化器
/// 注意：BinaryFormatter在部分平台已过时，建议使用其他方案
/// </summary>
public class BinarySerializer : ISerializer
{
    public string Serialize<T>(T data)
    {
        using (var ms = new MemoryStream())
        {
            var formatter = new BinaryFormatter();
            formatter.Serialize(ms, data);
            return Convert.ToBase64String(ms.ToArray());
        }
    }

    public T Deserialize<T>(string data)
    {
        var bytes = Convert.FromBase64String(data);
        using (var ms = new MemoryStream(bytes))
        {
            var formatter = new BinaryFormatter();
            return (T)formatter.Deserialize(ms);
        }
    }
}
```

### 4.2 文件存储（PC/主机）

```csharp
using System.IO;
using UnityEngine;

/// <summary>
/// 文件存储实现
/// </summary>
public class FileStorage : IStorage
{
    private readonly string saveDirectory;

    public FileStorage(string directory = "Saves")
    {
        saveDirectory = Path.Combine(Application.persistentDataPath, directory);
        Directory.CreateDirectory(saveDirectory);
    }

    public void Write(string key, string data)
    {
        var path = GetFilePath(key);
        File.WriteAllText(path, data);
    }

    public string Read(string key)
    {
        var path = GetFilePath(key);
        return File.Exists(path) ? File.ReadAllText(path) : null;
    }

    public void Delete(string key)
    {
        var path = GetFilePath(key);
        if (File.Exists(path))
        {
            File.Delete(path);
        }
    }

    public bool Exists(string key)
    {
        return File.Exists(GetFilePath(key));
    }

    public string[] GetSaveSlots()
    {
        var files = Directory.GetFiles(saveDirectory, "save_*.json");
        var slots = new string[files.Length];
        for (int i = 0; i < files.Length; i++)
        {
            slots[i] = Path.GetFileNameWithoutExtension(files[i]).Replace("save_", "");
        }
        return slots;
    }

    private string GetFilePath(string key)
    {
        return Path.Combine(saveDirectory, $"save_{key}.json");
    }
}
```

### 4.3 PlayerPrefs存储（简单场景）

```csharp
using UnityEngine;

/// <summary>
/// PlayerPrefs存储 - 适合小数据量

public class PlayerPrefsStorage : IStorage

    {
        PlayerPrefs.SetString(PREFIX + key, data);
        PlayerPrefs.Save();
    }

    public string Read(string key)
    {
        return PlayerPrefs.GetString(PREFIX + key, null);
    }

    public void Delete(string key)
    {
        PlayerPrefs.DeleteKey(PREFIX + key);
    }

    public bool Exists(string key)
    {
        return PlayerPrefs.HasKey(PREFIX + key);
    }

    public string[] GetSaveSlots()
    {
        // PlayerPrefs不支持枚举，需要额外管理
        return new string[0];
    }
}
```

## 五、存档管理器实现

### 5.1 完整实现

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 存档管理器
/// </summary>
public class SaveManager : MonoBehaviour
{
    public static SaveManager Instance { get; private set; }

    [Header("配置")]
    [SerializeField] private int maxSlots = 3;
    [SerializeField] private bool useEncryption = false;
    [SerializeField] private float autoSaveInterval = 300f;  // 5分钟

    private ISerializer serializer;
    private IStorage storage;
    private IEncryptor encryptor;

    private SaveData currentSave;
    private int currentSlot = -1;
    private float autoSaveTimer;

    public event Action<SaveData> OnSaveLoaded;
    public event Action OnSaveCompleted;
    public event Action OnSaveDeleted;

    private void Awake()
    {
        if (Instance != null && Instance != this)
        {
            Destroy(gameObject);
            return;
        }
        Instance = this;
        DontDestroyOnLoad(gameObject);

        // 初始化组件
        serializer = new JsonNetSerializer();
        storage = new FileStorage();
        encryptor = useEncryption ? new SimpleEncryptor() : null;
    }

    private void Update()
    {
        // 自动存档
        if (currentSlot >= 0 && autoSaveInterval > 0)
        {
            autoSaveTimer += Time.deltaTime;
            if (autoSaveTimer >= autoSaveInterval)
            {
                autoSaveTimer = 0;
                AutoSave();
            }
        }
    }

    /// <summary>
    /// 创建新存档
    /// </summary>
    public SaveData NewGame(int slot)
    {
        if (slot < 0 || slot >= maxSlots)
        {
            Debug.LogError($"无效的存档槽: {slot}");
            return null;
        }

        currentSave = new SaveData
        {
            slotIndex = slot,
            playTime = 0
        };
        currentSlot = slot;

        OnSaveLoaded?.Invoke(currentSave);
        return currentSave;
    }

    /// <summary>
    /// 保存游戏
    /// </summary>
    public bool Save(int? slot = null)
    {
        int targetSlot = slot ?? currentSlot;
        if (targetSlot < 0)
        {
            Debug.LogError("未指定存档槽");
            return false;
        }

        if (currentSave == null)
        {
            Debug.LogError("没有活动存档");
            return false;
        }

        try
        {
            // 更新数据
            currentSave.saveTime = DateTime.Now.ToString("yyyy-MM-dd HH:mm:ss");
            currentSave.slotIndex = targetSlot;

            // 收集当前游戏状态
            CollectGameData();

            // 序列化
            string data = serializer.Serialize(currentSave);

            // 加密
            if (encryptor != null)
            {
                data = encryptor.Encrypt(data);
            }

            // 存储
            storage.Write(targetSlot.ToString(), data);

            currentSlot = targetSlot;
            OnSaveCompleted?.Invoke();

            Debug.Log($"存档成功: 槽位 {targetSlot}");
            return true;
        }
        catch (Exception e)
        {
            Debug.LogError($"存档失败: {e.Message}");
            return false;
        }
    }

    /// <summary>
    /// 加载游戏
    /// </summary>
    public SaveData Load(int slot)
    {
        if (!storage.Exists(slot.ToString()))
        {
            Debug.LogError($"存档不存在: 槽位 {slot}");
            return null;
        }

        try
        {
            // 读取
            string data = storage.Read(slot.ToString());

            // 解密
            if (encryptor != null)
            {
                data = encryptor.Decrypt(data);
            }

            // 反序列化
            currentSave = serializer.Deserialize<SaveData>(data);
            currentSlot = slot;

            // 检查版本兼容性
            if (!CheckVersionCompatibility(currentSave.version))
            {
                currentSave = MigrateData(currentSave);
            }

            // 应用数据到游戏
            ApplyGameData();

            OnSaveLoaded?.Invoke(currentSave);
            Debug.Log($"读档成功: 槽位 {slot}");
            return currentSave;
        }
        catch (Exception e)
        {
            Debug.LogError($"读档失败: {e.Message}");
            return null;
        }
    }

    /// <summary>
    /// 删除存档
    /// </summary>
    public bool Delete(int slot)
    {
        if (!storage.Exists(slot.ToString()))
        {
            return false;
        }

        storage.Delete(slot.ToString());
        OnSaveDeleted?.Invoke();
        Debug.Log($"删除存档: 槽位 {slot}");
        return true;
    }

    /// <summary>
    /// 获取存档信息（用于UI显示）
    /// </summary>
    public SaveSlotInfo GetSlotInfo(int slot)
    {
        if (!storage.Exists(slot.ToString()))
        {
            return new SaveSlotInfo { exists = false, slotIndex = slot };
        }

        try
        {
            string data = storage.Read(slot.ToString());
            if (encryptor != null)
            {
                data = encryptor.Decrypt(data);
            }

            var save = serializer.Deserialize<SaveData>(data);
            return new SaveSlotInfo
            {
                exists = true,
                slotIndex = slot,
                saveTime = save.saveTime,
                playTime = save.playTime,
                level = save.player.level
            };
        }
        catch
        {
            return new SaveSlotInfo { exists = false, slotIndex = slot };
        }
    }

    /// <summary>
    /// 获取所有存档槽信息
    /// </summary>
    public SaveSlotInfo[] GetAllSlotInfo()
    {
        var slots = new SaveSlotInfo[maxSlots];
        for (int i = 0; i < maxSlots; i++)
        {
            slots[i] = GetSlotInfo(i);
        }
        return slots;
    }

    /// <summary>
    /// 自动存档
    /// </summary>
    private void AutoSave()
    {
        Save();
        Debug.Log("自动存档完成");
    }

    /// 收集游戏数据
    /// </summary>

    {

        // currentSave.player.position = Player.Instance.transform.position;
        // currentSave.playTime = GameManager.Instance.PlayTime;

    /// <summary>
    /// 应用游戏数据
    /// </summary>
    private void ApplyGameData()
    {
        // 将数据应用到游戏
        // Player.Instance.transform.position = currentSave.player.position;
        // GameManager.Instance.PlayTime = currentSave.playTime;

    /// <summary>
    /// 检查版本兼容性
    /// </summary>
    private bool CheckVersionCompatibility(int version)
    {
        return version == SaveData.CurrentVersion;
    }

    /// <summary>
    /// 迁移旧版本数据
```

## 六、加密实现

```csharp
using System;
using System.Text;
using System.Security.Cryptography;

/// <summary>
/// 加密接口
/// </summary>
public interface IEncryptor
{
    string Encrypt(string data);
    string Decrypt(string data);
}

/// <summary>
/// 简单加密（防修改）
/// </summary>
public class SimpleEncryptor : IEncryptor
{
    private readonly string key;
    private readonly byte[] keyBytes;

    public SimpleEncryptor(string key = "DefaultSecretKey123")
    {
        this.key = key;
        this.keyBytes = Encoding.UTF8.GetBytes(key);
    }

    public string Encrypt(string data)
    {
        var bytes = Encoding.UTF8.GetBytes(data);

        // XOR加密
        for (int i = 0; i < bytes.Length; i++)
        {
            bytes[i] ^= keyBytes[i % keyBytes.Length];
        }

        // 添加校验和
        var checksum = ComputeChecksum(bytes);
        var result = new byte[bytes.Length + 4];
        Buffer.BlockCopy(BitConverter.GetBytes(checksum), 0, result, 0, 4);
        Buffer.BlockCopy(bytes, 0, result, 4, bytes.Length);

        return Convert.ToBase64String(result);
    }

    public string Decrypt(string data)
    {
        var bytes = Convert.FromBase64String(data);

        // 提取校验和
        var storedChecksum = BitConverter.ToInt32(bytes, 0);
        var dataBytes = new byte[bytes.Length - 4];
        Buffer.BlockCopy(bytes, 4, dataBytes, 0, dataBytes.Length);

        // 验证校验和
        var actualChecksum = ComputeChecksum(dataBytes);
        if (storedChecksum != actualChecksum)
        {
            throw new Exception("存档校验失败，可能已被修改");
        }

        // XOR解密
        for (int i = 0; i < dataBytes.Length; i++)
        {
            dataBytes[i] ^= keyBytes[i % keyBytes.Length];
        }

        return Encoding.UTF8.GetString(dataBytes);
    }

    private int ComputeChecksum(byte[] data)
    {
        int sum = 0;
        foreach (var b in data)
        {
            sum = (sum * 31 + b) & 0x7FFFFFFF;
        }
        return sum;
    }
}
```
