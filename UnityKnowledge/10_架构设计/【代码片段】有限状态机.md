### 基础状态接口

```csharp
/// <summary>
/// 状态接口
/// </summary>
public interface IState
{
    void Enter();
    void Update();
    void FixedUpdate();
    void Exit();
}

/// <summary>
/// 带参数的状态接口
/// </summary>

{
    void Enter(TParam param);
    void Update();
    void FixedUpdate();

}
```

### 泛型状态机

```csharp
using System;
using System.Collections.Generic;

/// <summary>
/// 有限状态机 - 泛型实现
/// </summary>
/// <typeparam name="TState">状态标识类型（通常为enum）</typeparam>
public class FSM<TState> where TState : struct, Enum
{
    private readonly Dictionary<TState, IState> states;
    private readonly Dictionary<(TState, TState), Action> transitions;

    private TState currentStateKey;
    private IState currentState;

    public TState CurrentState => currentStateKey;
    public bool IsInState(TState state) => EqualityComparer<TState>.Default.Equals(currentStateKey, state);

    public FSM()
    {
        states = new Dictionary<TState, IState>();
        transitions = new Dictionary<(TState, TState), Action>();
    }

    /// <summary>
    /// 添加状态
    /// </summary>
    public FSM<TState> AddState(TState stateKey, IState state)
    {
        states[stateKey] = state;
        return this;
    }

    /// <summary>
    /// 添加状态转换回调
    /// </summary>
    public FSM<TState> AddTransition(TState from, TState to, Action onTransition)
    {
        transitions[(from, to)] = onTransition;
        return this;
    }

    /// <summary>
    /// 设置初始状态（不触发Enter）
    /// </summary>
    public void SetInitialState(TState initialState)
    {
        if (!states.TryGetValue(initialState, out var state))
        {
            throw new ArgumentException($"State {initialState} not found");
        }

        currentStateKey = initialState;
        currentState = state;
    }

    /// <summary>
    /// 切换状态
    /// </summary>
    public void ChangeState(TState newStateKey)
    {
        if (!states.TryGetValue(newStateKey, out var newState))
        {
            throw new ArgumentException($"State {newStateKey} not found");
        }

        // 相同状态不切换
        if (EqualityComparer<TState>.Default.Equals(currentStateKey, newStateKey))
        {
            return;
        }

        // 执行转换回调
        if (transitions.TryGetValue((currentStateKey, newStateKey), out var onTransition))
        {
            onTransition?.Invoke();
        }

        // 退出当前状态
        currentState?.Exit();

        // 切换状态
        currentStateKey = newStateKey;
        currentState = newState;

        // 进入新状态
        currentState?.Enter();
    }

    /// <summary>
    /// 更新当前状态
    /// </summary>
    public void Update()
    {
        currentState?.Update();
    }

    /// <summary>
    /// 固定更新当前状态
    /// </summary>
    public void FixedUpdate()
    {
        currentState?.FixedUpdate();
    }
}
```

### MonoBehaviour 状态机组件

```csharp
using UnityEngine;
using System;

/// <summary>
/// 可挂载的状态机组件
/// </summary>
public class FSMController : MonoBehaviour
{
    [SerializeField] private string initialState;

    private FSM<string> fsm;
    private bool isInitialized;

    public string CurrentState => fsm?.CurrentState;

    /// <summary>
    /// 初始化状态机
    /// </summary>
    public void Initialize(IEnumerable<KeyValuePair<string, IState>> states, string initial)
    {
        fsm = new FSM<string>();

        foreach (var kvp in states)
        {
            fsm.AddState(kvp.Key, kvp.Value);
        }

        fsm.SetInitialState(initial);
        isInitialized = true;
    }

    /// <summary>
    /// 添加状态
    /// </summary>
    public void AddState(string stateName, IState state)
    {
        fsm ??= new FSM<string>();
        fsm.AddState(stateName, state);
    }

    /// <summary>
    /// 切换状态
    /// </summary>
    public void ChangeState(string stateName)
    {
        if (!isInitialized)
        {
            fsm?.SetInitialState(stateName);
            isInitialized = true;
            return;
        }

        fsm.ChangeState(stateName);
    }

    private void Update()
    {
        fsm?.Update();
    }

    private void FixedUpdate()
    {
        fsm?.FixedUpdate();
    }
}
```

### 简化版状态机（无接口）

```csharp
using System;
using System.Collections.Generic;

/// <summary>
/// 轻量级状态机 - 使用Action委托
/// </summary>
public class LiteFSM<TState> where TState : struct, Enum
{
    private readonly Dictionary<TState, (Action enter, Action update, Action exit)> states;
    private TState currentState;
    private Action currentUpdate;

    public TState CurrentState => currentState;

    public LiteFSM()
    {
        states = new Dictionary<TState, (Action, Action, Action)>();
    }

    /// <summary>
    /// 添加状态
    /// </summary>
    public LiteFSM<TState> AddState(
        TState state,
        Action onEnter = null,
        Action onUpdate = null,
        Action onExit = null)
    {
        states[state] = (onEnter, onUpdate, onExit);
        return this;
    }

    /// <summary>
    /// 切换状态
    /// </summary>
    public void ChangeState(TState newState)
    {
        if (EqualityComparer<TState>.Default.Equals(currentState, newState))
            return;

        // 退出当前状态
        if (states.TryGetValue(currentState, out var oldState))
        {
            oldState.exit?.Invoke();
        }

        // 切换
        currentState = newState;

        // 进入新状态
        if (states.TryGetValue(newState, out var state))
        {
            state.enter?.Invoke();
            currentUpdate = state.update;
        }
        else
        {
            currentUpdate = null;
        }
    }

    /// <summary>
    /// 更新当前状态
    /// </summary>
    public void Update()
    {
        currentUpdate?.Invoke();
    }
}
```

## 使用示例

### 玩家控制器

```csharp
public enum PlayerState
{
    Idle,
    Run,
    Jump,
    Attack,
    Dead
}

public class PlayerController : MonoBehaviour
{
    [SerializeField] private float moveSpeed = 5f;
    [SerializeField] private float jumpForce = 10f;

    private FSM<PlayerState> fsm;
    private Rigidbody rb;
    private Animator animator;
    private bool isGrounded;

    private void Awake()
    {
        rb = GetComponent<Rigidbody>();
        animator = GetComponent<Animator>();

        InitializeFSM();
    }

    private void InitializeFSM()
    {
        fsm = new FSM<PlayerState>();

        fsm.AddState(PlayerState.Idle, new IdleState(this));
        fsm.AddState(PlayerState.Run, new RunState(this));
        fsm.AddState(PlayerState.Jump, new JumpState(this));
        fsm.AddState(PlayerState.Attack, new AttackState(this));
        fsm.AddState(PlayerState.Dead, new DeadState(this));

        fsm.SetInitialState(PlayerState.Idle);
    }

    private void Update()
    {
        fsm.Update();
        HandleInput();
    }

    private void FixedUpdate()
    {
        fsm.FixedUpdate();
    }

    private void HandleInput()
    {
        if (fsm.CurrentState == PlayerState.Dead) return;

        float h = Input.GetAxis("Horizontal");

        if (Input.GetButtonDown("Jump") && isGrounded)
        {
            fsm.ChangeState(PlayerState.Jump);
        }
        else if (Input.GetButtonDown("Fire1"))
        {
            fsm.ChangeState(PlayerState.Attack);
        }
        else if (Mathf.Abs(h) > 0.1f)
        {
            fsm.ChangeState(PlayerState.Run);
        }
        else if (fsm.CurrentState != PlayerState.Jump && fsm.CurrentState != PlayerState.Attack)
        {
            fsm.ChangeState(PlayerState.Idle);
        }
    }

    // 状态类可以访问这些方法
    public void Move(float direction)
    {
        rb.velocity = new Vector3(direction * moveSpeed, rb.velocity.y, 0);
    }

    public void Jump()
    {
        rb.velocity = new Vector3(rb.velocity.x, jumpForce, 0);
    }

    public void SetGrounded(bool grounded) => isGrounded = grounded;

    // 具体状态实现
    private class IdleState : IState
    {
        private readonly PlayerController player;

        public IdleState(PlayerController player) => this.player = player;

        public void Enter() => player.animator.SetBool("Idle", true);
        public void Update() { }
        public void FixedUpdate() { }
        public void Exit() => player.animator.SetBool("Idle", false);
    }

    private class RunState : IState
    {
        private readonly PlayerController player;

        public RunState(PlayerController player) => this.player = player;

        public void Enter() => player.animator.SetBool("Run", true);
        public void Update() { }
        public void FixedUpdate()
        {
            float h = Input.GetAxis("Horizontal");
            player.Move(h);
        }
        public void Exit() => player.animator.SetBool("Run", false);
    }

    private class JumpState : IState
    {
        private readonly PlayerController player;
        private bool hasJumped;

        public JumpState(PlayerController player) => this.player = player;

        public void Enter()
        {
            player.Jump();
            player.animator.SetTrigger("Jump");
            hasJumped = false;
        }
        public void Update()
        {
            if (player.isGrounded && hasJumped)
            {
                player.fsm.ChangeState(PlayerState.Idle);
            }
            hasJumped = true;
        }
        public void FixedUpdate() { }
        public void Exit() { }
    }

    private class AttackState : IState
    {
        private readonly PlayerController player;
        private float attackTimer;

        public AttackState(PlayerController player) => this.player = player;

        public void Enter()
        {
            player.animator.SetTrigger("Attack");
            attackTimer = 0.5f; // 攻击动画时长
        }
        public void Update()
        {
            attackTimer -= Time.deltaTime;
            if (attackTimer <= 0)
            {
                player.fsm.ChangeState(PlayerState.Idle);
            }
        }
        public void FixedUpdate() { }
        public void Exit() { }
    }

    private class DeadState : IState
    {
        private readonly PlayerController player;

        public DeadState(PlayerController player) => this.player = player;

        public void Enter() => player.animator.SetTrigger("Dead");
        public void Update() { }
        public void FixedUpdate() { }
        public void Exit() { }
    }
}
```

### 使用 LiteFSM 简化版

```csharp
public class EnemyAI : MonoBehaviour
{
    private enum State { Patrol, Chase, Attack }

    private LiteFSM<State> fsm;
    private Transform target;

    private void Awake()
    {
        fsm = new LiteFSM<State>()
            .AddState(State.Patrol, OnPatrolEnter, OnPatrolUpdate)
            .AddState(State.Chase, OnChaseEnter, OnChaseUpdate)
            .AddState(State.Attack, OnAttackEnter, OnAttackUpdate);

        fsm.ChangeState(State.Patrol);
    }

    private void Update()
    {
        fsm.Update();
    }

    private void OnPatrolEnter() { /* 开始巡逻 */ }
    private void OnPatrolUpdate()
    {
        if (Vector3.Distance(transform.position, target.position) < 5f)
        {
            fsm.ChangeState(State.Chase);
        }
    }

    private void OnChaseEnter() { /* 开始追击 */ }
    private void OnChaseUpdate()
    {
        float dist = Vector3.Distance(transform.position, target.position);
        if (dist < 1f)
        {
            fsm.ChangeState(State.Attack);
        }
        else if (dist > 10f)
        {
            fsm.ChangeState(State.Patrol);
        }
    }

    private void OnAttackEnter() { /* 开始攻击 */ }
    private void OnAttackUpdate()
    {
        if (Vector3.Distance(transform.position, target.position) > 1.5f)
        {
            fsm.ChangeState(State.Chase);
        }
    }
}
```

## 最佳实践

### DO ✅

- 使用 enum 作为状态标识，类型安全
- 每个状态类保持单一职责
- 在 Exit 中清理状态资源
- 使用转换回调处理跨状态逻辑

### DON'T ❌

- 不要在状态中直接修改其他状态
- 不要创建循环状态转换
- 不要在 Enter/Exit 中执行耗时操作
- 不要忘记调用 Update
