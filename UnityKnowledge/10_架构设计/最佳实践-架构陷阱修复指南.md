## 一、单例滥用修复

### 1.1 问题代码

```csharp
// ❌ 问题：什么都做成单例
public class PlayerSingleton : MonoBehaviour { ... }
public class EnemySingleton : MonoBehaviour { ... }
public class BulletSingleton : MonoBehaviour { ... }
public class UISingleton : MonoBehaviour { ... }
```

### 1.2 修复方案：服务定位器 + 接口

```csharp
// ✅ 修复：使用接口 + 服务注册

// 1. 定义接口
public interface IPlayerService
{
    Vector3 Position { get; }
    void TakeDamage(float damage);
}

public interface IEnemyService
{
    void Spawn(Vector3 position);
}

// 2. 实现类（普通类，不是单例）
public class PlayerService : IPlayerService
{
    private readonly Player player;

    public PlayerService(Player player)
    {
        this.player = player;
    }

    public Vector3 Position => player.transform.position;
    public void TakeDamage(float damage) => player.TakeDamage(damage);
}

// 3. 服务定位器
public static class Services
{
    private static readonly Dictionary<Type, object> services = new();

    public static void Register<T>(T service)
    {
        services[typeof(T)] = service;
    }

    public static T Get<T>()
    {
        return (T)services[typeof(T)];
    }

    public static void Clear()
    {
        services.Clear();
    }
}

// 4. 游戏初始化时注册
public class GameInitializer : MonoBehaviour
{
    [SerializeField] private Player player;

    private void Awake()
    {
        // 注册服务
        Services.Register<IPlayerService>(new PlayerService(player));
        Services.Register<IEnemyService>(new EnemyService());
    }

    private void OnDestroy()
    {
        Services.Clear();
    }
}

// 5. 使用
public class EnemyAI : MonoBehaviour
{
    private void Update()
    {
        var player = Services.Get<IPlayerService>();
        transform.LookAt(player.Position);
    }
}
```

### 1.3 测试对比

## 二、事件风暴修复

### 2.1 问题代码

```csharp
// ❌ 问题：事件触发事件，形成链式反应
public void OnPlayerDeath(PlayerDeathEvent e)
{
    EventBus.Publish(new ScoreChangedEvent());  // 事件1
}

public void OnScoreChanged(ScoreChangedEvent e)
{
    EventBus.Publish(new AchievementCheckEvent());  // 事件2

public void OnAchievementCheck(AchievementCheckEvent e)
```

### 2.2 修复方案：直接调用

```csharp
// ✅ 修复：直接调用，流程清晰

public class PlayerDeathHandler : MonoBehaviour
{
    private ScoreManager scoreManager;
    private AchievementSystem achievementSystem;
    private SaveManager saveManager;

    public void OnPlayerDeath(PlayerDeathEvent e)
    {
        // 直接调用，顺序清晰
        scoreManager.AddDeathScore(e.PlayerId);
        achievementSystem.CheckDeathAchievements(e.PlayerId);
        saveManager.Save();

        // 只在需要通知UI时用事件
        EventBus.Publish(new UIRefreshEvent());
    }
}

// 或者使用命令模式
public class PlayerDeathCommand : ICommand
{
    private readonly ScoreManager scoreManager;
    private readonly AchievementSystem achievementSystem;
    private readonly SaveManager saveManager;
    private readonly PlayerDeathEvent eventData;

    public void Execute()
    {
        scoreManager.AddDeathScore(eventData.PlayerId);
        achievementSystem.CheckDeathAchievements(eventData.PlayerId);
        saveManager.Save();
    }
}
```

## 三、Update泛滥修复

### 3.1 问题代码

```csharp
// ❌ 问题：每个物体都有Update
public class Bullet : MonoBehaviour
{
    private void Update()
    {
        transform.position += direction * speed * Time.deltaTime;
    }
}

// 1000个子弹 = 1000个Update调用
```

### 3.2 修复方案：管理器模式

```csharp
// ✅ 修复：统一管理

// 1. 子弹数据结构（纯数据）
public struct BulletData
{
    public Vector3 position;
    public Vector3 direction;
    public float speed;
    public bool isActive;
}

// 2. 子弹管理器（单个Update）
public class BulletManager : MonoBehaviour
{
    private BulletData[] bullets;
    private int activeCount;
    private const int MAX_BULLETS = 1000;

    private void Awake()
    {
        bullets = new BulletData[MAX_BULLETS];
    }

    public int Spawn(Vector3 position, Vector3 direction, float speed)
    {
        for (int i = 0; i < MAX_BULLETS; i++)
        {
            if (!bullets[i].isActive)
            {
                bullets[i] = new BulletData
                {
                    position = position,
                    direction = direction,
                    speed = speed,
                    isActive = true
                };
                return i;
            }
        }
        return -1;
    }

    private void Update()
    {
        // 一个Update更新所有子弹
        float dt = Time.deltaTime;
        for (int i = 0; i < MAX_BULLETS; i++)
        {
            if (bullets[i].isActive)
            {
                bullets[i].position += bullets[i].direction * bullets[i].speed * dt;

                // 边界检查
                if (IsOutOfBounds(bullets[i].position))
                {
                    bullets[i].isActive = false;
                }
            }
        }
    }

    private bool IsOutOfBounds(Vector3 pos)
    {
        return pos.magnitude > 100f;
    }
}

// 3. 视图层（只负责渲染）
public class BulletView : MonoBehaviour
{
    private int bulletIndex;
    private BulletManager manager;

    public void Initialize(BulletManager manager, int index)
    {
        this.manager = manager;
        this.bulletIndex = index;
    }

    private void LateUpdate()
    {
        // 只负责视觉更新
        // transform.position = manager.GetPosition(bulletIndex);
    }
}
```

### 4.2 修复方案：缓存引用

```csharp
// ✅ 修复方案1：Start时缓存
public class Enemy : MonoBehaviour
{
    private Transform player;

    private void Start()
    {
        player = GameObject.FindWithTag("Player")?.transform;
    }

    private void Update()
    {
        if (player != null)
        {
            transform.LookAt(player);
        }
    }
}

// ✅ 修复方案2：依赖注入
public class Enemy : MonoBehaviour
{
    private readonly IPlayerService playerService;

    // 通过构造函数或属性注入
    public void Initialize(IPlayerService service)
    {
        playerService = service;
    }

    private void Update()
    {
        if (playerService != null)
        {
            transform.LookAt(playerService.Position);
        }
    }
}

// ✅ 修复方案3：事件注册
public class Enemy : MonoBehaviour
{
    private Transform player;

    private void OnEnable()
    {
        EventBus.Subscribe<PlayerSpawnedEvent>(OnPlayerSpawned);
    }

    private void OnDisable()
    {
        EventBus.Unsubscribe<PlayerSpawnedEvent>(OnPlayerSpawned);
    }

    private void OnPlayerSpawned(PlayerSpawnedEvent e)
    {
        player = e.PlayerTransform;
    }

    private void Update()
    {
        if (player != null)
        {
            transform.LookAt(player);
        }
    }
}
```

```csharp
public enum AnimationType
```

## 六、协程嵌套地狱修复

### 6.1 问题代码

```csharp
// ❌ 问题：协程嵌套，难以控制和调试
private IEnumerator AttackSequence()
{
    yield return StartCoroutine(MoveToTarget());
    yield return StartCoroutine(PlayAttackAnim());
    yield return StartCoroutine(DealDamage());
    yield return StartCoroutine(PlayEffect());
    yield return StartCoroutine(ReturnToPosition());
}
```

### 6.2 修复方案：async/await (UniTask)

```csharp
// ✅ 修复方案：使用UniTask
using Cysharp.Threading.Tasks;

public class AttackSequence : MonoBehaviour
{
    private CancellationTokenSource cts;

    private void OnDestroy()
    {
        cts?.Cancel();
    }

    public async UniTask ExecuteAttackSequence()
    {
        cts = new CancellationTokenSource();

        try
        {
            await MoveToTarget(cts.Token);
            await PlayAttackAnim(cts.Token);
            await DealDamage(cts.Token);
            await PlayEffect(cts.Token);
            await ReturnToPosition(cts.Token);
        }
        catch (OperationCanceledException)
        {
            Debug.Log("攻击序列被取消");
        }
    }

    private async UniTask MoveToTarget(CancellationToken ct)
    {
        while (Vector3.Distance(transform.position, target) > 0.1f)
        {
            ct.ThrowIfCancellationRequested();
            transform.position = Vector3.MoveTowards(transform.position, target, speed * Time.deltaTime);
            await UniTask.Yield(ct);
        }
    }

    // 取消攻击
    public void CancelAttack()
    {
        cts?.Cancel();
    }
}
```

### 6.3 修复方案：状态机

```csharp
// ✅ 修复方案：使用状态机
public enum AttackState
{
    Idle,
    MovingToTarget,
    Attacking,
    PlayingEffect,
    Returning
}

public class AttackStateMachine : MonoBehaviour
{
    private AttackState currentState;
    private Vector3 originalPosition;
    private Transform target;

    private void Update()
    {
        switch (currentState)
        {
            case AttackState.Idle:
                // 等待指令
                break;

            case AttackState.MovingToTarget:
                UpdateMovingToTarget();
                break;

            case AttackState.Attacking:
                UpdateAttacking();
                break;

            case AttackState.PlayingEffect:
                UpdatePlayingEffect();
                break;

            case AttackState.Returning:
                UpdateReturning();
                break;
        }
    }

    public void StartAttack(Transform target)
    {
        this.target = target;
        originalPosition = transform.position;
        currentState = AttackState.MovingToTarget;
    }

    public void CancelAttack()
    {
        currentState = AttackState.Idle;
        transform.position = originalPosition;
    }

    private void UpdateMovingToTarget()
    {
        transform.position = Vector3.MoveTowards(
            transform.position,
            target.position,
            speed * Time.deltaTime
        );

        if (Vector3.Distance(transform.position, target.position) < 1f)
        {
            currentState = AttackState.Attacking;
        }
    }

    // ... 其他状态更新
}
```

## 七、过度抽象修复

### 7.1 问题代码

```csharp
// ❌ 问题：为了"未来可能"而过度抽象
public interface IEntity { int Id { get; } }
public interface IMovingEntity : IEntity { void Move(Vector3 dir); }
public interface IAttackingEntity : IMovingEntity { void Attack(); }
public interface IDamageableEntity : IEntity { void TakeDamage(float dmg); }

// 只有一个实现！
public class Player : MonoBehaviour, IAttackingEntity, IDamageableEntity
{
    // ...
}
```

### 7.2 修复方案：YAGNI原则

```csharp
// ✅ 修复：需要时再抽象

// 阶段1：只有一个Player时，直接写
public class Player : MonoBehaviour
{
    public void Move(Vector3 direction) { /* ... */ }
    public void Attack() { /* ... */ }
    public void TakeDamage(float damage) { /* ... */ }
}

// 阶段2：有Enemy时，提取共同接口
public interface IDamageable
{
    void TakeDamage(float damage);
    bool IsDead { get; }
}

public class Player : MonoBehaviour, IDamageable { /* ... */ }
public class Enemy : MonoBehaviour, IDamageable { /* ... */ }

// 阶段3：需要更多类型时，再扩展
public interface ICombatEntity : IDamageable
{
    void Attack(IDamageable target);
    float AttackRange { get; }
}
```

## 八、Inspector公开所有修复

### 8.1 问题代码

```csharp
// ❌ 问题：所有字段public
public class Player : MonoBehaviour
{
    public float health;
    public float maxHealth;
    public float speed;
    public List<Item> inventory;
    // 任何地方都能修改！
}
```

### 8.2 修复方案：SerializeField + 属性

```csharp
// ✅ 修复：封装 + Inspector可见
public class Player : MonoBehaviour
{
    [Header("属性")]
    [SerializeField] private float maxHealth = 100f;
    [SerializeField] private float speed = 5f;
    [SerializeField] private int inventorySize = 20;

    private float currentHealth;
    private List<Item> inventory;

    // 只读属性
    public float Health => currentHealth;
    public float MaxHealth => maxHealth;
    public bool IsDead => currentHealth <= 0;

    // 方法控制修改
    public void TakeDamage(float damage)
    {
        currentHealth = Mathf.Max(0, currentHealth - damage);
        OnHealthChanged?.Invoke(currentHealth);

        if (IsDead)
        {
            Die();
        }
    }

    public void Heal(float amount)
    {
        currentHealth = Mathf.Min(maxHealth, currentHealth + amount);
        OnHealthChanged?.Invoke(currentHealth);
    }

    public bool TryAddItem(Item item)
    {
        if (inventory.Count >= inventorySize) return false;
        inventory.Add(item);
        return true;
    }

    public event Action<float> OnHealthChanged;

    private void Die()
    {
        // 死亡逻辑
    }
}
```

## 九、修复速查表

| 反模式 | 问题 | 修复方案 |
|--------|------|----------|
| 单例滥用 | 全局状态混乱 | 服务定位器/依赖注入 |
| 事件风暴 | 调用链不可控 | 直接调用/命令模式 |
| Update泛滥 | 性能开销大 | 管理器统一更新 |
| 每帧Find | O(n)开销 | 缓存引用/注入 |
| 字符串滥用 | 无类型安全 | 使用枚举/常量 |
| 协程嵌套 | 难以控制 | async/await或状态机 |
| 过度抽象 | 复杂度增加 | YAGNI，需要时再抽象 |
| Inspector全开 | 破坏封装 | SerializeField + 属性 |
