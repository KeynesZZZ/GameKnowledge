### 1.2 通用策略管理器

```csharp
using System;
using System.Collections.Generic;

/// <summary>
/// 策略管理器
/// </summary>
public class StrategyManager<TKey, TStrategy, TContext>
{
    private readonly Dictionary<TKey, TStrategy> strategies;
    private readonly Dictionary<TKey, Func<TStrategy>> factories;

    public StrategyManager()
    {
        strategies = new Dictionary<TKey, TStrategy>();
        factories = new Dictionary<TKey, Func<TStrategy>>();
    }

    /// <summary>
    /// 注册策略实例
    /// </summary>
    public void Register(TKey key, TStrategy strategy)
    {
        strategies[key] = strategy;
    }

    /// <summary>
    /// 注册策略工厂（延迟创建）
    /// </summary>
    public void RegisterFactory(TKey key, Func<TStrategy> factory)
    {
        factories[key] = factory;
    }

    /// <summary>
    /// 获取策略
    /// </summary>
    public TStrategy Get(TKey key)
    {
        if (strategies.TryGetValue(key, out var strategy))
        {
            return strategy;
        }

        if (factories.TryGetValue(key, out var factory))
        {
            strategy = factory();
            strategies[key] = strategy;
            return strategy;
        }

        return default;
    }

    /// <summary>
    /// 检查策略是否存在
    /// </summary>
    public bool HasStrategy(TKey key)
    {
        return strategies.ContainsKey(key) || factories.ContainsKey(key);
    }
}
```

### 2.1 行为策略接口

```csharp
using UnityEngine;

/// <summary>
/// AI行为策略接口
/// </summary>
public interface IBehaviorStrategy
{
    /// <summary>
    /// 执行行为
    /// </summary>
    void Execute(Enemy enemy);

    /// <summary>
    /// 检查是否可以执行
    /// </summary>
    bool CanExecute(Enemy enemy);

    /// <summary>
    /// 行为名称（用于调试）
    /// </summary>
    string Name { get; }
}
```

### 2.2 具体策略实现

```csharp
using UnityEngine;

/// <summary>
/// 追击策略
/// </summary>
public class ChaseStrategy : IBehaviorStrategy
{
    private readonly float chaseSpeed;
    private readonly float stopDistance;

    public string Name => "Chase";

    public ChaseStrategy(float speed = 3f, float stopDist = 1f)
    {
        chaseSpeed = speed;
        stopDistance = stopDist;
    }

    public bool CanExecute(Enemy enemy)
    {
        return enemy.Target != null && enemy.DistanceToTarget > stopDistance;
    }

    public void Execute(Enemy enemy)
    {
        if (enemy.Target == null) return;

        Vector3 direction = (enemy.Target.position - enemy.transform.position).normalized;
        enemy.transform.position += direction * chaseSpeed * Time.deltaTime;
        enemy.transform.LookAt(enemy.Target);
    }
}

/// <summary>
/// 巡逻策略
/// </summary>
public class PatrolStrategy : IBehaviorStrategy
{
    private readonly List<Vector3> patrolPoints;
    private readonly float patrolSpeed;
    private int currentIndex;

    public string Name => "Patrol";

    public PatrolStrategy(List<Vector3> points, float speed = 2f)
    {
        patrolPoints = points;
        patrolSpeed = speed;
        currentIndex = 0;
    }

    public bool CanExecute(Enemy enemy)
    {
        return patrolPoints.Count > 0;
    }

    public void Execute(Enemy enemy)
    {
        if (patrolPoints.Count == 0) return;

        Vector3 target = patrolPoints[currentIndex];
        Vector3 direction = (target - enemy.transform.position).normalized;

        enemy.transform.position += direction * patrolSpeed * Time.deltaTime;

        if (Vector3.Distance(enemy.transform.position, target) < 0.5f)
        {
            currentIndex = (currentIndex + 1) % patrolPoints.Count;
        }
    }
}

/// <summary>
/// 攻击策略
/// </summary>
public class AttackStrategy : IBehaviorStrategy
{
    private readonly float attackRange;
    private readonly float attackCooldown;
    private float lastAttackTime;

    public string Name => "Attack";

    public AttackStrategy(float range = 2f, float cooldown = 1f)
    {
        attackRange = range;
        attackCooldown = cooldown;
    }

    public bool CanExecute(Enemy enemy)
    {
        return enemy.Target != null &&
               enemy.DistanceToTarget <= attackRange &&
               Time.time >= lastAttackTime + attackCooldown;
    }

    public void Execute(Enemy enemy)
    {
        if (enemy.Target == null) return;

        // 执行攻击
        enemy.PlayAttackAnimation();
        enemy.DealDamageToTarget(10);

        lastAttackTime = Time.time;
    }
}

/// <summary>
/// 逃跑策略
/// </summary>
public class FleeStrategy : IBehaviorStrategy
{
    private readonly float fleeSpeed;
    private readonly float fleeDistance;
    private readonly float healthThreshold;

    public string Name => "Flee";

    public FleeStrategy(float speed = 4f, float distance = 10f, float healthPercent = 0.3f)
    {
        fleeSpeed = speed;
        fleeDistance = distance;
        healthThreshold = healthPercent;
    }

    public bool CanExecute(Enemy enemy)
    {
        return enemy.HealthPercent <= healthThreshold;
    }

    public void Execute(Enemy enemy)
    {
        if (enemy.Target == null) return;

        // 逃离目标
        Vector3 fleeDirection = (enemy.transform.position - enemy.Target.position).normalized;
        Vector3 fleePosition = enemy.transform.position + fleeDirection * fleeDistance;

        enemy.transform.position = Vector3.MoveTowards(
            enemy.transform.position,
            fleePosition,
            fleeSpeed * Time.deltaTime
        );
    }
}
```

### 2.3 AI控制器

```csharp
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// AI控制器 - 使用策略模式
/// </summary>
public class AIController : MonoBehaviour
{
    [Header("配置")]
    [SerializeField] private float detectionRange = 10f;
    [SerializeField] private float attackRange = 2f;

    private Dictionary<string, IBehaviorStrategy> strategies;
    private IBehaviorStrategy currentStrategy;
    private Enemy enemy;

    private void Awake()
    {
        enemy = GetComponent<Enemy>();
        InitializeStrategies();
    }

    private void InitializeStrategies()
    {
        strategies = new Dictionary<string, IBehaviorStrategy>
        {
            { "Patrol", new PatrolStrategy(GetPatrolPoints(), 2f) },
            { "Chase", new ChaseStrategy(3f, attackRange) },
            { "Attack", new AttackStrategy(attackRange, 1f) },
            { "Flee", new FleeStrategy(4f, 10f, 0.3f) }
        };
    }

    private void Update()
    {
        // 策略选择逻辑
        SelectStrategy();
        currentStrategy?.Execute(enemy);
    }

    private void SelectStrategy()
    {
        // 优先级：逃跑 > 攻击 > 追击 > 巡逻
        if (strategies["Flee"].CanExecute(enemy))
        {
            SetStrategy("Flee");
        }
        else if (strategies["Attack"].CanExecute(enemy))
        {
            SetStrategy("Attack");
        }
        else if (strategies["Chase"].CanExecute(enemy))
        {
            SetStrategy("Chase");
        }
        else
        {
            SetStrategy("Patrol");
        }
    }

    public void SetStrategy(string strategyName)
    {
        if (strategies.TryGetValue(strategyName, out var strategy))
        {
            if (currentStrategy != strategy)
            {
                currentStrategy = strategy;
                Debug.Log($"AI策略切换: {strategy.Name}");
            }
        }
    }

    private List<Vector3> GetPatrolPoints()
    {
        // 获取巡逻点
        return new List<Vector3>
        {
            transform.position + Vector3.forward * 5,
            transform.position + Vector3.right * 5,
            transform.position + Vector3.back * 5,
            transform.position + Vector3.left * 5
        };
    }
}

/// <summary>
/// 敌人实体
/// </summary>
public class Enemy : MonoBehaviour
{
    [SerializeField] private float maxHealth = 100f;
    private float currentHealth;

    public Transform Target { get; set; }
    public float DistanceToTarget => Target != null ?
        Vector3.Distance(transform.position, Target.position) : float.MaxValue;
    public float HealthPercent => currentHealth / maxHealth;

    private void Awake()
    {
        currentHealth = maxHealth;
    }

    public void TakeDamage(float damage)
    {
        currentHealth = Mathf.Max(0, currentHealth - damage);
    }

    public void DealDamageToTarget(float damage)
    {
        if (Target != null)
        {
            var player = Target.GetComponent<Player>();
            player?.TakeDamage(damage);
        }
    }

    public void PlayAttackAnimation()
    {
        // 播放攻击动画
    }
}
```

```csharp
/// <summary>
/// 缓速效果
/// </summary>
public class SlowEffect : ISkillEffect
{
    private readonly float slowPercent;
    private readonly float duration;
    private float originalSpeed;

    public string Description => $"减缓 {slowPercent * 100}% 移动速度，持续 {duration} 秒";

    public SlowEffect(float percent, float dur)
    {
        slowPercent = percent;
        duration = dur;
    }

    public void Apply(Character target)
    {
        originalSpeed = target.MoveSpeed;
        target.MoveSpeed *= (1f - slowPercent);

        // 延迟移除
        target.StartCoroutine(RemoveAfterDuration(target));
    }

    public void Remove(Character target)
    {
        target.MoveSpeed = originalSpeed;
    }

    private System.Collections.IEnumerator RemoveAfterDuration(Character target)
    {
        yield return new WaitForSeconds(duration);
        Remove(target);
    }
}

/// <summary>
/// 护盾效果
/// </summary>
public class ShieldEffect : ISkillEffect
{
    private readonly float shieldAmount;
    private float currentShield;

    public string Description => $"获得 {shieldAmount} 点护盾";

    public ShieldEffect(float amount)
    {
        shieldAmount = amount;
    }

    public void Apply(Character target)
    {
        currentShield = shieldAmount;
        target.AddShield(currentShield);
    }

    public void Remove(Character target)
    {
        target.RemoveShield(currentShield);
    }
}

/// <summary>
/// 持续伤害效果（DOT）
/// </summary>
public class DotEffect : ISkillEffect
{
    private readonly float damagePerTick;
    private readonly float tickInterval;
    private readonly int totalTicks;
    private int currentTick;

    public string Description => $"每 {tickInterval} 秒造成 {damagePerTick} 伤害，共 {totalTicks} 次";

    public DotEffect(float damage, float interval, int ticks)
    {
        damagePerTick = damage;
        tickInterval = interval;
        totalTicks = ticks;
    }

    public void Apply(Character target)
    {
        currentTick = 0;
        target.StartCoroutine(ApplyDamageOverTime(target));
    }

    public void Remove(Character target)
    {
        // DOT效果由协程自然结束
    }

    private System.Collections.IEnumerator ApplyDamageOverTime(Character target)
    {
        while (currentTick < totalTicks)
        {
            yield return new WaitForSeconds(tickInterval);
            target.TakeDamage(damagePerTick);
            currentTick++;
        }
    }
}
```

### 3.3 复合技能效果

```csharp
using System.Collections.Generic;

/// <summary>
/// 复合效果 - 组合多个效果
/// </summary>
public class CompositeEffect : ISkillEffect
{
    private readonly List<ISkillEffect> effects;

    public string Description
    {
        get
        {
            var descriptions = new List<string>();
            foreach (var effect in effects)
            {
                descriptions.Add(effect.Description);
            }
            return string.Join(", ", descriptions);
        }
    }

    public CompositeEffect(params ISkillEffect[] effects)
    {
        this.effects = new List<ISkillEffect>(effects);
    }

    public void AddEffect(ISkillEffect effect)
    {
        effects.Add(effect);
    }

    public void Apply(Character target)
    {
        foreach (var effect in effects)
        {
            effect.Apply(target);
        }
    }

    public void Remove(Character target)
    {
        // 逆序移除
        for (int i = effects.Count - 1; i >= 0; i--)
        {
            effects[i].Remove(target);
        }
    }
}
```

### 4.2 具体实现

```csharp
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.AI;

/// <summary>
/// NavMesh寻路策略
/// </summary>
public class NavMeshPathfinding : IPathfindingStrategy
{
    public bool IsPathValid { get; private set; }

    public List<Vector3> FindPath(Vector3 start, Vector3 end)
    {
        var path = new NavMeshPath();
        IsPathValid = NavMesh.CalculatePath(start, end, NavMesh.AllAreas, path);

        if (IsPathValid)
        {
            return new List<Vector3>(path.corners);
        }

        return new List<Vector3>();
    }
}

/// <summary>
/// 直线寻路策略（简单场景）
/// </summary>
public class DirectPathfinding : IPathfindingStrategy
{
    public bool IsPathValid => true;

    public List<Vector3> FindPath(Vector3 start, Vector3 end)
    {
        return new List<Vector3> { start, end };
    }
}
```

### 4.3 使用示例

```csharp
public class MovementController : MonoBehaviour
{
    private IPathfindingStrategy pathfinding;
    private List<Vector3> currentPath;
    private int pathIndex;

    public void SetPathfindingStrategy(IPathfindingStrategy strategy)
    {
        pathfinding = strategy;
    }

    public void MoveTo(Vector3 destination)
    {
        currentPath = pathfinding.FindPath(transform.position, destination);
        pathIndex = 0;

        if (pathfinding.IsPathValid)
        {
            StartCoroutine(FollowPath());
        }
    }

    private System.Collections.IEnumerator FollowPath()
    {
        while (pathIndex < currentPath.Count)
        {
            Vector3 target = currentPath[pathIndex];
            transform.position = Vector3.MoveTowards(
                transform.position,
                target,
                5f * Time.deltaTime
            );

            if (Vector3.Distance(transform.position, target) < 0.1f)
            {
                pathIndex++;
            }

            yield return null;
        }
    }
}
```
