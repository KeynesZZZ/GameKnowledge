## 1. 依赖注入(DI)核心概念

### 什么是依赖？

```csharp
// 依赖：类A需要类B才能工作，就说A依赖B
public class Player
{
    // Player 依赖 AudioManager
    private AudioManager audioManager;

    public Player()
    {
        audioManager = new AudioManager();  // 硬编码依赖
    }
}
```

**问题**：
- 紧耦合，难以测试
- 难以替换实现
- 违反"依赖倒置原则"

### 依赖注入解决方案

```csharp
// 依赖注入：依赖从外部传入
public class Player
{
    private IAudioManager audioManager;

    // 构造函数注入
    public Player(IAudioManager audioManager)
    {
        this.audioManager = audioManager;
    }
}
```

## 2. 依赖注入的三种方式

```
┌─────────────────────────────────────────────────────────────┐
│                    依赖注入的三种方式                         │
│                                                             │
│  1. 构造函数注入（推荐）                                      │
│     ┌─────────────────────────────────────────┐            │
│     │ public Player(IAudioManager audio)       │            │
│     │ {                                        │            │
│     │     this.audioManager = audio;           │            │
│     │ }                                        │            │
│     └─────────────────────────────────────────┘            │
│     优点：依赖明确，创建时就必须提供                           │
│     缺点：依赖多时构造函数参数多                               │
│                                                             │
│  2. 属性注入                                                 │
│     ┌─────────────────────────────────────────┐            │
│     │ [Inject]                                 │            │
│     │ public IAudioManager AudioManager { set } │           │
│     └─────────────────────────────────────────┘            │
│     优点：可选依赖，灵活                                      │
│     缺点：依赖不明确，可能为null                              │
│                                                             │
│  3. 方法注入                                                 │
│     ┌─────────────────────────────────────────┐            │
│     │ public void SetAudio(IAudioManager audio) │           │
│     │ {                                        │            │
│     │     this.audioManager = audio;           │            │
│     │ }                                        │            │
│     └─────────────────────────────────────────┘            │
│     优点：可以在运行时更改                                    │
│     缺点：依赖不明确，可能忘记调用                            │
└─────────────────────────────────────────────────────────────┘
```

## 3. 完整的DI容器实现

```csharp
/// <summary>
/// 依赖注入容器 - 管理所有依赖的注册和解析
/// </summary>
public class DIContainer
{
    // 存储单例实例
    private readonly Dictionary<Type, object> instances = new();

    // 存储工厂方法
    private readonly Dictionary<Type, Func<DIContainer, object>> factories = new();

    // 存储类型映射（接口 -> 实现类）
    private readonly Dictionary<Type, Type> mappings = new();

    // 存储已解析的实例（用于检测循环依赖）
    private readonly HashSet<Type> resolving = new();

    #region 注册方法

    /// <summary>
    /// 注册单例（每次返回同一实例）
    /// </summary>
    public void RegisterSingleton<TInterface, TImplementation>()
        where TImplementation : TInterface
    {
        mappings[typeof(TInterface)] = typeof(TImplementation);
        factories[typeof(TInterface)] = container =>
        {
            if (!instances.TryGetValue(typeof(TInterface), out var instance))
            {
                instance = CreateInstance<TImplementation>(container);
                instances[typeof(TInterface)] = instance;
            }
            return instance;
        };
    }

    /// <summary>
    /// 注册单例（带工厂方法）
    /// </summary>
    public void RegisterSingleton<TInterface>(Func<DIContainer, TInterface> factory)
    {
        factories[typeof(TInterface)] = container =>
        {
            if (!instances.TryGetValue(typeof(TInterface), out var instance))
            {
                instance = factory(container);
                instances[typeof(TInterface)] = instance;
            }
            return instance;
        };
    }

    /// <summary>
    /// 注册瞬态（每次创建新实例）
    /// </summary>
    public void RegisterTransient<TInterface, TImplementation>()
        where TImplementation : TInterface
    {
        mappings[typeof(TInterface)] = typeof(TImplementation);
        factories[typeof(TInterface)] = container => CreateInstance<TImplementation>(container);
    }

    /// <summary>
    /// 注册已有实例
    /// </summary>
    public void RegisterInstance<TInterface>(TInterface instance)
    {
        instances[typeof(TInterface)] = instance;
        factories[typeof(TInterface)] = _ => instance;
    }

    #endregion

    #region 解析方法

    /// <summary>
    /// 解析依赖
    /// </summary>
    public T Resolve<T>()
    {
        return (T)Resolve(typeof(T));
    }

    private object Resolve(Type type)
    {
        // 检测循环依赖
        if (resolving.Contains(type))
        {
            throw new Exception($"检测到循环依赖: {type.Name}");
        }

        if (!factories.TryGetValue(type, out var factory))
        {
            throw new Exception($"未注册的依赖: {type.Name}");
        }

        resolving.Add(type);
        try
        {
            return factory(this);
        }
        finally
        {
            resolving.Remove(type);
        }
    }

    /// <summary>
    /// 尝试解析
    /// </summary>
    public bool TryResolve<T>(out T dependency)
    {
        try
        {
            dependency = Resolve<T>();
            return true;
        }
        catch
        {
            dependency = default;
            return false;
        }
    }

    #endregion

    #region 实例创建

    private T CreateInstance<T>(DIContainer container)
    {
        var type = typeof(T);
        var constructors = type.GetConstructors();

        // 找到参数最多的构造函数
        var constructor = constructors
            .OrderByDescending(c => c.GetParameters().Length)
            .First();

        var parameters = constructor.GetParameters();
        var args = new object[parameters.Length];

        for (int i = 0; i < parameters.Length; i++)
        {
            var paramType = parameters[i].ParameterType;
            args[i] = Resolve(paramType);
        }

        return (T)constructor.Invoke(args);
    }

    #endregion

    #region 辅助方法

    public bool IsRegistered<T>() => factories.ContainsKey(typeof(T));

    public void Clear()
    {
        instances.Clear();
        factories.Clear();
        mappings.Clear();
    }

    #endregion
}
```

## 4. 服务定位器模式

```csharp
/// <summary>
/// 服务定位器 - 全局访问服务的静态类
/// </summary>
public static class ServiceLocator
{
    private static DIContainer container;

    public static void Initialize(DIContainer diContainer)
    {
        container = diContainer;
    }

    public static T Get<T>()
    {
        if (container == null)
            throw new Exception("ServiceLocator未初始化");
        return container.Resolve<T>();
    }

    public static bool TryGet<T>(out T service)
    {
        if (container == null)
        {
            service = default;
            return false;
        }
        return container.TryResolve(out service);
    }

    public static bool Exists<T>() => container?.IsRegistered<T>() ?? false;
}
```

## 5. DI vs 服务定位器 对比

```
┌─────────────────────────────────────────────────────────────┐
│                   DI vs 服务定位器                           │
│                                                             │
│   依赖注入 (DI)                服务定位器 (Service Locator)  │
│   ┌─────────────────┐          ┌─────────────────┐        │
│   │     Player      │          │     Player      │        │
│   │  ┌───────────┐  │          │                 │        │
│   │  │AudioManager│←─┼── 外部   │  var audio =   │        │
│   │  └───────────┘  │   注入   │  ServiceLocator │        │
│   └─────────────────┘          │  .Get<Audio>()  │        │
│                                └─────────────────┘        │
│                                                             │
│   依赖通过构造函数传入          依赖在需要时主动获取          │
│   依赖关系明确                  依赖关系隐藏                  │
│   易于测试                      测试需要Mock Locator         │
│   推荐用于大型项目              适合快速开发                  │
└─────────────────────────────────────────────────────────────┘
```

| 对比项 | 依赖注入 | 服务定位器 |
|--------|----------|------------|
| 依赖可见性 | ✅ 明确（在构造函数中） | ❌ 隐藏（在方法内部） |
| 测试难度 | 低（易于Mock） | 中（需要Mock Locator） |
| 代码复杂度 | 较高 | 较低 |
| IDE支持 | 好（可自动补全） | 差（运行时才知道） |
| 适合场景 | 大型项目、团队协作 | 小型项目、快速原型 |

## 6. Unity中的最佳实践：混合使用

```csharp
/// <summary>
/// 游戏服务管理器 - 结合DI和Service Locator
/// </summary>
public class GameServices : MonoBehaviour
{
    [Header("Service Configuration")]
    [SerializeField] private bool useDebugServices = false;

    private static DIContainer container;
    public static DIContainer Container => container;

    private void Awake()
    {
        if (container != null)
        {
            Destroy(gameObject);
            return;
        }

        DontDestroyOnLoad(gameObject);
        InitializeServices();
    }

    private void InitializeServices()
    {
        container = new DIContainer();

        RegisterCoreServices();
        RegisterGameServices();

        ServiceLocator.Initialize(container);
        InjectAllInScene();
    }

    private void RegisterCoreServices()
    {
        var config = Resources.Load<GameConfig>("Data/GameConfig");
        container.RegisterInstance<IGameConfig>(config);
        container.RegisterSingleton<IAudioService, AudioService>();
        container.RegisterSingleton<ISaveSystem, SaveSystem>();
    }

    private void RegisterGameServices()
    {
        container.RegisterTransient<IBoardService, BoardService>();

        container.RegisterSingleton<ICombatService>(c =>
        {
            var config = c.Resolve<IGameConfig>();
            return new CombatService(config);
        });
    }

    private void InjectAllInScene()
    {
        var injectables = FindObjectsOfType<MonoBehaviour>(true)
            .OfType<IInjectable>();

        foreach (var injectable in injectables)
        {
            injectable.Inject(container);
        }
    }

    public static T Get<T>() => container.Resolve<T>();

    private void OnDestroy()
    {
        container?.Clear();
        container = null;
    }
}

/// <summary>
/// 可注入接口
/// </summary>
public interface IInjectable
{
    void Inject(DIContainer container);
}
```

## 7. 实战示例：三消游戏的服务架构

```csharp
// ============ 服务接口定义 ============

public interface IBoardService
{
    void Initialize(int width, int height);
    Gem GetGem(int x, int y);
    void SwapGems(Vector2Int a, Vector2Int b);
    List<Match> DetectMatches();
    event Action<List<Match>> OnMatchesFound;
}

public interface ICombatService
{
    void Initialize(List<HeroData> heroes, List<EnemyData> enemies);
    void ProcessMatch(Match match);
    int CalculateDamage(int matchCount, int combo);
    event Action<CombatResult> OnCombatComplete;
}

public interface IScoreService
{
    int CurrentScore { get; }
    int Combo { get; }
    void AddScore(int basePoints);
    void ResetCombo();
    event Action<ScoreChangedEventArgs> OnScoreChanged;
}

public interface IAudioService
{
    void PlayBGM(string clipName);
    void StopBGM();
    void PlaySFX(string clipName);
    void SetMasterVolume(float volume);
}

public interface ISaveSystem
{
    SaveData Load();
    void Save(SaveData data);
    void DeleteSave();
}

// ============ 服务实现 ============

public class BoardService : IBoardService
{
    private Gem[,] grid;
    private int width, height;

    public event Action<List<Match>> OnMatchesFound;

    private readonly IGameConfig config;
    private readonly IGemPool gemPool;

    // 构造函数注入
    public BoardService(IGameConfig config, IGemPool gemPool)
    {
        this.config = config;
        this.gemPool = gemPool;
    }

    public void Initialize(int width, int height)
    {
        this.width = width;
        this.height = height;
        grid = new Gem[width, height];
        FillBoard();
    }

    public Gem GetGem(int x, int y)
    {
        if (x < 0 || x >= width || y < 0 || y >= height)
            return null;
        return grid[x, y];
    }

    public void SwapGems(Vector2Int a, Vector2Int b)
    {
        (grid[a.x, a.y], grid[b.x, b.y]) = (grid[b.x, b.y], grid[a.x, a.y]);
    }

    public List<Match> DetectMatches()
    {
        var matches = new List<Match>();
        // 实现消除检测...
        OnMatchesFound?.Invoke(matches);
        return matches;
    }

    private void FillBoard()
    {
        for (int x = 0; x < width; x++)
        for (int y = 0; y < height; y++)
            grid[x, y] = gemPool.Get(Random.Range(0, 5));
    }
}

public class CombatService : ICombatService
{
    private List<HeroData> heroes;
    private List<EnemyData> enemies;

    public event Action<CombatResult> OnCombatComplete;

    private readonly IGameConfig config;
    private readonly IAudioService audioService;

    public CombatService(IGameConfig config, IAudioService audioService)
    {
        this.config = config;
        this.audioService = audioService;
    }

    public void Initialize(List<HeroData> heroes, List<EnemyData> enemies)
    {
        this.heroes = heroes;
        this.enemies = enemies;
    }

    public void ProcessMatch(Match match)
    {
        int damage = CalculateDamage(match.Count, 1);
        var target = enemies.FirstOrDefault(e => !e.IsDead);

        if (target != null)
        {
            target.TakeDamage(damage);
            audioService.PlaySFX("attack");

            OnCombatComplete?.Invoke(new CombatResult
            {
                Attacker = match.GemId,
                Target = target.Id,
                Damage = damage
            });
        }
    }

    public int CalculateDamage(int matchCount, int combo)
    {
        float multiplier = matchCount switch
        {
            >= 5 => config.DamageMultiplier5,
            4 => config.DamageMultiplier4,
            _ => config.DamageMultiplier3
        };

        return (int)(matchCount * 100 * multiplier * combo);
    }
}

// ============ 使用示例 ============

public class GameController : MonoBehaviour, IInjectable
{
    private IBoardService boardService;
    private ICombatService combatService;
    private IScoreService scoreService;
    private IAudioService audioService;

    public void Inject(DIContainer container)
    {
        boardService = container.Resolve<IBoardService>();
        combatService = container.Resolve<ICombatService>();
        scoreService = container.Resolve<IScoreService>();
        audioService = container.Resolve<IAudioService>();
    }

    private void OnEnable()
    {
        boardService.OnMatchesFound += OnMatchesFound;
        combatService.OnCombatComplete += OnCombatComplete;
        scoreService.OnScoreChanged += OnScoreChanged;
    }

    private void OnDisable()
    {
        boardService.OnMatchesFound -= OnMatchesFound;
        combatService.OnCombatComplete -= OnCombatComplete;
        scoreService.OnScoreChanged -= OnScoreChanged;
    }

    private void Start() => boardService.Initialize(7, 7);

    private void OnMatchesFound(List<Match> matches)
    {
        foreach (var match in matches)
        {
            combatService.ProcessMatch(match);
            scoreService.AddScore(match.Count * 100);
        }
    }

    private void OnCombatComplete(CombatResult result)
    {
        Debug.Log($"Combat: {result.Attacker} dealt {result.Damage} to {result.Target}");
    }

    private void OnScoreChanged(ScoreChangedEventArgs e)
    {
        Debug.Log($"Score: {e.NewScore}, Combo: {e.Combo}");
    }
}

// 使用服务定位器
public class PlayerController : MonoBehaviour
{
    private IAudioService audioService;

    private void Start() => audioService = ServiceLocator.Get<IAudioService>();

    public void Attack() => audioService.PlaySFX("attack");
}
```

## 8. 高级特性：自动注入

```csharp
/// <summary>
/// Inject特性 - 标记需要自动注入的字段
/// </summary>
[AttributeUsage(AttributeTargets.Field | AttributeTargets.Property)]
public class InjectAttribute : Attribute { }

/// <summary>
/// 自动注入器
/// </summary>
public static class AutoInjector
{
    public static void Inject(object target, DIContainer container)
    {
        var type = target.GetType();

        // 注入字段
        var fields = type.GetFields(BindingFlags.NonPublic | BindingFlags.Instance);
        foreach (var field in fields)
        {
            if (field.GetCustomAttribute<InjectAttribute>() != null)
            {
                var value = container.Resolve(field.FieldType);
                field.SetValue(target, value);
            }
        }

        // 注入属性
        var properties = type.GetProperties(BindingFlags.NonPublic | BindingFlags.Instance);
        foreach (var prop in properties)
        {
            if (prop.GetCustomAttribute<InjectAttribute>() != null && prop.CanWrite)
            {
                var value = container.Resolve(prop.PropertyType);
                prop.SetValue(target, value);
            }
        }
    }
}

// 使用示例
public class AutoInjectExample : MonoBehaviour
{
    [Inject] private IBoardService boardService;
    [Inject] private IScoreService scoreService;
    [Inject] private IAudioService audioService;

    private void Start()
    {
        AutoInjector.Inject(this, GameServices.Container);
        boardService.Initialize(7, 7);
    }
}
```

## 本课小结

### 核心知识点

| 知识点 | 核心要点 |
|--------|----------|
| 依赖注入 | 依赖从外部传入，不由类自己创建 |
| 三种注入方式 | 构造函数注入（推荐）、属性注入、方法注入 |
| DI容器 | 管理依赖的注册和解析 |
| 服务定位器 | 全局访问点获取服务 |
| 选择建议 | 大项目用DI，小项目/原型用服务定位器 |

### 最佳实践

| 场景 | 推荐方案 |
|------|----------|
| 核心依赖 | 构造函数注入 |
| 可选依赖 | 属性注入或服务定位器 |
| MonoBehaviour | 使用IInjectable接口 |
| 测试 | DI + Mock |
| 快速原型 | 服务定位器 |
