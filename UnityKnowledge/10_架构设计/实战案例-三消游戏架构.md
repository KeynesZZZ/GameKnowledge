### 3.1 棋盘系统 (BoardSystem)

```csharp
/// <summary>
/// 棋盘格子数据
/// </summary>
public struct Cell
{
    public int x;
    public int y;
    public GemType gemType;
    public bool isEmpty;
    public bool isMatched;

    public static Cell Empty(int x, int y) => new Cell
    {
        x = x, y = y,
        isEmpty = true,
        gemType = GemType.None
    };
}

/// <summary>
/// 宝石类型
/// </summary>
public enum GemType
{
    None = 0,
    Red = 1,    // 火焰 - 攻击
    Blue = 2,   // 冰霜 - 防御
    Green = 3,  // 自然 - 治疗
    Yellow = 4, // 雷电 - 魔法
    Purple = 5, // 暗影 - 特殊
}

/// <summary>
/// 棋盘系统
/// </summary>
public class BoardSystem
{
    private readonly int width = 8;
    private readonly int height = 8;
    private Cell[,] cells;

    public event Action<List<MatchResult>> OnMatchFound;
    public event Action<List<Cell>> OnCellsCleared;
    public event Action<List<FallData>> OnGemsFallen;
    public event Action<List<SpawnData>> OnGemsSpawned;

    public BoardSystem(int width, int height)
    {
        this.width = width;
        this.height = height;
        cells = new Cell[width, height];
        InitializeBoard();
    }

    private void InitializeBoard()
    {
        for (int x = 0; x < width; x++)
        {
            for (int y = 0; y < height; y++)
            {
                cells[x, y] = new Cell
                {
                    x = x,
                    y = y,
                    gemType = GetRandomGemType(),
                    isEmpty = false
                };
            }
        }

        // 确保初始无匹配
        while (HasAnyMatch())
        {
            ShuffleBoard();
        }
    }

    /// <summary>
    /// 交换宝石
    /// </summary>
    public bool TrySwap(int x1, int y1, int x2, int y2)
    {
        if (!IsValidPosition(x1, y1) || !IsValidPosition(x2, y2))
            return false;

        // 检查是否相邻
        if (!IsAdjacent(x1, y1, x2, y2))
            return false;

        // 执行交换
        Swap(x1, y1, x2, y2);

        // 检查是否产生匹配
        if (HasMatch())
        {
            return true;
        }
        else
        {
            // 无效交换，换回
            Swap(x1, y1, x2, y2);
            return false;
        }
    }

    private void Swap(int x1, int y1, int x2, int y2)
    {
        var temp = cells[x1, y1];
        cells[x1, y1] = cells[x2, y2];
        cells[x2, y2] = temp;
    }

    /// <summary>
    /// 处理匹配
    /// </summary>
    public List<MatchResult> ProcessMatches()
    {
        var matches = FindAllMatches();
        if (matches.Count > 0)
        {
            OnMatchFound?.Invoke(matches);
        }
        return matches;
    }

    /// <summary>
    /// 清除匹配的格子
    /// </summary>
    public void ClearMatchedCells(List<MatchResult> matches)
    {
        var clearedCells = new List<Cell>();

        foreach (var match in matches)
        {
            foreach (var pos in match.positions)
            {
                cells[pos.x, pos.y].isEmpty = true;
                cells[pos.x, pos.y].gemType = GemType.None;
                clearedCells.Add(cells[pos.x, pos.y]);
            }
        }

        OnCellsCleared?.Invoke(clearedCells);
    }

    /// <summary>
    /// 下落填充
    /// </summary>
    public void FillBoard()
    {
        var fallData = new List<FallData>();
        var spawnData = new List<SpawnData>();

        // 处理下落
        for (int x = 0; x < width; x++)
        {
            int emptyY = -1;

            for (int y = 0; y < height; y++)
            {
                if (cells[x, y].isEmpty && emptyY == -1)
                {
                    emptyY = y;
                }
                else if (!cells[x, y].isEmpty && emptyY != -1)
                {
                    // 下落
                    cells[x, emptyY] = cells[x, y];
                    cells[x, y] = Cell.Empty(x, y);

                    fallData.Add(new FallData
                    {
                        x = x,
                        fromY = y,
                        toY = emptyY,
                        gemType = cells[x, emptyY].gemType
                    });

                    emptyY++;
                }
            }

            // 生成新宝石
            int spawnCount = 0;
            for (int y = height - 1; y >= 0; y--)
            {
                if (cells[x, y].isEmpty)
                {
                    var newGem = GetRandomGemType();
                    cells[x, y] = new Cell
                    {
                        x = x, y = y,
                        gemType = newGem,
                        isEmpty = false
                    };

                    spawnData.Add(new SpawnData
                    {
                        x = x,
                        y = y,
                        gemType = newGem,
                        spawnHeight = height + spawnCount
                    });
                    spawnCount++;
                }
            }
        }

        if (fallData.Count > 0) OnGemsFallen?.Invoke(fallData);
        if (spawnData.Count > 0) OnGemsSpawned?.Invoke(spawnData);
    }

    private List<MatchResult> FindAllMatches()
    {
        var matches = new List<MatchResult>();

        // 横向匹配
        for (int y = 0; y < height; y++)
        {
            int matchStart = 0;
            for (int x = 1; x < width; x++)
            {
                if (cells[x, y].gemType == cells[matchStart, y].gemType && !cells[x, y].isEmpty)
                {
                    continue;
                }
                else
                {
                    if (x - matchStart >= 3)
                    {
                        matches.Add(CreateMatchResult(matchStart, y, x - 1, y, true));
                    }
                    matchStart = x;
                }
            }
            if (width - matchStart >= 3)
            {
                matches.Add(CreateMatchResult(matchStart, y, width - 1, y, true));
            }
        }

        // 纵向匹配（类似逻辑）
        // ...

        return matches;
    }

    private MatchResult CreateMatchResult(int x1, int y1, int x2, int y2, bool horizontal)
    {
        var positions = new List<Vector2Int>();
        var gemType = cells[x1, y1].gemType;

        if (horizontal)
        {
            for (int x = x1; x <= x2; x++)
            {
                positions.Add(new Vector2Int(x, y1));
            }
        }
        else
        {
            for (int y = y1; y <= y2; y++)
            {
                positions.Add(new Vector2Int(x1, y));
            }
        }

        return new MatchResult
        {
            gemType = gemType,
            count = positions.Count,
            positions = positions,
            isHorizontal = horizontal
        };
    }

    private bool IsAdjacent(int x1, int y1, int x2, int y2)
    {
        return (Mathf.Abs(x1 - x2) == 1 && y1 == y2) ||
               (Mathf.Abs(y1 - y2) == 1 && x1 == x2);
    }

    private bool IsValidPosition(int x, int y)
    {
        return x >= 0 && x < width && y >= 0 && y < height;
    }

    private bool HasMatch() => FindAllMatches().Count > 0;
    private bool HasAnyMatch() => HasMatch();

    private void ShuffleBoard()
    {
        // 随机重排
    }

    private GemType GetRandomGemType()
    {
        return (GemType)Random.Range(1, 6);
    }

    public Cell GetCell(int x, int y) => cells[x, y];
    public int Width => width;
    public int Height => height;
}

public struct MatchResult
{
    public GemType gemType;
    public int count;
    public List<Vector2Int> positions;
    public bool isHorizontal;
}

public struct FallData
{
    public int x;
    public int fromY;
    public int toY;
    public GemType gemType;
}

public struct SpawnData
{
    public int x;
    public int y;
    public GemType gemType;
    public int spawnHeight;
}
```

### 3.2 战斗系统 (CombatSystem)

```csharp
/// <summary>
/// 战斗单位数据
/// </summary>
public class CombatUnit
{
    public string id;
    public int maxHealth;
    public int currentHealth;
    public int attack;
    public int defense;
    public bool isPlayer;

    public void TakeDamage(int damage)
    {
        int actualDamage = Mathf.Max(1, damage - defense);
        currentHealth = Mathf.Max(0, currentHealth - actualDamage);
    }

    public void Heal(int amount)
    {
        currentHealth = Mathf.Min(maxHealth, currentHealth + amount);
    }

    public bool IsDead => currentHealth <= 0;
}

/// <summary>
/// 战斗系统
/// </summary>
public class CombatSystem
{
    private CombatUnit player;
    private List<CombatUnit> enemies;

    public event Action<CombatUnit, int> OnDamageDealt;
    public event Action<CombatUnit, int> OnHealApplied;
    public event Action<CombatUnit> OnUnitDied;
    public event Action OnCombatEnd;

    public void Initialize(CombatUnit playerUnit, List<CombatUnit> enemyUnits)
    {
        player = playerUnit;
        enemies = enemyUnits;
    }

    /// <summary>
    /// 处理匹配结果触发的战斗效果
    /// </summary>
    public void ProcessMatchResult(MatchResult match)
    {
        int effectPower = CalculateEffectPower(match);

        switch (match.gemType)
        {
            case GemType.Red:   // 火焰 = 攻击
                DealDamageToEnemies(effectPower);
                break;

            case GemType.Blue:  // 冰霜 = 防御
                ApplyDefenseBoost(effectPower);
                break;

            case GemType.Green: // 自然 = 治疗
                HealPlayer(effectPower);
                break;

            case GemType.Yellow: // 雷电 = 魔法
                DealMagicDamage(effectPower);
                break;

            case GemType.Purple: // 暗影 = 特殊
                TriggerSpecialEffect(effectPower);
                break;
        }

        CheckCombatEnd();
    }

    private int CalculateEffectPower(MatchResult match)
    {
        int basePower = match.count * 10;

        // 4消和5消加成
        if (match.count >= 5) basePower *= 3;
        else if (match.count >= 4) basePower *= 2;

        return basePower;
    }

    private void DealDamageToEnemies(int damage)
    {
        if (enemies.Count == 0) return;

        // 攻击第一个存活的敌人
        var target = enemies.FirstOrDefault(e => !e.IsDead);
        if (target != null)
        {
            target.TakeDamage(damage);
            OnDamageDealt?.Invoke(target, damage);

            if (target.IsDead)
            {
                OnUnitDied?.Invoke(target);
            }
        }
    }

    private void HealPlayer(int amount)
    {
        player.Heal(amount);
        OnHealApplied?.Invoke(player, amount);
    }

    private void ApplyDefenseBoost(int amount)
    {
        // 临时防御提升
    }

    private void DealMagicDamage(int damage)
    {
        // AOE伤害
        foreach (var enemy in enemies.Where(e => !e.IsDead))
        {
            enemy.TakeDamage(damage / 2);
            OnDamageDealt?.Invoke(enemy, damage / 2);
        }
    }

    private void TriggerSpecialEffect(int power)
    {
        // 特殊效果
    }

    private void CheckCombatEnd()
    {
        if (player.IsDead || enemies.All(e => e.IsDead))
        {
            OnCombatEnd?.Invoke();
        }
    }

    /// <summary>
    /// 敌人回合
    /// </summary>
    public void EnemyTurn()
    {
        foreach (var enemy in enemies.Where(e => !e.IsDead))
        {
            player.TakeDamage(enemy.attack);
            OnDamageDealt?.Invoke(player, enemy.attack);
        }

        CheckCombatEnd();
    }
}
```

### 3.3 游戏控制器 (GameController)

```csharp
using System.Collections;
using UnityEngine;

/// <summary>
/// 游戏状态
/// </summary>
public enum GameState
{
    Idle,           // 等待玩家输入
    Processing,     // 处理匹配中
    Combat,         // 战斗中
    EnemyTurn,      // 敌人回合
    Victory,        // 胜利
    Defeat          // 失败
}

/// <summary>
/// 游戏控制器
/// </summary>
public class GameController : MonoBehaviour
{
    [Header("配置")]
    [SerializeField] private int boardWidth = 8;
    [SerializeField] private int boardHeight = 8;

    private BoardSystem boardSystem;
    private CombatSystem combatSystem;
    private GameState currentState;

    public event Action<GameState> OnStateChanged;

    private void Start()
    {
        InitializeGame();
    }

    private void InitializeGame()
    {
        // 初始化系统
        boardSystem = new BoardSystem(boardWidth, boardHeight);
        combatSystem = new CombatSystem();

        // 订阅事件
        boardSystem.OnMatchFound += HandleMatchFound;
        boardSystem.OnCellsCleared += HandleCellsCleared;
        boardSystem.OnGemsFallen += HandleGemsFallen;

        combatSystem.OnCombatEnd += HandleCombatEnd;

        // 初始化战斗
        var player = CreatePlayerUnit();
        var enemies = CreateEnemyUnits();
        combatSystem.Initialize(player, enemies);

        SetState(GameState.Idle);
    }

    /// <summary>
    /// 玩家尝试交换
    /// </summary>
    public void TrySwap(int x1, int y1, int x2, int y2)
    {
        if (currentState != GameState.Idle) return;

        if (boardSystem.TrySwap(x1, y1, x2, y2))
        {
            StartCoroutine(ProcessBoardCoroutine());
        }
    }

    private IEnumerator ProcessBoardCoroutine()
    {
        SetState(GameState.Processing);

        while (true)
        {
            // 处理匹配
            var matches = boardSystem.ProcessMatches();

            if (matches.Count == 0)
            {
                break;
            }

            // 触发战斗效果
            SetState(GameState.Combat);
            foreach (var match in matches)
            {
                combatSystem.ProcessMatchResult(match);
                yield return new WaitForSeconds(0.3f);
            }

            // 清除匹配的格子
            boardSystem.ClearMatchedCells(matches);
            yield return new WaitForSeconds(0.2f);

            // 下落填充
            boardSystem.FillBoard();
            yield return new WaitForSeconds(0.3f);

            SetState(GameState.Processing);
        }

        // 敌人回合
        if (currentState != GameState.Victory && currentState != GameState.Defeat)
        {
            SetState(GameState.EnemyTurn);
            combatSystem.EnemyTurn();
            yield return new WaitForSeconds(0.5f);
        }

        SetState(GameState.Idle);
    }

    private void HandleMatchFound(System.Collections.Generic.List<MatchResult> matches)
    {
        // 播放消除动画
    }

    private void HandleCellsCleared(System.Collections.Generic.List<Cell> cells)
    {
        // 播放清除动画
    }

    private void HandleGemsFallen(System.Collections.Generic.List<FallData> fallData)
    {
        // 播放下落动画
    }

    private void HandleCombatEnd()
    {
        // 判断胜负
        if (combatSystem.PlayerIsDead)
        {
            SetState(GameState.Defeat);
        }
        else
        {
            SetState(GameState.Victory);
        }
    }

    private void SetState(GameState newState)
    {
        currentState = newState;
        OnStateChanged?.Invoke(newState);
    }

    private CombatUnit CreatePlayerUnit()
    {
        return new CombatUnit
        {
            id = "player",
            maxHealth = 100,
            currentHealth = 100,
            attack = 10,
            defense = 5,
            isPlayer = true
        };
    }

    private System.Collections.Generic.List<CombatUnit> CreateEnemyUnits()
    {
        return new System.Collections.Generic.List<CombatUnit>
        {
            new CombatUnit
            {
                id = "enemy_1",
                maxHealth = 50,
                currentHealth = 50,
                attack = 8,
                defense = 2,
                isPlayer = false
            }
        };
    }
}
```

## 四、架构总结

```
┌─────────────────────────────────────────────────────────────┐
│                    三消游戏架构要点                         │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  1. 数据与表现分离                                          │
│     └─ BoardSystem 只处理数据逻辑                          │
│     └─ BoardView 只处理视觉表现                            │
│                                                             │
│  2. 事件驱动通信                                            │
│     └─ 系统间通过事件解耦                                  │
│     └─ OnMatchFound, OnDamageDealt 等                      │
│                                                             │
│  3. 状态机控制流程                                          │
│     └─ GameState 控制游戏阶段                              │
│     └─ 状态转换清晰可预测                                  │
│                                                             │
│  4. 协程处理异步                                            │
│     └─ 匹配-消除-下落-战斗 的异步链                        │
│     └─ 等待动画完成后继续                                  │
│                                                             │
│  5. 可扩展设计                                              │
│     └─ 新增GemType只需修改对应逻辑                         │
│     └─ 新增系统通过事件订阅                                │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```
