# 【系统架构】音频系统

> 核心价值：高效管理游戏音频，支持音效池、BGM切换、音量分组

## 二、音频管理器实现

### 2.1 核心实现

```csharp
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 音频管理器
/// </summary>
public class AudioManager : MonoBehaviour
{
    public static AudioManager Instance { get; private set; }

    [Header("音源配置")]
    [SerializeField] private int sfxPoolSize = 10;
    [SerializeField] private int maxConcurrentSFX = 8;

    [Header("AudioMixer")]
    [SerializeField] private UnityEngine.Audio.AudioMixer audioMixer;

    [Header("默认设置")]
    [SerializeField] private float defaultMasterVolume = 1f;
    [SerializeField] private float defaultBGMVolume = 0.8f;
    [SerializeField] private float defaultSFXVolume = 1f;

    // BGM AudioSource
    private AudioSource bgmSource;
    private AudioSource bgmSourceSecondary;  // 用于渐变

    // SFX 对象池
    private List<AudioSource> sfxPool;
    private Queue<AudioSource> availableSources;

    // 当前状态
    private string currentBGM;
    private Coroutine bgmFadeCoroutine;

    // 音量
    private float masterVolume;
    private float bgmVolume;
    private float sfxVolume;

    public event Action<string> OnBGMStarted;
    public event Action OnBGMStopped;

    private void Awake()
    {
        if (Instance != null && Instance != this)
        {
            Destroy(gameObject);
            return;
        }
        Instance = this;
        DontDestroyOnLoad(gameObject);

        InitializeAudioSources();
        LoadVolumeSettings();
    }

    private void InitializeAudioSources()
    {
        // 创建BGM音源
        var bgmObject = new GameObject("BGM_Source");
        bgmObject.transform.SetParent(transform);
        bgmSource = bgmObject.AddComponent<AudioSource>();
        bgmSource.loop = true;
        bgmSource.playOnAwake = false;

        var bgmObject2 = new GameObject("BGM_Source_Secondary");
        bgmObject2.transform.SetParent(transform);
        bgmSourceSecondary = bgmObject2.AddComponent<AudioSource>();
        bgmSourceSecondary.loop = true;
        bgmSourceSecondary.playOnAwake = false;

        // 创建SFX对象池
        sfxPool = new List<AudioSource>();
        availableSources = new Queue<AudioSource>();

        var poolObject = new GameObject("SFX_Pool");
        poolObject.transform.SetParent(transform);

        for (int i = 0; i < sfxPoolSize; i++)
        {
            var source = poolObject.AddComponent<AudioSource>();
            source.playOnAwake = false;
            sfxPool.Add(source);
            availableSources.Enqueue(source);
        }
    }

    #region BGM控制

    /// <summary>
    /// 播放BGM
    /// </summary>
    public void PlayBGM(AudioClip clip, float fadeDuration = 1f)
    {
        if (clip == null) return;

        string clipName = clip.name;

        // 相同BGM，不重复播放
        if (currentBGM == clipName && bgmSource.isPlaying)
        {
            return;
        }

        if (bgmFadeCoroutine != null)
        {
            StopCoroutine(bgmFadeCoroutine);
        }

        bgmFadeCoroutine = StartCoroutine(CrossfadeBGM(clip, fadeDuration));
        currentBGM = clipName;
        OnBGMStarted?.Invoke(clipName);
    }

    private System.Collections.IEnumerator CrossfadeBGM(AudioClip newClip, float duration)
    {
        // 设置新BGM到次要音源
        bgmSourceSecondary.clip = newClip;
        bgmSourceSecondary.volume = 0f;
        bgmSourceSecondary.Play();

        float elapsed = 0f;
        float startVolume = bgmSource.volume;

        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;
            float t = elapsed / duration;

            // 淡出旧BGM，淡入新BGM
            bgmSource.volume = Mathf.Lerp(startVolume, 0f, t) * bgmVolume * masterVolume;
            bgmSourceSecondary.volume = Mathf.Lerp(0f, bgmVolume * masterVolume, t);

            yield return null;
        }

        // 交换音源
        var temp = bgmSource;
        bgmSource = bgmSourceSecondary;
        bgmSourceSecondary = temp;

        bgmSourceSecondary.Stop();
        bgmSourceSecondary.clip = null;

        bgmFadeCoroutine = null;
    }

    /// <summary>
    /// 停止BGM
    /// </summary>
    public void StopBGM(float fadeDuration = 1f)
    {
        if (!bgmSource.isPlaying) return;

        if (bgmFadeCoroutine != null)
        {
            StopCoroutine(bgmFadeCoroutine);
        }

        bgmFadeCoroutine = StartCoroutine(FadeOutBGM(fadeDuration));
    }

    private System.Collections.IEnumerator FadeOutBGM(float duration)
    {
        float elapsed = 0f;
        float startVolume = bgmSource.volume;

        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;
            float t = elapsed / duration;
            bgmSource.volume = Mathf.Lerp(startVolume, 0f, t);

            yield return null;
        }

        bgmSource.Stop();
        bgmSource.clip = null;
        currentBGM = null;
        OnBGMStopped?.Invoke();
        bgmFadeCoroutine = null;
    }

    /// <summary>
    /// 暂停BGM
    /// </summary>
    public void PauseBGM()
    {
        if (bgmSource.isPlaying)
        {
            bgmSource.Pause();
        }
    }

    /// <summary>
    /// 恢复BGM
    /// </summary>
    public void ResumeBGM()
    {
        if (bgmSource.clip != null && !bgmSource.isPlaying)
        {
            bgmSource.UnPause();
        }
    }

    #endregion

    #region SFX控制

    /// <summary>
    /// 播放音效
    /// </summary>
    public AudioSource PlaySFX(AudioClip clip, float volumeScale = 1f, float pitch = 1f)
    {
        if (clip == null) return null;

        var source = GetAvailableSource();
        if (source == null)
        {
            Debug.LogWarning("没有可用的AudioSource");
            return null;
        }

        source.clip = clip;
        source.volume = sfxVolume * masterVolume * volumeScale;
        source.pitch = pitch;
        source.spatialBlend = 0f;  // 2D
        source.Play();

        // 播放完成后自动归还
        StartCoroutine(ReturnSourceWhenFinished(source, clip.length / pitch));

        return source;
    }

    /// <summary>
    /// 播放3D音效
    /// </summary>
    public AudioSource PlaySFX3D(AudioClip clip, Vector3 position, float volumeScale = 1f, float maxDistance = 50f)
    {
        if (clip == null) return null;

        var source = GetAvailableSource();
        if (source == null) return null;

        source.transform.position = position;
        source.clip = clip;
        source.volume = sfxVolume * masterVolume * volumeScale;
        source.spatialBlend = 1f;  // 3D
        source.maxDistance = maxDistance;
        source.rolloffMode = AudioRolloffMode.Linear;
        source.Play();

        StartCoroutine(ReturnSourceWhenFinished(source, clip.length));

        return source;
    }

    /// <summary>
    /// 停止所有音效
    /// </summary>
    public void StopAllSFX()
    {
        foreach (var source in sfxPool)
        {
            if (source.isPlaying)
            {
                source.Stop();
                source.clip = null;
            }
        }
    }

    private AudioSource GetAvailableSource()
    {
        if (availableSources.Count > 0)
        {
            return availableSources.Dequeue();
        }

        // 池中没有可用，尝试找一个没在播放的
        foreach (var source in sfxPool)
        {
            if (!source.isPlaying)
            {
                return source;
            }
        }

        return null;
    }

    private System.Collections.IEnumerator ReturnSourceWhenFinished(AudioSource source, float duration)
    {
        yield return new WaitForSeconds(duration + 0.1f);

        if (source != null)
        {
            source.clip = null;
            source.spatialBlend = 0f;
            availableSources.Enqueue(source);
        }
    }

    #endregion

    #region 音量控制

    public void SetMasterVolume(float volume)
    {
        masterVolume = Mathf.Clamp01(volume);
        ApplyVolumes();
        SaveVolumeSettings();
    }

    public void SetBGMVolume(float volume)
    {
        bgmVolume = Mathf.Clamp01(volume);
        ApplyVolumes();
        SaveVolumeSettings();
    }

    public void SetSFXVolume(float volume)
    {
        sfxVolume = Mathf.Clamp01(volume);
        SaveVolumeSettings();
    }

    private void ApplyVolumes()
    {
        if (audioMixer != null)
        {
            audioMixer.SetFloat("MasterVolume", VolumeToDecibel(masterVolume));
            audioMixer.SetFloat("BGMVolume", VolumeToDecibel(bgmVolume));
            audioMixer.SetFloat("SFXVolume", VolumeToDecibel(sfxVolume));
        }
        else
        {
            bgmSource.volume = bgmVolume * masterVolume;
        }
    }

    private float VolumeToDecibel(float volume)
    {
        // 0-1 转换为 -80dB - 0dB
        return volume > 0 ? Mathf.Log10(volume) * 20f : -80f;
    }

    private void SaveVolumeSettings()
    {
        PlayerPrefs.SetFloat("MasterVolume", masterVolume);
        PlayerPrefs.SetFloat("BGMVolume", bgmVolume);
        PlayerPrefs.SetFloat("SFXVolume", sfxVolume);
        PlayerPrefs.Save();
    }

    private void LoadVolumeSettings()
    {
        masterVolume = PlayerPrefs.GetFloat("MasterVolume", defaultMasterVolume);
        bgmVolume = PlayerPrefs.GetFloat("BGMVolume", defaultBGMVolume);
        sfxVolume = PlayerPrefs.GetFloat("SFXVolume", defaultSFXVolume);
        ApplyVolumes();
    }

    public float MasterVolume => masterVolume;
    public float BGMVolume => bgmVolume;
    public float SFXVolume => sfxVolume;

    #endregion
}
```

## 三、音效配置系统

### 3.1 音效数据

```csharp
using UnityEngine;
using System.Collections.Generic;

/// <summary>
/// 音效配置数据
/// </summary>
[CreateAssetMenu(fileName = "AudioConfig", menuName = "Game/AudioConfig")]
public class AudioConfig : ScriptableObject
{
    public List<SFXEntry> sfxEntries = new List<SFXEntry>();
    public List<BGMEntry> bgmEntries = new List<BGMEntry>();

    [System.Serializable]
    public class SFXEntry
    {
        public string id;
        public AudioClip clip;
        [Range(0f, 2f)] public float volume = 1f;
        [Range(0.5f, 1.5f)] public float pitch = 1f;
        [Range(0f, 1f)] public float pitchVariation = 0f;
        public bool is3D;
    }

    [System.Serializable]
    public class BGMEntry
    {
        public string id;
        public AudioClip clip;
        [Range(0f, 1f)] public float volume = 0.8f;
    }

    public SFXEntry GetSFX(string id)
    {
        return sfxEntries.Find(s => s.id == id);
    }

    public BGMEntry GetBGM(string id)
    {
        return bgmEntries.Find(b => b.id == id);
    }
}
```

### 3.2 便捷播放接口

```csharp
/// <summary>
/// 音频播放器扩展
/// </summary>
public static class AudioPlayer
{
    private static AudioConfig config;

    public static void Initialize(AudioConfig audioConfig)
    {
        config = audioConfig;
    }

    public static void PlaySFX(string id)
    {
        if (config == null || AudioManager.Instance == null) return;

        var entry = config.GetSFX(id);
        if (entry != null)
        {
            float pitch = entry.pitch;
            if (entry.pitchVariation > 0)
            {
                pitch += Random.Range(-entry.pitchVariation, entry.pitchVariation);
            }

            AudioManager.Instance.PlaySFX(entry.clip, entry.volume, pitch);
        }
    }

    public static void PlaySFX3D(string id, Vector3 position)
    {
        if (config == null || AudioManager.Instance == null) return;

        var entry = config.GetSFX(id);
        if (entry != null && entry.is3D)
        {
            AudioManager.Instance.PlaySFX3D(entry.clip, position, entry.volume);
        }
    }

    public static void PlayBGM(string id, float fadeDuration = 1f)
    {
        if (config == null || AudioManager.Instance == null) return;

        var entry = config.GetBGM(id);
        if (entry != null)
        {
            // 这里可以设置entry.volume
            AudioManager.Instance.PlayBGM(entry.clip, fadeDuration);
        }
    }
}

// 使用示例
AudioPlayer.PlaySFX("Jump");
AudioPlayer.PlaySFX3D("Explosion", hitPoint);
AudioPlayer.PlayBGM("BattleTheme");
```

## 四、使用示例

```csharp
// 初始化
var audioManager = gameObject.AddComponent<AudioManager>();

// 播放BGM（带渐变）
AudioManager.Instance.PlayBGM(bgmClip, fadeDuration: 2f);

// 播放音效
AudioManager.Instance.PlaySFX(jumpClip);

// 播放3D音效
AudioManager.Instance.PlaySFX3D(explosionClip, explosionPosition);

// 音量控制
AudioManager.Instance.SetMasterVolume(0.8f);
AudioManager.Instance.SetBGMVolume(0.5f);
AudioManager.Instance.SetSFXVolume(1f);

// 暂停/恢复
AudioManager.Instance.PauseBGM();
AudioManager.Instance.ResumeBGM();

// 停止BGM（带渐变）
AudioManager.Instance.StopBGM(fadeDuration: 1f);
```
