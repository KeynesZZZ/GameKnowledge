## 一、核心概述

### 1.1 定义/作用

- **策略模式**是一种行为型设计模式，将算法/行为封装为独立策略，实现行为与调用方解耦
- **核心价值**：Unity中可快速切换AI行为、技能效果，降低代码耦合，便于扩展新策略

### 1.2 适用场景

- **场景1**：AI角色的多行为切换（巡逻/追击/攻击/逃跑）
- **场景2**：技能系统的多效果组合（伤害/治疗/减速/护盾）
- **场景3**：寻路算法的动态切换（NavMesh/A*/直线）
- **避坑点**：行为逻辑简单且无需扩展时，无需使用（避免过度设计）

### 2.2 通用策略管理器

```csharp
using System;
using System.Collections.Generic;

/// <summary>
/// 策略管理器 - 支持延迟创建
/// </summary>
public class StrategyManager<TKey, TStrategy, TContext>
{
    private readonly Dictionary<TKey, TStrategy> strategies;
    private readonly Dictionary<TKey, Func<TStrategy>> factories;

    public StrategyManager()
    {
        strategies = new Dictionary<TKey, TStrategy>();
        factories = new Dictionary<TKey, Func<TStrategy>>();
    }

    /// <summary>
    /// 注册策略实例
    /// </summary>
    public void Register(TKey key, TStrategy strategy)
    {
        strategies[key] = strategy;
    }

    /// <summary>
    /// 注册策略工厂（延迟创建）
    /// </summary>
    public void RegisterFactory(TKey key, Func<TStrategy> factory)
    {
        factories[key] = factory;
    }

    /// <summary>
    /// 获取策略
    /// </summary>
    public TStrategy Get(TKey key)
    {
        if (strategies.TryGetValue(key, out var strategy))
        {
            return strategy;
        }

        if (factories.TryGetValue(key, out var factory))
        {
            strategy = factory();
            strategies[key] = strategy;
            return strategy;
        }

        return default;
    }

    /// <summary>
    /// 检查策略是否存在
    /// </summary>
    public bool HasStrategy(TKey key)
    {
        return strategies.ContainsKey(key) || factories.ContainsKey(key);
    }
}
```

### 2.3 关键要点

- **泛型设计**：使用 `TKey`, `TStrategy`, `TContext` 实现高度复用
- **延迟创建**：通过工厂函数实现按需创建，减少初始化开销
- **单一职责**：每个策略类仅封装对应行为逻辑，不包含业务判断

public interface IBehaviorStrategy

    void Execute(Enemy enemy);

### 3.3 具体策略实现

```csharp
using UnityEngine;

/// <summary>
/// 追击策略
/// </summary>
public class ChaseStrategy : IBehaviorStrategy
{
    private readonly float chaseSpeed;
    private readonly float stopDistance;

    public string Name => "Chase";

    public ChaseStrategy(float speed = 3f, float stopDist = 1f)
    {
        chaseSpeed = speed;
        stopDistance = stopDist;
    }

    public bool CanExecute(Enemy enemy)
    {
        return enemy.Target != null && enemy.DistanceToTarget > stopDistance;
    }

    public void Execute(Enemy enemy)
    {
        if (enemy.Target == null) return;

        Vector3 direction = (enemy.Target.position - enemy.transform.position).normalized;
        enemy.transform.position += direction * chaseSpeed * Time.deltaTime;
        enemy.transform.LookAt(enemy.Target);
    }
}

/// <summary>
/// 巡逻策略
/// </summary>
public class PatrolStrategy : IBehaviorStrategy
{
    private readonly List<Vector3> patrolPoints;
    private readonly float patrolSpeed;
    private int currentIndex;

    public string Name => "Patrol";

    public PatrolStrategy(List<Vector3> points, float speed = 2f)
    {
        patrolPoints = points;
        patrolSpeed = speed;
        currentIndex = 0;
    }

    public bool CanExecute(Enemy enemy)
    {
        return patrolPoints.Count > 0;
    }

    public void Execute(Enemy enemy)
    {
        if (patrolPoints.Count == 0) return;

        Vector3 target = patrolPoints[currentIndex];
        Vector3 direction = (target - enemy.transform.position).normalized;

        enemy.transform.position += direction * patrolSpeed * Time.deltaTime;

        if (Vector3.Distance(enemy.transform.position, target) < 0.5f)
        {
            currentIndex = (currentIndex + 1) % patrolPoints.Count;
        }
    }
}

/// <summary>
/// 攻击策略
/// </summary>
public class AttackStrategy : IBehaviorStrategy
{
    private readonly float attackRange;
    private readonly float attackCooldown;
    private float lastAttackTime;

    public string Name => "Attack";

    public AttackStrategy(float range = 2f, float cooldown = 1f)
    {
        attackRange = range;
        attackCooldown = cooldown;
    }

    public bool CanExecute(Enemy enemy)
    {
        return enemy.Target != null &&
               enemy.DistanceToTarget <= attackRange &&
               Time.time >= lastAttackTime + attackCooldown;
    }

    public void Execute(Enemy enemy)
    {
        if (enemy.Target == null) return;

        // 执行攻击
        enemy.PlayAttackAnimation();
        enemy.DealDamageToTarget(10);

        lastAttackTime = Time.time;
    }
}

/// <summary>
/// 逃跑策略
/// </summary>
public class FleeStrategy : IBehaviorStrategy
{
    private readonly float fleeSpeed;
    private readonly float fleeDistance;
    private readonly float healthThreshold;

    public string Name => "Flee";

    public FleeStrategy(float speed = 4f, float distance = 10f, float healthPercent = 0.3f)
    {
        fleeSpeed = speed;
        fleeDistance = distance;
        healthThreshold = healthPercent;
    }

    public bool CanExecute(Enemy enemy)
    {
        return enemy.HealthPercent <= healthThreshold;
    }

    public void Execute(Enemy enemy)
    {
        if (enemy.Target == null) return;

        // 逃离目标
        Vector3 fleeDirection = (enemy.transform.position - enemy.Target.position).normalized;
        Vector3 fleePosition = enemy.transform.position + fleeDirection * fleeDistance;

        enemy.transform.position = Vector3.MoveTowards(
            enemy.transform.position,
            fleePosition,
            fleeSpeed * Time.deltaTime
        );
    }
}
```

### 3.4 AI控制器

```csharp
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// AI控制器 - 使用策略模式
/// </summary>
public class AIController : MonoBehaviour
{
    [Header("配置")]
    [SerializeField] private float detectionRange = 10f;
    [SerializeField] private float attackRange = 2f;

    private Dictionary<string, IBehaviorStrategy> strategies;
    private IBehaviorStrategy currentStrategy;
    private Enemy enemy;

    private void Awake()
    {
        enemy = GetComponent<Enemy>();
        InitializeStrategies();
    }

    private void InitializeStrategies()
    {
        strategies = new Dictionary<string, IBehaviorStrategy>
        {
            { "Patrol", new PatrolStrategy(GetPatrolPoints(), 2f) },
            { "Chase", new ChaseStrategy(3f, attackRange) },
            { "Attack", new AttackStrategy(attackRange, 1f) },
            { "Flee", new FleeStrategy(4f, 10f, 0.3f) }
        };
    }

    private void Update()
    {
        // 策略选择逻辑
        SelectStrategy();
        currentStrategy?.Execute(enemy);
    }

    private void SelectStrategy()
    {
        // 优先级：逃跑 > 攻击 > 追击 > 巡逻
        if (strategies["Flee"].CanExecute(enemy))
        {
            SetStrategy("Flee");
        }
        else if (strategies["Attack"].CanExecute(enemy))
        {
            SetStrategy("Attack");
        }
        else if (strategies["Chase"].CanExecute(enemy))
        {
            SetStrategy("Chase");
        }
        else
        {
            SetStrategy("Patrol");
        }
    }

    public void SetStrategy(string strategyName)
    {
        if (strategies.TryGetValue(strategyName, out var strategy))
        {
            if (currentStrategy != strategy)
            {
                currentStrategy = strategy;
                Debug.Log($"AI策略切换: {strategy.Name}");
            }
        }
    }

    private List<Vector3> GetPatrolPoints()
    {
        return new List<Vector3>
        {
            transform.position + Vector3.forward * 5,
            transform.position + Vector3.right * 5,
            transform.position + Vector3.back * 5,
            transform.position + Vector3.left * 5
        };
    }
}
```

## 五、常见问题/排错技巧

1. **策略切换无效果** → 排查 `CanExecute` 方法的条件是否写反，例：距离判断用 `>=` 而非 `<=`
2. **策略对象重复创建** → 排查是否在 `Update` 中注册策略，应移至 `Awake`/`Start`
3. **Unity中物体移动卡顿** → 排查是否将移动逻辑写在 `Update`，且未乘 `Time.deltaTime`
4. **策略状态未重置** → 在策略切换时调用重置方法，或在策略内部维护状态

## 六、总结/个人心得

- **核心收获**：策略模式的核心是"封装变化、面向接口编程"，在Unity中可大幅提升代码的扩展性和可维护性
- **个人技巧**：将常用策略封装为通用类，通过构造函数传参实现定制化，避免重复写相似策略
- **后续拓展**：结合事件总线实现策略的动态注册与注销，支持热更新策略
