## 二、输入管理器实现

### 2.1 基于新输入系统的实现

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.InputSystem;

/// <summary>
/// 输入管理器 - 基于Unity新输入系统
/// </summary>
public class InputManager : MonoBehaviour
{
    public static InputManager Instance { get; private set; }

    [Header("输入配置")]
    [SerializeField] private InputActionAsset inputActions;

    // 输入事件
    public event Action<Vector2> OnMove;
    public event Action OnJumpPressed;
    public event Action OnJumpReleased;
    public event Action OnAttackPressed;
    public event Action OnInteractPressed;
    public event Action OnPausePressed;

    // 当前输入状态
    public Vector2 MoveInput { get; private set; }
    public bool IsJumpHeld { get; private set; }
    public bool IsAttackHeld { get; private set; }

    // 输入启用状态
    public bool InputEnabled { get; private set; } = true;

    private InputAction moveAction;
    private InputAction jumpAction;
    private InputAction attackAction;
    private InputAction interactAction;
    private InputAction pauseAction;

    private void Awake()
    {
        if (Instance != null && Instance != this)
        {
            Destroy(gameObject);
            return;
        }
        Instance = this;
        DontDestroyOnLoad(gameObject);

        InitializeInputActions();
    }

    private void InitializeInputActions()
    {
        if (inputActions == null)
        {
            Debug.LogError("InputActionAsset未配置");
            return;
        }

        var gameplayMap = inputActions.FindActionMap("Gameplay");
        if (gameplayMap == null) return;

        moveAction = gameplayMap.FindAction("Move");
        jumpAction = gameplayMap.FindAction("Jump");
        attackAction = gameplayMap.FindAction("Attack");
        interactAction = gameplayMap.FindAction("Interact");
        pauseAction = gameplayMap.FindAction("Pause");

        // 绑定事件
        moveAction.performed += ctx => OnMoveInput(ctx);
        moveAction.canceled += ctx => OnMoveInput(ctx);

        jumpAction.performed += ctx => OnJumpInput(true);
        jumpAction.canceled += ctx => OnJumpInput(false);

        attackAction.performed += ctx => OnAttackInput(true);
        attackAction.canceled += ctx => OnAttackInput(false);

        interactAction.performed += ctx => OnInteractInput();
        pauseAction.performed += ctx => OnPauseInput();
    }

    private void OnEnable()
    {
        inputActions?.Enable();
    }

    private void OnDisable()
    {
        inputActions?.Disable();
    }

    private void OnMoveInput(InputAction.CallbackContext ctx)
    {
        if (!InputEnabled) return;

        MoveInput = ctx.ReadValue<Vector2>();
        OnMove?.Invoke(MoveInput);
    }

    private void OnJumpInput(bool pressed)
    {
        if (!InputEnabled) return;

        IsJumpHeld = pressed;
        if (pressed)
        {
            OnJumpPressed?.Invoke();
        }
        else
        {
            OnJumpReleased?.Invoke();
        }
    }

    private void OnAttackInput(bool pressed)
    {
        if (!InputEnabled) return;

        IsAttackHeld = pressed;
        if (pressed)
        {
            OnAttackPressed?.Invoke();
        }
    }

    private void OnInteractInput()
    {
        if (!InputEnabled) return;
        OnInteractPressed?.Invoke();
    }

    private void OnPauseInput()
    {
        // 暂停输入始终启用
        OnPausePressed?.Invoke();
    }

    /// <summary>
    /// 启用/禁用输入
    /// </summary>
    public void SetInputEnabled(bool enabled)
    {
        InputEnabled = enabled;
        if (!enabled)
        {
            MoveInput = Vector2.zero;
            IsJumpHeld = false;
            IsAttackHeld = false;
        }
    }

    /// <summary>
    /// 切换输入方案
    /// </summary>
    public void SwitchControlScheme(string scheme)
    {
        // 切换输入方案（如键盘/手柄）
        var playerInput = GetComponent<PlayerInput>();
        if (playerInput != null)
        {
            playerInput.SwitchCurrentControlScheme(scheme);
        }
    }
}
```

### 2.2 基于旧输入系统的实现

```csharp
using System;
using UnityEngine;

/// <summary>
/// 输入管理器 - 基于旧输入系统
/// </summary>
public class LegacyInputManager : MonoBehaviour
{
    public static LegacyInputManager Instance { get; private set; }

    [Header("按键配置")]
    [SerializeField] private KeyCode jumpKey = KeyCode.Space;
    [SerializeField] private KeyCode attackKey = KeyCode.Mouse0;
    [SerializeField] private KeyCode interactKey = KeyCode.E;
    [SerializeField] private KeyCode pauseKey = KeyCode.Escape;

    // 输入事件
    public event Action<Vector2> OnMove;
    public event Action OnJumpPressed;
    public event Action OnJumpReleased;
    public event Action OnAttackPressed;
    public event Action OnInteractPressed;
    public event Action OnPausePressed;

    // 当前状态
    public Vector2 MoveInput { get; private set; }
    public bool IsJumpHeld { get; private set; }
    public bool IsAttackHeld { get; private set; }

    public bool InputEnabled { get; private set; } = true;

    private void Awake()
    {
        if (Instance != null && Instance != this)
        {
            Destroy(gameObject);
            return;
        }
        Instance = this;
        DontDestroyOnLoad(gameObject);

        LoadKeyBindings();
    }

    private void Update()
    {
        if (!InputEnabled) return;

        // 移动输入
        Vector2 move = new Vector2(
            Input.GetAxisRaw("Horizontal"),
            Input.GetAxisRaw("Vertical")
        );

        if (move != MoveInput)
        {
            MoveInput = move;
            OnMove?.Invoke(move);
        }

        // 跳跃输入
        if (Input.GetKeyDown(jumpKey))
        {
            IsJumpHeld = true;
            OnJumpPressed?.Invoke();
        }
        if (Input.GetKeyUp(jumpKey))
        {
            IsJumpHeld = false;
            OnJumpReleased?.Invoke();
        }

        // 攻击输入
        if (Input.GetKeyDown(attackKey))
        {
            IsAttackHeld = true;
            OnAttackPressed?.Invoke();
        }
        if (Input.GetKeyUp(attackKey))
        {
            IsAttackHeld = false;
        }

        // 交互输入
        if (Input.GetKeyDown(interactKey))
        {
            OnInteractPressed?.Invoke();
        }

        // 暂停输入（始终检测）
        if (Input.GetKeyDown(pauseKey))
        {
            OnPausePressed?.Invoke();
        }
    }

    public void SetInputEnabled(bool enabled)
    {
        InputEnabled = enabled;
        if (!enabled)
        {
            MoveInput = Vector2.zero;
            IsJumpHeld = false;
            IsAttackHeld = false;
        }
    }

    #region 按键重映射

    public void SetJumpKey(KeyCode newKey)
    {
        jumpKey = newKey;
        SaveKeyBindings();
    }

    public void SetAttackKey(KeyCode newKey)
    {
        attackKey = newKey;
        SaveKeyBindings();
    }

    private void SaveKeyBindings()
    {
        PlayerPrefs.SetInt("JumpKey", (int)jumpKey);
        PlayerPrefs.SetInt("AttackKey", (int)attackKey);
        PlayerPrefs.SetInt("InteractKey", (int)interactKey);
        PlayerPrefs.Save();
    }

    private void LoadKeyBindings()
    {
        if (PlayerPrefs.HasKey("JumpKey"))
        {
            jumpKey = (KeyCode)PlayerPrefs.GetInt("JumpKey");
        }
        if (PlayerPrefs.HasKey("AttackKey"))
        {
            attackKey = (KeyCode)PlayerPrefs.GetInt("AttackKey");
        }
        if (PlayerPrefs.HasKey("InteractKey"))
        {
            interactKey = (KeyCode)PlayerPrefs.GetInt("InteractKey");
        }
    }

    #endregion
}
```

## 三、输入缓冲系统

### 3.1 输入缓冲（格斗/动作游戏）

```csharp
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 输入缓冲器 - 用于格斗/动作游戏的精确输入
/// </summary>
public class InputBuffer : MonoBehaviour
{
    [System.Serializable]
    public class BufferedInput
    {
        public string actionName;
        public float timestamp;
        public bool consumed;

        public BufferedInput(string name, float time)
        {
            actionName = name;
            timestamp = time;
            consumed = false;
        }

        public bool IsExpired(float bufferTime)
        {
            return Time.time - timestamp > bufferTime;
        }
    }

    [Header("配置")]
    [SerializeField] private float defaultBufferTime = 0.2f;

    private List<BufferedInput> buffer = new List<BufferedInput>();

    /// <summary>
    /// 记录输入
    /// </summary>
    public void RecordInput(string actionName)
    {
        buffer.Add(new BufferedInput(actionName, Time.time));
    }

    /// <summary>
    /// 检查是否有有效的缓冲输入
    /// </summary>
    public bool HasBufferedInput(string actionName, float? bufferTime = null)
    {
        float time = bufferTime ?? defaultBufferTime;

        // 清理过期输入
        buffer.RemoveAll(i => i.IsExpired(time));

        // 查找未消费的有效输入
        var input = buffer.Find(i => i.actionName == actionName && !i.consumed);
        return input != null;
    }

    /// <summary>
    /// 消费缓冲输入
    /// </summary>
    public bool ConsumeInput(string actionName, float? bufferTime = null)
    {
        float time = bufferTime ?? defaultBufferTime;
        buffer.RemoveAll(i => i.IsExpired(time));

        var input = buffer.Find(i => i.actionName == actionName && !i.consumed);
        if (input != null)
        {
            input.consumed = true;
            return true;
        }
        return false;
    }

    /// <summary>
    /// 清空缓冲
    /// </summary>
    public void ClearBuffer()
    {
        buffer.Clear();
    }

    private void LateUpdate()
    {
        // 定期清理过期输入
        buffer.RemoveAll(i => i.IsExpired(defaultBufferTime * 2));
    }
}

// 使用示例
public class PlayerCombat : MonoBehaviour
{
    private InputBuffer inputBuffer;

    private void OnAttackInput()
    {
        inputBuffer.RecordInput("Attack");
    }

    private void Update()
    {
        // 在攻击动画的特定窗口检查输入
        if (canCombo && inputBuffer.ConsumeInput("Attack"))
        {
            PerformComboAttack();
        }
    }
}
```

## 四、按键重映射UI

### 4.1 按键绑定界面

```csharp
using UnityEngine;
using UnityEngine.UI;
using TMPro;

/// <summary>
/// 按键绑定UI控制器
/// </summary>
public class KeyBindingUI : MonoBehaviour
{
    [System.Serializable]
    public class BindingSlot
    {
        public string actionName;
        public TMP_Text actionLabel;
        public Button keyButton;
        public TMP_Text keyLabel;
    }

    [Header("绑定槽")]
    [SerializeField] private BindingSlot[] bindings;

    [Header("等待按键提示")]
    [SerializeField] private GameObject waitingPanel;
    [SerializeField] private TMP_Text waitingText;

    private string currentRebindingAction;
    private bool isWaitingForKey;

    private void Start()
    {
        InitializeBindings();
        UpdateAllKeyLabels();
    }

    private void InitializeBindings()
    {
        foreach (var slot in bindings)
        {
            string action = slot.actionName;
            slot.keyButton.onClick.AddListener(() => StartRebinding(action));
        }
    }

    private void StartRebinding(string actionName)
    {
        currentRebindingAction = actionName;
        isWaitingForKey = true;

        waitingPanel.SetActive(true);
        waitingText.text = $"按下新的按键绑定 {actionName}...";

        // 禁用游戏输入
        InputManager.Instance.SetInputEnabled(false);
    }

    private void Update()
    {
        if (!isWaitingForKey) return;

        // 检测任意按键
        foreach (KeyCode key in System.Enum.GetValues(typeof(KeyCode)))
        {
            if (Input.GetKeyDown(key))
            {
                CompleteRebinding(key);
                break;
            }
        }

        // ESC取消
        if (Input.GetKeyDown(KeyCode.Escape))
        {
            CancelRebinding();
        }
    }

    private void CompleteRebinding(KeyCode newKey)
    {
        // 应用新绑定
        switch (currentRebindingAction)
        {
            case "Jump":
                LegacyInputManager.Instance.SetJumpKey(newKey);
                break;
            case "Attack":
                LegacyInputManager.Instance.SetAttackKey(newKey);
                break;
            // ... 其他动作
        }

        UpdateKeyLabel(currentRebindingAction, newKey);
        CancelRebinding();
    }

    private void CancelRebinding()
    {
        isWaitingForKey = false;
        currentRebindingAction = null;
        waitingPanel.SetActive(false);
        InputManager.Instance.SetInputEnabled(true);
    }

    private void UpdateAllKeyLabels()
    {
        // 更新所有按键显示
    }

    private void UpdateKeyLabel(string action, KeyCode key)
    {
        var slot = System.Array.Find(bindings, b => b.actionName == action);
        if (slot != null)
        {
            slot.keyLabel.text = key.ToString();
        }
    }

    /// <summary>
    /// 重置为默认
    /// </summary>
    public void ResetToDefault()
    {
        PlayerPrefs.DeleteKey("JumpKey");
        PlayerPrefs.DeleteKey("AttackKey");
        PlayerPrefs.DeleteKey("InteractKey");
        // 重新加载
        UnityEngine.SceneManagement.SceneManager.LoadScene(0);
    }
}
```

## 五、使用示例

```csharp
// 订阅输入事件
InputManager.Instance.OnJumpPressed += OnJump;
InputManager.Instance.OnAttackPressed += OnAttack;

private void OnJump()
{
    if (isGrounded)
    {
        Jump();
    }
}

private void OnAttack()
{
    if (canAttack)
    {
        Attack();
    }
}

// 查询输入状态
private void Update()
{
    Vector2 move = InputManager.Instance.MoveInput;
    Move(move);

    if (InputManager.Instance.IsJumpHeld && isGrounded)
    {
        // 持续按住跳跃
    }
}

// 禁用输入（如过场动画）
InputManager.Instance.SetInputEnabled(false);

// 恢复输入
InputManager.Instance.SetInputEnabled(true);
```
