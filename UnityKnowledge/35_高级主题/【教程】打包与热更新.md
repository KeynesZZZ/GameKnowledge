### 1.2 Addressables基础配置

```csharp
using UnityEngine;
using UnityEngine.AddressableAssets;
using UnityEngine.ResourceManagement.AsyncOperations;
using System.Collections.Generic;

/// <summary>
/// Addressables资源管理器
/// </summary>
public class AddressablesManager : MonoBehaviour
{
    private static AddressablesManager instance;
    public static AddressablesManager Instance => instance;

    // 缓存已加载的资源
    private Dictionary<string, AsyncOperationHandle> loadedAssets = new Dictionary<string, AsyncOperationHandle>();
    private Dictionary<string, int> referenceCounts = new Dictionary<string, int>();

    private void Awake()
    {
        instance = this;
    }

    /// <summary>
    /// 同步加载资源（实际上是异步，返回句柄）
    /// </summary>
    public AsyncOperationHandle<T> LoadAssetAsync<T>(string address) where T : Object
    {
        // 检查是否已加载
        if (loadedAssets.TryGetValue(address, out var existingHandle))
        {
            referenceCounts[address]++;
            return existingHandle.Convert<T>();
        }

        // 加载新资源
        var handle = Addressables.LoadAssetAsync<T>(address);
        loadedAssets[address] = handle;
        referenceCounts[address] = 1;

        return handle;
    }

    /// <summary>
    /// 加载并等待完成
    /// </summary>
    public System.Collections.IEnumerator LoadAssetCoroutine<T>(string address, System.Action<T> onComplete) where T : Object
    {
        var handle = LoadAssetAsync<T>(address);
        yield return handle;

        if (handle.Status == AsyncOperationStatus.Succeeded)
        {
            onComplete?.Invoke(handle.Result);
        }
        else
        {
            Debug.LogError($"Failed to load asset: {address}");
            onComplete?.Invoke(null);
        }
    }

    /// <summary>
    /// 释放资源
    /// </summary>
    public void ReleaseAsset(string address)
    {
        if (!referenceCounts.ContainsKey(address))
            return;

        referenceCounts[address]--;

        if (referenceCounts[address] <= 0)
        {
            if (loadedAssets.TryGetValue(address, out var handle))
            {
                Addressables.Release(handle);
                loadedAssets.Remove(address);
            }
            referenceCounts.Remove(address);
        }
    }

    /// <summary>
    /// 释放所有资源
    /// </summary>
    public void ReleaseAll()
    {
        foreach (var handle in loadedAssets.Values)
        {
            Addressables.Release(handle);
        }
        loadedAssets.Clear();
        referenceCounts.Clear();
    }
}
```

### 1.3 场景加载

```csharp
using UnityEngine;
using UnityEngine.AddressableAssets;
using UnityEngine.ResourceManagement.AsyncOperations;
using UnityEngine.ResourceManagement.ResourceProviders;
using System.Collections;

/// <summary>
/// Addressables场景管理器
/// </summary>
public class AddressablesSceneManager : MonoBehaviour
{
    [Header("Scene References")]
    [SerializeField] private AssetReference mainMenuScene;
    [SerializeField] private AssetReference gameScene;
    [SerializeField] private AssetReference loadingScene;

    private AsyncOperationHandle<SceneInstance> currentSceneHandle;
    private SceneInstance? currentScene;

    /// <summary>
    /// 切换场景（带加载界面）
    /// </summary>
    public IEnumerator LoadSceneAsync(AssetReference sceneReference, System.Action onComplete = null)
    {
        // 显示加载界面
        // LoadingScreen.Show();

        // 卸载当前场景
        if (currentScene.HasValue)
        {
            var unloadOp = Addressables.UnloadSceneAsync(currentSceneHandle);
            yield return unloadOp;
        }

        // 加载新场景
        var loadOp = Addressables.LoadSceneAsync(sceneReference, LoadSceneMode.Additive);
        loadOp.Completed += (op) =>
        {
            if (op.Status == AsyncOperationStatus.Succeeded)
            {
                currentSceneHandle = op;
                currentScene = op.Result;
            }
        };

        while (!loadOp.IsDone)
        {
            // 更新进度条
            float progress = loadOp.PercentComplete;
            // LoadingScreen.SetProgress(progress);
            yield return null;
        }

        // 隐藏加载界面
        // LoadingScreen.Hide();

        onComplete?.Invoke();
    }

    /// <summary>
    /// 预加载场景资源
    /// </summary>
    public IEnumerator PreloadSceneAssets(AssetReference sceneReference)
    {
        // 获取场景依赖
        var dependencies = Addressables.LoadAssetAsync<IList<Object>>(sceneReference);
        yield return dependencies;

        Debug.Log($"Preloaded {dependencies.Result.Count} dependencies for scene");
    }
}
```

### 1.4 资源更新检测

```csharp
using UnityEngine;
using UnityEngine.AddressableAssets;
using UnityEngine.ResourceManagement.AsyncOperations;
using System.Collections.Generic;

/// <summary>
/// Addressables更新管理器
/// </summary>
public class AddressablesUpdateManager : MonoBehaviour
{
    [Header("Settings")]
    [SerializeField] private bool checkForUpdatesOnStart = true;
    [SerializeField] private string updateLabel = "Remote";

    public System.Action<float> OnDownloadProgress;
    public System.Action<bool> OnUpdateComplete;

    private void Start()
    {
        if (checkForUpdatesOnStart)
        {
            CheckForUpdates();
        }
    }

    /// <summary>
    /// 检查资源更新
    /// </summary>
    public void CheckForUpdates()
    {
        StartCoroutine(CheckForUpdatesCoroutine());
    }

    private IEnumerator CheckForUpdatesCoroutine()
    {
        // 初始化Addressables（如果尚未初始化）
        var initOp = Addressables.InitializeAsync();
        yield return initOp;

        // 检查目录更新
        var checkOp = Addressables.CheckForCatalogUpdates();
        yield return checkOp;

        if (checkOp.Status == AsyncOperationStatus.Succeeded)
        {
            var catalogs = checkOp.Result;

            if (catalogs != null && catalogs.Count > 0)
            {
                Debug.Log($"Found {catalogs.Count} catalog updates");
                StartCoroutine(DownloadUpdates(catalogs));
            }
            else
            {
                Debug.Log("No updates available");
                OnUpdateComplete?.Invoke(false);
            }
        }
        else
        {
            Debug.LogError("Failed to check for updates");
            OnUpdateComplete?.Invoke(false);
        }
    }

    /// <summary>
    /// 下载更新
    /// </summary>
    private IEnumerator DownloadUpdates(List<string> catalogs)
    {
        // 更新目录
        var updateOp = Addressables.UpdateCatalogs(catalogs);
        yield return updateOp;

        if (updateOp.Status != AsyncOperationStatus.Succeeded)
        {
            Debug.LogError("Failed to update catalogs");
            OnUpdateComplete?.Invoke(false);
            yield break;
        }

        // 获取下载大小
        var sizeOp = Addressables.GetDownloadSizeAsync(updateLabel);
        yield return sizeOp;

        long totalSize = sizeOp.Result;
        Debug.Log($"Download size: {totalSize / 1024 / 1024} MB");

        if (totalSize > 0)
        {
            // 开始下载
            var downloadOp = Addressables.DownloadDependenciesAsync(updateLabel);

            while (!downloadOp.IsDone)
            {
                float progress = downloadOp.PercentComplete;
                OnDownloadProgress?.Invoke(progress);
                Debug.Log($"Download progress: {progress * 100:F0}%");
                yield return null;
            }

            if (downloadOp.Status == AsyncOperationStatus.Succeeded)
            {
                Debug.Log("Download complete!");
                OnUpdateComplete?.Invoke(true);
            }
            else
            {
                Debug.LogError("Download failed!");
                OnUpdateComplete?.Invoke(false);
            }

            Addressables.Release(downloadOp);
        }
        else
        {
            OnUpdateComplete?.Invoke(false);
        }

        Addressables.Release(sizeOp);
    }
}
```

### 2.2 版本管理

```csharp
using UnityEngine;
using System;

/// <summary>
/// 版本管理器
/// </summary>
public class VersionManager : MonoBehaviour
{
    [Serializable]
    public class VersionInfo
    {
        public string version;
        public string resourceVersion;
        public int codeVersion;
        public string updateUrl;
        public bool forceUpdate;
        public string updateMessage;
    }

    [Header("Current Version")]
    [SerializeField] private string currentVersion = "1.0.0";
    [SerializeField] private int currentCodeVersion = 1;

    [Header("Settings")]
    [SerializeField] private string versionCheckUrl = "https://your-server.com/version.json";

    public VersionInfo LatestVersion { get; private set; }

    /// <summary>
    /// 检查版本更新
    /// </summary>
    public IEnumerator CheckVersion(System.Action<VersionInfo, bool> onComplete)
    {
        using (UnityEngine.Networking.UnityWebRequest request = UnityEngine.Networking.UnityWebRequest.Get(versionCheckUrl))
        {
            yield return request.SendWebRequest();

            if (request.result == UnityEngine.Networking.UnityWebRequest.Result.Success)
            {
                try
                {
                    LatestVersion = JsonUtility.FromJson<VersionInfo>(request.downloadHandler.text);

                    bool needUpdate = CompareVersions(LatestVersion.version, currentVersion) > 0 ||
                                     LatestVersion.codeVersion > currentCodeVersion;

                    onComplete?.Invoke(LatestVersion, needUpdate);
                }
                catch (Exception e)
                {
                    Debug.LogError($"Failed to parse version info: {e.Message}");
                    onComplete?.Invoke(null, false);
                }
            }
            else
            {
                Debug.LogError($"Version check failed: {request.error}");
                onComplete?.Invoke(null, false);
            }
        }
    }

    /// <summary>
    /// 比较版本号
    /// </summary>
    private int CompareVersions(string version1, string version2)
    {
        string[] parts1 = version1.Split('.');
        string[] parts2 = version2.Split('.');

        int maxLength = Mathf.Max(parts1.Length, parts2.Length);

        for (int i = 0; i < maxLength; i++)
        {
            int v1 = i < parts1.Length ? int.Parse(parts1[i]) : 0;
            int v2 = i < parts2.Length ? int.Parse(parts2[i]) : 0;

            if (v1 > v2) return 1;
            if (v1 < v2) return -1;
        }

        return 0;
    }

    /// <summary>
    /// 打开应用商店
    /// </summary>
    public void OpenAppStore()
    {
#if UNITY_ANDROID
        // 打开Google Play
        Application.OpenURL($"market://details?id={Application.identifier}");
#elif UNITY_IOS
        // 打开App Store
        Application.OpenURL("itms-apps://itunes.apple.com/app/idYOUR_APP_ID");
#endif
    }
}
```

### 2.3 HybridCLR集成

```csharp
using UnityEngine;
using System;
using System.Collections;
using System.Collections.Generic;
using System.IO;
using System.Reflection;

#if ENABLE_HYBRIDCLR
using HybridCLR;
using HybridCLR.Data;
#endif

/// <summary>
/// HybridCLR热更新管理器
/// </summary>
public class HybridCLRMgr : MonoBehaviour
{
    /*
    ========== HybridCLR简介 ==========

    HybridCLR是Unity的C#热更新解决方案：
    - 支持全C#热更新
    - 无需学习Lua
    - 支持泛型、反射等特性
    - 性能接近原生

    ========== 工作流程 ==========

    1. 将热更新代码编译为DLL
    2. 将DLL打包为Addressables资源
    3. 运行时下载DLL
    4. 加载并执行热更新代码

    ========== 配置步骤 ==========

    1. 安装HybridCLR包
    2. 配置热更新程序集
    3. 生成AOT DLL和补充元数据
    4. 打包热更新DLL
    */

    [Header("Settings")]
    [SerializeField] private bool enableHotUpdate = true;
    [SerializeField] private string hotUpdateDllName = "HotUpdate";

    /// <summary>
    /// 加载热更新程序集
    /// </summary>
    public IEnumerator LoadHotUpdateAssembly(System.Action<Assembly> onComplete)
    {
        if (!enableHotUpdate)
        {
            onComplete?.Invoke(null);
            yield break;
        }

#if ENABLE_HYBRIDCLR
        // 加载AOT元数据
        yield return LoadAOTMetadata();

        // 加载热更新DLL
        var dllPath = Path.Combine(Application.persistentDataPath, "HotUpdate", $"{hotUpdateDllName}.dll");

        if (File.Exists(dllPath))
        {
            byte[] dllBytes = File.ReadAllBytes(dllPath);
            Assembly hotUpdateAssembly = Assembly.Load(dllBytes);

            Debug.Log("Hot update assembly loaded successfully");
            onComplete?.Invoke(hotUpdateAssembly);
        }
        else
        {
            // 从Addressables加载
            yield return LoadFromAddressables(onComplete);
        }
#else
        Debug.LogWarning("HybridCLR is not enabled");
        onComplete?.Invoke(null);
#endif
    }

#if ENABLE_HYBRIDCLR
    private IEnumerator LoadAOTMetadata()
    {
        // 加载AOT补充元数据
        HomologousImageManager imageManager = HomologousImageManager.Instance;

        string[] aotDlls = new string[]
        {
            "mscorlib",
            "System",
            "System.Core",
            "UnityEngine.CoreModule"
        };

        foreach (var dllName in aotDlls)
        {
            var handle = UnityEngine.AddressableAssets.Addressables.LoadAssetAsync<TextAsset>($"AOT/{dllName}");
            yield return handle;

            if (handle.Status == UnityEngine.ResourceManagement.AsyncOperations.AsyncOperationStatus.Succeeded)
            {
                byte[] dllBytes = handle.Result.bytes;
                imageManager.LoadImage(dllBytes);
                Debug.Log($"Loaded AOT metadata: {dllName}");
            }
        }
    }
#endif

    private IEnumerator LoadFromAddressables(System.Action<Assembly> onComplete)
    {
        var handle = UnityEngine.AddressableAssets.Addressables.LoadAssetAsync<TextAsset>($"HotUpdate/{hotUpdateDllName}.dll.bytes");
        yield return handle;

        if (handle.Status == UnityEngine.ResourceManagement.AsyncOperations.AsyncOperationStatus.Succeeded)
        {
            byte[] dllBytes = handle.Result.bytes;
            Assembly assembly = Assembly.Load(dllBytes);
            onComplete?.Invoke(assembly);
        }
        else
        {
            Debug.LogError($"Failed to load hot update DLL: {handle.OperationException}");
            onComplete?.Invoke(null);
        }
    }
}
```

```csharp
    /// <summary>
    /// 获取纹理总大小
    /// </summary>
    [MenuItem("Tools/Package/Analyze Textures")]
    public static void AnalyzeTextures()
    {
        var textures = Selection.GetFiltered<Texture2D>(SelectionMode.DeepAssets);
        long totalSize = 0;

        foreach (var tex in textures)
        {
            var importer = AssetImporter.GetAtPath(AssetDatabase.GetAssetPath(tex)) as TextureImporter;
            if (importer != null)
            {
                long size = EstimateTextureSize(tex, importer);
                totalSize += size;
                Debug.Log($"{tex.name}: {tex.width}x{tex.height} ({importer.GetAutomaticFormat("Android")}) = {size / 1024}KB");
            }
        }

        Debug.Log($"Total texture size: {totalSize / 1024 / 1024}MB");
    }

    private static long EstimateTextureSize(Texture2D tex, TextureImporter importer)
    {
        int maxTextureSize = importer.maxTextureSize;
        int actualWidth = Mathf.Min(tex.width, maxTextureSize);
        int actualHeight = Mathf.Min(tex.height, maxTextureSize);

        // 根据格式估算
        TextureFormat format = importer.GetAutomaticFormat("Android");
        int bytesPerPixel = GetBytesPerPixel(format);

        long size = (long)actualWidth * actualHeight * bytesPerPixel;

        if (importer.mipmapEnabled)
            size = (long)(size * 1.33);

        return size;
    }

    private static int GetBytesPerPixel(TextureFormat format)
    {
        return format switch
        {
            TextureFormat.ASTC_4x4 => 1,
            TextureFormat.ASTC_6x6 => 1,
            TextureFormat.ASTC_8x8 => 1,
            TextureFormat.ETC_RGB4 => 1,
            TextureFormat.ETC2_RGBA8 => 1,
            TextureFormat.RGBA32 => 4,
            TextureFormat.RGB24 => 3,
            _ => 4
        };
    }
}
```

```csharp
using UnityEngine;

/// <summary>
/// 分包加载管理器

{
    /*
    ========== 分包策略 ==========

    首包（必装）：
    ├── 启动场景
    ├── 主菜单UI
    ├── 核心游戏逻辑

    public class PackageInfo

        public string addressLabel;
        public long sizeBytes;
        public bool isDownloaded;
        public bool isRequired;
    }

    [Header("Packages")]
    [SerializeField] private List<PackageInfo> packages;

    public System.Action<string, float> OnPackageDownloadProgress;
    public System.Action<string> OnPackageDownloadComplete;

    /// <summary>
    /// 检查资源包状态
    /// </summary>
    public IEnumerator CheckPackageStatus()
    {
        foreach (var package in packages)
        {
            var handle = UnityEngine.AddressableAssets.Addressables.GetDownloadSizeAsync(package.addressLabel);
            yield return handle;

            if (handle.Status == UnityEngine.ResourceManagement.AsyncOperations.AsyncOperationStatus.Succeeded)
            {
                package.sizeBytes = handle.Result;
                package.isDownloaded = handle.Result == 0;
            }

            UnityEngine.AddressableAssets.Addressables.Release(handle);
        }
    }

    /// <summary>
    /// 下载资源包
    /// </summary>
    public IEnumerator DownloadPackage(string packageName)
    {
        var package = packages.Find(p => p.packageName == packageName);
        if (package == null) yield break;

        var handle = UnityEngine.AddressableAssets.Addressables.DownloadDependenciesAsync(package.addressLabel);

        while (!handle.IsDone)
        {
            OnPackageDownloadProgress?.Invoke(packageName, handle.PercentComplete);
            yield return null;
        }

        if (handle.Status == UnityEngine.ResourceManagement.AsyncOperations.AsyncOperationStatus.Succeeded)
        {
            package.isDownloaded = true;
            OnPackageDownloadComplete?.Invoke(packageName);
        }

        UnityEngine.AddressableAssets.Addressables.Release(handle);
    }

    /// <summary>
    /// 获取总下载大小
    /// </summary>
    public long GetTotalDownloadSize()
    {
        long total = 0;
        foreach (var package in packages)
        {
            if (!package.isDownloaded)
                total += package.sizeBytes;
        }
        return total;
    }
}
```

```csharp
using UnityEngine;
using UnityEditor;
using UnityEditor.Build.Reporting;
using System.IO;

/// <summary>
/// 自动化构建脚本
/// </summary>
public class CIIBuildScript
{
    private static string BUILD_OUTPUT_PATH = "Builds";

    public static void BuildAndroid()
    {
        // 从环境变量获取版本信息
        string version = Environment.GetEnvironmentVariable("BUILD_VERSION") ?? "1.0.0";
        int buildNumber = int.Parse(Environment.GetEnvironmentVariable("BUILD_NUMBER") ?? "1");

        // 设置版本
        PlayerSettings.bundleVersion = version;
        PlayerSettings.Android.bundleVersionCode = buildNumber;

        // 设置输出路径
        string outputPath = Path.Combine(BUILD_OUTPUT_PATH, "Android", $"Match3Game_v{version}_{buildNumber}.apk");

        // 构建选项
        BuildPlayerOptions options = new BuildPlayerOptions
        {
            scenes = GetEnabledScenes(),
            locationPathName = outputPath,
            target = BuildTarget.Android,
            options = BuildOptions.CompressWithLz4HC
        };

        // 执行构建
        BuildReport report = BuildPipeline.BuildPlayer(options);
        BuildSummary summary = report.summary;

        if (summary.result == BuildResult.Succeeded)
        {
            Debug.Log($"Build succeeded: {summary.totalSize / 1024 / 1024}MB");
            Debug.Log($"Output: {outputPath}");

            // 输出构建信息供CI使用
            File.WriteAllText(Path.Combine(BUILD_OUTPUT_PATH, "build_info.json"),
                $"{{\"version\":\"{version}\",\"buildNumber\":{buildNumber},\"size\":{summary.totalSize}}}");
        }
        else
        {
            Debug.LogError($"Build failed: {summary.totalErrors}");
            EditorApplication.Exit(1);
        }
    }

    public static void BuildIOS()
    {
        string version = Environment.GetEnvironmentVariable("BUILD_VERSION") ?? "1.0.0";
        int buildNumber = int.Parse(Environment.GetEnvironmentVariable("BUILD_NUMBER") ?? "1");

        PlayerSettings.bundleVersion = version;
        PlayerSettings.iOS.buildNumber = buildNumber.ToString();

        string outputPath = Path.Combine(BUILD_OUTPUT_PATH, "iOS", "Match3Game.xcodeproj");

        BuildPlayerOptions options = new BuildPlayerOptions
        {
            scenes = GetEnabledScenes(),
            locationPathName = outputPath,
            target = BuildTarget.iOS,
            options = BuildOptions.CompressWithLz4HC
        };

        BuildPipeline.BuildPlayer(options);
    }

    private static string[] GetEnabledScenes()
    {
        var scenes = new System.Collections.Generic.List<string>();
        foreach (var scene in EditorBuildSettings.scenes)
        {
            if (scene.enabled)
                scenes.Add(scene.path);
        }
        return scenes.ToArray();
    }
}
```
