### SceneChunk类

```csharp
/// <summary>
/// 场景Chunk - 场景的基本单位
/// </summary>
[System.Serializable]
public class SceneChunk
{
    [Header("基本信息")]
    [SerializeField] private string chunkId;
    [SerializeField] private string sceneName;
    [SerializeField] private Vector3 chunkPosition;
    [SerializeField] private float chunkSize = 100f;

    [Header("加载状态")]
    [SerializeField] private bool isLoaded;
    [SerializeField] private float loadPriority;

    [Header("场景引用")]
    [SerializeField] private Scene scene;
    [SerializeField] private AsyncOperation loadOperation;

    /// <summary>
    /// Chunk ID
    /// </summary>
    public string ChunkId => chunkId;

    /// <summary>
    /// 场景名称
    /// </summary>
    public string SceneName => sceneName;

    /// <summary>
    /// Chunk位置
    /// </summary>
    public Vector3 ChunkPosition => chunkPosition;

    /// <summary>
    /// Chunk大小
    /// </summary>
    public float ChunkSize => chunkSize;

    /// <summary>
    /// 是否已加载
    /// </summary>
    public bool IsLoaded => isLoaded;

    /// <summary>
    /// 加载优先级
    /// </summary>
    public float LoadPriority => loadPriority;

    /// <summary>
    /// 场景
    /// </summary>
    public Scene Scene => scene;

    /// <summary>
    /// 获取Chunk中心位置
    /// </summary>
    public Vector3 GetChunkCenter()
    {
        return chunkPosition + Vector3.one * (chunkSize / 2f);
    }

    /// <summary>
    /// 计算Chunk边界
    /// </summary>
    public Bounds GetChunkBounds()
    {
        Vector3 center = GetChunkCenter();
        Vector3 size = Vector3.one * chunkSize;
        return new Bounds(center, size);
    }

    /// <summary>
    /// 检查位置是否在Chunk内
    /// </summary>
    public bool ContainsPosition(Vector3 position)
    {
        Bounds bounds = GetChunkBounds();
        return bounds.Contains(position);
    }

    /// <summary>
    /// 计算与玩家距离
    /// </summary>
    public float GetDistanceToPlayer(Vector3 playerPosition)
    {
        return Vector3.Distance(playerPosition, GetChunkCenter());
    }
}
```

### SceneChunkManager类

```csharp
/// <summary>
/// 场景Chunk管理器
/// </summary>
public class SceneChunkManager : MonoBehaviour
{
    [Header("配置")]
    [SerializeField] private float loadDistance = 80f;
    [SerializeField] private float unloadDistance = 100f;
    [SerializeField] private int maxLoadedChunks = 5;

    [Header("玩家引用")]
    [SerializeField] private Transform player;

    // Chunk列表
    private List<SceneChunk> chunks = new List<SceneChunk>();

    // 已加载的Chunk
    private List<SceneChunk> loadedChunks = new List<SceneChunk>();

    /// <summary>
    /// 初始化Chunk
    /// </summary>
    public void InitializeChunks(List<SceneChunk> chunkList)
    {
        chunks = chunkList;
    }

    /// <summary>
    /// 更新Chunk状态（每帧调用）
    /// </summary>
    private void Update()
    {
        if (player == null)
            return;

        // 1. 检查需要加载的Chunk
        CheckChunksToLoad();

        // 2. 检查需要卸载的Chunk
        CheckChunksToUnload();

        // 3. 更新Chunk优先级
        UpdateChunkPriorities();
    }

    /// <summary>
    /// 检查需要加载的Chunk
    /// </summary>
    private void CheckChunksToLoad()
    {
        foreach (var chunk in chunks)
        {
            // 跳过已加载的
            if (chunk.IsLoaded)
                continue;

            // 跳过已加载的Chunk数量已达上限
            if (loadedChunks.Count >= maxLoadedChunks)
                break;

            // 检查是否在加载范围内
            float distance = chunk.GetDistanceToPlayer(player.position);
            if (distance < loadDistance)
            {
                // 加载Chunk
                LoadChunk(chunk);
            }
        }
    }

    /// <summary>
    /// 检查需要卸载的Chunk
    /// </summary>
    private void CheckChunksToUnload()
    {
        // 创建需要卸载的Chunk列表
        var chunksToUnload = new List<SceneChunk>();

        foreach (var chunk in loadedChunks)
        {
            // 检查是否超出卸载范围
            float distance = chunk.GetDistanceToPlayer(player.position);
            if (distance > unloadDistance)
            {
                chunksToUnload.Add(chunk);
            }
        }

        // 按距离排序（先卸载最远的）
        chunksToUnload.Sort((a, b) =>
            b.GetDistanceToPlayer(player.position).CompareTo(
                a.GetDistanceToPlayer(player.position)));

        // 卸载Chunk
        foreach (var chunk in chunksToUnload)
        {
            UnloadChunk(chunk);
        }
    }

    /// <summary>
    /// 更新Chunk优先级
    /// </summary>
    private void UpdateChunkPriorities()
    {
        foreach (var chunk in chunks)
        {
            // 计算距离优先级（距离越近，优先级越高）
            float distance = chunk.GetDistanceToPlayer(player.position);
            chunk.loadPriority = 1f - Mathf.Clamp01(distance / loadDistance);
        }
    }

    /// <summary>
    /// 加载Chunk
    /// </summary>
    private void LoadChunk(SceneChunk chunk)
    {
        if (chunk.IsLoaded)
            return;

        Debug.Log($"Loading chunk: {chunk.ChunkId}");

        // 加载场景
        StartCoroutine(LoadChunkAsync(chunk));
    }

    /// <summary>
    /// 异步加载Chunk
    /// </summary>
    private IEnumerator LoadChunkAsync(SceneChunk chunk)
    {
        // Additive模式加载场景
        var asyncLoad = SceneManager.LoadSceneAsync(chunk.SceneName, LoadSceneMode.Additive);
        asyncLoad.allowSceneActivation = false;

        chunk.loadOperation = asyncLoad;

        // 等待加载完成
        while (!asyncLoad.isDone)
        {
            // 更新Chunk加载进度
            chunk.loadPriority = asyncLoad.progress;

            yield return null;
        }

        // 激活场景
        asyncLoad.allowSceneActivation = true;

        // 获取场景引用
        chunk.scene = asyncLoad.scene;
        chunk.isLoaded = true;

        // 添加到已加载列表
        loadedChunks.Add(chunk);

        // 触发事件
        SceneEventSystem.Instance?.OnChunkLoaded(chunk);

        Debug.Log($"Chunk loaded: {chunk.ChunkId}");
    }

    /// <summary>
    /// 卸载Chunk
    /// </summary>
    private void UnloadChunk(SceneChunk chunk)
    {
        if (!chunk.IsLoaded)
            return;

        Debug.Log($"Unloading chunk: {chunk.ChunkId}");

        // 异步卸载场景
        StartCoroutine(UnloadChunkAsync(chunk));
    }

    /// <summary>
    /// 异步卸载Chunk
    /// </summary>
    private IEnumerator UnloadChunkAsync(SceneChunk chunk)
    {
        // 异步卸载场景
        var asyncUnload = SceneManager.UnloadSceneAsync(chunk.scene);

        // 等待卸载完成
        while (!asyncUnload.isDone)
        {
            yield return null;
        }

        // 清除Chunk引用
        chunk.isLoaded = false;
        chunk.scene = default;
        chunk.loadOperation = null;

        // 从已加载列表移除
        loadedChunks.Remove(chunk);

        // 触发事件
        SceneEventSystem.Instance?.OnChunkUnloaded(chunk);

        Debug.Log($"Chunk unloaded: {chunk.ChunkId}");
    }

    /// <summary>
    /// 获取所有Chunk
    /// </summary>
    public List<SceneChunk> GetAllChunks()
    {
        return new List<SceneChunk>(chunks);
    }

    /// <summary>
    /// 获取已加载的Chunk
    /// </summary>
    public List<SceneChunk> GetLoadedChunks()
    {
        return new List<SceneChunk>(loadedChunks);
    }
}
```

### SceneLoader类

```csharp
/// <summary>
/// 场景加载器
/// </summary>
public class SceneLoader : MonoBehaviour
{
    /// <summary>
    /// 加载Chunk
    /// </summary>
    public IEnumerator LoadChunk(SceneChunk chunk)
    {
        // 检查是否已加载
        if (chunk.IsLoaded)
        {
            Debug.LogWarning($"Chunk already loaded: {chunk.ChunkId}");
            yield break;
        }

        // 预加载资源
        yield return PreloadChunkResources(chunk);

        // 加载场景
        yield return LoadChunkScene(chunk);

        // 初始化Chunk
        InitializeChunk(chunk);
    }

    /// <summary>
    /// 预加载Chunk资源
    /// </summary>
    private IEnumerator PreloadChunkResources(SceneChunk chunk)
    {
        // 获取Chunk需要加载的资源列表
        var resources = SceneDataProvider.Instance?.GetChunkResources(chunk.ChunkId);

        if (resources == null || resources.Count == 0)
            yield break;

        // 加载资源
        foreach (var resource in resources)
        {
            // 根据资源类型加载
            switch (resource.type)
            {
                case "texture":
                    yield return LoadTexture(resource.path);
                    break;

                case "model":
                    yield return LoadModel(resource.path);
                    break;

                case "audio":
                    yield return LoadAudio(resource.path);
                    break;

                default:
                    Debug.LogWarning($"Unknown resource type: {resource.type}");
                    break;
            }
        }

        Debug.Log($"Preloaded {resources.Count} resources for chunk: {chunk.ChunkId}");
    }

    /// <summary>
    /// 加载Chunk场景
    /// </summary>
    private IEnumerator LoadChunkScene(SceneChunk chunk)
    {
        // Additive模式加载场景
        var asyncLoad = SceneManager.LoadSceneAsync(chunk.SceneName, LoadSceneMode.Additive);
        asyncLoad.allowSceneActivation = false;

        chunk.loadOperation = asyncLoad;

        // 等待加载完成
        while (!asyncLoad.isDone)
        {
            yield return null;
        }

        // 激活场景
        asyncLoad.allowSceneActivation = true;

        // 获取场景引用
        chunk.scene = asyncLoad.scene;
        chunk.isLoaded = true;

        Debug.Log($"Chunk scene loaded: {chunk.ChunkId}");
    }

    /// <summary>
    /// 初始化Chunk
    /// </summary>
    private void InitializeChunk(SceneChunk chunk)
    {
        if (!chunk.scene.IsValid())
            return;

        // 初始化场景中的游戏对象
        var rootObjects = chunk.scene.GetRootGameObjects();
        foreach (var root in rootObjects)
        {
            // 初始化游戏对象
            InitializeGameObject(root, chunk);
        }

        // 触发事件
        SceneEventSystem.Instance?.OnChunkInitialized(chunk);

        Debug.Log($"Chunk initialized: {chunk.ChunkId}");
    }

    /// <summary>
    /// 初始化游戏对象
    /// </summary>
    private void InitializeGameObject(GameObject obj, SceneChunk chunk)
    {
        // 获取所有MonoBehaviour
        var behaviours = obj.GetComponentsInChildren<MonoBehaviour>();

        foreach (var behaviour in behaviours)
        {
            // 检查是否是IChunkAware
            if (behaviour is IChunkAware chunkAware)
            {
                // 设置Chunk ID
                chunkAware.SetChunkId(chunk.ChunkId);
            }
        }
    }
}
```

```csharp
/// <summary>
/// 内存管理器
/// </summary>
public class MemoryManager : MonoBehaviour
{
    [Header("配置")]
    [SerializeField] private int maxMemoryMB = 500;
    [SerializeField] private float cleanupThreshold = 0.8f;

    // 场景缓存
    private Dictionary<string, Scene> sceneCache = new Dictionary<string, Scene>();

    // 内存占用跟踪
    private long currentMemoryUsage;

    /// <summary>
    /// 获取当前内存使用
    /// </summary>
    public long GetCurrentMemoryUsage()
    {
        return currentMemoryUsage;
    }

    /// <summary>
    /// 检查是否需要清理内存
    /// </summary>
    public bool NeedsCleanup()
    {
        long maxMemoryBytes = maxMemoryMB * 1024 * 1024;
        return currentMemoryUsage > maxMemoryBytes * cleanupThreshold;
    }

    /// <summary>
    /// 清理未使用的场景
    /// </summary>
    public void CleanupUnusedScenes()
    {
        // 获取所有已加载的但未激活的场景
        var unusedScenes = sceneCache.Values.Where(s => !s.IsValid()).ToList();

        foreach (var scene in unusedScenes)
        {
            // 卸载场景
            StartCoroutine(UnloadSceneAsync(scene));
        }

        Debug.Log($"Cleaned up {unusedScenes.Count} unused scenes");
    }

    /// <summary>
    /// 添加场景到缓存
    /// </summary>
    public void AddSceneToCache(Scene scene)
    {
        string sceneName = scene.name;
        sceneCache[sceneName] = scene;

        // 更新内存使用
        UpdateMemoryUsage();
    }

    /// <summary>
    /// 从缓存移除场景
    /// </summary>
    public void RemoveSceneFromCache(Scene scene)
    {
        string sceneName = scene.name;
        sceneCache.Remove(sceneName);

        // 更新内存使用
        UpdateMemoryUsage();
    }

    /// <summary>
    /// 更新内存使用
    /// </summary>
    private void UpdateMemoryUsage()
    {
        // 估算内存使用
        long estimatedMemory = sceneCache.Count * 50 * 1024 * 1024; // 50MB per scene

        currentMemoryUsage = estimatedMemory;

        // 检查是否需要清理
        if (NeedsCleanup())
        {
            Debug.LogWarning($"Memory usage high: {currentMemoryUsage / (1024 * 1024)}MB");

            // 自动清理
            CleanupUnusedScenes();
        }
    }

    /// <summary>
    /// 异步卸载场景
    /// </summary>
    private IEnumerator UnloadSceneAsync(Scene scene)
    {
        // 异步卸载场景
        var asyncUnload = SceneManager.UnloadSceneAsync(scene);

        // 等待卸载完成
        while (!asyncUnload.isDone)
        {
            yield return null;
        }

        // 从缓存移除
        string sceneName = scene.name;
        sceneCache.Remove(sceneName);

        Debug.Log($"Scene unloaded from cache: {sceneName}");
    }
}
```

```csharp
/// <summary>
/// 场景数据提供者
/// </summary>
public class SceneDataProvider : MonoBehaviour
{
    private static SceneDataProvider instance;

    // Chunk配置
    private Dictionary<string, SceneChunk> chunkConfigurations = new Dictionary<string, SceneChunk>();

    // 资源配置
    private Dictionary<string, List<SceneResource>> chunkResources = new Dictionary<string, List<SceneResource>>();

    /// <summary>
    /// 单例
    /// </summary>
    public static SceneDataProvider Instance => instance;

    private void Awake()
    {
        if (instance == null)
        {
            instance = this;
            DontDestroyOnLoad(gameObject);

            // 加载配置
            LoadChunkConfigurations();
        }
        else
        {
            Destroy(gameObject);
        }
    }

    /// <summary>
    /// 加载Chunk配置
    /// </summary>
    private void LoadChunkConfigurations()
    {
        // 加载Chunk配置文件
        var configJson = Resources.Load<TextAsset>("Scene/chunk_config");
        var config = JsonUtility.FromJson<SceneChunkConfig>(configJson.text);

        // 创建Chunk
        foreach (var chunkData in config.chunks)
        {
            var chunk = new SceneChunk
            {
                chunkId = chunkData.chunkId,
                sceneName = chunkData.sceneName,
                chunkPosition = chunkData.position,
                chunkSize = chunkData.size
            };

            chunkConfigurations[chunk.chunkId] = chunk;
        }

        // 加载资源配置
        var resourcesJson = Resources.Load<TextAsset>("Scene/chunk_resources");
        var resources = JsonUtility.FromJson<SceneResourcesConfig>(resourcesJson.text);

        foreach (var chunkResourceData in resources.chunkResources)
        {
            chunkResources[chunkResourceData.chunkId] = chunkResourceData.resources;
        }

        Debug.Log($"Loaded {chunkConfigurations.Count} chunk configurations");
    }

    /// <summary>
    /// 获取Chunk配置
    /// </summary>
    public SceneChunk GetChunk(string chunkId)
    {
        if (chunkConfigurations.TryGetValue(chunkId, out var chunk))
        {
            return chunk;
        }

        Debug.LogWarning($"Chunk not found: {chunkId}");
        return null;
    }

    /// <summary>
    /// 获取所有Chunk
    /// </summary>
    public List<SceneChunk> GetAllChunks()
    {
        return chunkConfigurations.Values.ToList();
    }

    /// <summary>
    /// 获取Chunk资源
    /// </summary>
    public List<SceneResource> GetChunkResources(string chunkId)
    {
        if (chunkResources.TryGetValue(chunkId, out var resources))
        {
            return resources;
        }

        return new List<SceneResource>();
    }
}

/// <summary>
```
```csharp
/// 场景流式加载器
/// </summary>
public class SceneStreamer : MonoBehaviour
{
    [Header("配置")]
    [SerializeField] private float loadDistance = 80f;
    [SerializeField] private float unloadDistance = 100f;
    [SerializeField] private int maxLoadedChunks = 5;
    [SerializeField] private int preloadAhead = 2;

    [Header("玩家")]
    [SerializeField] private Transform player;

    // 组件引用
    private SceneChunkManager chunkManager;
    private SceneLoader sceneLoader;
    private MemoryManager memoryManager;

    /// <summary>
    /// 初始化
    /// </summary>
    private void Start()
    {
        // 获取组件
        chunkManager = GetComponent<SceneChunkManager>();
        sceneLoader = GetComponent<SceneLoader>();
        memoryManager = GetComponent<MemoryManager>();

        // 初始化Chunk
        var chunks = SceneDataProvider.Instance.GetAllChunks();
        chunkManager.InitializeChunks(chunks);

        // 订阅事件
        SceneEventSystem.Instance.OnChunkLoaded += OnChunkLoaded;
        SceneEventSystem.Instance.OnChunkUnloaded += OnChunkUnloaded;

        Debug.Log("SceneStreamer initialized");
    }

    /// <summary>
    /// 更新（每帧调用）
    /// </summary>
    private void Update()
    {
        // 更新Chunk状态
        // (由chunkManager.Update()处理）
    }

    /// <summary>
    /// Chunk加载回调
    /// </summary>
    private void OnChunkLoaded(SceneChunk chunk)
    {
        Debug.Log($"Chunk loaded: {chunk.ChunkId}");

        // 检查内存使用
        if (memoryManager.NeedsCleanup())
        {
            memoryManager.CleanupUnusedScenes();
        }
    }

    /// <summary>
    /// Chunk卸载回调
    /// </summary>
    private void OnChunkUnloaded(SceneChunk chunk)
    {
        Debug.Log($"Chunk unloaded: {chunk.ChunkId}");
    }

    /// <summary>
    /// 获取加载统计
    /// </summary>
    public string GetLoadStatistics()
    {
        var loadedChunks = chunkManager.GetLoadedChunks();

        var stats = new StringBuilder();
        stats.AppendLine($"Loaded Chunks: {loadedChunks.Count}");
        stats.AppendLine($"Max Loaded Chunks: {maxLoadedChunks}");
        stats.AppendLine($"Load Distance: {loadDistance}m");
        stats.AppendLine($"Unload Distance: {unloadDistance}m");
        stats.AppendLine($"Memory Usage: {memoryManager.GetCurrentMemoryUsage() / (1024 * 1024)}MB");

        return stats.ToString();
    }
```

## 最佳实践

### DO ✅

- 使用Additive模式实现流式加载
- 合理设置加载/卸载距离
- 限制同时加载的Chunk数量
- 监控内存使用，及时清理
- 使用事件系统解耦组件
- 预加载Chunk资源
- 使用Chunk作为场景基本单位
- 异步加载/卸载场景
- 使用对象池管理场景对象

### DON'T ❌

- 不要使用Single模式流式加载
- 不要加载过多Chunk导致内存溢出
- 不要忽略内存清理
- 不要在主线程同步加载场景
- 不要忘记卸载不用的场景
- 不要混合使用Sync和Async API
- 不要忽略场景加载错误
- 不要在场景间传递复杂数据
- 不要过度预加载Chunk
