## 存档管理器

### SaveManager类

```csharp
/// <summary>
/// 存档管理器
/// </summary>
public class SaveManager : MonoBehaviour
{
    // 单例
    public static SaveManager Instance { get; private set; }

    // 组件引用
    [SerializeField] private LocalSaveManager localSaveManager;
    [SerializeField] private CloudSaveManager cloudSaveManager;
    [SerializeField] private ConflictResolver conflictResolver;
    [SerializeField] private SaveMerger saveMerger;

    // 配置
    [Header("配置")]
    [SerializeField] private SaveMode saveMode = SaveMode.CloudPreferred;
    [SerializeField] private bool autoMerge = true;
    [SerializeField] private bool enableOfflineMode = true;

    // 状态
    private SaveData localSaveData;
    private SaveData cloudSaveData;

    /// <summary>
    /// 存档模式
    /// </summary>
    public enum SaveMode
    {
        Local,           // 仅本地
        Cloud,           // 仅云端
        CloudPreferred,  // 优先云端，失败则本地
        CloudOnly,       // 仅云端
        Merge            // 自动合并
    }

    /// <summary>
    /// 初始化
    /// </summary>
    private void Awake()
    {
        if (Instance == null)
        {
            Instance = this;
            DontDestroyOnLoad(gameObject);
        }
        else
        {
            Destroy(gameObject);
            return;
        }

        // 初始化组件
        localSaveManager = GetComponent<LocalSaveManager>();
        cloudSaveManager = GetComponent<CloudSaveManager>();
        conflictResolver = GetComponent<ConflictResolver>();
        saveMerger = GetComponent<SaveMerger>();
    }

    /// <summary>
    /// 存档
    /// </summary>
    public async Task<bool> SaveGame(SaveData data)
    {
        Debug.Log("Saving game...");

        // 保存本地
        await localSaveManager.SaveAsync(data);

        // 保存云端(根据模式）
        if (saveMode == SaveMode.Cloud || saveMode == SaveMode.CloudPreferred)
        {
            try
            {
                await cloudSaveManager.SaveAsync(data);
                cloudSaveData = data;
                Debug.Log("Cloud save successful");
            }
            catch (Exception ex)
            {
                Debug.LogWarning($"Cloud save failed: {ex.Message}");

                if (saveMode == SaveMode.CloudPreferred)
                {
                    Debug.Log("Falling back to local save");
                    // 降级到本地
                    return true;
                }
                else if (saveMode == SaveMode.Cloud)
                {
                    Debug.LogError("Cloud save failed, not saving locally");
                    return false;
                }
            }
        }

        localSaveData = data;
        return true;
    }

    /// <summary>
    /// 加载存档
    /// </summary>
    public async Task<SaveData> LoadGame()
    {
        Debug.Log("Loading game...");

        SaveData data = null;

        // 根据模式加载
        switch (saveMode)
        {
            case SaveMode.Local:
                data = await localSaveManager.LoadAsync();
                break;

            case SaveMode.Cloud:
            case SaveMode.CloudPreferred:
            case SaveMode.CloudOnly:
                data = await LoadFromCloud();
                if (data == null && saveMode == SaveMode.CloudPreferred)
                {
                    // 云端失败，降级到本地
                    Debug.Log("Cloud load failed, loading from local");
                    data = await localSaveManager.LoadAsync();
                }
                break;

            case SaveMode.Merge:
                data = await LoadWithMerge();
                break;
        }

        if (data != null)
        {
            Debug.Log($"Game loaded from {saveMode}");
        }
        else
        {
            Debug.LogWarning("Failed to load game");
        }

        return data;
    }

    /// <summary>
    /// 从云端加载
    /// </summary>
    private async Task<SaveData> LoadFromCloud()
    {
        try
        {
            return await cloudSaveManager.LoadAsync();
        }
        catch (Exception ex)
        {
            Debug.LogWarning($"Cloud load failed: {ex.Message}");
            return null;
        }
    }

    /// <summary>
    /// 加载并合并存档
    /// </summary>
    private async Task<SaveData> LoadWithMerge()
    {
        // 加载本地存档
        var localData = await localSaveManager.LoadAsync();

        // 加载云端存档
        SaveData cloudData = null;
        try
        {
            cloudData = await cloudSaveManager.LoadAsync();
        }
        catch (Exception ex)
        {
            Debug.LogWarning($"Cloud load failed: {ex.Message}");
        }

        // 合并存档
        if (localData != null && cloudData != null)
        {
            // 检测冲突
            var conflict = conflictResolver.DetectConflict(localData, cloudData);

            if (conflict.hasConflict)
            {
                Debug.LogWarning($"Save conflict detected: {conflict.conflictType}");

                // 自动合并
                if (autoMerge)
                {
                    var mergedData = saveMerger.Merge(localData, cloudData);
                    data = mergedData;
                    Debug.Log("Auto-merged saves");
                }
                else
                {
                    // 使用本地存档
                    data = localData;
                    Debug.Log("Using local save (auto-merge disabled)");
                }
            }
            else
            {
                // 无冲突，使用云端存档(通常更新)
                data = cloudData;
                Debug.Log("Using cloud save (no conflict)");
            }
        }
        else if (localData != null)
        {
            // 只有本地存档
            data = localData;
            Debug.Log("Using local save (no cloud save)");
        }
        else if (cloudData != null)
        {
            // 只有云端存档
            data = cloudData;
            Debug.Log("Using cloud save (no local save)");
        }
        else
        {
            // 没有存档
            Debug.LogWarning("No save found");
            return null;
        }

        // 保存合并后的存档
        if (data != null)
        {
            await localSaveManager.SaveAsync(data);
            if (saveMode != SaveMode.Local)
            {
                try
                {
                    await cloudSaveManager.SaveAsync(data);
                }
                catch (Exception ex)
                {
                    Debug.LogWarning($"Failed to save merged data to cloud: {ex.Message}");
                }
            }
        }

        return data;
    }

    /// <summary>
    /// 同步存档
    /// </summary>
    public async Task SyncSaves()
    {
        Debug.Log("Syncing saves...");

        // 检查本地存档
        var localData = await localSaveManager.LoadAsync();

        // 检查云端存档
        SaveData cloudData = null;
        try
        {
            cloudData = await cloudSaveManager.LoadAsync();
        }
        catch (Exception ex)
        {
            Debug.LogWarning($"Cloud load failed during sync: {ex.Message}");
        }

        // 比较存档
        if (localData != null && cloudData != null)
        {
            // 检查是否需要上传
            if (localData.timestamp > cloudData.timestamp)
            {
                Debug.Log("Local save is newer, uploading to cloud");
                await cloudSaveManager.SaveAsync(localData);
            }
            else if (cloudData.timestamp > localData.timestamp)
            {
                Debug.Log("Cloud save is newer, downloading from cloud");
                await localSaveManager.SaveAsync(cloudData);
            }
            else
            {
                Debug.Log("Local and cloud saves are in sync");
            }
        }
        else if (localData != null)
        {
            // 只有本地存档，上传到云端
            Debug.Log("Uploading local save to cloud");
            try
            {
                await cloudSaveManager.SaveAsync(localData);
            }
            catch (Exception ex)
            {
                Debug.LogWarning($"Failed to upload to cloud: {ex.Message}");
            }
        }
        else if (cloudData != null)
        {
            // 只有云端存档，下载到本地
            Debug.Log("Downloading cloud save to local");
            await localSaveManager.SaveAsync(cloudData);
        }

        Debug.Log("Sync complete");
    }

    /// <summary>
    /// 删除存档
    /// </summary>
    public async Task DeleteSave()
    {
        Debug.Log("Deleting saves...");

        // 删除本地存档
        await localSaveManager.DeleteAsync();

        // 删除云端存档
        try
        {
            await cloudSaveManager.DeleteAsync();
        }
        catch (Exception ex)
        {
            Debug.LogWarning($"Cloud delete failed: {ex.Message}");
        }

        Debug.Log("Saves deleted");
    }
}
```

## 冲突解决器

### ConflictResolver类

```csharp
/// <summary>
/// 存档冲突解决器
/// </summary>
public class ConflictResolver : MonoBehaviour
{
    /// <summary>
    /// 冲突类型
    /// </summary>
    public enum ConflictType
    {
        NoConflict,          // 无冲突
        TimestampConflict,   // 时间戳冲突(同时修改)
        VersionConflict,      // 版本冲突(版本号不连续)
        DataConflict,          // 数据冲突(内容不同)
        VersionGap           // 版本冲突(版本号跳过)
    }

    /// <summary>
    /// 冲突信息
    /// </summary>
    public struct ConflictInfo
    {
        public ConflictType conflictType;
        public string description;
        public int localVersion;
        public int cloudVersion;
        public TimeSpan timeDifference;
    }

    /// <summary>
    /// 检测冲突
    /// </summary>
    public ConflictInfo DetectConflict(SaveData localData, SaveData cloudData)
    {
        var conflict = new ConflictInfo
        {
            conflictType = ConflictType.NoConflict,
            localVersion = localData != null ? localData.version : -1,
            cloudVersion = cloudData != null ? cloudData.version : -1
        };

        // 检查时间戳
        if (localData != null && cloudData != null)
        {
            var timeDiff = Math.Abs((localData.timestamp - cloudData.timestamp).TotalSeconds);

            // 时间差小于5秒，视为同时修改
            if (timeDiff < 5f)
            {
                conflict.conflictType = ConflictType.TimestampConflict;
                conflict.timeDifference = TimeSpan.FromSeconds(timeDiff);
                conflict.description = $"Simultaneous modification (time diff: {timeDiff:F2}s)";
                return conflict;
            }
        }

        // 检查版本号
        if (localData != null && cloudData != null)
        {
            // 版本号相同，但内容不同
            if (localData.version == cloudData.version)
            {
                // 比较数据内容
                string localJson = JsonUtility.ToJson(localData);
                string cloudJson = JsonUtility.ToJson(cloudData);

                if (localJson != cloudJson)
                {
                    conflict.conflictType = ConflictType.DataConflict;
                    conflict.description = $"Data conflict (version: {localData.version})";
                    return conflict;
                }
            }
            // 版本号不连续(跳过)
            else if (cloudData.version - localData.version > 1)
            {
                conflict.conflictType = ConflictType.VersionGap;
                conflict.description = $"Version gap (local: {localData.version}, cloud: {cloudData.version})";
                return conflict;
            }
            // 版本号递增
            else
            {
                conflict.conflictType = ConflictType.VersionConflict;
                conflict.description = $"Version conflict (local: {localData.version}, cloud: {cloudData.version})";
                return conflict;
            }
        }

        return conflict;
    }

    /// <summary>
    /// 自动解决冲突
    /// </summary>
    public SaveData AutoResolveConflict(SaveData localData, SaveData cloudData, ConflictType conflictType)
    {
        switch (conflictType)
        {
            case ConflictType.NoConflict:
                // 无冲突，使用云端存档(通常更新)
                return cloudData ?? localData;

            case ConflictType.TimestampConflict:
                // 时间戳冲突，使用最新的存档
                if (localData.timestamp > cloudData.timestamp)
                {
                    Debug.Log("Resolving: Using local save (newer timestamp)");
                    return localData;
                }
                else
                {
                    Debug.Log("Resolving: Using cloud save (newer timestamp)");
                    return cloudData;
                }

            case ConflictType.VersionConflict:
                // 版本冲突，使用版本号更大的
                if (localData.version > cloudData.version)
                {
                    Debug.Log($"Resolving: Using local save (version: {localData.version} > {cloudData.version})");
                    return localData;
                }
                else
                {
                    Debug.Log($"Resolving: Using cloud save (version: {cloudData.version} > {localData.version})");
                    return cloudData;
                }

            case ConflictType.VersionGap:
                // 版本跳过，询问用户(这里简化使用云端)
                Debug.Log($"Resolving: Using cloud save (version: {cloudData.version}, gap)");
                return cloudData;

            case ConflictType.DataConflict:
                // 数据冲突，需要手动合并
                Debug.LogWarning($"Data conflict detected, cannot auto-resolve (version: {cloudData.version})");
                return localData; // 默认返回本地

            default:
                Debug.LogWarning($"Unknown conflict type: {conflictType}");
                return localData;
        }
    }

    /// <summary>
    /// 手动解决冲突
    /// </summary>
    public SaveData ManualResolveConflict(ConflictInfo conflict, SaveData localData, SaveData cloudData, SaveResolution resolution)
    {
        switch (resolution)
        {
            case SaveResolution.UseLocal:
                Debug.Log("Manual resolution: Using local save");
                return localData;

            case SaveResolution.UseCloud:
                Debug.Log("Manual resolution: Using cloud save");
                return cloudData;

            case SaveResolution.UseLatest:
                if (localData.timestamp > cloudData.timestamp)
                {
                    Debug.Log("Manual resolution: Using local save (newer)");
                    return localData;
                }
                else
                {
                    Debug.Log("Manual resolution: Using cloud save (newer)");
                    return cloudData;
                }

            case SaveResolution.UseHighestVersion:
                if (localData.version > cloudData.version)
                {
                    Debug.Log($"Manual resolution: Using local save (higher version: {localData.version})");
                    return localData;
                }
                else
                {
                    Debug.Log($"Manual resolution: Using cloud save (higher version: {cloudData.version})");
                    return cloudData;
                }

            case SaveResolution.Merge:
                Debug.Log("Manual resolution: Merging saves");
                // 需要实现合并逻辑
                return MergeSaves(localData, cloudData);

            default:
                Debug.LogWarning($"Unknown resolution: {resolution}");
                return localData;
        }
    }

    /// <summary>
    /// 合并存档
    /// </summary>
    private SaveData MergeSaves(SaveData localData, SaveData cloudData)
    {
        // 简化实现: 使用云端版本，但保留本地数据
        var mergedData = new SaveData
        {
            version = cloudData.version,
            timestamp = DateTime.Now,
            playerName = cloudData.playerName,
            level = cloudData.level,
            health = cloudData.health,
            position = cloudData.position,
            inventory = cloudData.inventory,
            skills = cloudData.skills
        };

        // TODO: 实现更复杂的合并逻辑
        // 例如: 合并物品列表，去重等

        Debug.Log($"Merged saves (local: {localData.version}, cloud: {cloudData.version})");
        return mergedData;
    }

    /// <summary>
    /// 存档解决方案
    /// </summary>
    public enum SaveResolution
    {
        UseLocal,        // 使用本地存档
        UseCloud,        // 使用云端存档
        UseLatest,       // 使用最新存档(按时间戳)
        UseHighestVersion, // 使用最高版本号
        Merge            // 合并存档
    }
}
```

```csharp
/// <summary>

/// </summary>
public class SaveMerger : MonoBehaviour
{
    /// <summary>
    /// 合并策略
    /// </summary>
    public enum MergeStrategy
    {
        CloudPriority,    // 云端优先
        LocalPriority,    // 本地优先
        LatestTimestamp,  // 最新时间戳
        HighestVersion,    // 最高版本号
        Union,              // 并集(保留所有)
        Intersection        // 交集(保留共同)
    }

    /// <summary>
    /// 合并存档
    /// </summary>
    public SaveData Merge(SaveData localData, SaveData cloudData, MergeStrategy strategy = MergeStrategy.CloudPriority)
    {
        if (localData == null && cloudData == null)
        {
            Debug.LogWarning("Both saves are null");
            return null;
        }

        switch (strategy)
        {
            case MergeStrategy.CloudPriority:
                return MergeWithCloudPriority(localData, cloudData);

            case MergeStrategy.LocalPriority:
                return MergeWithLocalPriority(localData, cloudData);

            case MergeStrategy.LatestTimestamp:
                return MergeWithLatestTimestamp(localData, cloudData);

            case MergeStrategy.HighestVersion:
                return MergeWithHighestVersion(localData, cloudData);

            case MergeStrategy.Union:
                return MergeUnion(localData, cloudData);

        else
        {
            Debug.Log("Merged with latest timestamp (cloud)");
            return cloudData;
        }
    }

    /// <summary>
    /// 最高版本号合并
    /// </summary>
    private SaveData MergeWithHighestVersion(SaveData localData, SaveData cloudData)
    {
        if (localData == null)
            return cloudData;

        if (cloudData == null)
            return localData;

        // 选择版本号更高的存档
        if (localData.version > cloudData.version)
        {
            Debug.Log($"Merged with highest version (local: {localData.version})");
            return localData;
        }
        else
        {
            Debug.Log($"Merged with highest version (cloud: {cloudData.version})");
            return cloudData;
        }
    }

    /// <summary>
    /// 并集合并
    /// </summary>
    private SaveData MergeUnion(SaveData localData, SaveData cloudData)
    {
        if (localData == null)
            return cloudData;

        if (cloudData == null)
            return localData;

        var mergedData = new SaveData
        {
            version = Mathf.Max(localData.version, cloudData.version),
            timestamp = localData.timestamp > cloudData.timestamp ? localData.timestamp : cloudData.timestamp,
            playerName = localData.playerName,
            level = Mathf.Max(localData.level, cloudData.level),
            health = Mathf.Max(localData.health, cloudData.health),
            position = localData.position,
            inventory = MergeLists(localData.inventory, cloudData.inventory),
            skills = MergeDictionaries(localData.skills, cloudData.skills)
        };

        Debug.Log("Merged with union strategy");
        return mergedData;
    }

    /// <summary>
    /// 交集合并
    /// </summary>
    private SaveData MergeIntersection(SaveData localData, SaveData cloudData)
    {
        if (localData == null)
            return cloudData;

        if (cloudData == null)
            return localData;

        var mergedData = new SaveData
        {
            version = Mathf.Min(localData.version, cloudData.version),
            timestamp = localData.timestamp < cloudData.timestamp ? localData.timestamp : cloudData.timestamp,
            playerName = localData.playerName,
            level = Mathf.Min(localData.level, cloudData.level),
            health = Mathf.Min(localData.health, cloudData.health),
            position = localData.position,
            inventory = IntersectLists(localData.inventory, cloudData.inventory),
            skills = IntersectDictionaries(localData.skills, cloudData.skills)
        };

        Debug.Log("Merged with intersection strategy");
        return mergedData;
    }

    /// <summary>
    /// 合并列表
    /// </summary>
    private List<T> MergeLists<T>(List<T> list1, List<T> list2)
    {
        var merged = new List<T>(list1);

        foreach (var item in list2)
        {
            if (!merged.Contains(item))
            {
                merged.Add(item);
            }
        }

        return merged;
    }

    /// <summary>
    /// 交集列表
    /// </summary>
    private List<T> IntersectLists<T>(List<T> list1, List<T> list2)
    {
        var intersection = new List<T>();

        foreach (var item in list1)
        {
            if (list2.Contains(item))
            {
                intersection.Add(item);
            }
        }

        return intersection;
    }

    /// <summary>
    /// 合并字典
    /// </summary>
    private Dictionary<TKey, TValue> MergeDictionaries<TKey, TValue>(
        Dictionary<TKey, TValue> dict1,
        Dictionary<TKey, TValue> dict2)
    {
        var merged = new Dictionary<TKey, TValue>(dict1);

        foreach (var kvp in dict2)
        {
            if (!merged.ContainsKey(kvp.Key))
            {
                merged[kvp.Key] = kvp.Value;
            }
        }

        return merged;
    }

    /// <summary>
    /// 交集字典
    /// </summary>
    private Dictionary<TKey, TValue> IntersectDictionaries<TKey, TValue>(
        Dictionary<TKey, TValue> dict1,
        Dictionary<TKey, TValue> dict2)
    {
        var intersection = new Dictionary<TKey, TValue>();

        foreach (var kvp in dict1)
        {
            if (dict2.TryGetValue(kvp.Key, out var value))
            {
                intersection[kvp.Key] = value;
            }
        }

        return intersection;
    }
}
```

```csharp
/// <summary>

public class OfflineManager : MonoBehaviour
{
    // 单例
    public static OfflineManager Instance { get; private set; }

    // 离线队列
    [SerializeField] private int maxQueueSize = 10;

    // 离线状态
    private bool isOfflineMode;
    private Queue<OfflineOperation> offlineQueue = new Queue<OfflineOperation>();

    /// <summary>
    /// 离线操作
    /// </summary>
    public enum OperationType
    {
        Save,
        Load,
        Delete
    }

    /// <summary>
    /// 离线操作
    /// </summary>
    public class OfflineOperation
    {
        public OperationType type;
        public SaveData saveData;
        public string savePath;
        public DateTime timestamp;
    }

    private void Awake()
    {
        if (Instance == null)
        {
            Instance = this;
            DontDestroyOnLoad(gameObject);
        }
        else
        {
            Destroy(gameObject);
            return;
        }

        // 检查网络状态
        StartCoroutine(CheckNetworkStatus());
    }

    /// <summary>
    /// 检查网络状态
    /// </summary>
    private IEnumerator CheckNetworkStatus()
    {
        while (true)
        {
            bool wasOffline = isOfflineMode;
            isOfflineMode = Application.internetReachability == NetworkReachability.NotReachable;

            if (isOfflineMode && !wasOffline)
            {
                Debug.Log("Entered offline mode");
                OnOfflineModeEntered?.Invoke();
            }
            else if (!isOfflineMode && wasOffline)
            {
                Debug.Log("Exited offline mode");
                OnOfflineModeExited?.Invoke();

                // 同步离线队列
                yield return StartCoroutine(SyncOfflineQueue());
            }

            yield return new WaitForSeconds(1f);
        }
    }

    /// <summary>
    /// 离线模式进入事件
    /// </summary>
    public event Action OnOfflineModeEntered;

    /// <summary>
    /// 离线模式退出事件
    /// </summary>
    public event Action OnOfflineModeExited;

    /// <summary>
    /// 添加离线操作
    /// </summary>
    public void AddOfflineOperation(OperationType type, SaveData data, string path)
    {
        var operation = new OfflineOperation
        {
            type = type,
            saveData = data,
            savePath = path,
            timestamp = DateTime.Now
        };

        offlineQueue.Enqueue(operation);

        // 限制队列大小
        while (offlineQueue.Count > maxQueueSize)
        {
            offlineQueue.Dequeue();
        }

        Debug.Log($"Added offline operation: {type}, Queue size: {offlineQueue.Count}");
    }

    /// <summary>
    /// 同步离线队列
    /// </summary>
    private IEnumerator SyncOfflineQueue()
    {
        while (offlineQueue.Count > 0)
        {
            var operation = offlineQueue.Peek();

            switch (operation.type)
            {
                case OperationType.Save:
                    // 执行保存操作
                    bool success = await SaveToCloud(operation.saveData);
                    if (success)
                    {
                        offlineQueue.Dequeue();
                        Debug.Log($"Offline save completed: {operation.saveData.playerName}");
                    }
                    else
                    {
                        Debug.LogWarning($"Failed to save offline operation: {operation.saveData.playerName}");
                    }
                    break;

                case OperationType.Load:
                    // 离线模式下不支持加载
                    Debug.LogWarning("Load operation in offline mode skipped");
                    offlineQueue.Dequeue();
                    break;

                case OperationType.Delete:
                    // 执行删除操作
                    success = await DeleteFromCloud(operation.saveData);
                    if (success)
                    {
                        offlineQueue.Dequeue();
                        Debug.Log($"Offline delete completed");
                    }
                    break;
            }

            yield return null;
        }

        Debug.Log("Offline queue synced");
    }

    /// <summary>
    /// 保存到云端
    /// </summary>
    private async Task<bool> SaveToCloud(SaveData data)
    {
        try
        {
            await CloudAPIAdapter.Instance.SaveAsync(data);
            return true;
        }
        catch (Exception ex)
        {
            Debug.LogError($"Failed to save to cloud: {ex.Message}");
            return false;
        }
    }

    /// <summary>
    /// 从云端删除
    /// </summary>
    private async Task<bool> DeleteFromCloud(SaveData data)
    {
        try
        {
            await CloudAPIAdapter.Instance.DeleteAsync(data);
            return true;
        }
        catch (Exception ex)
        {
            Debug.LogError($"Failed to delete from cloud: {ex.Message}");
            return false;
        }
    }

    /// <summary>
    /// 获取离线队列大小
    /// </summary>
    public int GetOfflineQueueSize()
    {
        return offlineQueue.Count;
    }

    /// <summary>
    /// 清空离线队列
    /// </summary>
    public void ClearOfflineQueue()
    {
        offlineQueue.Clear();
        Debug.Log("Offline queue cleared");
    }
}
```

```csharp
/// <summary>
/// 云端存档完整流程

public class CloudSaveWorkflow : MonoBehaviour
{
    [SerializeField] private SaveManager saveManager;

    // 状态
    private SaveData currentSave;

    private void Start()
    {
        // 加载存档
        LoadSave();

    /// 加载存档
    /// </summary>
    private async void LoadSave()
    {
        currentSave = await saveManager.LoadGame();

        if (currentSave != null)
        {
            Debug.Log($"Save loaded: Level={currentSave.level}, Player={currentSave.playerName}");
        }

        {
            Debug.Log("No save found");
        }
    }

    /// </summary>
    public async void SaveGame(SaveData data)
    {
        bool success = await saveManager.SaveGame(data);

        if (success)
        {
            currentSave = data;
            Debug.Log($"Game saved: Level={data.level}, Player={data.playerName}");
        }
        else
        {
            Debug.LogError("Failed to save game");
        }
    }

    /// <summary>
    /// 同步存档
    /// </summary>
    public async void SyncSaves()
    {
        await saveManager.SyncSaves();
        Debug.Log("Saves synced");
    }

    /// <summary>
    /// 手动解决冲突
    /// </summary>
    public async Task ManualResolveConflict(ConflictResolver.ConflictInfo conflict, ConflictResolver.SaveResolution resolution)
    {
        var localData = await saveManager.GetComponent<LocalSaveManager>().LoadAsync();
        var cloudData = await saveManager.GetComponent<CloudSaveManager>().LoadAsync();

        var resolvedData = saveManager.GetComponent<ConflictResolver>().ManualResolveConflict(
            conflict, localData, cloudData, resolution);

        if (resolvedData != null)
        {
            currentSave = resolvedData;
            await saveManager.SaveGame(resolvedData);
            Debug.Log($"Conflict resolved: {conflict.conflictType}");
        }
    }
```
