## 战斗特效管理器

### CombatEffectManager类

```csharp
/// <summary>
/// 战斗特效管理器
/// </summary>
public class CombatEffectManager : MonoBehaviour
{
    // 单例
    public static CombatEffectManager Instance { get; private set; }

    // 配置
    [Header("配置")]
    [SerializeField] private int maxConcurrentEffects = 10;
    [SerializeField] private bool useObjectPooling = true;
    [SerializeField] private bool useLODSystem = true;

    // 特效类型定义
    [Header("特效类型")]
    [SerializeField] private CombatEffectAsset[] effectTypes;

    // LOD配置
    [Header("LOD配置")]
    [SerializeField] private Transform cameraTransform;
    [SerializeField] private float lod1Distance = 30f;
    [SerializeField] private float lod2Distance = 60f;
    [SerializeField] private float lod3Distance = 100f;

    // 组件引用
    private EffectPool effectPool;
    private LODManager lodManager;
    private PerformanceMonitor performanceMonitor;

    // 活跃特效实例
    private List<EffectInstance> activeEffects = new List<EffectInstance>();

    private void Awake()
    {
        if (Instance == null)
        {
            Instance = this;
            DontDestroyOnLoad(gameObject);
        }
        else
        {
            Destroy(gameObject);
            return;
        }

        // 初始化组件
        effectPool = GetComponent<EffectPool>();
        lodManager = GetComponent<LODManager>();
        performanceMonitor = GetComponent<PerformanceMonitor>();

        // 配置LOD
        if (useLODSystem && lodManager != null)
        {
            lodManager.SetLODLevels(lod1Distance, lod2Distance, lod3Distance);
        }
    }

    /// <summary>
    /// 播放特效
    /// </summary>
    public void PlayEffect(string effectTypeId, Vector3 position, Quaternion rotation)
    {
        // 查找特效类型定义
        var effectType = FindEffectType(effectTypeId);
        if (effectType == null)
        {
            Debug.LogError($"Effect type not found: {effectTypeId}");
            return;
        }

        // 检查并发数量限制
        if (activeEffects.Count >= maxConcurrentEffects)
        {
            // 移除最旧的特效
            RemoveOldestEffect();
        }

        // 获取特效实例
        EffectInstance instance;
        if (useObjectPooling && effectPool != null)
        {
            instance = effectPool.GetInstance(effectTypeId);
        }
        else
        {
            instance = CreateEffectInstance(effectType);
        }

        // 设置位置和旋转
        instance.transform.position = position;
        instance.transform.rotation = rotation;

        // 播放特效
        instance.Play();

        // 添加到活跃列表
        activeEffects.Add(instance);

        // 触发事件
        EffectEventSystem.Instance?.OnEffectPlayed(instance);

        Debug.Log($"Effect played: {effectTypeId}, Active: {activeEffects.Count}");
    }

    /// <summary>
    /// 停止特效
    /// </summary>
    public void StopEffect(EffectInstance instance)
    {
        if (instance == null)
            return;

        // 停止特效
        instance.Stop();

        // 从活跃列表移除
        activeEffects.Remove(instance);

        // 返回对象池
        if (useObjectPooling && effectPool != null)
        {
            effectPool.ReturnInstance(instance);
        }
        else
        {
            // 销毁实例
            Destroy(instance.gameObject);
        }

        // 触发事件
        EffectEventSystem.Instance?.OnEffectStopped(instance);

        Debug.Log($"Effect stopped: {instance.EffectTypeId}");
    }

    /// <summary>
    /// 停止所有特效
    /// </summary>
    public void StopAllEffects()
    {
        // 创建副本列表(避免在迭代中修改)
        var effectsToStop = new List<EffectInstance>(activeEffects);

        // 停止所有特效
        foreach (var effect in effectsToStop)
        {
            StopEffect(effect);
        }

        Debug.Log($"Stopped {effectsToStop.Count} effects");
    }

    /// <summary>
    /// 移除最旧的特效
    /// </summary>
    private void RemoveOldestEffect()
    {
        if (activeEffects.Count == 0)
            return;

        // 找到最旧的特效
        EffectInstance oldestEffect = activeEffects[0];

        // 停止特效
        StopEffect(oldestEffect);
    }

    /// <summary>
    /// 查找特效类型定义
    /// </summary>
    private CombatEffectAsset FindEffectType(string effectTypeId)
    {
        if (effectTypes == null)
            return null;

        return effectTypes.FirstOrDefault(e => e.effectTypeId == effectTypeId);
    }

    /// <summary>
    /// 创建特效实例
    /// </summary>
    private EffectInstance CreateEffectInstance(CombatEffectAsset effectType)
    {
        // 创建游戏对象
        var go = new GameObject(effectType.effectTypeId);
        go.transform.SetParent(transform);

        // 添加特效实例组件
        var instance = go.AddComponent<EffectInstance>();
        instance.Initialize(effectType);

        return instance;
    }

    /// <summary>
    /// 更新(每帧调用)
    /// </summary>
    private void Update()
    {
        // 更新LOD系统
        if (useLODSystem && lodManager != null && cameraTransform != null)
        {
            lodManager.UpdateLOD(activeEffects, cameraTransform.position);
        }

        // 更新特效
        foreach (var effect in activeEffects)
        {
            effect.UpdateEffect();
        }

        // 更新性能监控
        if (performanceMonitor != null)
        {
            performanceMonitor.UpdateEffects(activeEffects);
        }
    }

    /// <summary>
    /// 获取特效统计
    /// </summary>
    public string GetEffectStatistics()
    {
        var stats = new StringBuilder();
        stats.AppendLine($"Active Effects: {activeEffects.Count}");
        stats.AppendLine($"Max Concurrent: {maxConcurrentEffects}");
        stats.AppendLine($"Object Pooling: {useObjectPooling}");
        stats.AppendLine($"LOD System: {useLODSystem}");

        // 按类型统计
        var typeCounts = new Dictionary<string, int>();
        foreach (var effect in activeEffects)
        {
            string typeId = effect.EffectTypeId;
            if (!typeCounts.ContainsKey(typeId))
            {
                typeCounts[typeId] = 0;
            }
            typeCounts[typeId]++;
        }

        stats.AppendLine("\nEffect Types:");
        foreach (var kvp in typeCounts)
        {
            stats.AppendLine($"  {kvp.Key}: {kvp.Value}");
        }

        return stats.ToString();
    }
}
```

```csharp
/// <summary>
/// 战斗特效资源定义

[CreateAssetMenu(fileName = "CombatEffect", menuName = "Combat Effects/New Combat Effect", order = 1)]
public class CombatEffectAsset : ScriptableObject
{
    [Header("基本信息")]
    [SerializeField] private string effectTypeId = "FireEffect";
    [SerializeField] private string effectName = "火焰特效";
    [SerializeField] [TextArea] private string description;

    [Header("特效资源")]
    [SerializeField] private VisualEffectAsset vfxAsset;
    [SerializeField] private ParticleSystem particleSystem;
    [SerializeField] private GameObject prefab;

    [Header("特效配置")]
    [SerializeField] private int baseParticleCount = 1000;
    [SerializeField] private float baseLifetime = 3f;
    [SerializeField] private float baseSize = 1f;

    [Header("LOD配置")]
    [SerializeField] private int lod1ParticleCount = 500;
    [SerializeField] private int lod2ParticleCount = 200;
    [SerializeField] private int lod3ParticleCount = 50;

    [Header("碰撞配置")]
    [SerializeField] private bool enableCollision = false;
    [SerializeField] private float collisionRadius = 1f;
    [SerializeField] private LayerMask collisionLayer;

    [Header("音频配置")]
    [SerializeField] private AudioClip spawnSound;
    [SerializeField] private AudioClip loopSound;
    [SerializeField] private AudioClip stopSound;

    [Header("光照配置")]
    [SerializeField] private bool enableLighting = false;
    [SerializeField] private LightType lightType = LightType.Point;
    [SerializeField] private Color lightColor = Color.orange;
    [SerializeField] private float lightIntensity = 1f;
    [SerializeField] private float lightRange = 5f;

    /// <summary>
    /// 获取LOD粒子数
    /// </summary>
    public int GetLODParticleCount(int lodLevel)
    {
        switch (lodLevel)
        {
            case 0: return baseParticleCount;
            case 1: return lod1ParticleCount;
            case 2: return lod2ParticleCount;
            case 3: return lod3ParticleCount;
            default: return baseParticleCount;
        }
    }

    /// <summary>
    /// 获取LOD生命周期
    /// </summary>
    public float GetLODLifetime(int lodLevel)
    {
        return baseLifetime; // 生命周期通常不随LOD变化
    }

    /// <summary>
    /// 获取LOD尺寸
    /// </summary>
    public float GetLODSize(int lodLevel)
    {
        float sizeMultiplier = 1f;
        switch (lodLevel)
        {
            case 0: sizeMultiplier = 1f; break;
            case 1: sizeMultiplier = 0.8f; break;
            case 2: sizeMultiplier = 0.5f; break;
            case 3: sizeMultiplier = 0.3f; break;
        }
        return baseSize * sizeMultiplier;
    }
}
```

## 特效实例

### EffectInstance类

```csharp
/// <summary>
/// 特效实例
/// </summary>
public class EffectInstance : MonoBehaviour
{
    // 特效类型定义
    private CombatEffectAsset effectType;

    // 特效组件
    private VisualEffect vfxGraph;
    private ParticleSystem particleSystem;

    // LOD级别
    private int currentLODLevel = 0;

    // 状态
    private bool isPlaying;
    private float age;

    /// <summary>
    /// 特效类型ID
    /// </summary>
    public string EffectTypeId => effectType != null ? effectType.effectTypeId : "";

    /// <summary>
    /// 初始化
    /// </summary>
    public void Initialize(CombatEffectAsset asset)
    {
        effectType = asset;

        // 创建VFX Graph
        if (asset.vfxAsset != null)
        {
            vfxGraph = VisualEffect.Instantiate(asset.vfxAsset);
            vfxGraph.transform.SetParent(transform, false);
            vfxGraph.transform.localPosition = Vector3.zero;
            vfxGraph.transform.localRotation = Quaternion.identity;

            // 配置粒子数
            var spawnBlock = vfxGraph.GetBlock("Spawn");
            if (spawnBlock != null)
            {
                spawnBlock.SetFloat("Count", asset.baseParticleCount);
            }
        }

        // 创建Shuriken粒子系统
        if (asset.particleSystem != null)
        {
            particleSystem = Instantiate(asset.particleSystem);
            particleSystem.transform.SetParent(transform, false);
            particleSystem.transform.localPosition = Vector3.zero;
            particleSystem.transform.localRotation = Quaternion.identity;
        }

        // 加载预制体
        if (asset.prefab != null)
        {
            Instantiate(asset.prefab, transform);
        }

        // 配置碰撞
        if (asset.enableCollision)
        {
            ConfigureCollision(asset);
        }

        // 配置光照
        if (asset.enableLighting)
        {
            ConfigureLighting(asset);
        }
    }

    /// <summary>
    /// 播放特效
    /// </summary>
    public void Play()
    {
        if (isPlaying)
            return;

        // 重置状态
        age = 0f;
        currentLODLevel = 0;

        // 播放VFX Graph
        if (vfxGraph != null)
        {
            vfxGraph.Play();
        }

        // 播放粒子系统
        if (particleSystem != null)
        {
            var emission = particleSystem.emission;
            emission.rateOverTime = effectType.baseParticleCount;
            particleSystem.Play();
        }

        // 播放生成音效
        if (effectType.spawnSound != null)
        {
            AudioSource.PlayClipAtPoint(effectType.spawnSound, transform.position, 1f);
        }

        isPlaying = true;
    }

    /// <summary>
    /// 停止特效
    /// </summary>
    public void Stop()
    {
        if (!isPlaying)
            return;

        // 停止VFX Graph
        if (vfxGraph != null)
        {
            vfxGraph.Stop();
        }

        // 停止粒子系统
        if (particleSystem != null)
        {
            particleSystem.Stop();
        }

        // 播放停止音效
        if (effectType.stopSound != null)
        {
            AudioSource.PlayClipAtPoint(effectType.stopSound, transform.position, 1f);
        }

        isPlaying = false;
    }

    /// <summary>
    /// 更新特效(每帧调用)
    /// </summary>
    public void UpdateEffect()
    {
        if (!isPlaying)
            return;

        // 更新年龄
        age += Time.deltaTime;

        // 检查生命周期
        if (age >= effectType.baseLifetime)
        {
            CombatEffectManager.Instance?.StopEffect(this);
            return;
        }

        // 更新VFX Graph
        if (vfxGraph != null)
        {
            // 更新粒子数(根据LOD)
            var spawnBlock = vfxGraph.GetBlock("Spawn");
            if (spawnBlock != null)
            {
                int targetParticleCount = effectType.GetLODParticleCount(currentLODLevel);
                spawnBlock.SetFloat("Count", targetParticleCount);
            }
        }

        // 更新粒子系统
        if (particleSystem != null)
        {
            // 更新生命周期
            var main = particleSystem.main;
            if (main != null)
            {
                main.startLifetime = effectType.baseLifetime;
            }
        }
    }

    /// <summary>
    /// 设置LOD级别
    /// </summary>
    public void SetLODLevel(int lodLevel)
    {
        if (lodLevel < 0 || lodLevel > 3)
            return;

        currentLODLevel = lodLevel;

        // 更新VFX Graph
        if (vfxGraph != null)
        {
            var spawnBlock = vfxGraph.GetBlock("Spawn");
            if (spawnBlock != null)
            {
                int targetParticleCount = effectType.GetLODParticleCount(lodLevel);
                spawnBlock.SetFloat("Count", targetParticleCount);
            }
        }

        // 更新粒子系统
        if (particleSystem != null)
        {
            var emission = particleSystem.emission;
            emission.rateOverTime = effectType.GetLODParticleCount(lodLevel);
        }

        Debug.Log($"LOD level set to {lodLevel} for {EffectTypeId}");
    }

    /// <summary>
    /// 配置碰撞
    /// </summary>
    private void ConfigureCollision(CombatEffectAsset asset)
    {
        // (简化实现,实际需要配置VFX Graph的Collision Block)
        if (asset.enableCollision)
        {
            // VFX Graph碰撞配置
            if (vfxGraph != null)
            {
                var collisionBlock = vfxGraph.GetBlock("Collision");
                if (collisionBlock != null)
                {
                    collisionBlock.enabled = true;
                    collisionBlock.SetVector3("Collider", asset.collisionRadius * Vector3.one);
                }
            }
        }
    }

    /// <summary>
    /// 配置光照
    /// </summary>
    private void ConfigureLighting(CombatEffectAsset asset)
    {
        if (!asset.enableLighting)
            return;

        // 创建光源
        var light = gameObject.AddComponent<Light>();
        light.type = asset.lightType;
        light.color = asset.lightColor;
        light.intensity = asset.lightIntensity;
        light.range = asset.lightRange;
        light.shadows = LightShadows.Soft;

        // 设置光源模式
        if (vfxGraph != null)
        {
            light.renderMode = LightRenderMode.ForcePixel;
        }
    }
}
```

## 特效对象池

### EffectPool类

```csharp
/// <summary>
/// 特效对象池
/// </summary>
public class EffectPool : MonoBehaviour
{
    // 单例
    public static EffectPool Instance { get; private set; }

    // 对象池
    [Serializable]
    private class EffectPoolData
    {
        public string effectTypeId;
        public int poolSize;
        public GameObject prefab;
        public Queue<GameObject> pool = new();
    }

    [SerializeField] private EffectPoolData[] poolConfigs;

    // 活跃实例
    private Dictionary<string, EffectPoolData> pools = new();

    private void Awake()
    {
        if (Instance == null)
        {
            Instance = this;
            DontDestroyOnLoad(gameObject);
        }
        else
        {
            Destroy(gameObject);
            return;
        }

        // 初始化对象池
        InitializePools();
    }

    /// <summary>
    /// 初始化对象池
    /// </summary>
    private void InitializePools()
    {
        pools.Clear();

        // 为每种特效类型创建对象池
        foreach (var config in poolConfigs)
        {
            var poolData = new EffectPoolData
            {
                effectTypeId = config.effectTypeId,
                poolSize = config.poolSize,
                prefab = config.prefab
            };

            // 预创建对象
            for (int i = 0; i < poolData.poolSize; i++)
            {
                if (poolData.prefab != null)
                {
                    var obj = Instantiate(poolData.prefab, transform);
                    obj.SetActive(false);
                    poolData.pool.Enqueue(obj);
                }
            }

            pools[config.effectTypeId] = poolData;

            Debug.Log($"Initialized pool for {config.effectTypeId} with {poolData.poolSize} objects");
        }
    }

    /// <summary>
    /// 获取特效实例
    /// </summary>
    public EffectInstance GetInstance(string effectTypeId)
    {
        // 查找对象池
        if (!pools.TryGetValue(effectTypeId, out var poolData))
        {
            Debug.LogWarning($"Pool not found for {effectTypeId}");
            return CreateNewInstance(effectTypeId);
        }

        GameObject obj = null;

        // 从池中获取
        if (poolData.pool.Count > 0)
        {
            obj = poolData.pool.Dequeue();
            obj.SetActive(true);
            Debug.Log($"Effect {effectTypeId} retrieved from pool (Remaining: {poolData.pool.Count})");
        }
        // 池已空,创建新的
        else
        {
            if (poolData.prefab != null)
            {
                obj = Instantiate(poolData.prefab, transform);
                Debug.Log($"Effect {effectTypeId} created (pool empty)");
            }
        }

        if (obj == null)
        {
            Debug.LogError($"Failed to get instance for {effectTypeId}");
            return null;
        }

        // 获取或添加EffectInstance组件
        var instance = obj.GetComponent<EffectInstance>();
        if (instance == null)
        {
            instance = obj.AddComponent<EffectInstance>();
        }

        return instance;
    }

    /// <summary>
    /// 返回特效实例
    /// </summary>
    public void ReturnInstance(EffectInstance instance)
    {
        if (instance == null)
            return;

        string effectTypeId = instance.EffectTypeId;

        // 查找对象池
        if (!pools.TryGetValue(effectTypeId, out var poolData))
        {
            Debug.LogWarning($"Pool not found for {effectTypeId}, destroying instance");
            Destroy(instance.gameObject);
            return;
        }

        // 检查池是否已满
        if (poolData.pool.Count >= poolData.poolSize)
        {
            Debug.LogWarning($"Pool for {effectTypeId} is full, destroying instance");
            Destroy(instance.gameObject);
            return;
        }

        // 停止特效
        instance.Stop();

        // 禁用对象
        instance.gameObject.SetActive(false);

        // 放回池中
        poolData.pool.Enqueue(instance.gameObject);

        Debug.Log($"Effect {effectTypeId} returned to pool (Count: {poolData.pool.Count})");
    }

    /// <summary>
    /// 创建新实例
    /// </summary>
    private EffectInstance CreateNewInstance(string effectTypeId)
    {
        Debug.LogWarning($"Creating new instance for {effectTypeId} (pool not available)");

        // 创建游戏对象
        var go = new GameObject(effectTypeId);
        go.transform.SetParent(transform);

        // 添加EffectInstance组件
        var instance = go.AddComponent<EffectInstance>();

        // TODO: 需要从CombatEffectManager加载effectType
        // instance.Initialize(effectType);

        return instance;
    }

    /// <summary>
    /// 获取池统计
    /// </summary>
    public string GetPoolStatistics()
    {
        var stats = new StringBuilder();
        stats.AppendLine($"Effect Pool Statistics:");
        stats.AppendLine($"Total Pools: {pools.Count}");

        foreach (var kvp in pools)
        {
            string typeId = kvp.Key;
            var poolData = kvp.Value;

            stats.AppendLine($"\n{typeId}:");
            stats.AppendLine($"  Pool Size: {poolData.poolSize}");
            stats.AppendLine($"  Available: {poolData.pool.Count}");
            stats.AppendLine($"  In Use: {poolData.poolSize - poolData.pool.Count}");
        }

        return stats.ToString();
    }
}
```

```csharp
/// <summary>
/// 特效LOD管理器
/// </summary>
public class LODManager : MonoBehaviour
{
    // LOD距离配置
    [SerializeField] private float lod1Distance = 30f;
    [SerializeField] private float lod2Distance = 60f;
    [SerializeField] private float lod3Distance = 100f;

    // 更新频率
    [SerializeField] private float updateInterval = 0.5f;

    /// <summary>
    /// 设置LOD级别
    /// </summary>
    public void SetLODLevels(float lod1Dist, float lod2Dist, float lod3Dist)
    {
        lod1Distance = lod1Dist;
        lod2Distance = lod2Dist;
        lod3Distance = lod3Dist;
    }

    /// <summary>
    /// 更新LOD(每帧调用)
    /// </summary>
    public void UpdateLOD(List<EffectInstance> effects, Vector3 cameraPosition)
    {
        foreach (var effect in effects)
        {
            if (effect == null)
                continue;

            // 计算距离
            float distance = Vector3.Distance(effect.transform.position, cameraPosition);

            // 计算LOD级别
            int newLODLevel = CalculateLODLevel(distance);

            // 设置LOD级别
            if (newLODLevel != effect.currentLODLevel)
            {
                effect.SetLODLevel(newLODLevel);
            }
        }
    }

    /// <summary>
    /// 计算LOD级别
    /// </summary>
    private int CalculateLODLevel(float distance)
    {
        if (distance < lod1Distance)
        {
            return 0; // LOD0 - 完整质量
        }
        else if (distance < lod2Distance)
        {
            return 1; // LOD1 - 50%质量
        }
        else if (distance < lod3Distance)
        {
            return 2; // LOD2 - 20%质量
        }
        else
        {
            return 3; // LOD3 - 5%质量
        }
    }

    /// <summary>
    /// 获取LOD级别描述
    /// </summary>
    public string GetLODLevelDescription(int lodLevel)
    {
        switch (lodLevel)
        {
            case 0: return "LOD0 - 完整质量 (100% 粒子)";
            case 1: return "LOD1 - 高质量 (50% 粒子)";
            case 2: return "LOD2 - 中等质量 (20% 粒子)";
            case 3: return "LOD3 - 低质量 (5% 粒子)";
            default: return "Unknown LOD";
        }
    }
}
```

```csharp
/// <summary>
/// 特效性能监控器
/// </summary>
public class PerformanceMonitor : MonoBehaviour
{
    // 统计数据
    private int totalEffectsPlayed;
    private int totalEffectsStopped;
    private float totalEffectTime;

    // 性能指标
    [SerializeField] private float targetFrameTime = 5f;  // 目标: 60FPS

    /// <summary>
    /// 更新特效统计
    /// </summary>
    public void UpdateEffects(List<EffectInstance> activeEffects)
    {
        // 更新统计
        totalEffectsPlayed = activeEffects.Count;
        totalEffectTime += Time.deltaTime;

        // 检查性能
        if (Time.deltaTime > targetFrameTime)
        {
            Debug.LogWarning($"Effect performance warning: {Time.deltaTime:F2}ms > {targetFrameTime:F2}ms");
        }
    }

    /// <summary>
    /// 获取性能统计
    /// </summary>
    public string GetPerformanceStatistics()
    {
        var stats = new StringBuilder();
        stats.AppendLine("Effect Performance Statistics:");
        stats.AppendLine($"Total Effects Played: {totalEffectsPlayed}");
        stats.AppendLine($"Total Effects Stopped: {totalEffectsStopped}");
        stats.AppendLine($"Total Effect Time: {totalEffectTime:F2}s");
        stats.AppendLine($"Average Time Per Effect: {totalEffectsPlayed > 0 ? totalEffectTime / totalEffectsPlayed : 0:F3}s");
        stats.AppendLine($"Target Frame Time: {targetFrameTime:F2}ms");

        // 帧率计算
        float averageEffectFrameTime = totalEffectsPlayed > 0 ? totalEffectTime / totalEffectsPlayed : 0;
        float estimatedFPS = 1f / Mathf.Max(0.001f, averageEffectFrameTime);

        stats.AppendLine($"Estimated FPS (Effects Only): {estimatedFPS:F1}");

        return stats.ToString();
    }
}
```
