## 一、状态同步模型

### 基本原理

```
客户端A                    服务器                    客户端B
   │                        │                          │
   │──── 输入 ────────────>│                          │
   │                        │──── 状态广播 ───────────>│
   │                        │                          │
   │<────────────────── 状态广播 ──────────────────────│
   │                        │                          │
```

### 实现

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 网络状态同步管理器
/// </summary>
public class StateSynchronization : MonoBehaviour
{
    [Header("同步配置")]
    [SerializeField] private float syncInterval = 0.05f;  // 20次/秒
    [SerializeField] private float positionThreshold = 0.1f;
    [SerializeField] private float rotationThreshold = 5f;

    private float lastSyncTime;
    private readonly Dictionary<int, NetworkEntity> entities = new Dictionary<int, NetworkEntity>();

    /// <summary>
    /// 网络实体数据
    /// </summary>
    public struct EntityState
    {
        public int entityId;
        public Vector3 position;
        public Quaternion rotation;
        public Vector3 velocity;
        public float timestamp;
    }

    /// <summary>
    /// 网络实体
    /// </summary>
    public class NetworkEntity : MonoBehaviour
    {
        public int entityId;
        public bool isOwner;

        private Vector3 lastSentPosition;
        private Quaternion lastSentRotation;

        public EntityState GetState()
        {
            return new EntityState
            {
                entityId = entityId,
                position = transform.position,
                rotation = transform.rotation,
                velocity = GetComponent<Rigidbody>()?.velocity ?? Vector3.zero,
                timestamp = Time.time
            };
        }

        public bool HasStateChanged(float posThreshold, float rotThreshold)
        {
            if (Vector3.Distance(transform.position, lastSentPosition) > posThreshold)
                return true;

            if (Quaternion.Angle(transform.rotation, lastSentRotation) > rotThreshold)
                return true;

            return false;
        }

        public void MarkSynced()
        {
            lastSentPosition = transform.position;
            lastSentRotation = transform.rotation;
        }
    }

    /// <summary>
    /// 序列化状态数据
    /// </summary>
    public static byte[] SerializeState(EntityState state)
    {
        var data = new byte[48];  // 4 + 12 + 16 + 12 + 4
        int offset = 0;

        // Entity ID
        BitConverter.TryWriteBytes(data.AsSpan(offset), state.entityId); offset += 4;

        // Position
        BitConverter.TryWriteBytes(data.AsSpan(offset), state.position.x); offset += 4;
        BitConverter.TryWriteBytes(data.AsSpan(offset), state.position.y); offset += 4;
        BitConverter.TryWriteBytes(data.AsSpan(offset), state.position.z); offset += 4;

        // Rotation (quaternion)
        BitConverter.TryWriteBytes(data.AsSpan(offset), state.rotation.x); offset += 4;
        BitConverter.TryWriteBytes(data.AsSpan(offset), state.rotation.y); offset += 4;
        BitConverter.TryWriteBytes(data.AsSpan(offset), state.rotation.z); offset += 4;
        BitConverter.TryWriteBytes(data.AsSpan(offset), state.rotation.w); offset += 4;

        // Velocity
        BitConverter.TryWriteBytes(data.AsSpan(offset), state.velocity.x); offset += 4;
        BitConverter.TryWriteBytes(data.AsSpan(offset), state.velocity.y); offset += 4;
        BitConverter.TryWriteBytes(data.AsSpan(offset), state.velocity.z); offset += 4;

        // Timestamp
        BitConverter.TryWriteBytes(data.AsSpan(offset), state.timestamp);

        return data;
    }

    /// <summary>
    /// 反序列化状态数据
    /// </summary>
    public static EntityState DeserializeState(byte[] data, int offset = 0)
    {
        var state = new EntityState();

        state.entityId = BitConverter.ToInt32(data, offset); offset += 4;

        state.position = new Vector3(
            BitConverter.ToSingle(data, offset),
            BitConverter.ToSingle(data, offset + 4),
            BitConverter.ToSingle(data, offset + 8)
        ); offset += 12;

        state.rotation = new Quaternion(
            BitConverter.ToSingle(data, offset),
            BitConverter.ToSingle(data, offset + 4),
            BitConverter.ToSingle(data, offset + 8),
            BitConverter.ToSingle(data, offset + 12)
        ); offset += 16;

        state.velocity = new Vector3(
            BitConverter.ToSingle(data, offset),
            BitConverter.ToSingle(data, offset + 4),
            BitConverter.ToSingle(data, offset + 8)
        ); offset += 12;

        state.timestamp = BitConverter.ToSingle(data, offset);

        return state;
    }

    private void Update()
    {
        if (Time.time - lastSyncTime >= syncInterval)
        {
            SyncEntities();
            lastSyncTime = Time.time;
        }
    }

    private void SyncEntities()
    {
        foreach (var kvp in entities)
        {
            var entity = kvp.Value;

            // 只同步自己拥有的实体
            if (!entity.isOwner) continue;

            // 检查状态是否变化
            if (entity.HasStateChanged(positionThreshold, rotationThreshold))
            {
                SendStateToServer(entity.GetState());
                entity.MarkSynced();
            }
        }
    }

    private void SendStateToServer(EntityState state)
    {
        // TODO: 通过网络发送
        // var data = SerializeState(state);
        // networkClient.Send(data);
    }

    /// <summary>
    /// 处理接收到的状态更新
    /// </summary>
    public void OnStateReceived(EntityState state)
    {
        if (entities.TryGetValue(state.entityId, out var entity))
        {
            if (!entity.isOwner)  // 不覆盖自己控制的实体
            {
                ApplyState(entity, state);
            }
        }
    }

    /// <summary>
    /// 应用状态到实体
    /// </summary>
    private void ApplyState(NetworkEntity entity, EntityState state)
    {
        // 使用插值平滑过渡
        StartCoroutine(InterpolateToState(entity, state));
    }

    private System.Collections.IEnumerator InterpolateToState(NetworkEntity entity, EntityState state)
    {
        var startPos = entity.transform.position;
        var startRot = entity.transform.rotation;
        var targetPos = state.position;
        var targetRot = state.rotation;

        float duration = syncInterval;
        float elapsed = 0f;

        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;
            float t = elapsed / duration;

            entity.transform.position = Vector3.Lerp(startPos, targetPos, t);
            entity.transform.rotation = Quaternion.Slerp(startRot, targetRot, t);

            yield return null;
        }

        entity.transform.position = targetPos;
        entity.transform.rotation = targetRot;
    }
}
```

## 二、客户端预测

### 原理图

```
┌──────────────────────────────────────────────────────────────┐
│                      客户端预测流程                           │
├──────────────────────────────────────────────────────────────┤
│                                                              │
│  输入 ──> 本地立即执行 ──> 发送到服务器 ──> 接收服务器确认    │
│              │                                     │          │
│              v                                     v          │
│           预测状态 <─────────── 比较 ──────────  服务器状态   │
│                                    │                         │
│                              误差 > 阈值?                     │
│                               │        │                     │
│                              Yes       No                    │
│                               │        │                     │
│                           回滚修正    继续执行                │
│                                                              │
└──────────────────────────────────────────────────────────────┘
```

### 实现

```csharp
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 客户端预测系统
/// </summary>
public class ClientPrediction : MonoBehaviour
{
    [Header("预测配置")]
    [SerializeField] private float reconciliationThreshold = 2f;  // 2米误差触发修正
    [SerializeField] private int maxStoredInputs = 60;  // 存储60帧输入

    private struct PredictedState
    {
        public Vector3 position;
        public Quaternion rotation;
        public int inputSequence;
        public float timestamp;
    }

    private struct PlayerInput
    {
        public float horizontal;
        public float vertical;
        public bool jump;
        public int sequence;
        public float timestamp;
    }

    private readonly Queue<PlayerInput> pendingInputs = new Queue<PlayerInput>();
    private readonly List<PredictedState> predictedStates = new List<PredictedState>();
    private int inputSequence;

    private CharacterController controller;
    private Vector3 velocity;
    private bool isGrounded;

    public event Action<PlayerInput> OnInputSent;

    private void Awake()
    {
        controller = GetComponent<CharacterController>();
    }

    private void Update()
    {
        // 收集输入
        var input = CollectInput();

        // 本地预测执行
        ApplyInput(input);

        // 存储预测状态
        StorePrediction(input);

        // 发送输入到服务器
        SendInputToServer(input);
    }

    private PlayerInput CollectInput()
    {
        return new PlayerInput
        {
            horizontal = Input.GetAxis("Horizontal"),
            vertical = Input.GetAxis("Vertical"),
            jump = Input.GetButtonDown("Jump"),
            sequence = inputSequence++,
            timestamp = Time.time
        };
    }

    /// <summary>
    /// 本地预测执行输入
    /// </summary>
    private void ApplyInput(PlayerInput input)
    {
        // 移动
        Vector3 move = new Vector3(input.horizontal, 0, input.vertical);
        controller.Move(move * 5f * Time.deltaTime);

        // 重力
        isGrounded = controller.isGrounded;
        if (!isGrounded)
        {
            velocity.y += Physics.gravity.y * Time.deltaTime;
        }
        else if (velocity.y < 0)
        {
            velocity.y = -2f;
        }

        // 跳跃
        if (input.jump && isGrounded)
        {
            velocity.y = 5f;
        }

        controller.Move(velocity * Time.deltaTime);
    }

    private void StorePrediction(PlayerInput input)
    {
        predictedStates.Add(new PredictedState
        {
            position = transform.position,
            rotation = transform.rotation,
            inputSequence = input.sequence,
            timestamp = Time.time
        });

        // 限制存储数量
        if (predictedStates.Count > maxStoredInputs)
        {
            predictedStates.RemoveAt(0);
        }

        pendingInputs.Enqueue(input);
    }

    private void SendInputToServer(PlayerInput input)
    {
        OnInputSent?.Invoke(input);
    }

    /// <summary>
    /// 处理服务器确认的状态
    /// </summary>
    public void OnServerStateReceived(Vector3 serverPosition, int ackSequence)
    {
        // 移除已确认的输入
        while (pendingInputs.Count > 0 && pendingInputs.Peek().sequence <= ackSequence)
        {
            pendingInputs.Dequeue();
        }

        // 查找对应的预测状态
        var predictedState = predictedStates.Find(s => s.inputSequence == ackSequence);

        if (predictedState.position != Vector3.zero)
        {
            float error = Vector3.Distance(predictedState.position, serverPosition);

            if (error > reconciliationThreshold)
            {
                // 误差过大，需要修正
                Debug.Log($"状态修正: 误差 {error:F2}米");
                Reconcile(serverPosition, ackSequence);
            }
        }

        // 清理已确认的预测状态
        predictedStates.RemoveAll(s => s.inputSequence <= ackSequence);
    }

    /// <summary>
    /// 状态修正
    /// </summary>
    private void Reconcile(Vector3 serverPosition, int fromSequence)
    {
        // 1. 回滚到服务器状态
        transform.position = serverPosition;

        // 2. 重放未确认的输入
        foreach (var input in pendingInputs)
        {
            ApplyInput(input);
        }

        Debug.Log("状态修正完成");
    }
}
```

## 三、延迟补偿

### 原理

```
┌────────────────────────────────────────────────────────────┐
│                    延迟补偿原理                             │
├────────────────────────────────────────────────────────────┤
│                                                            │
│  时间线: ──────────────────────────────────────────────>   │
│                                                            │
│  客户端A (RTT=100ms)                                       │
│     │                                                      │
│     ├──── t=0 ──── 发送射击 ───────────────────────┐       │
│     │                                              │       │
│     │<──── t=100ms ── 收到确认                     │       │
│     │                                              │       │
│  服务器                                           │       │
│     │                                              │       │
│     ├──── t=50ms ──── 收到射击请求 ────────────────┘       │
│     │                                                      │
│     │   回滚客户端B到 t-50ms 时的位置                       │
│     │   进行命中判定                                        │
│     │                                                      │
│     ├──── t=50ms ──── 发送结果给双方                       │
│     │                                                      │
└────────────────────────────────────────────────────────────┘
```

### 实现

```csharp
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 服务器端延迟补偿
/// </summary>
public class LagCompensation : MonoBehaviour
{
    [Header("补偿配置")]
    [SerializeField] private float historyDuration = 1f;  // 存储1秒历史
    [SerializeField] private float snapshotInterval = 0.05f;  // 20Hz

    private struct PlayerSnapshot
    {
        public int playerId;
        public Vector3 position;
        public Quaternion rotation;
        public Bounds hitbox;
        public float timestamp;
    }

    private class PlayerHistory
    {
        public int playerId;
        public Transform transform;
        public Collider hitbox;
        public readonly List<PlayerSnapshot> snapshots = new List<PlayerSnapshot>();
    }

    private readonly Dictionary<int, PlayerHistory> playerHistories = new Dictionary<int, PlayerHistory>();

    /// <summary>
    /// 记录玩家快照
    /// </summary>
    private void RecordSnapshots()
    {
        float currentTime = Time.time;
        float cutoffTime = currentTime - historyDuration;

        foreach (var history in playerHistories.Values)
        {
            // 添加新快照
            history.snapshots.Add(new PlayerSnapshot
            {
                playerId = history.playerId,
                position = history.transform.position,
                rotation = history.transform.rotation,
                hitbox = history.hitbox.bounds,
                timestamp = currentTime
            });

            // 移除过期快照
            history.snapshots.RemoveAll(s => s.timestamp < cutoffTime);
        }
    }

    /// <summary>
    /// 获取指定时间点的玩家状态
    /// </summary>
    private PlayerSnapshot GetSnapshotAtTime(int playerId, float targetTime)
    {
        if (!playerHistories.TryGetValue(playerId, out var history))
        {
            return default;
        }

        // 查找最近的两个快照
        PlayerSnapshot before = default;
        PlayerSnapshot after = default;

        foreach (var snapshot in history.snapshots)
        {
            if (snapshot.timestamp <= targetTime)
            {
                before = snapshot;
            }
            else if (after.timestamp == 0)
            {
                after = snapshot;
                break;
            }
        }

        if (before.timestamp == 0) return after;
        if (after.timestamp == 0) return before;

        // 插值
        float t = (targetTime - before.timestamp) / (after.timestamp - before.timestamp);
        return InterpolateSnapshots(before, after, t);
    }

    private PlayerSnapshot InterpolateSnapshots(PlayerSnapshot a, PlayerSnapshot b, float t)
    {
        return new PlayerSnapshot
        {
            playerId = a.playerId,
            position = Vector3.Lerp(a.position, b.position, t),
            rotation = Quaternion.Slerp(a.rotation, b.rotation, t),
            hitbox = InterpolateBounds(a.hitbox, b.hitbox, t),
            timestamp = Mathf.Lerp(a.timestamp, b.timestamp, t)
        };
    }

    private Bounds InterpolateBounds(Bounds a, Bounds b, float t)
    {
        return new Bounds
        {
            center = Vector3.Lerp(a.center, b.center, t),
            size = Vector3.Lerp(a.size, b.size, t)
        };
    }

    /// <summary>
    /// 执行延迟补偿的命中检测
    /// </summary>
    public bool CheckHit(int shooterPlayerId, Vector3 origin, Vector3 direction, float latency, out int hitPlayerId)
    {
        hitPlayerId = -1;

        // 计算射击时的服务器时间
        float shootTime = Time.time - (latency / 2f);

        // 临时回滚所有玩家
        var originalStates = new Dictionary<int, (Vector3 pos, Quaternion rot)>();

        foreach (var history in playerHistories.Values)
        {
            if (history.playerId == shooterPlayerId) continue;

            originalStates[history.playerId] = (history.transform.position, history.transform.rotation);

            var snapshot = GetSnapshotAtTime(history.playerId, shootTime);
            history.transform.position = snapshot.position;
            history.transform.rotation = snapshot.rotation;
        }

        // 执行射线检测
        bool hit = Physics.Raycast(origin, direction, out var hitInfo, 100f);

        if (hit && hitInfo.collider != null)
        {
            // 查找被击中的玩家
            foreach (var history in playerHistories.Values)
            {
                if (history.hitbox == hitInfo.collider)
                {
                    hitPlayerId = history.playerId;
                    break;
                }
            }
        }

        // 恢复所有玩家位置
        foreach (var kvp in originalStates)
        {
            if (playerHistories.TryGetValue(kvp.Key, out var history))
            {
                history.transform.position = kvp.Value.pos;
                history.transform.rotation = kvp.Value.rot;
            }
        }

        return hit && hitPlayerId >= 0;
    }

    /// <summary>
    /// 注册玩家
    /// </summary>
    public void RegisterPlayer(int playerId, Transform transform, Collider hitbox)
    {
        playerHistories[playerId] = new PlayerHistory
        {
            playerId = playerId,
            transform = transform,
            hitbox = hitbox
        };
    }

    /// <summary>
    /// 注销玩家
    /// </summary>
    public void UnregisterPlayer(int playerId)
    {
        playerHistories.Remove(playerId);
    }
}
```

## 四、插值与平滑

### 远程实体插值

```csharp
using UnityEngine;

/// <summary>
/// 网络实体插值器
/// </summary>
public class NetworkInterpolator : MonoBehaviour
{
    [Header("插值配置")]
    [SerializeField] private float interpolationDelay = 0.1f;  // 100ms延迟
    [SerializeField] private float extrapolationLimit = 0.5f;  // 最大外推时间

    private struct StateSample
    {
        public Vector3 position;
        public Quaternion rotation;
        public Vector3 velocity;
        public double timestamp;
    }

    private readonly StateSample[] stateBuffer = new StateSample[20];
    private int bufferIndex;

    private double renderTime;

    /// <summary>
    /// 添加服务器状态样本
    /// </summary>
    public void AddState(Vector3 position, Quaternion rotation, Vector3 velocity, double serverTime)
    {
        var state = new StateSample
        {
            position = position,
            rotation = rotation,
            velocity = velocity,
            timestamp = serverTime
        };

        stateBuffer[bufferIndex] = state;
        bufferIndex = (bufferIndex + 1) % stateBuffer.Length;
    }

    private void Update()
    {
        // 计算插值目标时间
        renderTime = NetworkTime.Time - interpolationDelay;

        // 查找插值点
        if (TryFindInterpolationPoints(out var before, out var after, out float t))
        {
            // 插值位置
            transform.position = Vector3.Lerp(before.position, after.position, t);
            transform.rotation = Quaternion.Slerp(before.rotation, after.rotation, t);
        }
        else
        {
            // 没有足够数据，使用外推
            Extrapolate();
        }
    }

    private bool TryFindInterpolationPoints(out StateSample before, out StateSample after, out float t)
    {
        before = default;
        after = default;
        t = 0f;

        // 查找renderTime前后的两个样本
        for (int i = 0; i < stateBuffer.Length - 1; i++)
        {
            int idx = (bufferIndex - 2 - i + stateBuffer.Length) % stateBuffer.Length;
            int nextIdx = (idx + 1) % stateBuffer.Length;

            if (stateBuffer[idx].timestamp <= renderTime && stateBuffer[nextIdx].timestamp >= renderTime)
            {
                before = stateBuffer[idx];
                after = stateBuffer[nextIdx];

                double duration = after.timestamp - before.timestamp;
                if (duration > 0)
                {
                    t = (float)((renderTime - before.timestamp) / duration);
                }

                return true;
            }
        }

        return false;
    }

    private void Extrapolate()
    {
        // 找最新的状态
        var latest = stateBuffer[(bufferIndex - 1 + stateBuffer.Length) % stateBuffer.Length];

        if (latest.timestamp == 0) return;

        float timeSinceLastUpdate = (float)(NetworkTime.Time - latest.timestamp);

        if (timeSinceLastUpdate < extrapolationLimit)
        {
            // 线性外推
            transform.position = latest.position + latest.velocity * timeSinceLastUpdate;
        }
    }
}

/// <summary>
/// 网络时间（简化版）
/// </summary>
public static class NetworkTime
{
    private static double serverTimeOffset;

    public static double Time => UnityEngine.Time.timeAsDouble - serverTimeOffset;

    public static void SyncWithServer(double serverTime)
    {
        serverTimeOffset = UnityEngine.Time.timeAsDouble - serverTime;
    }
}
```

## 同步模型选择指南

| 游戏类型 | 推荐模型 | 关键因素 |
|----------|----------|----------|
| **FPS/TPS** | 服务器权威 + 预测回滚 | 公平性、响应速度 |
| **MOBA** | 服务器权威 + 状态同步 | 一致性、作弊防护 |
| **RTS** | 锁步同步 | 确定性、带宽优化 |
| **格斗游戏** | 预测回滚 (GGPO) | 极低延迟要求 |
| **MMORPG** | 状态同步 + 区域化 | 大规模、可靠性 |
| **休闲游戏** | 客户端权威 | 简单、低服务器成本 |
| **回合制** | 锁步/简单状态同步 | 无实时要求 |
