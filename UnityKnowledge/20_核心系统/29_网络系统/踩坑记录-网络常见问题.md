## 连接问题

### 1. 连接超时

**问题描述**: 客户端无法连接服务器，超时断开

**原因分析**:
- 防火墙阻止
- 服务器未启动或端口错误
- 网络NAT穿透问题
- IP地址错误

**解决方案**:

```csharp
// ✅ 推荐：带超时的连接
public async Task<bool> ConnectWithTimeout(string host, int port, int timeoutMs = 5000)
{
    var cts = new CancellationTokenSource(timeoutMs);

    try
    {
        var tcpClient = new TcpClient();
        await tcpClient.ConnectAsync(host, port, cts.Token);
        return true;
    }
    catch (OperationCanceledException)
    {
        Debug.LogError("连接超时");
        return false;
    }
    catch (SocketException ex)
    {
        Debug.LogError($"连接失败: {ex.SocketErrorCode}");
        return false;
    }
    finally
    {
        cts.Dispose();
    }
}
```

**检查清单**:
- [ ] 确认服务器IP和端口正确
- [ ] 检查防火墙规则（入站/出站）
- [ ] 测试本地回环 `127.0.0.1`
- [ ] 确认服务器已启动并监听

### 2. 连接频繁断开

**问题描述**: 连接建立后频繁断开重连

**原因分析**:
- 心跳超时
- 网络不稳定
- 服务器主动断开空闲连接
- 缓冲区溢出

**解决方案**:

```csharp
// ✅ 推荐：心跳保活
public class ConnectionKeeper
{
    private readonly float heartbeatInterval;
    private readonly float timeoutDuration;
    private float lastSendTime;
    private float lastReceiveTime;

    public void Update()
    {
        float now = Time.time;

        // 发送心跳
        if (now - lastSendTime >= heartbeatInterval)
        {
            SendHeartbeat();
            lastSendTime = now;
        }

        // 检测超时
        if (now - lastReceiveTime >= timeoutDuration)
        {
            Debug.LogWarning("连接超时");
            OnTimeout?.Invoke();
        }
    }

    public void OnDataReceived()
    {
        lastReceiveTime = Time.time;
    }
}
```

## 数据传输问题

### 4. 粘包/半包问题

**问题描述**: 消息粘连或截断，解析错误

**表现**:
```
发送: [消息A] [消息B] [消息C]
接收: [消息A+消息B] [消息C半截]
```

**解决方案**:

```csharp
// ✅ 推荐：长度前缀协议
// 协议格式: [4字节长度] [消息内容]

public class MessageFramer
{
    private readonly List<byte> buffer = new List<byte>();
    private int expectedLength = -1;

    public void AddData(byte[] data)
    {
        buffer.AddRange(data);

        while (buffer.Count >= 4)
        {
            if (expectedLength == -1)
            {
                // 读取消息长度
                expectedLength = BitConverter.ToInt32(buffer.ToArray(), 0);
            }

            if (buffer.Count >= 4 + expectedLength)
            {
                // 提取完整消息
                var message = buffer.GetRange(4, expectedLength).ToArray();

                // 移除已处理数据
                buffer.RemoveRange(0, 4 + expectedLength);
                expectedLength = -1;

                // 处理消息
                OnCompleteMessage?.Invoke(message);
            }
            else
            {
                // 数据不完整，等待更多数据
                break;
            }
        }
    }

    public event Action<byte[]> OnCompleteMessage;

    /// <summary>
    /// 打包消息
    /// </summary>
    public static byte[] FrameMessage(byte[] payload)
    {
        var result = new byte[4 + payload.Length];
        BitConverter.TryWriteBytes(result.AsSpan(), payload.Length);
        Buffer.BlockCopy(payload, 0, result, 4, payload.Length);
        return result;
    }
}
```

### 5. 大端/小端问题

**问题描述**: 不同平台字节序不一致，数据解析错误

**解决方案**:

```csharp
// ✅ 推荐：统一使用网络字节序（大端）
public static class NetworkByteOrder
{
    /// <summary>
    /// 写入int（大端）
    /// </summary>
    public static void WriteInt32BigEndian(Span<byte> buffer, int value)
    {
        buffer[0] = (byte)(value >> 24);
        buffer[1] = (byte)(value >> 16);
        buffer[2] = (byte)(value >> 8);
        buffer[3] = (byte)value;
    }

    /// <summary>
    /// 读取int（大端）
    /// </summary>
    public static int ReadInt32BigEndian(ReadOnlySpan<byte> buffer)
    {
        return (buffer[0] << 24) | (buffer[1] << 16) | (buffer[2] << 8) | buffer[3];
    }

    /// <summary>
    /// 写入float（大端）
    /// </summary>
    public static void WriteFloatBigEndian(Span<byte> buffer, float value)
    {
        var bytes = BitConverter.GetBytes(value);
        if (BitConverter.IsLittleEndian)
        {
            Array.Reverse(bytes);
        }
        bytes.CopyTo(buffer);
    }

    /// <summary>
    /// 读取float（大端）
    /// </summary>
    public static float ReadFloatBigEndian(ReadOnlySpan<byte> buffer)
    {
        var bytes = buffer.Slice(0, 4).ToArray();
        if (BitConverter.IsLittleEndian)
        {
            Array.Reverse(bytes);
        }
        return BitConverter.ToSingle(bytes, 0);
    }
}
```

```csharp
// ❌ 避免：频繁使用JsonUtility
public void SendPosition()
{
    var json = JsonUtility.ToJson(new { x = 1f, y = 2f, z = 3f });  // GC!
    Send(json);
}

// ✅ 推荐：二进制序列化
public struct PositionData
{
    public float x, y, z;

    public void Serialize(Span<byte> buffer)
    {
        BitConverter.TryWriteBytes(buffer.Slice(0, 4), x);
        BitConverter.TryWriteBytes(buffer.Slice(4, 4), y);
        BitConverter.TryWriteBytes(buffer.Slice(8, 4), z);
    }

    public static PositionData Deserialize(ReadOnlySpan<byte> buffer)
    {
        return new PositionData
        {
            x = BitConverter.ToSingle(buffer.Slice(0, 4)),
            y = BitConverter.ToSingle(buffer.Slice(4, 4)),
            z = BitConverter.ToSingle(buffer.Slice(8, 4))
        };
    }
}

// ✅ 推荐：使用Span<T>避免分配
public void SerializePosition(Vector3 pos, Span<byte> buffer)
{
    BitConverter.TryWriteBytes(buffer, pos.x);
    BitConverter.TryWriteBytes(buffer.Slice(4), pos.y);
    BitConverter.TryWriteBytes(buffer.Slice(8), pos.z);
}
```

## 同步问题

### 7. 位置同步抖动

**问题描述**: 远程玩家位置跳动、不流畅

**原因分析**:
- 网络延迟波动
- 插值算法不当
- 同步频率过低

**解决方案**:

```csharp
// ✅ 推荐：带延迟缓冲的插值
public class SmoothInterpolation : MonoBehaviour
{
    [SerializeField] private float interpolationDelay = 0.1f;  // 100ms缓冲

    private readonly Queue<TimedSnapshot> snapshots = new Queue<TimedSnapshot>();

    private struct TimedSnapshot
    {
        public Vector3 position;
        public Quaternion rotation;
        public double timestamp;
    }

    public void AddSnapshot(Vector3 position, Quaternion rotation, double serverTime)
    {
        snapshots.Enqueue(new TimedSnapshot
        {
            position = position,
            rotation = rotation,
            timestamp = serverTime
        });

        // 限制缓冲区大小
        while (snapshots.Count > 20)
        {
            snapshots.Dequeue();
        }
    }

    private void Update()
    {
        double renderTime = NetworkTime.Time - interpolationDelay;

        // 找插值点
        var arr = snapshots.ToArray();
        for (int i = 0; i < arr.Length - 1; i++)
        {
            if (arr[i].timestamp <= renderTime && arr[i + 1].timestamp >= renderTime)
            {
                float t = (float)((renderTime - arr[i].timestamp) /
                                  (arr[i + 1].timestamp - arr[i].timestamp));

                transform.position = Vector3.Lerp(arr[i].position, arr[i + 1].position, t);
                transform.rotation = Quaternion.Slerp(arr[i].rotation, arr[i + 1].rotation, t);
                return;
            }
        }
    }
}
```

### 8. 时钟不同步

**问题描述**: 客户端与服务器时间不一致，影响同步

**解决方案**:

```csharp
// ✅ 推荐：NTP风格的时间同步
public class NetworkTimeSynchronizer : MonoBehaviour
{
    private double serverTimeOffset;
    private double roundTripTime;
    private int samplesCount;

    public double ServerTime => Time.timeAsDouble + serverTimeOffset;
    public double RTT => roundTripTime;

    public IEnumerator SyncTime()
    {
        // 发送多次请求取平均
        for (int i = 0; i < 5; i++)
        {
            double sendTime = Time.timeAsDouble;

            // 发送ping请求
            SendPing();

            // 等待响应
            yield return new WaitUntil(() => pongReceived);

            double receiveTime = Time.timeAsDouble;
            double serverTime = receivedServerTime;

            // 计算RTT
            double currentRTT = receiveTime - sendTime;

            // 计算时间偏移
            double currentOffset = serverTime - (sendTime + currentRTT / 2);

            // 加权平均
            roundTripTime = (roundTripTime * samplesCount + currentRTT) / (samplesCount + 1);
            serverTimeOffset = (serverTimeOffset * samplesCount + currentOffset) / (samplesCount + 1);
            samplesCount++;

            pongReceived = false;

            yield return new WaitForSeconds(0.5f);
        }

        Debug.Log($"时间同步完成: RTT={roundTripTime * 1000:F1}ms, Offset={serverTimeOffset * 1000:F1}ms");
    }

    private bool pongReceived;
    private double receivedServerTime;

    private void OnPongReceived(double serverTime)
    {
        receivedServerTime = serverTime;
        pongReceived = true;
    }

    private void SendPing()
    {
        // 发送ping到服务器
    }
}
```

## 安全问题

### 9. 数据未验证

**问题描述**: 直接信任服务器数据导致崩溃

**解决方案**:

```csharp
// ❌ 避免：直接使用网络数据
public void OnDamageReceived(byte[] data)
{
    int damage = BitConverter.ToInt32(data, 0);
    currentHealth -= damage;  // 危险！可能为负数
}

// ✅ 推荐：数据验证
public void OnDamageReceived(byte[] data)
{
    if (data.Length < 4)
    {
        Debug.LogError("数据长度不足");
        return;
    }

    int damage = BitConverter.ToInt32(data, 0);

    // 验证范围
    if (damage < 0 || damage > MaxDamage)
    {
        Debug.LogError($"无效的伤害值: {damage}");
        return;
    }

    currentHealth -= damage;
}
```

### 10. 重放攻击

**问题描述**: 恶意玩家重复发送有效消息

**解决方案**:

```csharp
// ✅ 推荐：序列号验证
public class MessageValidator
{
    private int lastReceivedSequence = -1;

    public bool ValidateMessage(int sequence)
    {
        // 序列号必须递增
        if (sequence <= lastReceivedSequence)
        {
            Debug.LogWarning($"检测到重放攻击: seq={sequence}, last={lastReceivedSequence}");
            return false;
        }

        lastReceivedSequence = sequence;
        return true;
    }
}

// ✅ 推荐：时间戳验证
public class TimestampValidator
{
    private const float MaxTimeDelta = 5f;  // 5秒有效期

    public bool ValidateTimestamp(float timestamp)
    {
        float delta = Mathf.Abs(Time.time - timestamp);
        return delta <= MaxTimeDelta;
    }
}
```

```csharp
// ✅ 推荐：后台线程 + 主线程回调
public class NetworkThread : MonoBehaviour
{
    private readonly ConcurrentQueue<Action> mainThreadActions = new ConcurrentQueue<Action>();

    private void Update()
    {
        while (mainThreadActions.TryDequeue(out var action))
        {
            action?.Invoke();
        }
    }

    public void OnMessageReceived(string msg)
    {
        // 在后台线程
        mainThreadActions.Enqueue(() =>
        {
            // 在主线程执行
            HandleMessage(msg);
        });
    }
}
```

### 12. 内存泄漏

**问题描述**: 网络缓冲区持续增长

**解决方案**:

```csharp
// ❌ 避免：无限增长的缓冲区
private List<byte> buffer = new List<byte>();

public void OnDataReceived(byte[] data)
{
    buffer.AddRange(data);  // 无限增长！
}

// ✅ 推荐：固定大小缓冲区 + 循环利用
public class CircularBuffer
{
    private readonly byte[] buffer;
    private int writeIndex;
    private int readIndex;
    private int count;

    public CircularBuffer(int capacity)
    {
        buffer = new byte[capacity];
    }

    public bool Write(byte[] data)
    {
        if (count + data.Length > buffer.Length)
            return false;

        foreach (byte b in data)
        {
            buffer[writeIndex] = b;
            writeIndex = (writeIndex + 1) % buffer.Length;
            count++;
        }

        return true;
    }

    public int Read(byte[] output, int offset, int length)
    {
        int read = 0;
        while (read < length && count > 0)
        {
            output[offset + read] = buffer[readIndex];
            readIndex = (readIndex + 1) % buffer.Length;
            count--;
            read++;
        }
        return read;
    }
}
```
