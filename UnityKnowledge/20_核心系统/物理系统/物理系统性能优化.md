## 刚体优化

### 刚体设置

```csharp
// 优化刚体配置
public class OptimizedRigidbody : MonoBehaviour
{
    private Rigidbody rb;

    private void Awake()
    {
        rb = GetComponent<Rigidbody>();
        ConfigureRigidbody();
    }

    private void ConfigureRigidbody()
    {
        // 睡眠阈值 - 让不动的物体更快进入睡眠
        rb.sleepThreshold = 0.005f;  // 默认0.005，可适当提高

        // 睡眠模式
        rb.sleepMode = RigidbodySleepMode.StartAwake;

        // 插值 - 仅对需要平滑的物体使用
        rb.interpolation = RigidbodyInterpolation.None;  // 默认

        // 碰撞检测 - 仅高速物体使用Continuous
        rb.collisionDetectionMode = CollisionDetectionMode.Discrete;
    }

    // 对于需要平滑移动的玩家
    public void ConfigureForPlayer()
    {
        rb.interpolation = RigidbodyInterpolation.Interpolate;
        rb.collisionDetectionMode = CollisionDetectionMode.Continuous;
    }

    // 对于高速弹道
    public void ConfigureForProjectile()
    {
        rb.collisionDetectionMode = CollisionDetectionMode.ContinuousDynamic;
    }
}
```

### 刚体睡眠管理

```csharp
public class RigidbodySleepManager : MonoBehaviour
{
    [SerializeField] private Rigidbody[] rigidbodies;
    [SerializeField] private float wakeDistance = 50f;

    private Transform player;

    private void Start()
    {
        player = GameObject.FindGameObjectWithTag("Player").transform;
    }

    private void Update()
    {
        float distToPlayer = Vector3.Distance(transform.position, player.position);

        if (distToPlayer > wakeDistance)
        {
            // 远离玩家时强制睡眠
            ForceSleepAll();
        }
    }

    private void ForceSleepAll()
    {
        foreach (var rb in rigidbodies)
        {
            if (rb != null && !rb.IsSleeping())
            {
                rb.Sleep();
            }
        }
    }

    public void WakeAll()
    {
        foreach (var rb in rigidbodies)
        {
            if (rb != null)
            {
                rb.WakeUp();
            }
        }
    }
}
```

## 碰撞体优化

### 碰撞体选择

| 类型 | 精度 | 性能 | 使用场景 |
|------|------|------|----------|
| **BoxCollider** | 低 | 最好 | 建筑、家具 |
| **SphereCollider** | 低 | 很好 | 球形物体、触发器 |
| **CapsuleCollider** | 中 | 好 | 角色、胶囊体 |
| **MeshCollider** | 高 | 差 | 复杂形状（尽量避免） |

## 物理时间优化

### Fixed Timestep

```csharp
// Project Settings > Time > Fixed Timestep
// 默认: 0.02 (50Hz)

// 调整建议：
// - 高精度需求（格斗游戏）: 0.01 (100Hz)
// - 普通游戏: 0.02 (50Hz)
// - 性能优先: 0.033 (30Hz)

// 运行时调整
public class PhysicsTimeManager : MonoBehaviour
{
    [SerializeField] private float defaultFixedTimestep = 0.02f;
    [SerializeField] private float performanceFixedTimestep = 0.033f;

    public void SetPerformanceMode()
    {
        Time.fixedDeltaTime = performanceFixedTimestep;
    }

    public void SetDefaultMode()
    {
        Time.fixedDeltaTime = defaultFixedTimestep;
    }
}
```

### 射线检测

public class RaycastOptimizer : MonoBehaviour
{
    [SerializeField] private LayerMask raycastLayers;
    private RaycastHit[] hits = new RaycastHit[10];

    // 使用NonAlloc版本避免GC
    public bool RaycastNonAlloc(Vector3 origin, Vector3 direction, float maxDistance)
    {
        int hitCount = Physics.RaycastNonAlloc(origin, direction, hits, maxDistance, raycastLayers);

        for (int i = 0; i < hitCount; i++)
        {
            if (hits[i].collider != null)
            {
                return true;
            }
        }

        return false;
    }

### 视野检测

```csharp
public class FieldOfView : MonoBehaviour
{
    [SerializeField] private float viewRadius = 10f;
    [SerializeField] private float viewAngle = 90f;
    [SerializeField] private LayerMask targetMask;
    [SerializeField] private LayerMask obstacleMask;
    [SerializeField] private float checkInterval = 0.1f;

    private Collider[] targets = new Collider[20];
    private float lastCheckTime;

    public List<Transform> VisibleTargets { get; } = new List<Transform>();

    private void Update()
    {
        if (Time.time - lastCheckTime < checkInterval)
            return;

        lastCheckTime = Time.time;
        FindVisibleTargets();
    }

    private void FindVisibleTargets()
    {
        VisibleTargets.Clear();

        int targetCount = Physics.OverlapSphereNonAlloc(
            transform.position,
            viewRadius,
            targets,
            targetMask
        );

        for (int i = 0; i < targetCount; i++)
        {
            var target = targets[i].transform;
            Vector3 dirToTarget = (target.position - transform.position).normalized;

            // 角度检测
            if (Vector3.Angle(transform.forward, dirToTarget) < viewAngle / 2)
            {
                float distToTarget = Vector3.Distance(transform.position, target.position);

                // 遮挡检测
                if (!Physics.Raycast(transform.position, dirToTarget, distToTarget, obstacleMask))
                {
                    VisibleTargets.Add(target);
                }
            }
        }
    }
}
```

public class IndependentPhysicsSimulation : MonoBehaviour
