## 2. 游戏管理器

```csharp
using UnityEngine;
using System;

/// <summary>
/// 游戏状态
/// </summary>
public enum GameState
{
    None,
    Loading,
    MainMenu,
    HeroSelect,
    Playing,
    Paused,
    Victory,
    Defeat
}

/// <summary>
/// 游戏管理器 - 主入口
/// </summary>
public class GameManager : MonoSingleton<GameManager>
{
    [Header("References")]
    [SerializeField] private BoardController boardController;
    [SerializeField] private CombatManager combatManager;

    [Header("Config")]
    [SerializeField] private GemConfig gemConfig;
    [SerializeField] private HeroConfig heroConfig;
    [SerializeField] private LevelConfig levelConfig;

    public GameState CurrentState { get; private set; }

    // 子系统
    public UIManager UI => UIManager.Instance;
    public SaveManager Save => SaveManager.Instance;
    public AudioManager Audio => AudioManager.Instance;
    public PoolManager Pool => PoolManager.Instance;

    public event Action<GameState> OnStateChanged;

    protected override void Awake()
    {
        base.Awake();
        InitializeSystems();
    }

    private void Start()
    {
        SetState(GameState.Loading);
        LoadConfigs();
        Save.Initialize();
        SetState(GameState.MainMenu);
    }

    private void InitializeSystems()
    {
        // 初始化各个子系统
        // 大多数子系统使用MonoSingleton自动初始化

        // 订阅事件
        EventBus.Subscribe<CombatStateChangedEvent>(OnCombatStateChanged);
        EventBus.Subscribe<RunEndedEvent>(OnRunEnded);
    }

    private void LoadConfigs()
    {
        ConfigManager.PreloadAll();
    }

    /// <summary>
    /// 切换游戏状态
    /// </summary>
    public void SetState(GameState newState)
    {
        if (CurrentState == newState) return;

        var previousState = CurrentState;
        CurrentState = newState;

        HandleStateChange(previousState, newState);
        OnStateChanged?.Invoke(newState);
    }

    private void HandleStateChange(GameState from, GameState to)
    {
        switch (to)
        {
            case GameState.MainMenu:
                Time.timeScale = 1f;
                UI.CloseAll();
                UI.OpenScreen<MainMenuScreen>();
                break;

            case GameState.HeroSelect:
                UI.CloseAll();
                UI.OpenScreen<HeroSelectScreen>();
                break;

            case GameState.Playing:
                Time.timeScale = 1f;
                UI.CloseAll();
                UI.OpenScreen<GameHUD>();
                break;

            case GameState.Paused:
                Time.timeScale = 0f;
                UI.OpenScreen<PausePopup>();
                break;

            case GameState.Victory:
            case GameState.Defeat:
                Time.timeScale = 0f;
                UI.OpenScreen<ResultScreen>(new ResultScreenData
                {
                    IsVictory = to == GameState.Victory
                });
                break;
        }
    }

    // ========== 游戏流程 ==========

    /// <summary>
    /// 开始新游戏
    /// </summary>
    public void StartNewGame(int heroId)
    {
        RoguelikeManager.Instance.StartNewRun(heroId);
        SetState(GameState.Playing);
    }

    /// <summary>
    /// 暂停游戏
    /// </summary>
    public void PauseGame()
    {
        if (CurrentState == GameState.Playing)
            SetState(GameState.Paused);
    }

    /// <summary>
    /// 继续游戏
    /// </summary>
    public void ResumeGame()
    {
        if (CurrentState == GameState.Paused)
            SetState(GameState.Playing);
    }

    /// <summary>
    /// 返回主菜单
    /// </summary>
    public void ReturnToMainMenu()
    {
        RoguelikeManager.Instance.EndRun(false);
        SetState(GameState.MainMenu);
    }

    // ========== 事件处理 ==========

    private void OnCombatStateChanged(CombatStateChangedEvent e)
    {
        if (e.State == CombatState.Victory)
        {
            // 检查是否通关
            if (RoguelikeManager.Instance.CurrentRun.CurrentFloor >= 10)
            {
                SetState(GameState.Victory);
            }
        }
        else if (e.State == CombatState.Defeat)
        {
            SetState(GameState.Defeat);
        }
    }

    private void OnRunEnded(RunEndedEvent e)
    {
        // 保存统计数据
        Save.Save();
    }

    private void OnDestroy()
    {
        EventBus.Unsubscribe<CombatStateChangedEvent>(OnCombatStateChanged);
        EventBus.Unsubscribe<RunEndedEvent>(OnRunEnded);
    }
}

public struct CombatStateChangedEvent { public CombatState State; }
public struct RunEndedEvent { public bool Victory; public int Floor; }
```

### 3.1 三消与战斗的连接

```csharp
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 三消-战斗连接器
/// </summary>
public class MatchCombatBridge : MonoBehaviour
{
    [SerializeField] private BoardController boardController;
    [SerializeField] private CombatManager combatManager;

    private void Start()
    {
        // 订阅棋盘事件
        boardController.OnMatchFound += OnMatchFound;
        boardController.OnTurnComplete += OnTurnComplete;
        boardController.OnComboChanged += OnComboChanged;
    }

    private void OnMatchFound(List<MatchGroup> matches, int score)
    {
        foreach (var match in matches)
        {
            // 将棋子类型转换为元素类型
            var element = (ElementType)match.GemType;

            // 计算伤害
            int matchCount = match.Count;

            // 通知战斗系统
            combatManager.ProcessMatchDamage(
                (GemType)element,
                matchCount,
                boardController.ComboCount
            );

            // 增加英雄技能能量
            var hero = RoguelikeManager.Instance.CurrentRun?.SelectedHero;
            if (hero != null)
            {
                hero.AddSkillEnergy(element, matchCount);
            }
        }
    }

    private void OnTurnComplete()
    {
        // 触发敌人回合
        combatManager.OnPlayerTurnComplete();
    }

    private void OnComboChanged(int combo)
    {
        // 更新UI显示
        EventBus.Publish(new ComboChangedEvent { Combo = combo });
    }

    private void OnDestroy()
    {
        if (boardController != null)
        {
            boardController.OnMatchFound -= OnMatchFound;
            boardController.OnTurnComplete -= OnTurnComplete;
            boardController.OnComboChanged -= OnComboChanged;
        }
    }
}
```

### 3.2 UI与游戏状态的同步

```csharp
using UnityEngine;
using UnityEngine.UI;

/// <summary>
/// 游戏HUD - 整合所有游戏内UI
/// </summary>
public class GameHUD : UIScreen
{
    [Header("Top Bar")]
    [SerializeField] private Text floorText;
    [SerializeField] private Text goldText;
    [SerializeField] private Text energyText;

    [Header("Hero Status")]
    [SerializeField] private Image heroPortrait;
    [SerializeField] private Slider hpSlider;
    [SerializeField] private Text hpText;
    [SerializeField] private Slider energySlider;
    [SerializeField] private Image skillIcon;
    [SerializeField] private Button skillButton;

    [Header("Enemy Status")]
    [SerializeField] private Transform enemyContainer;
    [SerializeField] private GameObject enemyStatusPrefab;

    [Header("Combat Info")]
    [SerializeField] private Text comboText;
    [SerializeField] private Text scoreText;
    [SerializeField] private GameObject comboPanel;

    [Header("Buttons")]
    [SerializeField] private Button pauseButton;
    [SerializeField] private Button inventoryButton;

    private RunData runData;

    protected override void OnOpen(object param)
    {
        runData = RoguelikeManager.Instance.CurrentRun;

        // 绑定按钮
        pauseButton.onClick.AddListener(OnPauseClicked);
        inventoryButton.onClick.AddListener(OnInventoryClicked);
        skillButton.onClick.AddListener(OnSkillClicked);

        // 订阅事件
        EventBus.Subscribe<FloorChangedEvent>(OnFloorChanged);
        EventBus.Subscribe<GoldChangedEvent>(OnGoldChanged);
        EventBus.Subscribe<HeroHPChangedEvent>(OnHeroHPChanged);
        EventBus.Subscribe<HeroEnergyChangedEvent>(OnHeroEnergyChanged);
        EventBus.Subscribe<ComboChangedEvent>(OnComboChanged);
        EventBus.Subscribe<ScoreChangedEvent>(OnScoreChanged);
        EventBus.Subscribe<CombatStartedEvent>(OnCombatStarted);
        EventBus.Subscribe<EnemyDiedEvent>(OnEnemyDied);

        // 初始化显示
        InitializeDisplay();
    }

    protected override void OnClose()
    {
        pauseButton.onClick.RemoveListener(OnPauseClicked);
        inventoryButton.onClick.RemoveListener(OnInventoryClicked);
        skillButton.onClick.RemoveListener(OnSkillClicked);

        EventBus.Unsubscribe<FloorChangedEvent>(OnFloorChanged);
        EventBus.Unsubscribe<GoldChangedEvent>(OnGoldChanged);
        EventBus.Unsubscribe<HeroHPChangedEvent>(OnHeroHPChanged);
        EventBus.Unsubscribe<HeroEnergyChangedEvent>(OnHeroEnergyChanged);
        EventBus.Unsubscribe<ComboChangedEvent>(OnComboChanged);
        EventBus.Unsubscribe<ScoreChangedEvent>(OnScoreChanged);
        EventBus.Unsubscribe<CombatStartedEvent>(OnCombatStarted);
        EventBus.Unsubscribe<EnemyDiedEvent>(OnEnemyDied);
    }

    private void InitializeDisplay()
    {
        if (runData == null) return;

        // 楼层
        floorText.text = $"Floor {runData.CurrentFloor}";

        // 金币
        goldText.text = runData.Gold.ToString();

        // 英雄信息
        var hero = runData.SelectedHero;
        if (hero != null)
        {
            UpdateHeroHP(hero.CurrentHP, hero.MaxHP);
            UpdateHeroEnergy(hero.SkillEnergy, hero.MaxSkillEnergy);
        }
    }

    private void UpdateHeroHP(int current, int max)
    {
        hpSlider.value = (float)current / max;
        hpText.text = $"{current}/{max}";
    }

    private void UpdateHeroEnergy(int current, int max)
    {
        energySlider.value = (float)current / max;
        skillButton.interactable = current >= max;
    }

    // ========== 事件处理 ==========

    private void OnFloorChanged(FloorChangedEvent e)
    {
        floorText.text = $"Floor {e.Floor}";
    }

    private void OnGoldChanged(GoldChangedEvent e)
    {
        goldText.text = e.Amount.ToString();
    }

    private void OnHeroHPChanged(HeroHPChangedEvent e)
    {
        UpdateHeroHP(e.Current, e.Max);
    }

    private void OnHeroEnergyChanged(HeroEnergyChangedEvent e)
    {
        UpdateHeroEnergy(e.Current, e.Max);
    }

    private void OnComboChanged(ComboChangedEvent e)
    {
        if (e.Combo > 1)
        {
            comboPanel.SetActive(true);
            comboText.text = $"{e.Combo}x Combo!";
        }
        else
        {
            comboPanel.SetActive(false);
        }
    }

    private void OnScoreChanged(ScoreChangedEvent e)
    {
        scoreText.text = e.Score.ToString("N0");
    }

    private void OnCombatStarted(CombatStartedEvent e)
    {
        // 创建敌人状态UI
        foreach (Transform child in enemyContainer)
            Destroy(child.gameObject);

        foreach (var enemy in e.Enemies)
        {
            var statusGO = Instantiate(enemyStatusPrefab, enemyContainer);
            var statusUI = statusGO.GetComponent<EnemyStatusUI>();
            statusUI.Initialize(enemy);
        }
    }

    private void OnEnemyDied(EnemyDiedEvent e)
    {
        // 移除敌人状态UI
    }

    // ========== 按钮处理 ==========

    private void OnPauseClicked()
    {
        GameManager.Instance.PauseGame();
    }

    private void OnInventoryClicked()
    {
        UIManager.Instance.OpenScreen<InventoryPopup>();
    }

    private void OnSkillClicked()
    {
        var hero = runData?.SelectedHero;
        if (hero != null && hero.SkillEnergy >= hero.MaxSkillEnergy)
        {
            CombatManager.Instance.UseHeroSkill(hero.Id);
        }
    }
}
```

### 4.1 对象池管理器

```csharp
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 通用对象池管理器
/// </summary>
public class PoolManager : MonoSingleton<PoolManager>
{
    private Dictionary<string, ObjectPool> pools = new Dictionary<string, ObjectPool>();

    /// <summary>
    /// 初始化对象池
    /// </summary>
    public void InitializePool(string key, GameObject prefab, int initialSize = 10, int maxSize = 50)
    {
        if (pools.ContainsKey(key)) return;

        var pool = new ObjectPool(prefab, initialSize, maxSize, transform);
        pools[key] = pool;
    }

    /// <summary>
    /// 获取对象
    /// </summary>
    public GameObject Get(string key, Vector3 position, Quaternion rotation)
    {
        if (pools.TryGetValue(key, out var pool))
        {
            var obj = pool.Get();
            obj.transform.position = position;
            obj.transform.rotation = rotation;
            return obj;
        }
        return null;
    }

    /// <summary>
    /// 归还对象
    /// </summary>
    public void Return(string key, GameObject obj)
    {
        if (pools.TryGetValue(key, out var pool))
        {
            pool.Return(obj);
        }
        else
        {
            Destroy(obj);
        }
    }

    /// <summary>
    /// 预热所有池
    /// </summary>
    public void Warmup()
    {
        foreach (var pool in pools.Values)
        {
            pool.Warmup();
        }
    }

    /// <summary>
    /// 清空所有池
    /// </summary>
    public void ClearAll()
    {
        foreach (var pool in pools.Values)
        {
            pool.Clear();
        }
        pools.Clear();
    }
}

/// <summary>
/// 对象池
/// </summary>
public class ObjectPool
{
    private GameObject prefab;
    private Transform parent;
    private Queue<GameObject> available = new Queue<GameObject>();
    private HashSet<GameObject> inUse = new HashSet<GameObject>();
    private int maxSize;

    public ObjectPool(GameObject prefab, int initialSize, int maxSize, Transform parent)
    {
        this.prefab = prefab;
        this.maxSize = maxSize;
        this.parent = parent;
    }

    public void Warmup()
    {
        for (int i = 0; i < 5; i++)
        {
            var obj = CreateNew();
            obj.SetActive(false);
            available.Enqueue(obj);
        }
    }

    public GameObject Get()
    {
        GameObject obj;

        if (available.Count > 0)
        {
            obj = available.Dequeue();
        }
        else if (inUse.Count < maxSize)
        {
            obj = CreateNew();
        }
        else
        {
            obj = null;
        }

        if (obj != null)
        {
            obj.SetActive(true);
            inUse.Add(obj);
        }

        return obj;
    }

    public void Return(GameObject obj)
    {
        if (inUse.Remove(obj))
        {
            obj.SetActive(false);
            obj.transform.SetParent(parent);
            available.Enqueue(obj);
        }
    }

    private GameObject CreateNew()
    {
        var obj = Instantiate(prefab, parent);
        obj.SetActive(false);
        return obj;
    }

    public void Clear()
    {
        foreach (var obj in available)
            Object.Destroy(obj);
        foreach (var obj in inUse)
            Object.Destroy(obj);

        available.Clear();
        inUse.Clear();
    }
}
```

### 4.2 游戏优化配置

```csharp
using UnityEngine;

/// <summary>
/// 游戏优化设置
/// </summary>
public class GameOptimizer : MonoBehaviour
{
    [Header("Performance Settings")]
    [SerializeField] private int targetFrameRate = 60;
    [SerializeField] private bool useVSync = false;

    [Header("Quality Settings")]
    [SerializeField] private bool reduceParticlesOnLowFPS = true;
    [SerializeField] private bool reduceShadowsOnLowFPS = true;
    [SerializeField] private int lowFPSThreshold = 30;

    [Header("Memory Settings")]
    [SerializeField] private float gcInterval = 60f;

    private float lastGCTime;
    private float fps;
    private int frameCount;
    private float fpsUpdateTime;

    private void Start()
    {
        ApplySettings();
    }

    private void ApplySettings()
    {
        Application.targetFrameRate = targetFrameRate;
        QualitySettings.vSyncCount = useVSync ? 1 : 0;
    }

    private void Update()
    {
        UpdateFPS();
        CheckPerformance();
        CheckGC();
    }

    private void UpdateFPS()
    {
        frameCount++;
        fpsUpdateTime += Time.deltaTime;

        if (fpsUpdateTime >= 1f)
        {
            fps = frameCount / fpsUpdateTime;
            frameCount = 0;
            fpsUpdateTime = 0f;
        }
    }

    private void CheckPerformance()
    {
        if (fps < lowFPSThreshold)
        {
            if (reduceParticlesOnLowFPS)
            {
                ReduceParticleQuality();
            }

            if (reduceShadowsOnLowFPS)
            {
                QualitySettings.shadows = ShadowQuality.Disable;
            }
        }
    }

    private void ReduceParticleQuality()
    {
        var particles = FindObjectsOfType<ParticleSystem>();
        foreach (var ps in particles)
        {
            var main = ps.main;
            if (main.maxParticles > 50)
            {
                main.maxParticles = 50;
            }
        }
    }

    private void CheckGC()
    {
        if (Time.time - lastGCTime > gcInterval)
        {
            System.GC.Collect();
            lastGCTime = Time.time;
        }
    }

    /// <summary>
    /// 卸载未使用的资源
    /// </summary>
    public void UnloadUnusedAssets()
    {
        Resources.UnloadUnusedAssets();
        System.GC.Collect();
    }
}
```

```csharp
/// <summary>
/// 游戏启动器
/// </summary>
public class GameBootstrap : MonoBehaviour
{
    [Header("Bootstrap Settings")]
    [SerializeField] private bool showSplashScreen = true;
    [SerializeField] private float minLoadingTime = 2f;

    private IEnumerator Start()
    {
        // 1. 显示加载画面
        UIManager.Instance.OpenScreen<LoadingScreen>();

        // 2. 初始化系统
        yield return InitializeSystems();

        // 3. 加载配置
        yield return LoadConfigs();

        // 4. 预热对象池
        yield return WarmupPools();

        // 5. 加载存档
        yield return LoadSaveData();

        // 6. 最小加载时间
        yield return new WaitForSecondsRealtime(minLoadingTime);

        // 7. 进入主菜单
        UIManager.Instance.CloseScreen<LoadingScreen>();
        GameManager.Instance.SetState(GameState.MainMenu);
    }

    private IEnumerator InitializeSystems()
    {
        // 各系统已通过MonoSingleton自动初始化
        yield return null;
    }

    private IEnumerator LoadConfigs()
    {
        ConfigManager.PreloadAll();
        yield return null;
    }

    private IEnumerator WarmupPools()
    {
        // 预热棋子对象池
        PoolManager.Instance.InitializePool("Gem", Resources.Load<GameObject>("Prefabs/Gem"), 64, 100);

        // 预热特效对象池
        PoolManager.Instance.InitializePool("MatchEffect", Resources.Load<GameObject>("Prefabs/MatchEffect"), 20, 50);

        PoolManager.Instance.Warmup();
        yield return null;
    }

    private IEnumerator LoadSaveData()
    {
        SaveManager.Instance.Initialize();
        yield return null;
    }
}
```
