## 2. 核心数据结构

### 2.1 战斗单位基类

```csharp
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 战斗单位类型
/// </summary>
public enum UnitType
{
    Hero,
    Enemy
}

/// <summary>
/// 元素类型（对应棋子颜色）
/// </summary>
public enum ElementType
{
    Fire = 1,    // 红色
    Water = 2,   // 蓝色
    Nature = 3,  // 绿色
    Light = 4,   // 黄色
    Dark = 5     // 紫色
}

/// <summary>
/// 战斗单位基类
/// </summary>
public abstract class CombatUnit
{
    public int Id { get; protected set; }
    public string Name { get; protected set; }
    public UnitType UnitType { get; protected set; }
    public ElementType Element { get; protected set; }

    // 基础属性
    public int MaxHP { get; protected set; }
    public int CurrentHP { get; protected set; }
    public int BaseAttack { get; protected set; }
    public int BaseDefense { get; protected set; }

    // Buff影响的属性
    public virtual int Attack => BaseAttack + GetBuffValue(BuffStatType.Attack);
    public virtual int Defense => BaseDefense + GetBuffValue(BuffStatType.Defense);

    // Buff列表
    protected List<BuffInstance> buffs = new List<BuffInstance>();

    public bool IsDead => CurrentHP <= 0;

    public event System.Action<int, int> OnHPChanged; // current, max
    public event System.Action OnDeath;

    public virtual void Initialize(CombatUnitData data)
    {
        Id = data.Id;
        Name = data.Name;
        MaxHP = data.MaxHP;
        CurrentHP = MaxHP;
        BaseAttack = data.Attack;
        BaseDefense = data.Defense;
        Element = data.Element;
    }

    public virtual void TakeDamage(int damage)
    {
        int actualDamage = Mathf.Max(1, damage - Defense);
        CurrentHP = Mathf.Max(0, CurrentHP - actualDamage);

        OnHPChanged?.Invoke(CurrentHP, MaxHP);

        if (IsDead)
            OnDeath?.Invoke();
    }

    public virtual void Heal(int amount)
    {
        CurrentHP = Mathf.Min(MaxHP, CurrentHP + amount);
        OnHPChanged?.Invoke(CurrentHP, MaxHP);
    }

    // ========== Buff系统 ==========

    public void AddBuff(BuffInstance buff)
    {
        buffs.Add(buff);
        buff.OnApply(this);
    }

    public void RemoveBuff(BuffInstance buff)
    {
        buffs.Remove(buff);
        buff.OnRemove(this);
    }

    public void TickBuffs()
    {
        for (int i = buffs.Count - 1; i >= 0; i--)
        {
            buffs[i].Tick(this);
            if (buffs[i].IsExpired)
            {
                buffs.RemoveAt(i);
            }
        }
    }

    protected int GetBuffValue(BuffStatType statType)
    {
        int total = 0;
        foreach (var buff in buffs)
        {
            if (buff.StatType == statType)
                total += buff.Value;
        }
        return total;
    }
}

/// <summary>
/// 战斗单位数据（ScriptableObject配置）
/// </summary>
[System.Serializable]
public class CombatUnitData
{
    public int Id;
    public string Name;
    public int MaxHP;
    public int Attack;
    public int Defense;
    public ElementType Element;
}
```

### 2.2 英雄类

```csharp
using System.Collections.Generic;

/// <summary>
/// 英雄单位
/// </summary>
public class Hero : CombatUnit
{
    public int Level { get; private set; }
    public int Experience { get; private set; }
    public List<int> Skills { get; private set; }
    public int SelectedSkillIndex { get; private set; }

    // 技能冷却
    private Dictionary<int, int> skillCooldowns = new Dictionary<int, int>();

    // 技能充能（通过消除对应元素积累）
    public int SkillEnergy { get; private set; }
    public int MaxSkillEnergy { get; private set; }

    // 克制元素
    public ElementType StrongAgainst { get; private set; }
    public float BonusDamageMultiplier { get; private set; }

    public event System.Action<int> OnSkillEnergyChanged;
    public event System.Action<int, int> OnSkillCooldownChanged; // skillId, remainingTurns

    public Hero()
    {
        UnitType = UnitType.Hero;
    }

    public void Initialize(HeroConfig.HeroData data, int level = 1)
    {
        base.Initialize(new CombatUnitData
        {
            Id = data.Id,
            Name = data.Name,
            MaxHP = data.BaseHP,
            Attack = data.BaseAttack,
            Defense = data.BaseDefense,
            Element = (ElementType)data.Id // 简化：用ID对应元素
        });

        Level = level;
        Skills = new List<int> { data.SkillId };
        MaxSkillEnergy = data.SkillCooldown;
        StrongAgainst = (ElementType)data.StrongAgainstGemId;
        BonusDamageMultiplier = data.BonusDamage;
    }

    /// <summary>
    /// 增加技能能量（通过消除对应元素）
    /// </summary>
    public void AddSkillEnergy(ElementType element, int amount)
    {
        if (element == Element)
        {
            SkillEnergy = Mathf.Min(MaxSkillEnergy, SkillEnergy + amount);
            OnSkillEnergyChanged?.Invoke(SkillEnergy);

            if (SkillEnergy >= MaxSkillEnergy)
            {
                // 技能就绪
                EventBus.Publish(new SkillReadyEvent { HeroId = Id });
            }
        }
    }

    /// <summary>
    /// 使用技能
    /// </summary>
    public bool UseSkill(int skillId)
    {
        if (SkillEnergy < MaxSkillEnergy)
            return false;

        SkillEnergy = 0;
        OnSkillEnergyChanged?.Invoke(SkillEnergy);
        return true;
    }

    /// <summary>
    /// 计算对目标的伤害
    /// </summary>
    public int CalculateDamage(CombatUnit target, int matchCount)
    {
        float baseDamage = Attack * GetMatchMultiplier(matchCount);

        // 元素克制
        if (target.Element == StrongAgainst)
            baseDamage *= BonusDamageMultiplier;

        return Mathf.RoundToInt(baseDamage);
    }

    private float GetMatchMultiplier(int matchCount)
    {
        return matchCount switch
        {
            3 => 1.0f,
            4 => 1.5f,
            5 => 2.0f,
            _ => 1.0f + (matchCount - 3) * 0.25f
        };
    }

    public void TickCooldowns()
    {
        var keys = new List<int>(skillCooldowns.Keys);
        foreach (var skillId in keys)
        {
            if (skillCooldowns[skillId] > 0)
            {
                skillCooldowns[skillId]--;
                OnSkillCooldownChanged?.Invoke(skillId, skillCooldowns[skillId]);
            }
        }
    }
}

public struct SkillReadyEvent
{
    public int HeroId;
}
```

### 2.3 敌人类

```csharp
using System.Collections.Generic;

/// <summary>
/// 敌人行为模式
/// </summary>
public enum EnemyBehavior
{
    Aggressive,   // 优先攻击
    Defensive,    // 优先防御
    Support,      // 优先治疗/增益
    Random        // 随机
}

/// <summary>
/// 敌人单位
/// </summary>
public class Enemy : CombatUnit
{
    public int Floor { get; private set; }  // 出现层数
    public EnemyBehavior Behavior { get; private set; }
    public List<EnemySkill> Skills { get; private set; }

    // 行动计数器
    public int ActionInterval { get; private set; }
    public int CurrentActionCount { get; private set; }

    public event System.Action<Enemy, EnemyAction> OnActionDecided;

    public Enemy()
    {
        UnitType = UnitType.Enemy;
    }

    public void Initialize(EnemyData data)
    {
        base.Initialize(data);
        Floor = data.Floor;
        Behavior = data.Behavior;
        ActionInterval = data.ActionInterval;
        CurrentActionCount = ActionInterval;
        Skills = data.Skills;
    }

    /// <summary>
    /// 敌人回合开始
    /// </summary>
    public EnemyAction DecideAction(List<CombatUnit> targets)
    {
        CurrentActionCount--;

        if (CurrentActionCount > 0)
        {
            return new EnemyAction { Type = EnemyActionType.Wait };
        }

        CurrentActionCount = ActionInterval;

        // 根据行为模式决定行动
        EnemyAction action = Behavior switch
        {
            EnemyBehavior.Aggressive => DecideAggressiveAction(targets),
            EnemyBehavior.Defensive => DecideDefensiveAction(),
            EnemyBehavior.Support => DecideSupportAction(),
            _ => DecideRandomAction(targets)
        };

        OnActionDecided?.Invoke(this, action);
        return action;
    }

    private EnemyAction DecideAggressiveAction(List<CombatUnit> targets)
    {
        // 选择血量最低的目标
        CombatUnit target = null;
        int minHP = int.MaxValue;

        foreach (var t in targets)
        {
            if (t.CurrentHP < minHP && !t.IsDead)
            {
                minHP = t.CurrentHP;
                target = t;
            }
        }

        return new EnemyAction
        {
            Type = EnemyActionType.Attack,
            Target = target,
            Value = Attack
        };
    }

    private EnemyAction DecideDefensiveAction()
    {
        // 如果有防御技能且血量低于50%
        if (CurrentHP < MaxHP * 0.5f && HasSkill(EnemySkillType.Heal))
        {
            return new EnemyAction
            {
                Type = EnemyActionType.Heal,
                Target = this,
                Value = Mathf.RoundToInt(MaxHP * 0.2f)
            };
        }

        return new EnemyAction
        {
            Type = EnemyActionType.Defend,
            Value = Mathf.RoundToInt(Defense * 0.5f)
        };
    }

    private EnemyAction DecideSupportAction()
    {
        // TODO: 实现支援行为
        return new EnemyAction { Type = EnemyActionType.Wait };
    }

    private EnemyAction DecideRandomAction(List<CombatUnit> targets)
    {
        float roll = Random.value;

        if (roll < 0.7f && targets.Count > 0)
        {
            var target = targets[Random.Range(0, targets.Count)];
            return new EnemyAction
            {
                Type = EnemyActionType.Attack,
                Target = target,
                Value = Attack
            };
        }
        else
        {
            return new EnemyAction
            {
                Type = EnemyActionType.Defend,
                Value = Mathf.RoundToInt(Defense * 0.5f)
            };
        }
    }

    private bool HasSkill(EnemySkillType type)
    {
        return Skills.Exists(s => s.Type == type);
    }
}

/// <summary>
/// 敌人数据
/// </summary>

    public List<EnemySkill> Skills;
}

/// <summary>
/// 敌人技能

{
```

## 3. 伤害计算系统

```csharp
using UnityEngine;

/// <summary>
/// 伤害计算器
/// </summary>
public static class DamageCalculator
{
    /// <summary>
    /// 计算最终伤害
    /// </summary>
    public static DamageResult Calculate(CombatUnit attacker, CombatUnit defender, int matchCount, ElementType gemElement)
    {
        var result = new DamageResult();

        // 基础伤害
        float baseDamage = attacker.Attack * GetMatchMultiplier(matchCount);
        result.BaseDamage = Mathf.RoundToInt(baseDamage);

        // 元素加成
        float elementMultiplier = GetElementMultiplier(attacker.Element, defender.Element);
        result.ElementBonus = Mathf.RoundToInt(baseDamage * (elementMultiplier - 1));

        // 连击加成
        float comboMultiplier = 1f;
        result.ComboBonus = Mathf.RoundToInt(baseDamage * (comboMultiplier - 1));

        // 防御减免
        float finalDamage = baseDamage * elementMultiplier * comboMultiplier;
        int defense = defender.Defense;
        result.DefenseReduction = Mathf.Min(defense, Mathf.RoundToInt(finalDamage * 0.5f));

        // 最终伤害
        result.FinalDamage = Mathf.Max(1, Mathf.RoundToInt(finalDamage) - result.DefenseReduction);

        // 暴击（可选）
        if (Random.value < 0.1f) // 10%暴击率
        {
            result.IsCritical = true;
            result.FinalDamage = Mathf.RoundToInt(result.FinalDamage * 1.5f);
        }

        return result;
    }

    /// <summary>
    /// 匹配数量伤害倍率
    /// </summary>
    private static float GetMatchMultiplier(int matchCount)
    {
        return matchCount switch
        {
            3 => 1.0f,
            4 => 1.5f,
            5 => 2.0f,
            6 => 2.5f,
            _ => 1.0f + (matchCount - 3) * 0.5f
        };
    }

    /// <summary>
    /// 元素克制倍率
    /// </summary>
    private static float GetElementMultiplier(ElementType attacker, ElementType defender)
    {
        // 克制关系：火 > 自然 > 水 > 火, 光 <-> 暗
        var advantages = new Dictionary<ElementType, ElementType>
        {
            { ElementType.Fire, ElementType.Nature },
            { ElementType.Nature, ElementType.Water },
            { ElementType.Water, ElementType.Fire },
        };

        if (advantages.TryGetValue(attacker, out var strongAgainst) && defender == strongAgainst)
            return 1.5f;

        if (advantages.TryGetValue(defender, out var weakAgainst) && attacker == weakAgainst)
            return 0.75f;

        // 光暗互克
        if (attacker == ElementType.Light && defender == ElementType.Dark)
            return 1.5f;
        if (attacker == ElementType.Dark && defender == ElementType.Light)
            return 1.5f;

        return 1.0f;
    }

    /// <summary>
    /// 计算治疗量
    /// </summary>
    public static int CalculateHeal(CombatUnit healer, int baseAmount)
    {
        return Mathf.RoundToInt(baseAmount * (1 + healer.Attack * 0.01f));
    }
}

/// <summary>
/// 伤害结果
/// </summary>
public struct DamageResult
{
    public int BaseDamage;
    public int ElementBonus;
    public int ComboBonus;
    public int DefenseReduction;
    public int FinalDamage;
    public bool IsCritical;

    public override string ToString()
    {
        return $"Base:{BaseDamage} Elem:+{ElementBonus} Combo:+{ComboBonus} Def:-{DefenseReduction} = {FinalDamage}{(IsCritical ? " CRIT!" : "")}";
    }
}
```

## 4. Buff/Debuff系统

```csharp
using System.Collections.Generic;

/// <summary>
/// Buff类型
/// </summary>
public enum BuffType
{
    // 属性增益
    AttackUp,
    DefenseUp,
    SpeedUp,

    // 属性减益
    AttackDown,
    DefenseDown,
    SpeedDown,

    // 特殊效果
    Poison,      // 毒（每回合扣血）
    Burn,        // 灼烧（每回合扣血+降低攻击）
    Freeze,      // 冻结（跳过回合）
    Stun,        // 眩晕（跳过回合）
    Shield,      // 护盾（吸收伤害）
    Regeneration // 再生（每回合回血）
}

/// <summary>
/// Buff影响的属性类型
/// </summary>
public enum BuffStatType
{
    None,
    Attack,
    Defense,
    Speed
}

/// <summary>
/// Buff实例
/// </summary>
public class BuffInstance
{
    public BuffType Type { get; private set; }
    public BuffStatType StatType { get; private set; }
    public int Value { get; private set; }
    public int Duration { get; private set; }
    public int RemainingTurns { get; private set; }
    public bool IsExpired => RemainingTurns <= 0;

    public CombatUnit Source { get; private set; }

    public BuffInstance(BuffType type, int value, int duration, CombatUnit source = null)
    {
        Type = type;
        Value = value;
        Duration = duration;
        RemainingTurns = duration;
        Source = source;
        StatType = GetStatType(type);
    }

    public void OnApply(CombatUnit target)
    {
        // 应用时的效果
        switch (Type)
        {
            case BuffType.Shield:
                // 添加护盾值
                break;
            case BuffType.Freeze:
            case BuffType.Stun:
                // 标记跳过回合
                break;
        }
    }

    public void Tick(CombatUnit target)
    {
        RemainingTurns--;

        // 每回合效果
        switch (Type)
        {
            case BuffType.Poison:
                target.TakeDamage(Value);
                break;
            case BuffType.Burn:
                target.TakeDamage(Value / 2);
                break;
            case BuffType.Regeneration:
                target.Heal(Value);
                break;
        }
    }

    public void OnRemove(CombatUnit target)
    {
        // 移除时的效果
    }

    private BuffStatType GetStatType(BuffType type)
    {
        return type switch
        {
            BuffType.AttackUp or BuffType.AttackDown => BuffStatType.Attack,
            BuffType.DefenseUp or BuffType.DefenseDown => BuffStatType.Defense,
            BuffType.SpeedUp or BuffType.SpeedDown => BuffStatType.Speed,
            _ => BuffStatType.None
        };
    }
}

    }

    /// <summary>
    /// 创建防御增益

    public static BuffInstance CreateDefenseUp(int value, int duration)
    {
        return new BuffInstance(BuffType.DefenseUp, value, duration);
    }

    /// <summary>
    /// 创建中毒
```

## 5. 战斗管理器

```csharp
using System.Collections;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 战斗状态
/// </summary>
public enum CombatState
{
    Idle,
    PlayerTurn,
    EnemyTurn,
    Victory,
    Defeat
}

/// <summary>
/// 战斗管理器
/// </summary>
public class CombatManager : MonoSingleton<CombatManager>
{
    [Header("References")]
    [SerializeField] private BoardController boardController;

    [Header("Combat Settings")]
    [SerializeField] private int maxTurns = 20;

    // 战斗单位
    private List<Hero> heroes = new List<Hero>();
    private List<Enemy> enemies = new List<Enemy>();

    // 状态
    public CombatState State { get; private set; }
    public int CurrentTurn { get; private set; }
    public int ComboCount { get; private set; }

    // 事件
    public event System.Action<CombatState> OnStateChanged;
    public event System.Action<int> OnTurnChanged;
    public event System.Action<Hero> OnHeroDied;
    public event System.Action<Enemy> OnEnemyDied;
    public event System.Action<CombatUnit, DamageResult> OnDamageDealt;

    private void Start()
    {
        // 订阅棋盘事件
        if (boardController != null)
        {
            boardController.OnScoreChanged += OnMatchScored;
            boardController.OnComboChanged += OnComboChanged;
            boardController.OnTurnComplete += OnPlayerTurnComplete;
        }
    }

    /// <summary>
    /// 开始战斗
    /// </summary>
    public void StartBattle(List<Hero> heroList, List<Enemy> enemyList)
    {
        heroes = new List<Hero>(heroList);
        enemies = new List<Enemy>(enemyList);

        CurrentTurn = 1;
        ComboCount = 0;

        // 订阅死亡事件
        foreach (var hero in heroes)
            hero.OnDeath += () => OnUnitDeath(hero);
        foreach (var enemy in enemies)
            enemy.OnDeath += () => OnUnitDeath(enemy);

        SetState(CombatState.PlayerTurn);
        OnTurnChanged?.Invoke(CurrentTurn);
    }

    /// <summary>
    /// 匹配得分时触发
    /// </summary>
    private void OnMatchScored(int score)
    {
        // 伤害计算在 OnComboChanged 中处理
    }

    /// <summary>
    /// 连击变化
    /// </summary>
    private void OnComboChanged(int combo)
    {
        ComboCount = combo;
    }

    /// <summary>
    /// 玩家回合结束
    /// </summary>
    private void OnPlayerTurnComplete()
    {
        // 检查战斗状态
        if (CheckBattleEnd()) return;

        // 开始敌人回合
        StartCoroutine(EnemyTurn());
    }

    /// <summary>
    /// 敌人回合
    /// </summary>
    private IEnumerator EnemyTurn()
    {
        SetState(CombatState.EnemyTurn);

        foreach (var enemy in enemies)
        {
            if (enemy.IsDead) continue;

            yield return new WaitForSeconds(0.5f);

            var action = enemy.DecideAction(new List<CombatUnit>(heroes));
            yield return ExecuteEnemyAction(enemy, action);

            if (CheckBattleEnd()) yield break;
        }

        // 回合结束处理
        EndTurn();
    }

    /// <summary>
    /// 执行敌人行动
    /// </summary>
    private IEnumerator ExecuteEnemyAction(Enemy enemy, EnemyAction action)
    {
        switch (action.Type)
        {
            case EnemyActionType.Attack:
                if (action.Target != null && !action.Target.IsDead)
                {
                    var result = DamageCalculator.Calculate(enemy, action.Target, 3, enemy.Element);
                    action.Target.TakeDamage(result.FinalDamage);
                    OnDamageDealt?.Invoke(enemy, result);
                }
                break;

            case EnemyActionType.Heal:
                enemy.Heal(action.Value);
                break;

            case EnemyActionType.Defend:
                enemy.AddBuff(BuffFactory.CreateDefenseUp(action.Value, 1));
                break;
        }

        yield return new WaitForSeconds(0.3f);
    }

    /// <summary>
    /// 回合结束
    /// </summary>
    private void EndTurn()
    {
        CurrentTurn++;
        OnTurnChanged?.Invoke(CurrentTurn);

        // Tick所有单位的Buff
        foreach (var hero in heroes)
            if (!hero.IsDead) hero.TickBuffs();
        foreach (var enemy in enemies)
            if (!enemy.IsDead) enemy.TickBuffs();

        // 检查回合限制
        if (CurrentTurn > maxTurns)
        {
            SetState(CombatState.Defeat);
            return;
        }

        SetState(CombatState.PlayerTurn);
    }

    /// <summary>
    /// 处理消除伤害
    /// </summary>
    public void ProcessMatchDamage(GemType gemType, int matchCount, int combo)
    {
        if (State != CombatState.PlayerTurn) return;

        // 找到对应元素的英雄
        var element = (ElementType)gemType;
        var hero = heroes.Find(h => h.Element == element && !h.IsDead);

        if (hero == null) return;

        // 选择目标（优先攻击血量最低的敌人）
        Enemy target = null;
        int minHP = int.MaxValue;
        foreach (var enemy in enemies)
        {
            if (!enemy.IsDead && enemy.CurrentHP < minHP)
            {
                minHP = enemy.CurrentHP;
                target = enemy;
            }
        }

        if (target == null) return;

        // 计算伤害
        var result = DamageCalculator.Calculate(hero, target, matchCount, element);

        // 连击加成
        if (combo > 1)
        {
            result.FinalDamage = Mathf.RoundToInt(result.FinalDamage * (1 + (combo - 1) * 0.25f));
        }

        // 造成伤害
        target.TakeDamage(result.FinalDamage);
        OnDamageDealt?.Invoke(hero, result);

        // 增加英雄技能能量
        hero.AddSkillEnergy(element, matchCount);
    }

    /// <summary>
    /// 使用英雄技能
    /// </summary>
    public void UseHeroSkill(int heroId)
    {
        var hero = heroes.Find(h => h.Id == heroId);
        if (hero == null || hero.IsDead) return;

        if (!hero.UseSkill(0)) return; // 假设技能ID为0

        // 执行技能效果（简化）
        foreach (var enemy in enemies)
        {
            if (!enemy.IsDead)
            {
                int damage = Mathf.RoundToInt(hero.Attack * 3f);
                enemy.TakeDamage(damage);
            }
        }
    }

    /// <summary>
    /// 单位死亡
    /// </summary>
    private void OnUnitDeath(CombatUnit unit)
    {
        if (unit is Hero hero)
        {
            OnHeroDied?.Invoke(hero);
        }
        else if (unit is Enemy enemy)
        {
            OnEnemyDied?.Invoke(enemy);
        }

        CheckBattleEnd();
    }

    /// <summary>
    /// 检查战斗是否结束
    /// </summary>
    private bool CheckBattleEnd()
    {
        bool allHeroesDead = heroes.TrueForAll(h => h.IsDead);
        bool allEnemiesDead = enemies.TrueForAll(e => e.IsDead);

        if (allHeroesDead)
        {
            SetState(CombatState.Defeat);
            return true;
        }

        if (allEnemiesDead)
        {
            SetState(CombatState.Victory);
            return true;
        }

        return false;
    }

    private void SetState(CombatState state)
    {
        State = state;
        OnStateChanged?.Invoke(state);
    }

    // ========== 辅助方法 ==========

    public List<Hero> GetAliveHeroes() => heroes.FindAll(h => !h.IsDead);
    public List<Enemy> GetAliveEnemies() => enemies.FindAll(e => !e.IsDead);
}
```

```csharp
using System.Collections.Generic;

/// <summary>
/// 技能目标类型
/// </summary>
public enum SkillTargetType
{
    SingleEnemy,
    AllEnemies,
    Self,
    AllAllies,
    RandomEnemy
}

/// <summary>
/// 技能效果类型
/// </summary>
public enum SkillEffectType
{
    Damage,
    Heal,
    Buff,
    Debuff,
    Special
}

/// <summary>
/// 技能数据
/// </summary>
[System.Serializable]
public class SkillData
{
    public int Id;
    public string Name;
    public string Description;
    public SkillTargetType TargetType;
    public SkillEffectType EffectType;
    public int BaseValue;
    public float Multiplier;
    public int Cooldown;
    public int EnergyCost;

    // Buff效果（如果有）
    public BuffType BuffType;
    public int BuffValue;
    public int BuffDuration;
}

/// <summary>
/// 技能系统
/// </summary>
public class SkillSystem
{
    private Dictionary<int, SkillData> skills;

    public SkillSystem()
    {
        skills = new Dictionary<int, SkillData>();
        LoadSkills();
    }

    private void LoadSkills()
    {
        // 从配置加载技能
        // 这里使用硬编码示例
        skills[1] = new SkillData
        {
            Id = 1,
            Name = "火焰风暴",
            Description = "对所有敌人造成150%攻击力的火焰伤害",
            TargetType = SkillTargetType.AllEnemies,
            EffectType = SkillEffectType.Damage,
            BaseValue = 0,
            Multiplier = 1.5f,
            Cooldown = 3,
            EnergyCost = 10
        };

        skills[2] = new SkillData
        {
            Id = 2,
            Name = "治愈之泉",
            Description = "恢复全体友方30%生命值",
            TargetType = SkillTargetType.AllAllies,
            EffectType = SkillEffectType.Heal,
            BaseValue = 0,
            Multiplier = 0.3f,
            Cooldown = 4,
            EnergyCost = 12
        };

        skills[3] = new SkillData
        {
            Id = 3,
            Name = "毒液喷射",
            Description = "对单体敌人造成100%伤害并施加中毒",
            TargetType = SkillTargetType.SingleEnemy,
            EffectType = SkillEffectType.Damage,
            BaseValue = 0,
            Multiplier = 1.0f,
            Cooldown = 2,
            EnergyCost = 8,
            BuffType = BuffType.Poison,
            BuffValue = 10,
            BuffDuration = 3
        };
    }

    /// <summary>
    /// 执行技能
    /// </summary>
    public void ExecuteSkill(int skillId, CombatUnit caster, List<CombatUnit> targets)
    {
        if (!skills.TryGetValue(skillId, out var skill)) return;

        foreach (var target in targets)
        {
            switch (skill.EffectType)
            {
                case SkillEffectType.Damage:
                    int damage = Mathf.RoundToInt(caster.Attack * skill.Multiplier + skill.BaseValue);
                    target.TakeDamage(damage);
                    break;

                case SkillEffectType.Heal:
                    int heal = Mathf.RoundToInt(target.MaxHP * skill.Multiplier + skill.BaseValue);
                    target.Heal(heal);
                    break;

                case SkillEffectType.Buff:
                    target.AddBuff(new BuffInstance(skill.BuffType, skill.BuffValue, skill.BuffDuration, caster));
                    break;

                case SkillEffectType.Debuff:
                    target.AddBuff(new BuffInstance(skill.BuffType, skill.BuffValue, skill.BuffDuration, caster));
                    break;
            }
        }
    }

    public SkillData GetSkill(int skillId)
    {
        return skills.TryGetValue(skillId, out var skill) ? skill : null;
    }
}
```

## 本课小结

### 核心知识点

| 知识点 | 核心要点 |
|--------|----------|
| 战斗单位 | Hero/Enemy基类设计 |
| 属性系统 | 攻击、防御、元素克制 |
| 伤害计算 | 基础伤害 × 元素加成 × 连击加成 - 防御 |
| Buff系统 | 属性修改、持续效果、回合计数 |
| 敌人AI | 行为模式、目标选择 |
| 技能系统 | 目标类型、效果类型、冷却 |

### 三消战斗流程

```
玩家消除棋子
     ↓
匹配检测 → 元素类型 → 对应英雄攻击
     ↓
伤害计算（攻击 × 匹配加成 × 元素克制 - 防御）
     ↓
敌人受伤害
     ↓
连击累加伤害
     ↓
玩家回合结束
     ↓
敌人回合（根据AI决策行动）
     ↓
Tick所有Buff
     ↓
下一回合
```

### 元素克制关系

```
火 → 自然 → 水 → 火（三角克制）
光 ↔ 暗（互相克制）
```
