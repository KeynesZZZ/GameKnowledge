```csharp
/// <summary>

/// </summary>
public static class Prefs
{

    // ========== 基础类型 ==========

    public static void SetInt(string key, int value)
        => PlayerPrefs.SetInt(KEY_PREFIX + key, value);

    public static int GetInt(string key, int defaultValue = 0)
        => PlayerPrefs.GetInt(KEY_PREFIX + key, defaultValue);

    public static void SetFloat(string key, float value)
        => PlayerPrefs.SetFloat(KEY_PREFIX + key, value);

    public static float GetFloat(string key, float defaultValue = 0f)
        => PlayerPrefs.GetFloat(KEY_PREFIX + key, defaultValue);

    public static void SetString(string key, string value)
        => PlayerPrefs.SetString(KEY_PREFIX + key, value);

    public static string GetString(string key, string defaultValue = "")
        => PlayerPrefs.GetString(KEY_PREFIX + key, defaultValue);

    public static void SetBool(string key, bool value)
        => PlayerPrefs.SetInt(KEY_PREFIX + key, value ? 1 : 0);

    public static bool GetBool(string key, bool defaultValue = false)
        => PlayerPrefs.GetInt(KEY_PREFIX + key, defaultValue ? 1 : 0) == 1;

    // ========== 扩展类型 ==========

    public static void SetVector3(string key, Vector3 value)
    {
        SetFloat($"{key}_x", value.x);
        SetFloat($"{key}_y", value.y);
        SetFloat($"{key}_z", value.z);
    }

    public static Vector3 GetVector3(string key, Vector3 defaultValue = default)
    {
        return new Vector3(
            GetFloat($"{key}_x", defaultValue.x),
            GetFloat($"{key}_y", defaultValue.y),
            GetFloat($"{key}_z", defaultValue.z)
        );
    }

    public static void SetColor(string key, Color value)
    {
        SetFloat($"{key}_r", value.r);
        SetFloat($"{key}_g", value.g);
        SetFloat($"{key}_b", value.b);
        SetFloat($"{key}_a", value.a);
    }

    public static Color GetColor(string key, Color defaultValue = default)
    {
        return new Color(
            GetFloat($"{key}_r", defaultValue.r),
            GetFloat($"{key}_g", defaultValue.g),
            GetFloat($"{key}_b", defaultValue.b),
            GetFloat($"{key}_a", defaultValue.a)
        );
    }

    // ========== 工具方法 ==========

    public static bool HasKey(string key)
        => PlayerPrefs.HasKey(KEY_PREFIX + key);

    public static void DeleteKey(string key)
        => PlayerPrefs.DeleteKey(KEY_PREFIX + key);

    public static void DeleteAll()
        => PlayerPrefs.DeleteAll();

    public static void Save()
        => PlayerPrefs.Save();
}
```

## 3. JSON序列化系统

### 3.1 基础JSON存档

```csharp
using UnityEngine;
using System.IO;
using System;

/// <summary>
/// JSON存档系统
/// </summary>
public class JsonSaveSystem
{
    private readonly string savePath;
    private readonly string fileExtension = ".json";

    public JsonSaveSystem(string subFolder = "Saves")
    {
        savePath = Path.Combine(Application.persistentDataPath, subFolder);
        if (!Directory.Exists(savePath))
            Directory.CreateDirectory(savePath);
    }

    /// <summary>
    /// 保存对象到JSON文件
    /// </summary>
    public bool Save<T>(string fileName, T data)
    {
        try
        {
            string filePath = GetFilePath(fileName);
            string json = JsonUtility.ToJson(data, true);
            File.WriteAllText(filePath, json);
            Debug.Log($"Saved to: {filePath}");
            return true;
        }
        catch (Exception e)
        {
            Debug.LogError($"Save failed: {e.Message}");
            return false;
        }
    }

    /// <summary>
    /// 从JSON文件加载对象
    /// </summary>
    public T Load<T>(string fileName) where T : new()
    {
        string filePath = GetFilePath(fileName);

        if (!File.Exists(filePath))
        {
            Debug.LogWarning($"File not found: {filePath}");
            return new T();
        }

        try
        {
            string json = File.ReadAllText(filePath);
            return JsonUtility.FromJson<T>(json);
        }
        catch (Exception e)
        {
            Debug.LogError($"Load failed: {e.Message}");
            return new T();
        }
    }

    /// <summary>
    /// 检查存档是否存在
    /// </summary>
    public bool Exists(string fileName)
        => File.Exists(GetFilePath(fileName));

    /// <summary>
    /// 删除存档
    /// </summary>
    public void Delete(string fileName)
    {
        string filePath = GetFilePath(fileName);
        if (File.Exists(filePath))
            File.Delete(filePath);
    }

    /// <summary>
    /// 获取所有存档文件
    /// </summary>
    public string[] GetAllSaveFiles()
    {
        if (!Directory.Exists(savePath))
            return Array.Empty<string>();

        var files = Directory.GetFiles(savePath, $"*{fileExtension}");
        for (int i = 0; i < files.Length; i++)
            files[i] = Path.GetFileNameWithoutExtension(files[i]);
        return files;
    }

    private string GetFilePath(string fileName)
        => Path.Combine(savePath, fileName + fileExtension);
}
```

### 3.2 带加密的JSON存档

```csharp
using System.Security.Cryptography;
using System.Text;

/// <summary>
/// 加密JSON存档系统
/// </summary>
public class EncryptedJsonSaveSystem : JsonSaveSystem
{
    private readonly byte[] key;
    private readonly byte[] iv;

    public EncryptedJsonSaveSystem(string subFolder = "Saves", string encryptionKey = "DefaultKey123456")
        : base(subFolder)
    {
        // 从密钥生成AES密钥和IV
        using (var sha256 = SHA256.Create())
        {
            byte[] hash = sha256.ComputeHash(Encoding.UTF8.GetBytes(encryptionKey));
            key = new byte[16];
            iv = new byte[16];
            Array.Copy(hash, key, 16);
            Array.Copy(hash, 16, iv, 0, 16);
        }
    }

    public new bool Save<T>(string fileName, T data)
    {
        try
        {
            string json = JsonUtility.ToJson(data, false);
            string encrypted = Encrypt(json);

            string filePath = GetFilePath(fileName);
            File.WriteAllText(filePath, encrypted);
            return true;
        }
        catch (Exception e)
        {
            Debug.LogError($"Encrypted save failed: {e.Message}");
            return false;
        }
    }

    public new T Load<T>(string fileName) where T : new()
    {
        string filePath = GetFilePath(fileName);

        if (!File.Exists(filePath))
            return new T();

        try
        {
            string encrypted = File.ReadAllText(filePath);
            string json = Decrypt(encrypted);
            return JsonUtility.FromJson<T>(json);
        }
        catch (Exception e)
        {
            Debug.LogError($"Encrypted load failed: {e.Message}");
            return new T();
        }
    }

    private string Encrypt(string plainText)
    {
        using (var aes = Aes.Create())
        {
            aes.Key = key;
            aes.IV = iv;

            var encryptor = aes.CreateEncryptor(aes.Key, aes.IV);
            using (var ms = new MemoryStream())
            using (var cs = new CryptoStream(ms, encryptor, CryptoStreamMode.Write))
            using (var sw = new StreamWriter(cs))
            {
                sw.Write(plainText);
                sw.Flush();
                cs.FlushFinalBlock();
                return Convert.ToBase64String(ms.ToArray());
            }
        }
    }

    private string Decrypt(string cipherText)
    {
        byte[] buffer = Convert.FromBase64String(cipherText);

        using (var aes = Aes.Create())
        {
            aes.Key = key;
            aes.IV = iv;

            var decryptor = aes.CreateDecryptor(aes.Key, aes.IV);
            using (var ms = new MemoryStream(buffer))
            using (var cs = new CryptoStream(ms, decryptor, CryptoStreamMode.Read))
            using (var sr = new StreamReader(cs))
            {
                return sr.ReadToEnd();
            }
        }
    }

    private string GetFilePath(string fileName)
        => Path.Combine(savePath, fileName + ".dat"); // 使用.dat扩展名
}
```

## 4. ScriptableObject配置管理

### 4.1 配置数据基类

```csharp
using UnityEngine;
using System.Collections.Generic;

/// <summary>
/// ScriptableObject配置基类
/// </summary>
public abstract class ConfigDataBase : ScriptableObject
{
    public abstract void Validate();
}

/// <summary>
/// 配置数据管理器
/// </summary>
public static class ConfigManager
{
    private static Dictionary<string, ConfigDataBase> configs = new Dictionary<string, ConfigDataBase>();

    /// <summary>
    /// 加载配置
    /// </summary>
    public static T Load<T>() where T : ConfigDataBase
    {
        string key = typeof(T).Name;

        if (configs.TryGetValue(key, out var config))
            return config as T;

        var loaded = Resources.Load<T>($"Configs/{key}");
        if (loaded != null)
        {
            loaded.Validate();
            configs[key] = loaded;
        }
        return loaded;
    }

    /// <summary>
    /// 预加载所有配置
    /// </summary>
    public static void PreloadAll()
    {
        var configs = Resources.LoadAll<ConfigDataBase>("Configs");
        foreach (var config in configs)
        {
            config.Validate();
            ConfigManager.configs[config.GetType().Name] = config;
        }
    }

    /// <summary>
    /// 清除缓存
    /// </summary>
    public static void ClearCache()
    {
        configs.Clear();
    }
}
```

```csharp
using UnityEngine;
using System;

/// <summary>
/// 存档管理器
/// </summary>
public class SaveManager : MonoSingleton<SaveManager>
{
    [Header("Settings")]
    [SerializeField] private bool useEncryption = true;
    [SerializeField] private string encryptionKey = "Match3Roguelike2024";

    public PlayerSaveData CurrentData { get; private set; }

    private JsonSaveSystem saveSystem;
    private const string SAVE_FILE = "player_save";
    private const string LEVEL_PROGRESS_FILE = "level_progress";

    public event System.Action<PlayerSaveData> OnDataChanged;

    protected override void Awake()
    {
        base.Awake();
        saveSystem = useEncryption
            ? new EncryptedJsonSaveSystem("Saves", encryptionKey)
            : new JsonSaveSystem("Saves");
    }

    /// <summary>
    /// 初始化 - 加载或创建新存档
    /// </summary>
    public void Initialize()
    {
        if (saveSystem.Exists(SAVE_FILE))
        {
            CurrentData = saveSystem.Load<PlayerSaveData>(SAVE_FILE);

            // 版本迁移
            if (CurrentData.Version < GetCurrentVersion())
            {
                MigrateData(CurrentData.Version);
            }
        }
        else
        {
            CreateNewSave();
        }
    }

    /// <summary>
    /// 创建新存档
    /// </summary>
    public void CreateNewSave()
    {
        CurrentData = new PlayerSaveData
        {
            PlayerId = Guid.NewGuid().ToString(),
            PlayerName = "Player",
            LastSaveTime = DateTime.Now,
            Energy = 100,
            EnergyLastUpdate = DateTime.Now,
            Coins = 1000,
            Gems = 50
        };
        CurrentData.UnlockedHeroes.Add(1); // 初始英雄

        Save();
    }

    /// <summary>
    /// 保存当前数据
    /// </summary>
    public void Save()
    {
        if (CurrentData == null) return;

        CurrentData.LastSaveTime = DateTime.Now;
        saveSystem.Save(SAVE_FILE, CurrentData);
        OnDataChanged?.Invoke(CurrentData);
        Debug.Log("Game Saved");
    }

    /// <summary>
    /// 自动保存（定时调用）
    /// </summary>
    public void AutoSave()
    {
        Save();
    }

    /// <summary>
    /// 删除存档
    /// </summary>
    public void DeleteSave()
    {
        saveSystem.Delete(SAVE_FILE);
        CurrentData = null;
    }

    // ========== 便捷方法 ==========

    public void AddCoins(int amount)
    {
        CurrentData.Coins += amount;
        Save();
    }

    public bool SpendCoins(int amount)
    {
        if (CurrentData.Coins >= amount)
        {
            CurrentData.Coins -= amount;
            Save();
            return true;
        }
        return false;
    }

    public void AddGems(int amount)
    {
        CurrentData.Gems += amount;
        Save();
    }

    public bool SpendGems(int amount)
    {
        if (CurrentData.Gems >= amount)
        {
            CurrentData.Gems -= amount;
            Save();
            return true;
        }
        return false;
    }

    public void SetCurrentLevel(int level)
    {
        CurrentData.CurrentLevel = level;
        if (level > CurrentData.HighestLevel)
            CurrentData.HighestLevel = level;
        Save();
    }

    public void UnlockHero(int heroId)
    {
        if (!CurrentData.UnlockedHeroes.Contains(heroId))
        {
            CurrentData.UnlockedHeroes.Add(heroId);
            Save();
        }
    }

    public void SelectHero(int heroId)
    {
        if (CurrentData.UnlockedHeroes.Contains(heroId))
        {
            CurrentData.SelectedHeroId = heroId;
            Save();
        }
    }

    // ========== 能量系统 ==========

    public void UpdateEnergy()
    {
        var now = DateTime.Now;

        int energyGain = (int)(elapsed.TotalMinutes / 5);

        }

    }
```

```csharp
    private readonly JsonSaveSystem saveSystem;

    private Dictionary<int, LevelProgressData> levelProgress;

    public LevelProgressManager(JsonSaveSystem saveSystem)
    {
        this.saveSystem = saveSystem;
        LoadProgress();
    }

    private void LoadProgress()
    {
        levelProgress = new Dictionary<int, LevelProgressData>();

        var data = saveSystem.Load<LevelProgressList>(FILE_NAME);
        foreach (var progress in data.Levels)
        {
            levelProgress[progress.LevelId] = progress;
        }
    }

    public LevelProgressData GetProgress(int levelId)
    {
        if (!levelProgress.TryGetValue(levelId, out var progress))
        {
            progress = new LevelProgressData { LevelId = levelId };
            levelProgress[levelId] = progress;
        }
        return progress;
    }

    public void UpdateProgress(int levelId, int score, int stars)
    {
        var progress = GetProgress(levelId);

        bool changed = false;

        if (score > progress.HighScore)
        {
            progress.HighScore = score;
            changed = true;
        }

        if (stars > progress.Stars)
        {
            progress.Stars = stars;
            changed = true;
        }

        if (!progress.Completed && stars > 0)
        {
            progress.Completed = true;
            changed = true;
        }

        if (changed)
        {
            SaveProgress();
        }
    }

    private void SaveProgress()
    {
        var data = new LevelProgressList
        {
            Levels = new List<LevelProgressData>(levelProgress.Values).ToArray()
        };
        saveSystem.Save(FILE_NAME, data);
    }
}

[Serializable]
public class LevelProgressList
{
    public LevelProgressData[] Levels;
}
```

```csharp
/// <summary>

    [SerializeField] private HeroConfig heroConfig;

    {
        base.Awake();
        LoadConfigs();

        SaveManager.Instance.Initialize();
        levelProgressManager = new LevelProgressManager(
            new EncryptedJsonSaveSystem("Saves", "Match3Key2024"));
    }

    private void LoadConfigs()
    {
        if (gemConfig == null) gemConfig = ConfigManager.Load<GemConfig>();
        if (levelConfig == null) levelConfig = ConfigManager.Load<LevelConfig>();
        if (heroConfig == null) heroConfig = ConfigManager.Load<HeroConfig>();
    }

        => Levels.GetLevel(SaveManager.CurrentData.CurrentLevel);

    {

        // 计算奖励
        int coinReward = score / 10;
        SaveManager.AddCoins(coinReward);

        // 解锁下一关
        if (stars > 0 && levelId >= SaveManager.CurrentData.CurrentLevel)

            SaveManager.SetCurrentLevel(levelId + 1);
        }

        // 更新总星数
        UpdateTotalStars();
    }

        int total = 0;
```

## 本课小结

### 核心知识点

| 知识点 | 核心要点 |
|--------|----------|
| PlayerPrefs | 简单键值存储，适合游戏设置 |
| JSON序列化 | 可读性好，适合存档 |
| 加密存储 | AES加密保护敏感数据 |
| ScriptableObject | 静态配置数据，Inspector友好 |
| 存档系统 | 完整的数据结构和管理器 |

### 数据存储选型

```
┌─────────────────────────────────────────────────────────────┐
│                    数据存储选型指南                           │
│                                                             │
│  游戏设置（音量、画质）    → PlayerPrefs                    │
│  静态配置（物品、技能）    → ScriptableObject               │
│  玩家存档（进度、资源）    → JSON + 加密                    │
│  关卡进度（星星、分数）    → JSON                           │
│  运行时缓存               → 内存Dictionary                  │
└─────────────────────────────────────────────────────────────┘
```

### 存档最佳实践

```
1. 定期自动保存（5分钟间隔）
2. 关键节点保存（过关、购买等）
3. 版本迁移机制
4. 云存档备份（可选）
5. 数据校验（防止篡改）
```
