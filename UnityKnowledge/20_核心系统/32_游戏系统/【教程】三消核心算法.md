```csharp
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 棋子类型
/// </summary>
public enum GemType
{
    None = 0,
    Red = 1,
    Blue = 2,
    Green = 3,
    Yellow = 4,
    Purple = 5,
    Special = 6  // 特殊棋子
}

/// <summary>
/// 特殊棋子类型
/// </summary>
public enum SpecialType
{
    None,
    Horizontal,   // 横向消除
    Vertical,     // 纵向消除
    Bomb,         // 炸弹（范围消除）
    Rainbow       // 彩虹（消除所有同类）
}

/// <summary>
/// 棋盘格子
/// </summary>
public class Cell
{
    public int Row;
    public int Column;
    public GemType GemType;
    public SpecialType SpecialType;
    public bool IsEmpty;
    public bool IsBlocked;  // 被障碍物阻挡
    public bool IsMatched;  // 已标记消除

    public Gem GemObject;   // 关联的棋子GameObject

    public Cell(int row, int column)
    {
        Row = row;
        Column = column;
        GemType = GemType.None;
        SpecialType = SpecialType.None;
        IsEmpty = true;
    }

    public void SetGem(GemType type, SpecialType special = SpecialType.None)
    {
        GemType = type;
        SpecialType = special;
        IsEmpty = type == GemType.None;
    }

    public void Clear()
    {
        GemType = GemType.None;
        SpecialType = SpecialType.None;
        IsEmpty = true;
        IsMatched = false;

        if (GemObject != null)
        {
            UnityEngine.Object.Destroy(GemObject.gameObject);
            GemObject = null;
        }
    }
}

/// <summary>
/// 棋盘数据
/// </summary>
public class BoardData
{
    public int Width { get; private set; }
    public int Height { get; private set; }
    public Cell[,] Cells { get; private set; }

    public BoardData(int width, int height)
    {
        Width = width;
        Height = height;
        Cells = new Cell[height, width];

        for (int row = 0; row < height; row++)
        {
            for (int col = 0; col < width; col++)
            {
                Cells[row, col] = new Cell(row, col);
            }
        }
    }

    public Cell GetCell(int row, int col)
    {
        if (row < 0 || row >= Height || col < 0 || col >= Width)
            return null;
        return Cells[row, col];
    }

    public bool IsValidPosition(int row, int col)
    {
        return row >= 0 && row < Height && col >= 0 && col < Width;
    }

    public List<Cell> GetNeighbors(int row, int col)
    {
        var neighbors = new List<Cell>();
        int[][] directions = { new[] { 0, 1 }, new[] { 0, -1 }, new[] { 1, 0 }, new[] { -1, 0 } };

        foreach (var dir in directions)
        {
            var cell = GetCell(row + dir[0], col + dir[1]);
            if (cell != null && !cell.IsBlocked)
                neighbors.Add(cell);
        }

        return neighbors;
    }
}
```

## 2. 匹配检测算法

### 2.1 BFS匹配检测

```csharp
using System.Collections.Generic;

/// <summary>
/// 匹配检测器 - BFS实现
/// </summary>
public class MatchDetector
{
    private BoardData board;
    private bool[,] visited;

    public MatchDetector(BoardData board)
    {
        this.board = board;
    }

    /// <summary>
    /// 检测所有匹配
    /// </summary>
    public List<MatchGroup> FindAllMatches()
    {
        var matches = new List<MatchGroup>();
        visited = new bool[board.Height, board.Width];

        for (int row = 0; row < board.Height; row++)
        {
            for (int col = 0; col < board.Width; col++)
            {
                var cell = board.GetCell(row, col);
                if (cell == null || cell.IsEmpty || cell.IsBlocked || visited[row, col])
                    continue;

                var match = FindMatchAt(row, col);
                if (match != null && match.Count >= 3)
                {
                    matches.Add(match);

                    // 标记已访问
                    foreach (var c in match.Cells)
                        visited[c.Row, c.Column] = true;
                }
            }
        }

        return matches;
    }

    /// <summary>
    /// 检测指定位置的匹配
    /// </summary>
    private MatchGroup FindMatchAt(int row, int col)
    {
        var cell = board.GetCell(row, col);
        if (cell == null || cell.IsEmpty) return null;

        var matchGroup = new MatchGroup(cell.GemType);
        var queue = new Queue<Cell>();
        var localVisited = new HashSet<string>();

        queue.Enqueue(cell);
        localVisited.Add($"{row},{col}");

        while (queue.Count > 0)
        {
            var current = queue.Dequeue();
            matchGroup.AddCell(current);

            var neighbors = board.GetNeighbors(current.Row, current.Column);
            foreach (var neighbor in neighbors)
            {
                string key = $"{neighbor.Row},{neighbor.Column}";
                if (localVisited.Contains(key)) continue;
                if (neighbor.GemType != cell.GemType) continue;

                localVisited.Add(key);
                queue.Enqueue(neighbor);
            }
        }

        return matchGroup.Count >= 3 ? matchGroup : null;
    }

    /// <summary>
    /// 检测交换后的匹配（不实际交换）
    /// </summary>
    public bool HasMatchAfterSwap(int row1, int col1, int row2, int col2)
    {
        var cell1 = board.GetCell(row1, col1);
        var cell2 = board.GetCell(row2, col2);

        if (cell1 == null || cell2 == null) return false;

        // 临时交换
        var tempType = cell1.GemType;
        var tempSpecial = cell1.SpecialType;
        cell1.GemType = cell2.GemType;
        cell1.SpecialType = cell2.SpecialType;
        cell2.GemType = tempType;
        cell2.SpecialType = tempSpecial;

        // 检测匹配
        bool hasMatch = false;

        var match1 = FindMatchAt(row1, col1);
        var match2 = FindMatchAt(row2, col2);

        hasMatch = (match1 != null && match1.Count >= 3) ||
                   (match2 != null && match2.Count >= 3);

        // 还原
        tempType = cell1.GemType;
        tempSpecial = cell1.SpecialType;
        cell1.GemType = cell2.GemType;
        cell1.SpecialType = cell2.SpecialType;
        cell2.GemType = tempType;
        cell2.SpecialType = tempSpecial;

        return hasMatch;
    }

    /// <summary>
    /// 查找可能的移动
    /// </summary>
    public List<SwapMove> FindPossibleMoves()
    {
        var moves = new List<SwapMove>();

        for (int row = 0; row < board.Height; row++)
        {
            for (int col = 0; col < board.Width; col++)
            {
                // 检查右侧交换
                if (col < board.Width - 1)
                {
                    if (HasMatchAfterSwap(row, col, row, col + 1))
                        moves.Add(new SwapMove(row, col, row, col + 1));
                }

                // 检查下方交换
                if (row < board.Height - 1)
                {
                    if (HasMatchAfterSwap(row, col, row + 1, col))
                        moves.Add(new SwapMove(row, col, row + 1, col));
                }
            }
        }

        return moves;
    }

    /// <summary>
    /// 检测是否有可用的移动
    /// </summary>
    public bool HasPossibleMoves()
    {
        return FindPossibleMoves().Count > 0;
    }
}

/// <summary>
/// 匹配组
/// </summary>
public class MatchGroup
{
    public GemType GemType { get; private set; }
    public List<Cell> Cells { get; private set; }

    public int Count => Cells.Count;
    public bool IsHorizontal { get; private set; }
    public bool IsVertical { get; private set; }
    public bool IsLShape { get; private set; }

    public MatchGroup(GemType gemType)
    {
        GemType = gemType;
        Cells = new List<Cell>();
    }

    public void AddCell(Cell cell)
    {
        if (!Cells.Contains(cell))
            Cells.Add(cell);
    }

    public void AnalyzeShape()
    {
        int minRow = int.MaxValue, maxRow = int.MinValue;
        int minCol = int.MaxValue, maxCol = int.MinValue;

        foreach (var cell in Cells)
        {
            minRow = Mathf.Min(minRow, cell.Row);
            maxRow = Mathf.Max(maxRow, cell.Row);
            minCol = Mathf.Min(minCol, cell.Column);
            maxCol = Mathf.Max(maxCol, cell.Column);
        }

        int rowSpan = maxRow - minRow + 1;
        int colSpan = maxCol - minCol + 1;

        IsHorizontal = rowSpan == 1 && colSpan >= 3;
        IsVertical = colSpan == 1 && rowSpan >= 3;
        IsLShape = rowSpan > 1 && colSpan > 1;
    }

    public SpecialType DetermineSpecialType()
    {
        AnalyzeShape();

        if (Count >= 5)
        {
            // 5个及以上生成彩虹
            return SpecialType.Rainbow;
        }
        else if (IsLShape)
        {
            // L型生成炸弹
            return SpecialType.Bomb;
        }
        else if (Count == 4)
        {
            // 4个生成线消
            return IsHorizontal ? SpecialType.Horizontal : SpecialType.Vertical;
        }

        return SpecialType.None;
    }
}

/// <summary>
/// 交换移动
/// </summary>
public struct SwapMove
{
    public int Row1, Col1;
    public int Row2, Col2;

    public SwapMove(int row1, int col1, int row2, int col2)
    {
        Row1 = row1;
        Col1 = col1;
        Row2 = row2;
        Col2 = col2;
    }
}
```

### 2.2 DFS匹配检测（备选）

```csharp
/// <summary>
/// 匹配检测器 - DFS实现
/// </summary>
public class MatchDetectorDFS
{
    private BoardData board;
    private bool[,] visited;

    public MatchDetectorDFS(BoardData board)
    {
        this.board = board;
    }

    public List<MatchGroup> FindAllMatches()
    {
        var matches = new List<MatchGroup>();
        visited = new bool[board.Height, board.Width];

        for (int row = 0; row < board.Height; row++)
        {
            for (int col = 0; col < board.Width; col++)
            {
                if (visited[row, col]) continue;

                var cell = board.GetCell(row, col);
                if (cell == null || cell.IsEmpty || cell.IsBlocked) continue;

                var group = new MatchGroup(cell.GemType);
                DFS(row, col, cell.GemType, group);

                if (group.Count >= 3)
                {
                    matches.Add(group);
                    foreach (var c in group.Cells)
                        visited[c.Row, c.Column] = true;
                }
            }
        }

        return matches;
    }

    private void DFS(int row, int col, GemType targetType, MatchGroup group)
    {
        if (!board.IsValidPosition(row, col)) return;

        var cell = board.GetCell(row, col);
        if (cell == null || cell.IsEmpty || cell.IsBlocked) return;
        if (cell.GemType != targetType) return;
        if (group.Cells.Contains(cell)) return;

        group.AddCell(cell);

        // 四个方向
        DFS(row + 1, col, targetType, group);
        DFS(row - 1, col, targetType, group);
        DFS(row, col + 1, targetType, group);
        DFS(row, col - 1, targetType, group);
    }
}
```

## 3. 消除动画系统

```csharp
using System.Collections;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 消除动画控制器
/// </summary>
public class MatchAnimationController : MonoBehaviour
{
    [Header("Animation Settings")]
    [SerializeField] private float dissolveDuration = 0.3f;
    [SerializeField] private float delayBetweenCells = 0.05f;
    [SerializeField] private float specialEffectDuration = 0.5f;

    [Header("Particle Effects")]
    [SerializeField] private ParticleSystem matchParticle;
    [SerializeField] private ParticleSystem bombParticle;
    [SerializeField] private ParticleSystem rainbowParticle;

    private Queue<IEnumerator> animationQueue = new Queue<IEnumerator>();
    private bool isAnimating = false;

    /// <summary>
    /// 播放消除动画
    /// </summary>
    public IEnumerator PlayMatchAnimation(List<MatchGroup> matches, System.Action onComplete)
    {
        foreach (var match in matches)
        {
            yield return AnimateMatchGroup(match);
        }

        onComplete?.Invoke();
    }

    private IEnumerator AnimateMatchGroup(MatchGroup match)
    {
        // 确定是否生成特殊棋子
        var specialType = match.DetermineSpecialType();
        Cell specialCell = null;

        if (specialType != SpecialType.None)
        {
            // 选择中心位置生成特殊棋子
            specialCell = GetCenterCell(match.Cells);
        }

        // 播放消除动画
        float delay = 0f;
        foreach (var cell in match.Cells)
        {
            if (cell == specialCell) continue;  // 特殊棋子位置暂不消除

            StartCoroutine(AnimateCellDissolve(cell, delay));

            // 播放粒子效果
            if (matchParticle != null)
            {
                var particle = Instantiate(matchParticle, GetWorldPosition(cell), Quaternion.identity);
                particle.Play();
                Destroy(particle.gameObject, particle.main.duration);
            }

            delay += delayBetweenCells;
        }

        yield return new WaitForSeconds(delay + dissolveDuration);

        // 标记消除
        foreach (var cell in match.Cells)
        {
            if (cell != specialCell)
                cell.IsMatched = true;
        }

        // 生成特殊棋子
        if (specialCell != null && specialType != SpecialType.None)
        {
            yield return AnimateSpecialGemCreation(specialCell, specialType, match.GemType);
        }
    }

    private IEnumerator AnimateCellDissolve(Cell cell, float delay)
    {
        yield return new WaitForSeconds(delay);

        if (cell.GemObject == null) yield break;

        var materialController = cell.GemObject.GetComponent<GemMaterialController>();
        if (materialController != null)
        {
            materialController.StartDissolve();
            yield return new WaitForSeconds(dissolveDuration);
        }
    }

    private IEnumerator AnimateSpecialGemCreation(Cell cell, SpecialType specialType, GemType gemType)
    {
        // 播放特殊效果
        ParticleSystem particlePrefab = specialType switch
        {
            SpecialType.Bomb => bombParticle,
            SpecialType.Rainbow => rainbowParticle,
            _ => matchParticle
        };

        if (particlePrefab != null)
        {
            var particle = Instantiate(particlePrefab, GetWorldPosition(cell), Quaternion.identity);
            particle.Play();
            Destroy(particle.gameObject, particle.main.duration);
        }

        yield return new WaitForSeconds(specialEffectDuration);

        // 创建特殊棋子（实际逻辑在BoardController中处理）
        cell.SpecialType = specialType;
    }

    private Cell GetCenterCell(List<Cell> cells)
    {
        // 计算中心点
        float avgRow = 0, avgCol = 0;
        foreach (var cell in cells)
        {
            avgRow += cell.Row;
            avgCol += cell.Column;
        }
        avgRow /= cells.Count;
        avgCol /= cells.Count;

        // 找最接近中心的格子
        Cell closest = cells[0];
        float minDist = float.MaxValue;

        foreach (var cell in cells)
        {
            float dist = Mathf.Abs(cell.Row - avgRow) + Mathf.Abs(cell.Column - avgCol);
            if (dist < minDist)
            {
                minDist = dist;
                closest = cell;
            }
        }

        return closest;
    }

    private Vector3 GetWorldPosition(Cell cell)
    {
        // 转换为世界坐标
        return new Vector3(cell.Column, -cell.Row, 0);
    }
}
```

## 4. 下落填充算法

### 4.1 基础下落算法

```csharp
using System.Collections.Generic;

/// <summary>
/// 下落填充控制器
/// </summary>
public class FallController
{
    private BoardData board;
    private GemType[] availableGemTypes;

    public FallController(BoardData board, GemType[] gemTypes)
    {
        this.board = board;
        this.availableGemTypes = gemTypes;
    }

    /// <summary>
    /// 执行下落（返回所有下落动作）
    /// </summary>
    public List<FallAction> ExecuteFall()
    {
        var actions = new List<FallAction>();

        // 逐列处理
        for (int col = 0; col < board.Width; col++)
        {
            var columnActions = ProcessColumn(col);
            actions.AddRange(columnActions);
        }

        return actions;
    }

    private List<FallAction> ProcessColumn(int col)
    {
        var actions = new List<FallAction>();
        int emptyCount = 0;

        // 从底部向上扫描
        for (int row = 0; row < board.Height; row++)
        {
            var cell = board.GetCell(row, col);

            if (cell.IsBlocked) continue;

            if (cell.IsMatched || cell.IsEmpty)
            {
                emptyCount++;
            }
            else if (emptyCount > 0)
            {
                // 需要下落
                int targetRow = row - emptyCount;
                actions.Add(new FallAction
                {
                    FromRow = row,
                    FromCol = col,
                    ToRow = targetRow,
                    ToCol = col,
                    GemType = cell.GemType,
                    SpecialType = cell.SpecialType
                });

                // 更新棋盘数据
                var targetCell = board.GetCell(targetRow, col);
                targetCell.SetGem(cell.GemType, cell.SpecialType);
                targetCell.GemObject = cell.GemObject;
                cell.Clear();
            }
        }

        // 填充新棋子
        for (int i = 0; i < emptyCount; i++)
        {
            int spawnRow = board.Height - emptyCount + i;
            var newGemType = GetRandomGemType();

            actions.Add(new FallAction
            {
                FromRow = board.Height + i,  // 从棋盘上方生成
                FromCol = col,
                ToRow = spawnRow,
                ToCol = col,
                GemType = newGemType,
                SpecialType = SpecialType.None,
                IsNewGem = true
            });

            var cell = board.GetCell(spawnRow, col);
            cell.SetGem(newGemType);
        }

        return actions;
    }

    private GemType GetRandomGemType()
    {
        return availableGemTypes[Random.Range(0, availableGemTypes.Length)];
    }
}

/// <summary>
/// 下落动作
/// </summary>
public struct FallAction
{
    public int FromRow, FromCol;
    public int ToRow, ToCol;
    public GemType GemType;
    public SpecialType SpecialType;
    public bool IsNewGem;
}
```

### 4.2 下落动画

```csharp
using System.Collections;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 下落动画控制器
/// </summary>
public class FallAnimationController : MonoBehaviour
{
    [Header("Animation Settings")]
    [SerializeField] private float fallSpeed = 10f;
    [SerializeField] private float bounceHeight = 0.2f;
    [SerializeField] private float bounceDuration = 0.15f;
    [SerializeField] private float delayBetweenColumns = 0.1f;

    /// <summary>
    /// 播放下落动画
    /// </summary>
    public IEnumerator PlayFallAnimation(List<FallAction> actions, System.Action onComplete)
    {
        // 按列分组
        var actionsByColumn = GroupByColumn(actions);

        // 逐列播放
        foreach (var kvp in actionsByColumn)
        {
            yield return AnimateColumn(kvp.Value);
            yield return new WaitForSeconds(delayBetweenColumns);
        }

        onComplete?.Invoke();
    }

    private Dictionary<int, List<FallAction>> GroupByColumn(List<FallAction> actions)
    {
        var dict = new Dictionary<int, List<FallAction>>();
        foreach (var action in actions)
        {
            if (!dict.ContainsKey(action.ToCol))
                dict[action.ToCol] = new List<FallAction>();
            dict[action.ToCol].Add(action);
        }
        return dict;
    }

    private IEnumerator AnimateColumn(List<FallAction> actions)
    {
        var coroutines = new List<Coroutine>();

        foreach (var action in actions)
        {
            var gem = CreateOrGetGem(action);
            if (gem != null)
            {
                var coroutine = StartCoroutine(AnimateGemFall(gem, action));
                coroutines.Add(coroutine);
            }
        }

        // 等待所有动画完成
        foreach (var coroutine in coroutines)
            yield return coroutine;
    }

    private GameObject CreateOrGetGem(FallAction action)
    {
        // 根据action创建或获取棋子GameObject
        // 这里简化处理，实际需要从对象池获取
        return null;
    }

    private IEnumerator AnimateGemFall(GameObject gem, FallAction action)
    {
        Vector3 startPos = new Vector3(action.ToCol, -action.FromRow, 0);
        Vector3 endPos = new Vector3(action.ToCol, -action.ToRow, 0);

        gem.transform.position = startPos;

        float distance = Vector3.Distance(startPos, endPos);
        float duration = distance / fallSpeed;
        float elapsed = 0f;

        // 下落
        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;
            float t = elapsed / duration;
            gem.transform.position = Vector3.Lerp(startPos, endPos, t);
            yield return null;
        }

        gem.transform.position = endPos;

        // 弹跳效果
        yield return BounceAnimation(gem.transform, endPos);
    }

    private IEnumerator BounceAnimation(Transform gemTransform, Vector3 originalPos)
    {
        Vector3 bouncePos = originalPos + Vector3.up * bounceHeight;

        float elapsed = 0f;
        while (elapsed < bounceDuration)
        {
            elapsed += Time.deltaTime;
            float t = elapsed / bounceDuration;

            // 使用正弦曲线
            float bounce = Mathf.Sin(t * Mathf.PI);
            gemTransform.position = Vector3.Lerp(originalPos, bouncePos, bounce);

            yield return null;
        }

        gemTransform.position = originalPos;
    }
}
```

## 5. 完整消除流程

```csharp
using System.Collections;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 棋盘控制器 - 协调所有消除逻辑
/// </summary>
public class BoardController : MonoBehaviour
{
    [Header("References")]
    [SerializeField] private GemConfig gemConfig;

    [Header("Board Settings")]
    [SerializeField] private int width = 8;
    [SerializeField] private int height = 8;
    [SerializeField] private float cellSize = 1f;
    [SerializeField] private GemType[] availableGemTypes;

    private BoardData boardData;
    private MatchDetector matchDetector;
    private FallController fallController;
    private MatchAnimationController matchAnimController;
    private FallAnimationController fallAnimController;

    private bool isProcessing = false;
    private int comboCount = 0;

    public event System.Action<int> OnScoreChanged;
    public event System.Action<int> OnComboChanged;
    public event System.Action OnTurnComplete;

    private void Awake()
    {
        InitializeBoard();
    }

    private void InitializeBoard()
    {
        boardData = new BoardData(width, height);
        matchDetector = new MatchDetector(boardData);
        fallController = new FallController(boardData, availableGemTypes);
        matchAnimController = GetComponent<MatchAnimationController>();
        fallAnimController = GetComponent<FallAnimationController>();

        // 初始填充（确保没有初始匹配）
        FillBoardWithoutMatches();
    }

    /// <summary>
    /// 初始填充棋盘（避免初始匹配）
    /// </summary>
    private void FillBoardWithoutMatches()
    {
        for (int row = 0; row < height; row++)
        {
            for (int col = 0; col < width; col++)
            {
                var cell = boardData.GetCell(row, col);
                GemType gemType;

                do
                {
                    gemType = GetRandomGemType();
                } while (WouldCreateMatch(row, col, gemType));

                cell.SetGem(gemType);
                CreateGemObject(cell);
            }
        }
    }

    private bool WouldCreateMatch(int row, int col, GemType type)
    {
        // 检查水平
        if (col >= 2)
        {
            var left1 = boardData.GetCell(row, col - 1);
            var left2 = boardData.GetCell(row, col - 2);
            if (left1.GemType == type && left2.GemType == type)
                return true;
        }

        // 检查垂直
        if (row >= 2)
        {
            var up1 = boardData.GetCell(row - 1, col);
            var up2 = boardData.GetCell(row - 2, col);
            if (up1.GemType == type && up2.GemType == type)
                return true;
        }

        return false;
    }

    /// <summary>
    /// 尝试交换棋子
    /// </summary>
    public bool TrySwap(int row1, int col1, int row2, int col2)
    {
        if (isProcessing) return false;

        // 检查是否相邻
        if (!IsAdjacent(row1, col1, row2, col2)) return false;

        // 检查是否会产生匹配
        if (!matchDetector.HasMatchAfterSwap(row1, col1, row2, col2))
            return false;

        // 执行交换
        StartCoroutine(ProcessSwap(row1, col1, row2, col2));
        return true;
    }

    private bool IsAdjacent(int r1, int c1, int r2, int c2)
    {
        int dr = Mathf.Abs(r1 - r2);
        int dc = Mathf.Abs(c1 - c2);
        return (dr == 1 && dc == 0) || (dr == 0 && dc == 1);
    }

    private IEnumerator ProcessSwap(int row1, int col1, int row2, int col2)
    {
        isProcessing = true;
        comboCount = 0;

        // 交换动画
        yield return AnimateSwap(row1, col1, row2, col2);

        // 更新数据
        SwapCells(row1, col1, row2, col2);

        // 处理消除链
        yield return ProcessMatchChain();

        isProcessing = false;
        OnTurnComplete?.Invoke();
    }

    private IEnumerator AnimateSwap(int r1, int c1, int r2, int c2)
    {
        var cell1 = boardData.GetCell(r1, c1);
        var cell2 = boardData.GetCell(r2, c2);

        Vector3 pos1 = GetWorldPosition(r1, c1);
        Vector3 pos2 = GetWorldPosition(r2, c2);

        float duration = 0.2f;
        float elapsed = 0f;

        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;
            float t = elapsed / duration;

            if (cell1.GemObject != null)
                cell1.GemObject.transform.position = Vector3.Lerp(pos1, pos2, t);
            if (cell2.GemObject != null)
                cell2.GemObject.transform.position = Vector3.Lerp(pos2, pos1, t);

            yield return null;
        }
    }

    private void SwapCells(int r1, int c1, int r2, int c2)
    {
        var cell1 = boardData.GetCell(r1, c1);
        var cell2 = boardData.GetCell(r2, c2);

        var tempType = cell1.GemType;
        var tempSpecial = cell1.SpecialType;
        var tempObject = cell1.GemObject;

        cell1.GemType = cell2.GemType;
        cell1.SpecialType = cell2.SpecialType;
        cell1.GemObject = cell2.GemObject;

        cell2.GemType = tempType;
        cell2.SpecialType = tempSpecial;
        cell2.GemObject = tempObject;
    }

    private IEnumerator ProcessMatchChain()
    {
        while (true)
        {
            // 检测匹配
            var matches = matchDetector.FindAllMatches();
            if (matches.Count == 0) break;

            comboCount++;
            OnComboChanged?.Invoke(comboCount);

            // 计算分数
            int score = CalculateScore(matches, comboCount);
            OnScoreChanged?.Invoke(score);

            // 播放消除动画
            yield return matchAnimController.PlayMatchAnimation(matches, null);

            // 清除已消除的棋子
            ClearMatchedCells(matches);

            // 执行下落
            var fallActions = fallController.ExecuteFall();

            // 播放下落动画
            yield return fallAnimController.PlayFallAnimation(fallActions, null);

            // 更新棋子引用
            UpdateGemReferences(fallActions);

            yield return new WaitForSeconds(0.1f);
        }

        // 检查是否死局
        if (!matchDetector.HasPossibleMoves())
        {
            yield return ShuffleBoard();
        }
    }

    private int CalculateScore(List<MatchGroup> matches, int combo)
    {
        int totalScore = 0;

        foreach (var match in matches)
        {
            int baseScore = match.Count * 10;

            // 连击加成
            float comboMultiplier = 1f + (combo - 1) * 0.5f;

            // 特殊棋子加成
            float specialMultiplier = match.DetermineSpecialType() != SpecialType.None ? 2f : 1f;

            totalScore += Mathf.RoundToInt(baseScore * comboMultiplier * specialMultiplier);
        }

        return totalScore;
    }

    private void ClearMatchedCells(List<MatchGroup> matches)
    {
        foreach (var match in matches)
        {
            foreach (var cell in match.Cells)
            {
                if (cell.SpecialType != SpecialType.None)
                {
                    // 触发特殊效果
                    TriggerSpecialEffect(cell);
                }

                cell.Clear();
            }
        }
    }

    private void TriggerSpecialEffect(Cell cell)
    {
        // 根据特殊类型触发效果
        switch (cell.SpecialType)
        {
            case SpecialType.Horizontal:
                // 消除整行
                for (int c = 0; c < boardData.Width; c++)
                {
                    var target = boardData.GetCell(cell.Row, c);
                    if (!target.IsEmpty)
                        target.IsMatched = true;
                }
                break;

            case SpecialType.Vertical:
                // 消除整列
                for (int r = 0; r < boardData.Height; r++)
                {
                    var target = boardData.GetCell(r, cell.Column);
                    if (!target.IsEmpty)
                        target.IsMatched = true;
                }
                break;

            case SpecialType.Bomb:
                // 范围消除
                for (int dr = -1; dr <= 1; dr++)
                {
                    for (int dc = -1; dc <= 1; dc++)
                    {
                        var target = boardData.GetCell(cell.Row + dr, cell.Column + dc);
                        if (target != null && !target.IsEmpty)
                            target.IsMatched = true;
                    }
                }
                break;

            case SpecialType.Rainbow:
                // 消除所有同类型棋子（需要在上下文中确定类型）
                break;
        }
    }

    private void UpdateGemReferences(List<FallAction> actions)
    {
        foreach (var action in actions)
        {
            var cell = boardData.GetCell(action.ToRow, action.ToCol);
            if (action.IsNewGem)
            {
                CreateGemObject(cell);
            }
        }
    }

    private IEnumerator ShuffleBoard()
    {
        // 收集所有棋子类型
        var gems = new List<GemType>();
        for (int row = 0; row < height; row++)
        {
            for (int col = 0; col < width; col++)
            {
                var cell = boardData.GetCell(row, col);
                if (!cell.IsEmpty && !cell.IsBlocked)
                {
                    gems.Add(cell.GemType);
                }
            }
        }

        // 随机打乱
        for (int i = gems.Count - 1; i > 0; i--)
        {
            int j = Random.Range(0, i + 1);
            (gems[i], gems[j]) = (gems[j], gems[i]);
        }

        // 重新分配
        int index = 0;
        for (int row = 0; row < height; row++)
        {
            for (int col = 0; col < width; col++)
            {
                var cell = boardData.GetCell(row, col);
                if (!cell.IsBlocked)
                {
                    cell.SetGem(gems[index++]);
                }
            }
        }

        // 播放重洗动画
        yield return new WaitForSeconds(0.5f);

        // 如果重洗后仍然死局，再次重洗
        if (!matchDetector.HasPossibleMoves())
        {
            yield return ShuffleBoard();
        }
    }
```

## 本课小结

### 核心算法

| 算法 | 用途 | 复杂度 |
|------|------|--------|
| BFS匹配检测 | 查找相连的同色棋子 | O(n²) |
| DFS匹配检测 | 备选实现 | O(n²) |
| 下落算法 | 棋子下落和填充 | O(n²) |
| 死局检测 | 检查可用移动 | O(n²) |
| 洗牌算法 | 重排棋盘 | O(n) |

### 消除流程

```
1. 玩家交换棋子
     ↓
2. 检测是否产生匹配
     ↓
3. 播放交换动画
     ↓
4. 更新棋盘数据
     ↓
5. 循环处理：
   ├── 检测所有匹配
   ├── 计算分数（含连击加成）
   ├── 播放消除动画
   ├── 生成特殊棋子（如果需要）
   ├── 清除已消除棋子
   ├── 执行下落填充
   └── 播放下落动画
     ↓
6. 检查死局，必要时洗牌
     ↓
7. 回合结束
```

### 特殊棋子规则

| 数量 | 形状 | 生成类型 |
|------|------|----------|
| 3个 | 直线 | 普通消除 |
| 4个 | 直线 | 横/纵向消除 |
| 5个 | 直线 | 彩虹棋子 |
| L/T型 | 拐角 | 炸弹 |
