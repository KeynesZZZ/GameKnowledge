```csharp
// Animation Clip组件配置
public class AnimationClipConfig : ScriptableObject
{
    [Header("动画属性")]
    [SerializeField] private AnimationClip animationClip;

    [Header("动画设置")]
    [SerializeField] private bool loop = true;
    [SerializeField] private WrapMode wrapMode = WrapMode.Loop;
    [SerializeField] private float animationSpeed = 1f;

    [Header("优化设置")]
    [SerializeField] private bool optimize = true;
    [SerializeField] private int compressionRate = 50;
    [SerializeField] private int maxFrameError = 1;

    /// <summary>
    /// 创建优化后的动画片段
    /// </summary>
    public AnimationClip CreateOptimizedClip()
    {
        if (animationClip == null)
            return null;

        // 优化动画片段
        var optimizedClip = new AnimationClip();
        optimizedClip.SetCurve("", 0, animationClip.GetCurve("", 0));

        // 设置动画属性
        AnimationUtility.SetAnimationEvents(optimizedClip,
            new AnimationEvent[0], // 无事件
            new AnimationClipSettings
            {
                loopTime = loop ? animationClip.length : 0f,
                cycleOffset = 0f
            });

        return optimizedClip;
    }
}
```

### 创建简单状态机

```csharp
// 简单角色控制器
public class SimpleCharacterController : MonoBehaviour
{
    [Header("动画控制")]
    [SerializeField] private Animator animator;

    // 动画参数
    private int moveSpeedHash;
    private int attackHash;
    private int deadHash;

    private void Awake()
    {
        // 获取Animator组件
        animator = GetComponent<Animator>();

        // 获取动画参数Hash
        moveSpeedHash = Animator.StringToHash("MoveSpeed");
        attackHash = Animator.StringToHash("Attack");
        deadHash = Animator.StringToHash("Dead");
    }

    private void Update()
    {
        // 检查动画状态
        AnimatorStateInfo stateInfo = animator.GetCurrentAnimatorStateInfo(0);
        Debug.Log($"当前动画状态: {stateInfo.shortNameHash}");

        // 检查动画是否在播放
        bool isPlaying = animator.GetCurrentAnimatorStateInfo(0, out _, out _) != 0;
        Debug.Log($"动画是否在播放: {isPlaying}");
    }

    /// <summary>
    /// 设置移动速度
    /// </summary>
    public void SetMoveSpeed(float speed)
    {
        if (animator != null)
        {
            animator.SetFloat(moveSpeedHash, speed);
        }
    }

    /// <summary>
    /// 触发攻击
    /// </summary>
    public void Attack()
    {
        if (animator != null)
        {
            animator.SetTrigger(attackHash);
        }
    }

    /// <summary>
    /// 触发死亡
    /// </summary>
    public void Die()
    {
        if (animator != null)
```

## 动画过渡 (Transitions)

### 过渡条件

```csharp
// 过渡控制器
public class TransitionController : MonoBehaviour
{
    [Header("过渡控制")]
    [SerializeField] private Animator animator;
    [SerializeField] private float transitionDuration = 0.2f;

    // 过渡条件
    [SerializeField] private bool useExitTime = true;
    [SerializeField] private bool useFixedDuration = false;
    [SerializeField] private float fixedDuration = 0.25f;

    /// <summary>
    /// 播放动画（带过渡）
    /// </summary>
    public void PlayAnimationWithTransition(string animationName)
    {
        if (animator != null)
        {
            // 检查当前动画是否还在播放
            AnimatorStateInfo stateInfo = animator.GetCurrentAnimatorStateInfo(0);

            if (stateInfo.fullPathHash != Animator.StringToHash(animationName))
            {
                // 当前动画与目标动画不同，需要过渡
                if (useExitTime)
                {
                    // 使用ExitTime过渡
                    animator.CrossFade(animationName, transitionDuration);
                }
                else if (useFixedDuration)
                {
                    // 使用固定时长过渡
                    animator.CrossFadeInFixedTime(animationName, fixedDuration);
                }
                else
                {
                    // 直接切换（无过渡）
                    animator.Play(animationName);
                }
            }
        }
    }

    /// <summary>
    /// 播放动画（带混合）
    /// </summary>
    public void PlayAnimationWithBlend(string animationName, float blendDuration)
    {
        if (animator != null)
        {
            // 使用混合过渡
            animator.CrossFade(animationName, blendDuration);
        }
    }

    /// <summary>
    /// 播放动画（带过渡条件）
    /// </summary>
    public void PlayAnimationWithTransitionConditions(string animationName)
    {
        if (animator != null)
        {
            // 检查是否可以过渡
            if (animator.IsInTransition(0))
            {
                // 当前正在过渡中，直接切换
                animator.Play(animationName);
            }
            else
            {
                // 没有在过渡中，使用混合过渡
                animator.CrossFade(animationName, transitionDuration);
            }
        }
    }
}
```

```csharp
public class BlendTreeController : MonoBehaviour

    [SerializeField] private string directionParam = "Direction";

        {
            // 设置速度参数
            animator.SetFloat(speedParam, speed);
            currentSpeed = speed;
        }
    }

    /// 设置移动方向

        SetMoveSpeed(0f);
```

### 参数控制器

```csharp
// 参数控制器
public class AnimatorParameterController : MonoBehaviour
{
    [Header("参数控制")]
    [SerializeField] private Animator animator;

    // 参数Hash缓存
    private Dictionary<string, int> floatParamHashes = new Dictionary<string, int>();
    private Dictionary<string, int> intParamHashes = new Dictionary<string, int>();
    private Dictionary<string, int> boolParamHashes = new Dictionary<string, int>();
    private Dictionary<string, int> triggerParamHashes = new Dictionary<string, int>();

    private void Awake()
    {
        // 获取Animator组件
        animator = GetComponent<Animator>();

        // 缓存参数Hash
        CacheParameterHashes();
    }

    /// <summary>
    /// 缓存参数Hash
    /// </summary>
    private void CacheParameterHashes()
    {
        // 缓存Float参数
        CacheParam("MoveSpeed");
        CacheParam("AttackSpeed");
        CacheParam("Gravity");

        // 缓存Int参数
        CacheParam("StateID");
        CacheParam("ComboCount");
        CacheParam("Level");

        // 缓存Bool参数
        CacheParam("IsGrounded");
        CacheParam("IsDead");
        CacheParam("IsAttacking");

        // 缓存Trigger参数
        CacheParam("Attack");
        CacheParam("Jump");
        CacheParam("Dash");
        CacheParam("Hit");
    }

    /// <summary>
    /// 缓存参数Hash
    /// </summary>
    private void CacheParam(string paramName)
    {
        // Float
        int floatHash = animator.GetParameter(AnimatorControllerParameterType.Float, paramName, out _);
        if (floatHash != 0)
        {
            floatParamHashes[paramName] = floatHash;
        }

        // Int
        int intHash = animator.GetParameter(AnimatorControllerParameterType.Int, paramName, out _);
        if (intHash != 0)
        {
            intParamHashes[paramName] = intHash;
        }

        // Bool
        int boolHash = animator.GetParameter(AnimatorControllerParameterType.Bool, paramName, out _);
        if (boolHash != 0)
        {
            boolParamHashes[paramName] = boolHash;
        }

        // Trigger
        int triggerHash = animator.GetParameter(AnimatorControllerParameterType.Trigger, paramName, out _);
        if (triggerHash != 0)
        {
            triggerParamHashes[paramName] = triggerHash;
        }
    }

    /// <summary>
    /// 设置Float参数
    /// </summary>
    public void SetFloat(string paramName, float value)
    {
        if (animator != null && floatParamHashes.ContainsKey(paramName))
        {
            animator.SetFloat(floatParamHashes[paramName], value);
        }
    }

    /// <summary>
    /// 设置Int参数
    /// </summary>
    public void SetInt(string paramName, int value)
    {
        if (animator != null && intParamHashes.ContainsKey(paramName))
        {
            animator.SetInteger(intParamHashes[paramName], value);
        }
    }

    /// <summary>
    /// 设置Bool参数
    /// </summary>
    public void SetBool(string paramName, bool value)
    {
        if (animator != null && boolParamHashes.ContainsKey(paramName))
        {
            animator.SetBool(boolParamHashes[paramName], value);
        }
    }

    /// <summary>
    /// 触发Trigger参数
    /// </summary>
    public void SetTrigger(string triggerName)
    {
        if (animator != null && triggerParamHashes.ContainsKey(triggerName))
        {
            animator.SetTrigger(triggerParamHashes[triggerName]);
        }
    }

    /// <summary>
    /// 重置Trigger参数
    /// </summary>
    public void ResetTrigger(string triggerName)
    {
        if (animator != null && triggerParamHashes.ContainsKey(triggerName))
        {
            animator.ResetTrigger(triggerParamHashes[triggerName]);
        }
    }
}
```

### 动画事件监听器

```csharp
// 动画事件监听器
public class AnimationEventListener : MonoBehaviour
{
    // 动画事件回调
    public event Action<AnimationEvent> OnAnimationEventFired;

    private void OnAnimationEvent(AnimationEvent animationEvent)
    {
        // 动画事件触发
        Debug.Log($"Animation event fired: {animationEvent.intParameter}");

        // 触发回调
        OnAnimationEventFired?.Invoke(animationEvent);
    }

    /// <summary>
    /// 播放粒子特效
    /// </summary>
    public void PlayParticleEffect()
    {
        // 播放粒子特效
        var particleSystem = GetComponent<ParticleSystem>();
        if (particleSystem != null)
        {
            particleSystem.Play();
        }
    }

    /// <summary>
    /// 播放音效
    /// </summary>
    public void PlaySoundEffect()
    {
        // 播放音效
        var audioSource = GetComponent<AudioSource>();
        if (audioSource != null)
        {
            audioSource.Play();
        }
    }

    /// <summary>
    /// 发送消息
    /// </summary>
    public void SendMessage(string message, SendMessageOptions options)
    {
        // 发送消息到其他GameObject
        var gameObjects = FindObjectsOfType<GameObject>();
        foreach (var go in gameObjects)
        {
            go.SendMessage(message, options);
        }
    }
}
```

## 最佳实践

### DO ✅

- 使用Animator Controller控制角色动画
- 使用Hash代替字符串设置参数
- 使用CrossFade实现平滑过渡
- 使用Blend Tree实现复杂的动画混合
- 使用动画事件处理动画关键时刻
- 合理设置动画层权重
- 使用动画压缩减少文件大小

### DON'T ❌

- 不要在Update中频繁设置参数
- 不要忽略动画层权重
- 不要忘记重置Trigger参数
- 不要使用字符串参数（性能问题）
- 不要在Animator中添加过多参数
- 不要忽略动画过渡条件
- 不要滥用动画事件
