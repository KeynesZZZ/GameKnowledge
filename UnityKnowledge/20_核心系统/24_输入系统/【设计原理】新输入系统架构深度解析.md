```csharp
        {
            action.Enable();
        }

        m_Enabled = true;
    }

    /// <summary>
    /// 禁用动作映射
    /// </summary>
    public void Disable()
    {
        if (!m_Enabled)
            return;

        {
            action.Disable();
        }

        m_Enabled = false;
    }
```

```csharp
        Canceled    // 输入取消
    }

    /// </summary>

    {
        var binding = new InputBinding(path);
        m_Bindings.Add(binding);
        return binding;
    }

    /// <summary>
    /// 添加复合绑定
    /// </summary>
    public InputBinding AddCompositeBinding(string name)
    {
        var binding = InputBindingComposite.CreateBinding(name);
        m_Bindings.Add(binding);
        return binding;
    }

    /// <summary>
    /// 启用动作
    /// </summary>
    public void Enable()
    {
        foreach (var binding in m_Bindings)
        {
            binding.Enable();
        }
    }

    /// <summary>
    /// 禁用动作
    /// </summary>
    public void Disable()
    {
        foreach (var binding in m_Bindings)
        {
            binding.Disable();
        }
    }

    /// <summary>
    /// 触发回调
    /// </summary>
    public void Trigger(InputAction.CallbackContext context)
    {
        m_CurrentPhase = context.phase;

        if (m_Interaction != null)
        {
            m_Interaction.Process(this, context);
        }
    }
}
```

## Input Processor机制

### 处理器工作原理

```
原始输入
    ↓
InputProcessor[] (处理器链)
    ├─> DeadzoneProcessor
    ├─> NormalizeProcessor
    ├─> InvertProcessor
    └─> CustomProcessor
    ↓
处理后输入
    ↓
InputAction (动作)
```

### 常用处理器实现

```csharp
// 1. DeadzoneProcessor - 死区处理器
public class DeadzoneProcessor : InputProcessor<float>
{
    [Tooltip("死区大小")]
    public float minOrDefault = 0.125f;
    [Tooltip("死区最大值")]
    public float maxOrDefault = float.NegativeInfinity;

    public override float Process(float value, InputControl control)
    {
        var min = minOrDefault;
        var max = maxOrDefault;

        if (max == float.NegativeInfinity)
            max = min;

        // 计算死区
        if (Mathf.Abs(value) < min)
            return 0f;

        // 归一化
        if (Mathf.Abs(value) < max)
        {
            var sign = Mathf.Sign(value);
            value = (Mathf.Abs(value) - min) / (max - min);
            return sign * value;
        }

        return value;
    }
}

// 2. NormalizeProcessor - 归一化处理器
public class NormalizeProcessor : InputProcessor<float>
{
    [Tooltip("归一化模式")]
    public NormalizeMode normalize = NormalizeMode.Normalize;

    public enum NormalizeMode
    {
        Normalize,      // 归一化到[0,1]
        Normalize2,     // 归一化到[-1,1]
        Normalize3     // 归一化到[0,1]并保留符号
    }

    public override float Process(float value, InputControl control)
    {
        switch (normalize)
        {
            case NormalizeMode.Normalize:
                return Mathf.Clamp01(value);

            case NormalizeMode.Normalize2:
                return Mathf.Clamp(value, -1f, 1f);

            case NormalizeMode.Normalize3:
                return Mathf.Clamp01(Mathf.Abs(value)) * Mathf.Sign(value);

            default:
                return value;
        }
    }
}

// 3. InvertProcessor - 反转处理器
public class InvertProcessor : InputProcessor<float>
{
    [Tooltip("反转模式")]
    public InvertMode invert = InvertMode.Invert;

    public enum InvertMode
    {
        Invert,     // 完全反转
        InvertX,    // 仅反转X轴
        InvertY     // 仅反转Y轴
    }

    public override float Process(float value, InputControl control)
    {
        switch (invert)
        {
            case InvertMode.Invert:
                return -value;

            case InvertMode.InvertX:
                if (control.layout.IndexOfChild(Vector2.x) != -1)
                    return -value;
                return value;

            case InvertMode.InvertY:
                if (control.layout.IndexOfChild(Vector2.y) != -1)
                    return -value;
                return value;

            default:
                return value;
        }
    }
}

// 4. 自定义处理器
public class ScaleProcessor : InputProcessor<float>
{
    [Tooltip("缩放系数")]
    public float scale = 2f;

    [Tooltip("最小值")]
    public float min = 0f;

    [Tooltip("最大值")]
    public float max = 1f;

    public override float Process(float value, InputControl control)
    {
        // 应用缩放
        value *= scale;

        // 限制范围
        value = Mathf.Clamp(value, min, max);

        return value;
    }
}

// 5. StickDeadzoneProcessor - 摇杆死区处理器
public class StickDeadzoneProcessor : InputProcessor<Vector2>
{
    [Tooltip("死区大小")]
    public float deadzoneMin = 0.125f;

    [Tooltip("死区最大值")]
    public float deadzoneMax = 0.9f;

    [Tooltip("死区模式")]
    public DeadzoneMode deadzoneMode = DeadzoneMode.Circular;

    public enum DeadzoneMode
    {
        Circular,   // 圆形死区
        Axial       // 轴向死区
    }

    public override Vector2 Process(Vector2 value, InputControl control)
    {
        switch (deadzoneMode)
        {
            case DeadzoneMode.Circular:
                return ProcessCircularDeadzone(value);
            case DeadzoneMode.Axial:
                return ProcessAxialDeadzone(value);
            default:
                return value;
        }
    }

    private Vector2 ProcessCircularDeadzone(Vector2 value)
    {
        float magnitude = value.magnitude;

        // 在死区内
        if (magnitude < deadzoneMin)
            return Vector2.zero;

        // 归一化
        if (magnitude < deadzoneMax)
        {
            float newMagnitude = (magnitude - deadzoneMin) / (deadzoneMax - deadzoneMin);
            return value.normalized * newMagnitude;
        }

        return value;
    }

    private Vector2 ProcessAxialDeadzone(Vector2 value)
    {
        // X轴死区
        if (Mathf.Abs(value.x) < deadzoneMin)
            value.x = 0;
        else if (Mathf.Abs(value.x) < deadzoneMax)
        {
            float sign = Mathf.Sign(value.x);
            value.x = (Mathf.Abs(value.x) - deadzoneMin) / (deadzoneMax - deadzoneMin) * sign;
        }

        // Y轴死区
        if (Mathf.Abs(value.y) < deadzoneMin)
            value.y = 0;
        else if (Mathf.Abs(value.y) < deadzoneMax)
        {
            float sign = Mathf.Sign(value.y);
            value.y = (Mathf.Abs(value.y) - deadzoneMin) / (deadzoneMax - deadzoneMin) * sign;
        }

        return value;
    }
}
```

```csharp
// 处理器链执行
public class ProcessorChain : MonoBehaviour
{
    [SerializeField] private InputAction moveAction;

    private void OnEnable()
    {
        // 配置处理器链
        var bindings = moveAction.bindings;

        // 第1个处理器: 死区
        bindings[0].addProcessor<StickDeadzoneProcessor>();

        // 第2个处理器: 归一化
        bindings[0].addProcessor<NormalizeProcessor>();

        // 第3个处理器: 缩放
        bindings[0].addProcessor<ScaleProcessor>();

        // 执行顺序: 原始值 → 死区 → 归一化 → 缩放 → 最终值
    }

    private void OnDisable()
    {
        moveAction.Disable();
    }
}
```

## Interaction系统

### Interaction工作原理

```
输入事件
    ↓
IInputInteraction (交互处理器)
    ├─> PressInteraction
    ├─> HoldInteraction
    ├─> DoublePressInteraction
    ├─> MultiTapInteraction
    └─> SlowTapInteraction
    ↓
InputActionState (动作状态)
    ├─> Started
    ├─> Performed
    └─> Canceled
    ↓
回调触发
```

### 常用Interaction实现

```csharp
// 1. PressInteraction - 按压交互
public class PressInteraction : IInputInteraction
{
    public float pressPoint = 0.5f;
    public float pressOnlyBehavior = PressOnlyBehavior.ReleaseOnly;

    public enum PressOnlyBehavior
    {
        ReleaseOnly,   // 仅在释放时触发
        PressOnly,     // 仅在按下时触发
        PressAndRelease // 按下和释放都触发
    }

    public void Process(InputAction action, InputAction.CallbackContext context)
    {
        switch (context.phase)
        {
            case InputActionPhase.Started:
                if (context.control.value >= pressPoint)
                {
                    action.Trigger(context);
                }
                break;

            case InputActionPhase.Performed:
                if (context.control.value >= pressPoint)
                {
                    action.Trigger(context);
                }
                break;

            case InputActionPhase.Canceled:
                if (context.control.value < pressPoint)
                {
                    action.Trigger(context);
                }
                break;
        }
    }
}

// 2. HoldInteraction - 长按交互
public class HoldInteraction : IInputInteraction
{
    public float duration = 0.4f;
    public float pressPoint = 0.5f;
    public bool pressOnly = false;

    private float m_StartTime;

    public void Process(InputAction action, InputAction.CallbackContext context)
    {
        switch (context.phase)
        {
            case InputActionPhase.Started:
                if (context.control.value >= pressPoint)
                {
                    m_StartTime = Time.realtimeSinceStartup;
                }
                break;

            case InputActionPhase.Performed:
                if (context.control.value >= pressPoint)
                {
                    float heldTime = Time.realtimeSinceStartup - m_StartTime;
                    if (heldTime >= duration)
                    {
                        action.Trigger(context);
                    }
                }
                break;

            case InputActionPhase.Canceled:
                if (context.control.value < pressPoint)
                {
                    float heldTime = Time.realtimeSinceStartup - m_StartTime;
                    if (heldTime >= duration)
                    {
                        action.Trigger(context);
                    }
                }
                break;
        }
    }
}

// 3. DoublePressInteraction - 双击交互
public class DoublePressInteraction : IInputInteraction
{
    public float doublePressTime = 0.2f;
    public float pressPoint = 0.5f;

    private float m_LastPressTime;
    private int m_PressCount;

    public void Process(InputAction action, InputAction.CallbackContext context)
    {
        switch (context.phase)
        {
            case InputActionPhase.Started:
                if (context.control.value >= pressPoint)
                {
                    float currentTime = Time.realtimeSinceStartup;

                    if (currentTime - m_LastPressTime <= doublePressTime)
                    {
                        m_PressCount++;
                    }
                    else
                    {
                        m_PressCount = 1;
                    }

                    m_LastPressTime = currentTime;

                    if (m_PressCount >= 2)
                    {
                        action.Trigger(context);
                        m_PressCount = 0;
                    }
                }
                break;
        }
    }
}

// 4. SlowTapInteraction - 慢点击交互
public class SlowTapInteraction : IInputInteraction
{
    public float pressPoint = 0.5f;
    public float slowTapTime = 0.5f;
    public float speedLimit = 0.4f;

    private Vector2 m_StartPosition;
    private float m_StartTime;

    public void Process(InputAction action, InputAction.CallbackContext context)
    {
        switch (context.phase)
        {
            case InputActionPhase.Started:
                if (context.control.value >= pressPoint)
                {
                    m_StartPosition = context.control.ReadValueAsObject() as Vector2? ?? Vector2.zero;
                    m_StartTime = Time.realtimeSinceStartup;
                }
                break;

            case InputActionPhase.Canceled:
                if (context.control.value < pressPoint)
                {
                    float duration = Time.realtimeSinceStartup - m_StartTime;
                    Vector2 currentPosition = context.control.ReadValueAsObject() as Vector2? ?? Vector2.zero;
                    float distance = Vector2.Distance(m_StartPosition, currentPosition) / duration;

                    if (duration >= slowTapTime && distance < speedLimit)
                    {
                        action.Trigger(context);
                    }
                }
                break;
        }
    }
}

// 5. MultiTapInteraction - 多次点击交互
public class MultiTapInteraction : IInputInteraction
{
    public float tapTime = 0.2f;
    public int tapCount = 3;
    public float pressPoint = 0.5f;

    private float m_LastTapTime;
    private int m_TapCount;

    public void Process(InputAction action, InputAction.CallbackContext context)
    {
        switch (context.phase)
        {
            case InputActionPhase.Started:
                if (context.control.value >= pressPoint)
                {
                    float currentTime = Time.realtimeSinceStartup;

                    if (currentTime - m_LastTapTime <= tapTime)
                    {
                        m_TapCount++;
                    }
                    else
                    {
                        m_TapCount = 1;
                    }

                    m_LastTapTime = currentTime;

                    if (m_TapCount >= tapCount)
                    {
                        action.Trigger(context);
                        m_TapCount = 0;
                    }
                }
                break;
        }
    }
}
```

## Binding机制

### Binding解析

```
Binding路径格式:
<设备>/<控件>/<修饰符>

示例:
<Keyboard>/w          - 键盘W键
<Mouse>/leftButton    - 鼠标左键
<Gamepad>/buttonSouth - 手柄下方按钮
<Joystick>/stick      - 摇杆

复合绑定:
2DVector              - 2D向量
Axis                  - 轴向
OneModifier           - 单修饰符
TwoModifiers          - 双修饰符
```

### Binding实现

```csharp
// InputBinding.cs (简化版)
public struct InputBinding : IEquatable<InputBinding>
{
    public string path;              // 绑定路径
    public string interactions;      // 交互配置
    public string processors;        // 处理器配置
    public string groups;            // 控制组
    public string action;            // 动作名称

    /// <summary>
    /// 解析绑定路径
    /// </summary>
    public bool ParseBinding(string path, out InputControl control)
    {
        // 解析设备类型
        var parts = path.Split('/');

        if (parts.Length < 2)
        {
            control = null;
            return false;
        }

        string deviceType = parts[0].Substring(1, parts[0].Length - 2);
        string controlPath = string.Join("/", parts, 1, parts.Length - 1);

        // 查找设备
        var device = InputSystem.devices.FirstOrDefault(d => d.layout.name == deviceType);
        if (device == null)
        {
            control = null;
            return false;
        }

        // 查找控件
        control = device.TryGetChildControl(controlPath);
        return control != null;
    }

    /// <summary>
    /// 启用绑定
    /// </summary>
    public void Enable()
    {
        if (ParseBinding(path, out var control))
        {
            // 绑定回调
            control.AddListener(callbackContext => {
                // 触发Action
            });
        }
    }
}
```

```csharp
// 多平台输入适配
public class MultiPlatformInput : MonoBehaviour
{
    [SerializeField] private InputActionAsset inputActions;

    private void Start()
    {
        // 检测当前平台
        if (Application.platform == RuntimePlatform.WindowsPlayer ||
            Application.platform == RuntimePlatform.OSXPlayer)
        {
            // PC平台: 键盘+鼠标
            ApplyControlScheme("KeyboardMouse");
        }
        else if (Application.platform == RuntimePlatform.IPhonePlayer ||
                 Application.platform == RuntimePlatform.Android)
        {
            // 移动平台: 触摸+手柄
            ApplyControlScheme("Touch");
        }
        else if (Gamepad.current != null)
        {
            // 手柄连接
            ApplyControlScheme("Gamepad");
        }

        inputActions.Enable();
    }

    private void ApplyControlScheme(string schemeName)
    {
        var schemes = inputActions.controlSchemes;
        foreach (var scheme in schemes)
        {
            if (scheme.name == schemeName)
            {
                scheme.Apply(inputActions);
                break;
            }
        }
    }
}
```
