```csharp
// InputActionAsset创建工具
public class InputActionAssetCreator : MonoBehaviour
{
    [MenuItem("Tools/Create Input Actions")]
    public static void CreateInputActions()
    {
        // 1. 创建InputActionAsset
        var inputActions = ScriptableObject.CreateInstance<InputActionAsset>();
        inputActions.name = "GameInputActions";

        // 2. 创建Game Map
        var gameMap = inputActions.AddActionMap("Game");

        // 3. 创建Move Action
        var moveAction = gameMap.AddAction("Move", InputActionType.Value);
        moveAction.AddBinding("<Keyboard>/w").With("Name", "Up");
        moveAction.AddBinding("<Keyboard>/s").With("Name", "Down");
        moveAction.AddBinding("<Keyboard>/a").With("Name", "Left");
        moveAction.AddBinding("<Keyboard>/d").With("Name", "Right");
        moveAction.AddBinding("<Gamepad>/leftStick");

        // 4. 创建Jump Action
        var jumpAction = gameMap.AddAction("Jump", InputActionType.Button);
        jumpAction.AddBinding("<Keyboard>/space");
        jumpAction.AddBinding("<Gamepad>/buttonSouth");

        // 5. 创建Attack Action
        var attackAction = gameMap.AddAction("Attack", InputActionType.Button);
        attackAction.AddBinding("<Mouse>/leftButton");
        attackAction.AddBinding("<Gamepad>/buttonRight");

        // 6. 创建Sprint Action
        var sprintAction = gameMap.AddAction("Sprint", InputActionType.Button);
        sprintAction.AddBinding("<Keyboard>/leftShift");
        sprintAction.AddBinding("<Gamepad>/buttonWest");

        // 7. 创建UI Map
        var uiMap = inputActions.AddActionMap("UI");

        // 8. 创建Navigate Action
        var navigateAction = uiMap.AddAction("Navigate", InputActionType.Value);
        navigateAction.AddBinding("<Keyboard>/upArrow");
        navigateAction.AddBinding("<Keyboard>/downArrow");
        navigateAction.AddBinding("<Keyboard>/leftArrow");
        navigateAction.AddBinding("<Keyboard>/rightArrow");
        navigateAction.AddBinding("<Gamepad>/dpad");

        // 9. 创建Submit Action
        var submitAction = uiMap.AddAction("Submit", InputActionType.Button);
        submitAction.AddBinding("<Keyboard>/enter");
        submitAction.AddBinding("<Gamepad>/buttonSouth");

        // 10. 创建Cancel Action
        var cancelAction = uiMap.AddAction("Cancel", InputActionType.Button);
        cancelAction.AddBinding("<Keyboard>/escape");
        cancelAction.AddBinding("<Gamepad>/buttonEast");

        // 11. 保存资源
        AssetDatabase.CreateAsset(inputActions, "Assets/Input/GameInputActions.inputactions");
        AssetDatabase.Refresh();

        Debug.Log("Input Actions created successfully!");
    }
}
```

### UnifiedInputManager实现

```csharp
/// <summary>
/// 统一输入管理器
/// </summary>
public class UnifiedInputManager : MonoBehaviour
{
    // 单例
    public static UnifiedInputManager Instance { get; private set; }

    // InputActionAsset
    [SerializeField] private InputActionAsset inputActions;

    // Action Maps
    public InputActionMap gameMap { get; private set; }
    public InputActionMap uiMap { get; private set; }
    public InputActionMap menuMap { get; private set; }

    // Game Actions
    public InputAction moveAction => gameMap.FindAction("Move");
    public InputAction jumpAction => gameMap.FindAction("Jump");
    public InputAction attackAction => gameMap.FindAction("Attack");
    public InputAction sprintAction => gameMap.FindAction("Sprint");
    public InputAction interactAction => gameMap.FindAction("Interact");

    // UI Actions
    public InputAction navigateAction => uiMap.FindAction("Navigate");
    public InputAction submitAction => uiMap.FindAction("Submit");
    public InputAction cancelAction => uiMap.FindAction("Cancel");

    // Menu Actions
    public InputAction selectAction => menuMap.FindAction("Select");
    public InputAction backAction => menuMap.FindAction("Back");

    // 当前控制方案
    public InputControlScheme currentScheme { get; private set; }

    // 当前设备
    public InputDevice currentDevice { get; private set; }

    private void Awake()
    {
        if (Instance == null)
        {
            Instance = this;
            DontDestroyOnLoad(gameObject);
        }
        else
        {
            Destroy(gameObject);
            return;
        }

        InitializeInputActions();
        DetectInputScheme();
    }

    /// <summary>
    /// 初始化输入配置
    /// </summary>
    private void InitializeInputActions()
    {
        // 加载InputActionAsset
        if (inputActions == null)
        {
            inputActions = Resources.Load<InputActionAsset>("Input/GameInputActions");
        }

        // 获取Action Maps
        gameMap = inputActions.FindActionMap("Game");
        uiMap = inputActions.FindActionMap("UI");
        menuMap = inputActions.FindActionMap("Menu");

        // 启用Game Map
        gameMap.Enable();
    }

    /// <summary>
    /// 检测输入方案
    /// </summary>
    private void DetectInputScheme()
    {
        // 检查手柄
        if (Gamepad.current != null)
        {
            currentDevice = Gamepad.current;
            currentScheme = inputActions.controlSchemes.FirstOrDefault(s => s.name == "Gamepad");
            Debug.Log($"Using gamepad: {currentDevice.name}");
        }
        // 检查触摸
        else if (Touchscreen.current != null)
        {
            currentDevice = Touchscreen.current;
            currentScheme = inputActions.controlSchemes.FirstOrDefault(s => s.name == "Touch");
            Debug.Log($"Using touchscreen");
        }
        // 默认键盘鼠标
        else
        {
            currentDevice = Keyboard.current;
            currentScheme = inputActions.controlSchemes.FirstOrDefault(s => s.name == "KeyboardMouse");
            Debug.Log($"Using keyboard & mouse");
        }
    }

    /// <summary>
    /// 切换到Game Map
    /// </summary>
    public void SwitchToGameMap()
    {
        uiMap.Disable();
        menuMap.Disable();
        gameMap.Enable();
    }

    /// <summary>
    /// 切换到UI Map
    /// </summary>
    public void SwitchToUIMap()
    {
        gameMap.Disable();
        menuMap.Disable();
        uiMap.Enable();
    }

    /// <summary>
    /// 切换到Menu Map
    /// </summary>
    public void SwitchToMenuMap()
    {
        gameMap.Disable();
        uiMap.Disable();
        menuMap.Enable();
    }

    /// <summary>
    /// 获取当前设备类型
    /// </summary>
    public DeviceType GetDeviceType()
    {
        if (currentDevice is Gamepad)
            return DeviceType.Gamepad;
        else if (currentDevice is Touchscreen)
            return DeviceType.Touchscreen;
        else
            return DeviceType.KeyboardMouse;
    }

    public enum DeviceType
    {
        KeyboardMouse,
        Gamepad,
        Touchscreen
    }
}
```

### PlayerController完整实现

```csharp
/// <summary>
/// 玩家控制器
/// </summary>
public class PlayerController : MonoBehaviour
{
    [Header("References")]
    [SerializeField] private CharacterController characterController;
    [SerializeField] private Camera playerCamera;

    [Header("Movement Settings")]
    [SerializeField] private float moveSpeed = 5f;
    [SerializeField] private float sprintSpeed = 8f;
    [SerializeField] private float jumpForce = 10f;
    [SerializeField] private float gravity = -20f;

    [Header("Input Settings")]
    [SerializeField] private float lookSensitivity = 2f;
    [SerializeField] private float deadzone = 0.1f;

    // 状态
    private Vector3 velocity;
    private Vector2 moveInput;
    private Vector2 lookInput;
    private bool isJumping;
    private bool isSprinting;

    // 相机旋转
    private float verticalRotation = 0f;

    private void Awake()
    {
        if (characterController == null)
            characterController = GetComponent<CharacterController>();

        if (playerCamera == null)
            playerCamera = Camera.main;
    }

    private void Start()
    {
        // 订阅输入事件
        SubscribeToInput();
    }

    /// <summary>
    /// 订阅输入事件
    /// </summary>
    private void SubscribeToInput()
    {
        var inputManager = UnifiedInputManager.Instance;

        // 移动
        inputManager.moveAction.performed += OnMovePerformed;
        inputManager.moveAction.canceled += OnMoveCanceled;

        // 跳跃
        inputManager.jumpAction.performed += OnJumpPerformed;

        // 冲刺
        inputManager.sprintAction.performed += OnSprintPerformed;
        inputManager.sprintAction.canceled += OnSprintCanceled;
    }

    /// <summary>
    /// 取消订阅输入事件
    /// </summary>
    private void UnsubscribeFromInput()
    {
        var inputManager = UnifiedInputManager.Instance;

        inputManager.moveAction.performed -= OnMovePerformed;
        inputManager.moveAction.canceled -= OnMoveCanceled;

        inputManager.jumpAction.performed -= OnJumpPerformed;

        inputManager.sprintAction.performed -= OnSprintPerformed;
        inputManager.sprintAction.canceled -= OnSprintCanceled;
    }

    private void OnEnable()
    {
        SubscribeToInput();
    }

    private void OnDisable()
    {
        UnsubscribeFromInput();
    }

    /// <summary>
    /// 移动输入
    /// </summary>
    private void OnMovePerformed(InputAction.CallbackContext context)
    {
        moveInput = context.ReadValue<Vector2>();

        // 应用死区
        if (moveInput.magnitude < deadzone)
        {
            moveInput = Vector2.zero;
        }
        else
        {
            moveInput = moveInput.normalized;
        }
    }

    private void OnMoveCanceled(InputAction.CallbackContext context)
    {
        moveInput = Vector2.zero;
    }

    /// <summary>
    /// 跳跃输入
    /// </summary>
    private void OnJumpPerformed(InputAction.CallbackContext context)
    {
        if (characterController.isGrounded)
        {
            velocity.y = jumpForce;
            isJumping = true;
        }
    }

    /// <summary>
    /// 冲刺输入
    /// </summary>
    private void OnSprintPerformed(InputAction.CallbackContext context)
    {
        isSprinting = true;
    }

    private void OnSprintCanceled(InputAction.CallbackContext context)
    {
        isSprinting = false;
    }

    private void Update()
    {
        HandleMovement();
        HandleGravity();
    }

    /// <summary>
    /// 处理移动
    /// </summary>
    private void HandleMovement()
    {
        // 计算移动方向
        Vector3 moveDirection = transform.right * moveInput.x + transform.forward * moveInput.y;

        // 应用速度
        float currentSpeed = isSprinting ? sprintSpeed : moveSpeed;
        characterController.Move(moveDirection * currentSpeed * Time.deltaTime);
    }

    /// <summary>
    /// 处理重力
    /// </summary>
    private void HandleGravity()
    {
        if (characterController.isGrounded && velocity.y < 0)
        {
            velocity.y = -2f;
        }

        velocity.y += gravity * Time.deltaTime;
        characterController.Move(velocity * Time.deltaTime);
    }
}
```

```csharp
/// <summary>
/// 虚拟摇杆

public class VirtualJoystick : MonoBehaviour, IPointerDownHandler, IPointerUpHandler, IDragHandler
{
    [Header("Visuals")]
    [SerializeField] private RectTransform background;
    [SerializeField] private RectTransform handle;
    [SerializeField] private float handleRange = 50f;

    // 当前值
    private Vector2 value;

    // 事件
    public event Action<Vector2> OnValueChanged;

    /// <summary>
    /// 按下
    /// </summary>
    public void OnPointerDown(PointerEventData eventData)
    {
        background.position = eventData.position;
        handle.position = eventData.position;
        value = Vector2.zero;
        OnValueChanged?.Invoke(value);
    }

    /// <summary>
    /// 拖拽
    /// </summary>
    public void OnDrag(PointerEventData eventData)
    {
        // 计算方向
        Vector2 direction = eventData.position - (Vector2)background.position;

        // 限制范围
        direction = Vector2.ClampMagnitude(direction, handleRange);

        // 更新手柄位置
        handle.position = background.position + (Vector3)direction;

        // 计算归一化值
        value = direction / handleRange;
        OnValueChanged?.Invoke(value);
    }

    /// <summary>
    /// 抬起
    /// </summary>
    public void OnPointerUp(PointerEventData eventData)
    {
        handle.position = background.position;
        value = Vector2.zero;
        OnValueChanged?.Invoke(value);
    }
}

/// <summary>
```
```csharp
/// 输入绑定UI
/// </summary>
public class InputBindingUI : MonoBehaviour
{
    [Header("References")]
    [SerializeField] private GameObject bindingItemPrefab;
    [SerializeField] private Transform content;

    [Header("Input Actions")]
    [SerializeField] private InputActionAsset inputActions;

    // 当前正在绑定的动作
    private InputAction rebindingAction;

    private void Start()
    {
        PopulateBindings();
    }

    /// <summary>
    /// 填充绑定列表
    /// </summary>
    private void PopulateBindings()
    {
        // 清空现有项
        foreach (Transform child in content)
        {
            Destroy(child.gameObject);
        }

        // 遍历所有Action
        foreach (var map in inputActions.actionMaps)
        {
            // 只显示Game Map
            if (map.name != "Game")
                continue;

            foreach (var action in map.actions)
            {
                // 创建绑定项
                var bindingItem = Instantiate(bindingItemPrefab, content);
                var bindingUI = bindingItem.GetComponent<BindingItemUI>();

                // 设置UI
                bindingUI.SetActionName(action.name);
                bindingUI.SetBindings(GetBindingDisplayNames(action));
                bindingUI.OnRebindClicked += () => StartRebinding(action);
            }
        }
    }

    /// <summary>
    /// 获取绑定显示名称
    /// </summary>
    private List<string> GetBindingDisplayNames(InputAction action)
    {
        var displayNames = new List<string>();

        foreach (var binding in action.bindings)
        {
            if (!string.IsNullOrEmpty(binding.path))
            {
                // 获取显示名称
                var displayName = GetBindingDisplayName(binding);
                displayNames.Add(displayName);
            }
        }

        return displayNames;
    }

    /// <summary>
    /// 获取绑定显示名称
    /// </summary>
    private string GetBindingDisplayName(InputBinding binding)
    {
        // 解析路径
        var parts = binding.path.Split('/');

        if (parts.Length < 2)
            return binding.path;

        string deviceType = parts[0].Substring(1, parts[0].Length - 2);
        string controlPath = string.Join("/", parts, 1, parts.Length - 1);

        // 简化显示
        if (deviceType == "Keyboard")
        {
            return controlPath.ToUpper();
        }
        else if (deviceType == "Mouse")
        {
            return controlPath.Replace("Button", "").ToUpper();
        }
        else if (deviceType == "Gamepad")
        {
            return controlPath.ToUpper();
        }

        return binding.path;
    }

    /// <summary>
    /// 开始重新绑定
    /// </summary>
    private void StartRebinding(InputAction action)
    {
        rebindingAction = action;

        // 禁用所有Action
        inputActions.Disable();

        // 启用重新绑定
        var rebindOperation = action.PerformInteractiveRebinding()
            .WithControlsExcluding("<Mouse>/position")
            .OnComplete(operation =>
            {
                // 重新绑定完成
                operation.Dispose();
                rebindingAction = null;

                // 重新启用所有Action
                inputActions.Enable();

                // 刷新UI
                PopulateBindings();

                Debug.Log($"Rebinding complete for {action.name}");
            })
            .Start();
    }
}

/// <summary>
/// 绑定项UI
/// </summary>
public class BindingItemUI : MonoBehaviour
{
    [Header("UI Elements")]
    [SerializeField] private TextMeshProUGUI actionNameText;
    [SerializeField] private TextMeshProUGUI bindingsText;
    [SerializeField] private Button rebindButton;

    // 事件
    public event Action OnRebindClicked;

    /// <summary>
    /// 设置动作名称
    /// </summary>
    public void SetActionName(string name)
    {
        actionNameText.text = name;
    }

    /// <summary>
    /// 设置绑定显示
    /// </summary>
    public void SetBindings(List<string> bindings)
    {
        bindingsText.text = string.Join(" + ", bindings);
    }

    /// <summary>
    /// 重新绑定按钮点击
    /// </summary>
    public void OnRebindButtonClick()
    {
        OnRebindClicked?.Invoke();
    }
}
```

## 最佳实践

### DO ✅

- 使用InputActionAsset统一管理输入
- 合理分组Action Map（Game、UI、Menu）
- 为多平台配置多种Binding
- 使用事件而非轮询
- 实现输入重绑定功能
- 添加虚拟控制器支持移动端
- 使用统一的输入管理器
- 控制Action数量在合理范围

### DON'T ❌

- 不要混合使用Legacy Input和Input System
- 不要忽略设备切换处理
- 不要硬编码输入键位
- 不要在Update中直接读取输入
- 不要忘记禁用不用的Action Map
- 不要忽略输入冲突
- 不要过度使用Processor
- 不要忽略移动端虚拟控制器
