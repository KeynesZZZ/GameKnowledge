512×512 Plane → 圆筒（从边缘开始卷）

像真的卷纸一样：从左边缘 / 底边缘 / 外边缘逐步卷起，已卷部分贴合圆弧，未卷部分保持平面

一、"从边缘卷"和"全局 Lerp"的区别

之前的做法是全局 lerp：所有顶点同时从平面位置向圆筒位置过渡，中间状态是一个"均匀弯曲的弧面"。但这不像真实的卷纸——真实卷纸时，已经卷过去的部分完全贴在圆筒上，还没卷到的部分还是平的，两者之间有一条"卷曲前沿"。

左：全局 Lerp（均匀弯曲） · 右：从边缘卷（前沿推进，左侧已卷，右侧仍平）

二、核心数学：卷曲前沿 + 分段处理

1定义卷曲前沿位置 f

参数 t ∈ [0, 1] 控制卷曲进度。以"沿 X 轴从左边缘开始卷"为例，前沿位置 f 表示"已经卷到了 X 轴上的哪个位置"：

前沿位置：f = -W/2 + t × W = -256 + t × 512 t = 0 → f = -256（最左边，还没开始卷） t = 0.5 → f = 0 （卷到正中间） t = 1 → f = +256 （卷到最右边，完全卷完）

2每个顶点的分段判断

对于每个顶点的 x 坐标，和前沿 f 比较，分两种情况：

if (x ≤ f) → 已卷区域：这个顶点已经贴在圆筒弧面上 if (x > f) → 未卷区域：这个顶点还在平面上，但要跟着前沿平移

3已卷区域的坐标计算

已卷部分的顶点，其 x 坐标到前沿 f 的距离就是它在圆筒弧面上走过的弧长。从前沿处开始，沿圆筒表面"贴"上去：

弧长（从前沿量起）：arcLen = f - x （x ≤ f，所以 arcLen ≥ 0） 半径：R = W / (2π) ≈ 81.49 弧长对应的角度：θ = arcLen / R 前沿在圆筒上的切点位置（圆筒和平面相切的地方）： 切点 X = f 切点 Z = 0 （和平面齐平） 从切点沿圆弧走角度 θ： newX = f - R × sin(θ) ← 从前沿往左（已卷方向）走 newZ = R - R × cos(θ) ← 向上抬起（离开平面） newY = y ← 高度不变
从边缘卷的几何关系：圆筒在前沿处与平面相切，已卷部分贴在圆弧上

4未卷区域的坐标计算

未卷部分保持平面（Z=0），但要注意：随着卷曲推进，平面的未卷部分不动，因为前沿就是圆筒和平面的切点，未卷部分自然还在原位。

if (x > f): newX = x ← 保持原位 newZ = 0 ← 还在平面上 newY = y ← 高度不变

5为什么是 R - R×cos(θ) 而不是别的？

想象圆筒放在平面上方，圆心在 (f, R)。圆筒在 f 处和平面相切（切点 Z=0）。从切点沿圆弧走角度 θ：

圆心位置：(f, R) ← 在前沿正上方，距平面 R 圆弧上的点（从切点逆时针走 θ）： X = 圆心X - R×sin(θ) = f - R×sin(θ) Z = 圆心Z - R×cos(θ) = R - R×cos(θ) 验证： θ=0 → X=f, Z=0 ✓ 切点，在平面上 θ=π → X=f, Z=2R ✓ 圆筒顶部 θ=2π → X=f, Z=0 ✓ 绕一圈回到切点

三、三种卷曲方向

模式卷曲方向前沿轴弧长计算最终形状
沿 X 卷从左边缘 → 右f 沿 X 轴推进arcLen = f - x竖直圆筒（轴 = Y）
沿 Y 卷从底边缘 → 上f 沿 Y 轴推进arcLen = f - y水平圆筒（轴 = X）
从中心卷从外边缘 → 中心f 沿径向收缩arcLen = r - f漏斗/碗形向上卷起

沿 X 轴卷（从左边缘开始）

f = -256 + t × 512 // 前沿从左到右 R = 512 / (2π) 对每个顶点 (x, y): if (x ≤ f): arcLen = f - x θ = arcLen / R newX = f - R × sin(θ) newZ = R - R × cos(θ) newY = y else: newX = x, newZ = 0, newY = y

沿 Y 轴卷（从底边缘开始）

f = -256 + t × 512 // 前沿从下到上 R = 512 / (2π) 对每个顶点 (x, y): if (y ≤ f): arcLen = f - y θ = arcLen / R newY = f - R × sin(θ) newZ = R - R × cos(θ) newX = x else: newX = x, newZ = 0, newY = y

从中心卷（从外边缘向内收）

rMax = √(256² + 256²) ≈ 362 f = rMax × (1 - t) // 前沿从外向内收缩 R = rMax / π 对每个顶点 (x, y): r = √(x² + y²) φ = atan2(y, x) if (r ≥ f): // 已超过前沿 → 已卷 arcLen = r - f θ = arcLen / R newR = f + R × sin(θ) // 注意：这里是从前沿向外"翻"上去 → 但为了向上卷，我们让它在径向收缩的同时抬高： flatR = f // 水平投影收缩到前沿位置 newX = (f × cos(φ)) - (R×sin(θ)) × cos(φ) // 简化处理 ... 实际用极坐标更直观： horizR = f - R × sin(θ) // 水平径向距离（向内卷） newX = horizR × cos(φ) newY = horizR × sin(φ) newZ = R - R × cos(θ) // 向上抬起 else: newX = x, newY = y, newZ = 0

using UnityEngine; public enum CurlDir { AlongX, AlongY, FromCenter } [RequireComponent(typeof(MeshFilter))] public class EdgeCurl : MonoBehaviour { [Range(0f, 1f)] public float t = 0f; [Range(8, 128)] public int segments = 64; public float planeSize = 512f; public CurlDir dir = CurlDir.AlongX; Vector3[] baseVerts; Mesh mesh; void Start() { mesh = GenPlane(planeSize, planeSize, segments, segments); GetComponent<MeshFilter>().mesh = mesh; baseVerts = mesh.vertices.Clone() as Vector3[]; } void Update() => Deform(t); void Deform(float t) { float W = planeSize; float R = W / (2f * Mathf.PI); var nv = new Vector3[baseVerts.Length]; for (int i = 0; i < baseVerts.Length; i++) { float x = baseVerts[i].x, y = baseVerts[i].y; switch (dir) { case CurlDir.AlongX: { float f = -W/2f + t * W; // 前沿位置 if (x <= f) { float arc = f - x; float th = arc / R; nv[i] = new Vector3( f - R * Mathf.Sin(th), // 贴圆弧 y, R - R * Mathf.Cos(th)); // 抬起 } else { nv[i] = new Vector3(x, y, 0); // 保持平面 } break; } case CurlDir.AlongY: { float f = -W/2f + t * W; if (y <= f) { float arc = f - y; float th = arc / R; nv[i] = new Vector3( x, f - R * Mathf.Sin(th), R - R * Mathf.Cos(th)); } else { nv[i] = new Vector3(x, y, 0); } break; } case CurlDir.FromCenter: { float rMax = Mathf.Sqrt(W*W/4f + W*W/4f); float Rc = rMax / Mathf.PI; float f = rMax * (1f - t); // 前沿从外向内 float r = Mathf.Sqrt(x*x + y*y); float phi = Mathf.Atan2(y, x); if (r >= f) { float arc = r - f; float th = arc / Rc; float hr = f - Rc * Mathf.Sin(th); nv[i] = new Vector3( hr * Mathf.Cos(phi), hr * Mathf.Sin(phi), Rc - Rc * Mathf.Cos(th)); } else { nv[i] = new Vector3(x, y, 0); } break; } } } mesh.vertices = nv; mesh.RecalculateNormals(); mesh.RecalculateBounds(); } Mesh GenPlane(float w, float h, int sx, int sy) { var m = new Mesh(); var v = new Vector3[(sx+1)*(sy+1)]; var uv = new Vector2[(sx+1)*(sy+1)]; var tri = new int[sx*sy*6]; for (int iy=0;iy<=sy;iy++) for (int ix=0;ix<=sx;ix++){ int id=iy*(sx+1)+ix; float u=(float)ix/sx, vv=(float)iy/sy; v[id]=new Vector3(u*w-w/2f, vv*h-h/2f, 0); uv[id]=new Vector2(u,vv); } int ti=0; for(int iy=0;iy<sy;iy++) for(int ix=0;ix<sx;ix++){ int a=iy*(sx+1)+ix,b=a+1,c=a+(sx+1),d=c+1; tri[ti++]=a;tri[ti++]=c;tri[ti++]=b; tri[ti++]=b;tri[ti++]=c;tri[ti++]=d; } m.vertices=v; m.uv=uv; m.triangles=tri; m.RecalculateNormals(); m.RecalculateBounds(); return m; } }
