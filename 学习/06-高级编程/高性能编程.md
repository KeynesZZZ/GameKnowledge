### 1.1 避免 GC 分配


### 1.2 使用 Span<T> 避免分配

```csharp
using System;

// ❌ 子字符串产生新分配
string ExtractDomainBad(string url)
{
    int start = url.IndexOf("://") + 3;
    int end = url.IndexOf('/', start);
    return url.Substring(start, end - start);  // 新字符串
}

// ✅ 使用 Span 零分配
ReadOnlySpan<char> ExtractDomainGood(ReadOnlySpan<char> url)
{
    int start = url.IndexOf("://".AsSpan()) + 3;
    int end = url.Slice(start).IndexOf('/');
    return url.Slice(start, end);
}

// 使用
string url = "https://example.com/path";
var domain = ExtractDomainGood(url);
string result = new string(domain);  // 只在最后需要时分配
```

### 1.3 ArrayPool<T> 复用数组

```csharp
using System.Buffers;

// ❌ 频繁分配大数组
void ProcessDataBad(int size)
{
    byte[] buffer = new byte[size];  // 大对象堆分配
    Process(buffer);
    // GC 回收
}

// ✅ 使用数组池
void ProcessDataGood(int size)
{
    byte[] buffer = ArrayPool<byte>.Shared.Rent(size);
    try
    {
        Process(buffer.AsSpan(0, size));  // 注意：实际长度可能大于 size
    }
    finally
    {
        ArrayPool<byte>.Shared.Return(buffer);
    }
}
```

### 1.4 对象池模式

```csharp
using System.Collections.Concurrent;

public class ObjectPool<T> where T : new()
{
    private readonly ConcurrentBag<T> _pool = new();
    private readonly int _maxSize;

    public ObjectPool(int maxSize = 100)
    {
        _maxSize = maxSize;
    }

    public T Get()
    {
        if (_pool.TryTake(out var item))
            return item;
        return new T();
    }

    public void Return(T item)
    {
        if (_pool.Count < _maxSize)
            _pool.Add(item);
    }
}

// 使用
public class ParticleSystem : MonoBehaviour
{
    private ObjectPool<ParticleData> particlePool = new(1000);

    public void SpawnParticle()
    {
        ParticleData particle = particlePool.Get();
        // 使用...
    }

    public void RecycleParticle(ParticleData particle)
    {
        particle.Reset();  // 重置状态
        particlePool.Return(particle);
    }
}
```

## 2. 缓存友好编程

### 2.1 数据布局优化

```csharp
// ❌ AoS (Array of Structures) - 缓存不友好
struct ParticleAoS
{
    public Vector3 position;
    public Vector3 velocity;
    public Color color;
    public float life;
}

ParticleAoS[] particlesAoS;

// 访问 position 时，会加载整个 struct 到 cache line
// 包含不需要的 velocity、color、life

// ✅ SoA (Structure of Arrays) - 缓存友好
class ParticleSoA
{
    public Vector3[] positions;
    public Vector3[] velocities;
    public Color[] colors;
    public float[] lifes;
}

// 只加载需要的数据
void UpdatePositions(ParticleSoA particles, float dt)
{
    for (int i = 0; i < particles.positions.Length; i++)
    {
        particles.positions[i] += particles.velocities[i] * dt;
        // 只访问 positions 和 velocities，cache 命中率高
    }
}
```

### 2.2 内存对齐

```csharp
using System.Runtime.InteropServices;

// ❌ 未对齐
struct BadAlignment
{
    public byte a;     // 1 byte
    // 3 bytes padding
    public int b;      // 4 bytes
    public byte c;     // 1 byte
    // 3 bytes padding
    public int d;      // 4 bytes
}
// 总大小: 16 bytes

// ✅ 对齐优化
struct GoodAlignment
{
    public int b;      // 4 bytes
    public int d;      // 4 bytes
    public byte a;     // 1 byte
    public byte c;     // 1 byte
    // 2 bytes padding
}
// 总大小: 12 bytes
```

## 3. SIMD 向量化

### 3.1 使用 System.Numerics

```csharp
using System.Numerics;
using System.Runtime.CompilerServices;

// ❌ 标量计算
void AddScalar(float[] a, float[] b, float[] result, int count)
{
    for (int i = 0; i < count; i++)
    {
        result[i] = a[i] + b[i];
    }
}

// ✅ SIMD 向量化
void AddSimd(float[] a, float[] b, float[] result, int count)
{
    int vectorSize = Vector<float>.Count;  // 4 (32位) 或 8 (64位)
    int i = 0;

    // SIMD 处理
    for (; i <= count - vectorSize; i += vectorSize)
    {
        var va = new Vector<float>(a, i);
        var vb = new Vector<float>(b, i);
        (va + vb).CopyTo(result, i);
    }

    // 处理剩余元素
    for (; i < count; i++)
    {
        result[i] = a[i] + b[i];
    }
}
```

### 3.2 Unity Mathematics (Burst 优化)

```csharp
using Unity.Mathematics;
using Unity.Burst;
using Unity.Jobs;
using Unity.Collections;

[BurstCompile]
public struct SimdJob : IJobParallelFor
{
    [ReadOnly] public NativeArray<float3> positions;
    [ReadOnly] public NativeArray<float3> velocities;
    public NativeArray<float3> results;
    public float deltaTime;

    public void Execute(int index)
    {
        // float3 自动使用 SIMD
        results[index] = positions[index] + velocities[index] * deltaTime;
    }
}

// 对比
// 标量: 100万次 ~10ms
// SIMD: 100万次 ~1ms (10x 提升)
```

## 4. 字符串优化

### 4.1 避免字符串拼接

```csharp
// ❌ 多次分配
string message = "Player ";
message += playerName;
message += " scored ";
message += score;
message += " points!";

// ✅ StringBuilder
var sb = new StringBuilder(100);
sb.Append("Player ");
sb.Append(playerName);

// ✅ 字符串插值（编译时优化）

Debug.Log($"Player {playerName} scored {score}");
```

### 4.2 字符串比较

```csharp

if (input == "yes") { }

if (string.Equals(input, "yes", StringComparison.OrdinalIgnoreCase)) { }

// ✅ 对于固定值，使用 switch（编译器优化）

{
    case "yes":
```

## 5. 集合优化

### 5.1 选择正确的集合

| 场景 | 推荐集合 | 原因 |
|------|----------|------|
| 频繁遍历 | Array / List<T> | 连续内存 |
| 频繁查找 | Dictionary<K,V> | O(1) 查找 |
| 频繁插入/删除头部 | LinkedList<T> | O(1) 操作 |
| 先进先出 | Queue<T> | 语义清晰 |
| 后进先出 | Stack<T> | 语义清晰 |
| 去重 | HashSet<T> | O(1) 查找 |

### 5.2 List<T> 优化

```csharp
// ❌ 未预分配
var list = new List<int>();
for (int i = 0; i < 10000; i++)
{
    list.Add(i);  // 多次扩容
}

// ✅ 预分配容量
var list = new List<int>(10000);
for (int i = 0; i < 10000; i++)
{
    list.Add(i);  // 无扩容
}

// ✅ 避免频繁 Clear
list.Clear();  // 保留容量
// 而不是
list = new List<int>();  // 重新分配
```

### 5.3 Dictionary 优化

```csharp
// ❌ 重复 TryGetValue
if (dict.TryGetValue(key, out var value))
{
    dict[key] = value + 1;  // 再次查找
}

// ✅ 单次查找
if (dict.TryGetValue(key, out var value))
{
    dict[key] = value + 1;  // 使用索引器仍然会查找
}

// ✅ 使用 CollectionsMarshal (.NET 5+)
ref int valueRef = ref System.Runtime.InteropServices.CollectionsMarshal.GetValueRefOrAddDefault(dict, key, out bool exists);
if (exists)
{
    valueRef++;
}
```

### 6.2 空间分区

```csharp
// ❌ O(n²) 碰撞检测
void CheckCollisionsBruteForce(List<Enemy> enemies)
{
    for (int i = 0; i < enemies.Count; i++)
    {
        for (int j = i + 1; j < enemies.Count; j++)
        {
            if (Distance(enemies[i], enemies[j]) < collisionRange)
            {
                HandleCollision(enemies[i], enemies[j]);
            }
        }
    }
}

// ✅ 网格分区 O(n)
class SpatialGrid
{
    private Dictionary<Vector2Int, List<Enemy>> cells = new();
    private float cellSize = 10f;

    public void AddEnemy(Enemy enemy)
    {
        Vector2Int cell = GetCell(enemy.position);
        if (!cells.ContainsKey(cell))
            cells[cell] = new List<Enemy>();
        cells[cell].Add(enemy);
    }

    public List<Enemy> GetNearbyEnemies(Vector3 position)
    {
        var result = new List<Enemy>();
        Vector2Int centerCell = GetCell(position);

        // 只检查周围 9 个格子
        for (int x = -1; x <= 1; x++)
        {
            for (int y = -1; y <= 1; y++)
            {
                Vector2Int cell = centerCell + new Vector2Int(x, y);
                if (cells.TryGetValue(cell, out var enemies))
                {
                    result.AddRange(enemies);
                }
            }
        }
        return result;
    }

    private Vector2Int GetCell(Vector3 position)
    {
        return new Vector2Int(
            Mathf.FloorToInt(position.x / cellSize),
            Mathf.FloorToInt(position.z / cellSize)
        );
    }
}
```

## 7. Profiling 与测量

### 7.1 高精度计时

```csharp
using System.Diagnostics;

public class Benchmark
{
    public static void Measure(string name, Action action, int iterations = 1000)
    {
        // 预热
        action();

        // 测量
        var sw = Stopwatch.StartNew();
        for (int i = 0; i < iterations; i++)
        {
            action();
        }
        sw.Stop();

        double avgMs = sw.Elapsed.TotalMilliseconds / iterations;
        UnityEngine.Debug.Log($"{name}: {avgMs:F4} ms/iter");
    }
}

// 使用
Benchmark.Measure("List Add", () =>
{
    var list = new List<int>(1000);
    for (int i = 0; i < 1000; i++)
        list.Add(i);
});
```

### 7.2 内存测量

```csharp
public class MemoryProfiler
{
    public static long GetUsedMemory()
    {
        return GC.GetTotalMemory(forceFullCollection: false);
    }

    public static void MeasureAllocation(string name, Action action)
    {
        long before = GetUsedMemory();
        action();
        long after = GetUsedMemory();

        long allocated = after - before;
        UnityEngine.Debug.Log($"{name}: {allocated / 1024} KB allocated");
    }
}
```

### 优化优先级

| 优先级 | 优化方向 | 效果 |
|--------|----------|------|
| 1 | 算法优化 | O(n²) → O(n) |
| 2 | 内存分配减少 | GC 暂停消除 |
| 3 | 缓存友好布局 | 2-10x |
| 4 | SIMD 向量化 | 4-8x |

### 性能对比

| 优化 | 原始 | 优化后 | 提升 |
|------|------|--------|------|
| StringBuilder vs + | 100ms | 1ms | 100x |
| Span vs Substring | 50ms | 5ms | 10x |
| SoA vs AoS | 20ms | 2ms | 10x |
| SIMD vs 标量 | 10ms | 1ms | 10x |
