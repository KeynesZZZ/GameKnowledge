> 第6课 | 脚本与架构模块（完结）

## 1. 为什么需要良好的架构？

### 没有架构的问题

```
❌ 代码耦合严重，改一处bug多处
❌ 无法复用，每个项目重写
❌ 难以测试，依赖关系混乱
❌ 团队协作困难，代码风格不一
❌ 维护成本高，新人看不懂
```

### 良好架构的好处

```
✅ 高内聚低耦合
✅ 代码可复用
✅ 易于单元测试
✅ 团队协作顺畅
✅ 维护成本低
```

## 2. 分层架构原理

**核心思想**：将代码按职责分层，上层依赖下层，下层不知道上层

```
┌─────────────────────────────────────────────┐
│              表现层 (Presentation)           │
│         UI、动画、音效、输入处理              │
├─────────────────────────────────────────────┤
│              逻辑层 (Game Logic)             │
│         游戏规则、状态机、系统交互            │
├─────────────────────────────────────────────┤
│              数据层 (Data)                   │
│         配置、存档、网络数据                  │
├─────────────────────────────────────────────┤
│              基础设施层 (Core/Infrastructure) │
│         工具类、对象池、事件总线              │
└─────────────────────────────────────────────┘
              ↑ 依赖方向（从上到下）
```

## 3. MVC/MVP/MVVM 在Unity中的应用

### MVC模式

```
┌─────────────────────────────────────────────┐
│                    MVC                       │
│                                             │
│   ┌─────────┐      ┌─────────┐            │
│   │  View   │←────→│Controller│            │
│   │  (视图)  │      │ (控制器) │            │
│   └────┬────┘      └────┬────┘            │
│        │                │                  │
│        │    ┌─────────┐ │                  │
│        └───→│  Model  │←─┘                │
│             │  (模型)  │                    │
│             └─────────┘                    │
└─────────────────────────────────────────────┘

Model：数据和业务逻辑
View：UI显示
Controller：协调Model和View
```

```csharp
// ============ Model ============
public class ScoreModel
{
    public int CurrentScore { get; private set; }
    public int HighScore { get; private set; }

    public event Action<int> OnScoreChanged;

    public void AddScore(int amount)
    {
        CurrentScore += amount;
        if (CurrentScore > HighScore)
        {
            HighScore = CurrentScore;
        }
        OnScoreChanged?.Invoke(CurrentScore);
    }

    public void Reset()
    {
        CurrentScore = 0;
        OnScoreChanged?.Invoke(CurrentScore);
    }
}

// ============ View ============
public class ScoreView : MonoBehaviour
{
    [SerializeField] private Text scoreText;
    [SerializeField] private Text highScoreText;

    public void UpdateScore(int score)
    {
        scoreText.text = $"Score: {score}";
    }

    public void UpdateHighScore(int highScore)
    {
        highScoreText.text = $"Best: {highScore}";
    }

    public void PlayScoreAnimation()
    {
        // 播放得分动画
    }
}

// ============ Controller ============
public class ScoreController : MonoBehaviour
{
    private ScoreModel model;
    private ScoreView view;

    private void Awake()
    {
        model = new ScoreModel();
        view = GetComponent<ScoreView>();

        // 绑定Model到View
        model.OnScoreChanged += OnScoreChanged;
    }

    private void Start()
    {
        // 初始化View
        view.UpdateScore(model.CurrentScore);
        view.UpdateHighScore(model.HighScore);
    }

    private void OnScoreChanged(int score)
    {
        view.UpdateScore(score);
        view.UpdateHighScore(model.HighScore);
        view.PlayScoreAnimation();
    }

    // 外部调用
    public void AddScore(int amount)
    {
        model.AddScore(amount);
    }

    public void ResetScore()
    {
        model.Reset();
    }
}
```

### MVP模式

```
┌─────────────────────────────────────────────┐
│                    MVP                       │
│                                             │
│   ┌─────────┐      ┌─────────┐            │
│   │  View   │←────→│Presenter│            │
│   │  (视图)  │      │(展示器) │            │
│   └─────────┘      └────┬────┘            │
│        ↑                │                  │
│        │    ┌─────────┐ │                  │
│        └───│  Model  │←─┘                 │
│            │  (模型)  │                    │
│            └─────────┘                    │
└─────────────────────────────────────────────┘

View：被动视图，只负责显示和转发用户操作
Presenter：处理View的事件，更新Model，更新View
Model：纯数据和业务逻辑

关键区别：View不知道Model的存在
```

```csharp
// ============ Model ============
public class GameModel
{
    public int Score { get; private set; }
    public int Combo { get; private set; }
    public bool IsPlaying { get; private set; }

    public event Action<ModelChangedEventArgs> OnModelChanged;

    public void StartGame()
    {
        Score = 0;
        Combo = 0;
        IsPlaying = true;
        NotifyChanged(nameof(IsPlaying));
    }

    public void AddMatchScore(int matchCount)
    {
        Combo++;
        int points = matchCount * 100 * Combo;
        Score += points;
        NotifyChanged(nameof(Score), nameof(Combo));
    }

    public void ResetCombo()
    {
        Combo = 0;
        NotifyChanged(nameof(Combo));
    }

    public void EndGame()
    {
        IsPlaying = false;
        NotifyChanged(nameof(IsPlaying));
    }

    private void NotifyChanged(params string[] properties)
    {
        OnModelChanged?.Invoke(new ModelChangedEventArgs(properties));
    }
}

public class ModelChangedEventArgs
{
    public string[] ChangedProperties { get; }

    public ModelChangedEventArgs(params string[] properties)
    {
        ChangedProperties = properties;
    }

    public bool HasChanged(string property) =>
        System.Array.IndexOf(ChangedProperties, property) >= 0;
}

// ============ View Interface ============
public interface IGameView
{
    void ShowScore(int score);
    void ShowCombo(int combo);
    void ShowGameStart();
    void ShowGameOver(int finalScore);
    event Action OnStartClicked;
    event Action<Vector2Int> OnGemSelected;
}

// ============ View ============
public class GameView : MonoBehaviour, IGameView
{
    [SerializeField] private Text scoreText;
    [SerializeField] private Text comboText;
    [SerializeField] private GameObject gameOverPanel;
    [SerializeField] private Button startButton;

    public event Action OnStartClicked;
    public event Action<Vector2Int> OnGemSelected;

    private void Awake()
    {
        startButton.onClick.AddListener(() => OnStartClicked?.Invoke());
    }

    public void ShowScore(int score)
    {
        scoreText.text = $"Score: {score}";
    }

    public void ShowCombo(int combo)
    {
        comboText.text = combo > 1 ? $"x{combo} Combo!" : "";
    }

    public void ShowGameStart()
    {
        gameOverPanel.SetActive(false);
    }

    public void ShowGameOver(int finalScore)
    {
        gameOverPanel.SetActive(true);
    }

    private void Update()
    {
        if (Input.GetMouseButtonDown(0))
        {
            var gridPos = ConvertToGridPosition(Input.mousePosition);
            OnGemSelected?.Invoke(gridPos);
        }
    }

    private Vector2Int ConvertToGridPosition(Vector3 mousePosition)
    {
        // 实现省略
        return Vector2Int.zero;
    }
}

// ============ Presenter ============
public class GamePresenter : MonoBehaviour
{
    private IGameView view;
    private GameModel model;

    private void Awake()
    {
        view = GetComponent<IGameView>();
        model = new GameModel();

        view.OnStartClicked += OnStartClicked;
        view.OnGemSelected += OnGemSelected;
        model.OnModelChanged += OnModelChanged;
    }

    private void OnDestroy()
    {
        view.OnStartClicked -= OnStartClicked;
        view.OnGemSelected -= OnGemSelected;
        model.OnModelChanged -= OnModelChanged;
    }

    private void OnStartClicked()
    {
        model.StartGame();
        view.ShowGameStart();
    }

    private void OnGemSelected(Vector2Int position)
    {
        if (!model.IsPlaying) return;
        // 处理棋子选择逻辑
        model.AddMatchScore(3);
    }

    private void OnModelChanged(ModelChangedEventArgs e)
    {
        if (e.HasChanged(nameof(GameModel.Score)))
            view.ShowScore(model.Score);

        if (e.HasChanged(nameof(GameModel.Combo)))
            view.ShowCombo(model.Combo);

        if (e.HasChanged(nameof(GameModel.IsPlaying)) && !model.IsPlaying)
            view.ShowGameOver(model.Score);
    }
}
```

## 4. 依赖注入基础

### 问题：硬编码的依赖关系

```csharp
// ❌ 硬依赖，难以测试和替换
public class Player : MonoBehaviour
{
    private AudioManager audioManager;

    private void Start()
    {
        audioManager = AudioManager.Instance;  // 直接引用单例
    }

    public void Attack()
    {
        audioManager.PlaySFX("attack");
    }
}
```

### 解决方案：依赖注入

```csharp
// ✅ 依赖注入，松耦合
public class Player : MonoBehaviour
{
    private IAudioManager audioManager;

    // 构造函数注入
    public void Construct(IAudioManager audioManager)
    {
        this.audioManager = audioManager;
    }

    // 属性注入
    [Inject]
    public IAudioManager AudioManager { set => audioManager = value; }

    // 方法注入
    public void SetAudioManager(IAudioManager audioManager)
    {
        this.audioManager = audioManager;
    }

    public void Attack()
    {
        audioManager.PlaySFX("attack");
    }
}
```

### 简单的依赖注入容器

```csharp
/// <summary>
/// 简单的DI容器
/// </summary>
public class DIContainer
{
    private readonly Dictionary<Type, object> instances = new();
    private readonly Dictionary<Type, Func<object>> factories = new();

    public void RegisterSingleton<TInterface, TImplementation>()
        where TImplementation : TInterface, new()
    {
        factories[typeof(TInterface)] = () =>
        {
            if (!instances.TryGetValue(typeof(TInterface), out var instance))
            {
                instance = new TImplementation();
                instances[typeof(TInterface)] = instance;
            }
            return instance;
        };
    }

    public void RegisterInstance<TInterface>(TInterface instance)
    {
        instances[typeof(TInterface)] = instance;
    }

    public void RegisterFactory<TInterface>(Func<TInterface> factory)
    {
        factories[typeof(TInterface)] = () => factory();
    }

    public T Resolve<T>()
    {
        var type = typeof(T);

        if (instances.TryGetValue(type, out var instance))
            return (T)instance;

        if (factories.TryGetValue(type, out var factory))
            return (T)factory();

        throw new Exception($"Dependency not registered: {type.Name}");
    }

    public bool TryResolve<T>(out T dependency)
    {
        try
        {
            dependency = Resolve<T>();
            return true;
        }
        catch
        {
            dependency = default;
            return false;
        }
    }
}

// 使用示例
public class GameInstaller : MonoBehaviour
{
    [SerializeField] private GameConfig gameConfig;

    private void Awake()
    {
        var container = new DIContainer();

        container.RegisterInstance<IGameConfig>(gameConfig);
        container.RegisterSingleton<IAudioManager, AudioManager>();
        container.RegisterSingleton<ISaveSystem, SaveSystem>();
        container.RegisterSingleton<ICombatSystem, CombatSystem>();

        GameDependencies.Container = container;
    }
}

public static class GameDependencies
{
    public static DIContainer Container { get; set; }
    public static T Get<T>() => Container.Resolve<T>();
}

// 在代码中使用
public class Player : MonoBehaviour
{
    private IAudioManager audioManager;
    private ICombatSystem combatSystem;

    private void Start()
    {
        audioManager = GameDependencies.Get<IAudioManager>();
        combatSystem = GameDependencies.Get<ICombatSystem>();
    }
}
```

```csharp
public class UIManager : MonoSingleton<UIManager>
{
    [SerializeField] private Transform uiRoot;
    private Dictionary<Type, UIBase> uiCache = new();

    public T Open<T>() where T : UIBase
    {
        var type = typeof(T);
        if (!uiCache.TryGetValue(type, out var ui))
        {
            var prefab = Resources.Load<T>($"UI/{type.Name}");
            ui = Instantiate(prefab, uiRoot);
            uiCache[type] = ui;
        }
        ui.Show();
        return ui as T;
    }

    public void Close<T>() where T : UIBase
    {
        if (uiCache.TryGetValue(typeof(T), out var ui))
            ui.Hide();
    }
}

public abstract class UIBase : MonoBehaviour
{
    public virtual void Show() => gameObject.SetActive(true);
    public virtual void Hide() => gameObject.SetActive(false);
}

// ============ Game Entry Point ============

public class GameEntryPoint : MonoBehaviour

        Services.Register<ICombatService>(new CombatService());
        Services.Register<ISaveSystem>(new SaveSystem());
        Services.Register<IAudioService>(new AudioService());
        Services.Get<ISaveSystem>().Load();

    private void InitializeGame()
    {
        var boardService = Services.Get<IBoardService>();
        boardService.Initialize(config.boardWidth, config.boardHeight);

        var combatService = Services.Get<ICombatService>();
        combatService.Initialize(heroes);

        var boardView = FindObjectOfType<BoardView>();
        boardView.Initialize(boardService);
    }
}
```

## 本课小结

### 核心知识点

| 知识点 | 核心要点 |
|--------|----------|
| 分层架构 | 表现层→逻辑层→数据层→基础设施层 |
| MVC | Model-View-Controller，经典模式 |
| MVP | Model-View-Presenter，View被动 |
| MVVM | Model-View-ViewModel，数据绑定 |
| 依赖注入 | 控制反转，松耦合 |

### 架构选择指南

| 项目规模 | 推荐架构 |
|----------|----------|
| 小型原型 | 简单分层 |
| 中型项目 | MVC/MVP |
| 大型项目 | MVP + 依赖注入 |
| 团队协作 | MVP + DI + 严格分层 |
