## 2. Memory Profiler工具

### 2.1 安装与使用

```
安装步骤：
1. Window > Package Manager
2. 选择 Unity Registry
3. 搜索 "Memory Profiler"
4. 点击 Install

使用方法：
1. Window > Analysis > Memory Profiler
2. 点击 "Capture" 捕获内存快照
3. 对比不同时间点的快照
```

### 2.2 内存快照分析

```csharp
using UnityEngine;
using Unity.Collections;
using Unity.Collections.LowLevel.Unsafe;

/// <summary>
/// 内存分析辅助工具
/// </summary>
public static class MemoryAnalyzer
{
    /// <summary>
    /// 获取对象托管大小（近似值）
    /// </summary>
    public static long GetManagedSize(object obj)
    {
        if (obj == null) return 0;

        // 使用Unsafe获取大小
        return UnsafeUtility.SizeOf(obj.GetType());
    }

    /// <summary>
    /// 分析纹理内存
    /// </summary>
    public static void AnalyzeTextures()
    {
        var textures = Resources.FindObjectsOfTypeAll<Texture>();
        long totalMemory = 0;

        foreach (var tex in textures)
        {
            long memory = tex.width * tex.height *
                         (tex.graphicsFormat.ToString().Contains("RGBA") ? 4 : 2);
            totalMemory += memory;

            Debug.Log($"Texture: {tex.name}, " +
                     $"Size: {tex.width}x{tex.height}, " +
                     $"Memory: {memory / 1024}KB");
        }

        Debug.Log($"Total Texture Memory: {totalMemory / 1024 / 1024}MB");
    }
}
```

### 3.3 UI渲染调试

```csharp
using UnityEngine;
using UnityEngine.UI;

/// <summary>
/// UI性能分析器
/// </summary>
public static class UIPerformanceAnalyzer
{
    /// <summary>
    /// 分析Canvas性能
    /// </summary>
    public static void AnalyzeCanvas(Canvas canvas)
    {
        var graphicComponents = canvas.GetComponentsInChildren<Graphic>(true);
        int totalVertices = 0;
        int totalDrawCalls = 0;

        foreach (var graphic in graphicComponents)
        {
            if (graphic.isActiveAndEnabled)
            {
                totalVertices += graphic.GetComponent<CanvasRenderer>()?.vertexCount ?? 0;
            }
        }

        // 获取Canvas的DrawCall数量
        var canvasRenderer = canvas.GetComponent<CanvasRenderer>();
        if (canvasRenderer != null)
        {
            // 注意：这是近似值
            totalDrawCalls = canvas.transform.childCount;
        }

        Debug.Log($"Canvas: {canvas.name}");
        Debug.Log($"  Active Graphics: {graphicComponents.Length}");
        Debug.Log($"  Total Vertices: {totalVertices}");
        Debug.Log($"  Estimated DrawCalls: {totalDrawCalls}");
    }

    /// <summary>
    /// 检测UI重叠问题
    /// </summary>
    public static void CheckOverdraw(Canvas canvas)
    {
        var graphics = canvas.GetComponentsInChildren<Graphic>(true);

        foreach (var graphic in graphics)
        {
            if (!graphic.raycastTarget && graphic.color.a < 0.01f)
            {
                Debug.LogWarning($"可能的不可见UI: {graphic.name}", graphic);
            }
        }
    }
}
```

## 4. 性能测试自动化

### 4.1 性能测试框架

```csharp
using UnityEngine;
using System.Collections;
using System.Diagnostics;

/// <summary>
/// 性能测试运行器
/// </summary>
public class PerformanceTestRunner : MonoBehaviour
{
    [Header("Test Settings")]
    [SerializeField] private int warmupFrames = 60;
    [SerializeField] private int testFrames = 300;
    [SerializeField] private float targetFrameTime = 16.67f; // 60 FPS

    private PerformanceTestResult result;

    public void RunTest(string testName, System.Action testAction)
    {
        StartCoroutine(RunTestCoroutine(testName, testAction));
    }

    private IEnumerator RunTestCoroutine(string testName, System.Action testAction)
    {
        result = new PerformanceTestResult
        {
            TestName = testName,
            FrameTimes = new float[testFrames]
        };

        // 预热
        for (int i = 0; i < warmupFrames; i++)
        {
            testAction?.Invoke();
            yield return null;
        }

        // 强制GC
        System.GC.Collect();
        yield return null;

        // 正式测试
        var stopwatch = Stopwatch.StartNew();

        for (int i = 0; i < testFrames; i++)
        {
            float frameStart = Time.realtimeSinceStartup;

            testAction?.Invoke();
            yield return null;

            result.FrameTimes[i] = (Time.realtimeSinceStartup - frameStart) * 1000f;
        }

        stopwatch.Stop();
        CalculateResults();
        PrintResults();
    }

    private void CalculateResults()
    {
        float sum = 0;
        float max = 0;
        float min = float.MaxValue;

        foreach (var time in result.FrameTimes)
        {
            sum += time;
            if (time > max) max = time;
            if (time < min) min = time;
        }

        result.AverageFrameTime = sum / result.FrameTimes.Length;
        result.MaxFrameTime = max;
        result.MinFrameTime = min;

        // 计算标准差
        float variance = 0;
        foreach (var time in result.FrameTimes)
        {
            variance += Mathf.Pow(time - result.AverageFrameTime, 2);
        }
        result.StandardDeviation = Mathf.Sqrt(variance / result.FrameTimes.Length);
    }

    private void PrintResults()
    {
        UnityEngine.Debug.Log($@"
========== Performance Test Result ==========
Test: {result.TestName}
Average Frame Time: {result.AverageFrameTime:F2}ms
Min Frame Time: {result.MinFrameTime:F2}ms
Max Frame Time: {result.MaxFrameTime:F2}ms
Standard Deviation: {result.StandardDeviation:F2}ms
Target: {targetFrameTime:F2}ms
Status: {(result.AverageFrameTime <= targetFrameTime ? "PASS" : "FAIL")}
=============================================");
    }
}

public class PerformanceTestResult
{
    public string TestName;
    public float[] FrameTimes;
    public float AverageFrameTime;
    public float MinFrameTime;
    public float MaxFrameTime;
    public float StandardDeviation;
}
```

### 4.2 三消游戏性能基准测试

```csharp
using UnityEngine;

/// <summary>
/// 三消游戏性能测试用例
/// </summary>
public class Match3PerformanceTests : MonoBehaviour
{
    private PerformanceTestRunner testRunner;
    private BoardController boardController;

    private void Start()
    {
        testRunner = GetComponent<PerformanceTestRunner>();
        boardController = FindObjectOfType<BoardController>();

        // 运行测试
        RunAllTests();
    }

    private void RunAllTests()
    {
        // 测试1：棋盘生成
        testRunner.RunTest("Board_Generation", () =>
        {
            boardController.GenerateBoard();
        });

        // 测试2：匹配检测
        testRunner.RunTest("Match_Detection", () =>
        {
            boardController.DetectMatches();
        });

        // 测试3：消除动画
        testRunner.RunTest("Match_Animation", () =>
        {
            boardController.PlayMatchAnimation();
        });

        // 测试4：棋子下落
        testRunner.RunTest("Tile_Falling", () =>
        {
            boardController.ApplyGravity();
        });
    }
}
```

## 5. 性能监控面板

### 5.1 运行时性能监控

```csharp
using UnityEngine;
using UnityEngine.UI;

/// <summary>
/// 运行时性能监控面板
/// </summary>
public class PerformanceMonitor : MonoBehaviour
{
    [Header("UI References")]
    [SerializeField] private Text fpsText;
    [SerializeField] private Text memoryText;
    [SerializeField] private Text drawCallText;

    [Header("Settings")]
    [SerializeField] private float updateInterval = 0.5f;
    [SerializeField] private bool showInRelease = false;

    private float accum;
    private int frames;
    private float timeLeft;
    private float currentFps;

    private void Start()
    {
#if !UNITY_EDITOR
        if (!showInRelease)
        {
            gameObject.SetActive(false);
            return;
        }
#endif
        timeLeft = updateInterval;
    }

    private void Update()
    {
        UpdateFPS();
        UpdateMemory();
        UpdateDrawCalls();
    }

    private void UpdateFPS()
    {
        timeLeft -= Time.deltaTime;
        accum += Time.timeScale / Time.deltaTime;
        frames++;

        if (timeLeft <= 0f)
        {
            currentFps = accum / frames;
            fpsText.text = $"FPS: {currentFps:F0}";

            // 颜色指示
            if (currentFps >= 55)
                fpsText.color = Color.green;
            else if (currentFps >= 30)
                fpsText.color = Color.yellow;
            else
                fpsText.color = Color.red;

            timeLeft = updateInterval;
            accum = 0f;
            frames = 0;
        }
    }

    private void UpdateMemory()
    {
        long totalMemory = System.GC.GetTotalMemory(false) / 1024 / 1024;
        long usedHeap = UnityEngine.Profiling.Profiler.GetTotalAllocatedMemoryLong() / 1024 / 1024;

        memoryText.text = $"Memory: {usedHeap}MB / {totalMemory}MB";
    }

    private void UpdateDrawCalls()
    {
        // Unity 2020+ 可用
        int drawCalls = UnityEngine.Rendering.GraphicsSettings.currentRenderPipeline ??
            UnityEngine.Graphics.activeTier != GraphicsTier.Tier1 ? 0 : 0;

        // 使用Unity Stats（需要相机）
        var camera = Camera.main;
        if (camera != null)
        {
            // 注意：这是近似值
            drawCallText.text = $"Batches: {UnityStats.batches}";
        }
    }
}

/// <summary>
/// Unity统计信息（Editor Only）
/// </summary>
public static class UnityStats
{
    public static int batches => UnityEngine.Profiling.Profiler.GetStat(
        UnityEngine.Profiling.ProfilerCategory.Render, "Batches Count");
}
```

## 6. 性能优化工作流

### 6.1 优化流程图

```
┌─────────────────────────────────────────────────────────────┐
│                    性能优化工作流                            │
│                                                             │
│   1. 确定目标                                               │
│   └── 设定帧率、内存、包体目标                              │
│                                                             │
│   2. 建立基准                                               │
│   ├── 使用Profiler记录当前性能                              │
│   └── 记录各模块耗时占比                                    │
│                                                             │
│   3. 识别瓶颈                                               │
│   ├── CPU > 16ms?  → 优化脚本/物理                         │
│   ├── GPU > 16ms?  → 优化渲染/Shader                       │
│   └── 内存增长?     → 检查泄漏                             │
│                                                             │
│   4. 针对优化                                               │
│   ├── 只优化最大的瓶颈                                      │
│   └── 一次只改一个变量                                      │
│                                                             │
│   5. 验证效果                                               │
│   ├── 对比优化前后数据                                      │
│   └── 确保无副作用                                          │
│                                                             │
│   6. 持续监控                                               │
│   └── 建立性能回归测试                                      │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

## 本课小结

### 核心知识点

| 工具 | 用途 | 关键指标 |
|------|------|----------|
| CPU Profiler | CPU性能分析 | 帧时间、GC分配 |
| Memory Profiler | 内存分析 | 内存快照、泄漏检测 |
| Frame Debugger | 渲染调试 | DrawCall、合批分析 |
| 自定义监控 | 运行时监控 | FPS、内存实时显示 |

### 性能分析原则

```
1. 先测量，后优化
2. 优化最大瓶颈
3. 设定性能预算
4. 持续监控回归
```
