## 2. 棋子Shader核心代码

### 完整的棋子Shader

```hlsl
Shader "Match3/Complete/GemShader"
{
    Properties
    {
        _BaseMap ("Base Map", 2D) = "white" {}
        _BaseColor ("Base Color", Color) = (1, 1, 1, 1)

        // 效果参数
        _Highlight ("Highlight", Range(0, 1)) = 0
        _HighlightColor ("Highlight Color", Color) = (1, 1, 0.5, 1)
        _Selected ("Selected", Float) = 0
        _SelectedPulse ("Selected Pulse", Float) = 0
        _Dissolve ("Dissolve", Range(0, 1)) = 0
        _DissolveEdge ("Dissolve Edge", Range(0, 0.5)) = 0.1
        _DissolveColor ("Dissolve Color", Color) = (1, 0.5, 0, 1)
    }

    SubShader
    {
        Tags
        {
            "RenderType" = "Transparent"
            "Queue" = "Transparent"
            "RenderPipeline" = "UniversalPipeline"
        }

        Blend SrcAlpha OneMinusSrcAlpha
        ZWrite Off

        Pass
        {
            HLSLPROGRAM
            #pragma vertex Vertex
            #pragma fragment Fragment

            #include "Packages/com.unity.render-pipelines.universal/ShaderLibrary/Core.hlsl"

            struct Attributes
            {
                float4 positionOS : POSITION;
                float2 uv : TEXCOORD0;
                float4 color : COLOR;
            };

            struct Varyings
            {
                float4 positionCS : SV_POSITION;
                float2 uv : TEXCOORD0;
                float4 vertexColor : TEXCOORD1;
            };

            TEXTURE2D(_BaseMap);
            SAMPLER(sampler_BaseMap);

            CBUFFER_START(UnityPerMaterial)
                float4 _BaseMap_ST;
                half4 _BaseColor;
                half _Highlight;
                half4 _HighlightColor;
                half _Selected;
                half _SelectedPulse;
                half _Dissolve;
                half _DissolveEdge;
                half4 _DissolveColor;
            CBUFFER_END

            Varyings Vertex(Attributes input)
            {
                Varyings output;
                output.positionCS = TransformObjectToHClip(input.positionOS.xyz);
                output.uv = TRANSFORM_TEX(input.uv, _BaseMap);
                output.vertexColor = input.color;

                // 选中缩放动画
                if (_Selected > 0.5)
                {
                    float scale = 1.0 + sin(_SelectedPulse * 6.28) * 0.1;
                    float3 center = float3(0, 0, 0);
                    float3 localPos = (input.positionOS.xyz - center) * scale + center;
                    output.positionCS = TransformObjectToHClip(localPos);
                }

                return output;
            }

            half4 Fragment(Varyings input) : SV_Target
            {
                half4 baseColor = SAMPLE_TEXTURE2D(_BaseMap, sampler_BaseMap, input.uv);
                baseColor *= _BaseColor * input.vertexColor;

                // 高亮效果
                half3 highlight = _HighlightColor.rgb * _Highlight * 0.5;
                baseColor.rgb += highlight;

                // 选中边缘发光
                if (_Selected > 0.5)
                {
                    float dist = distance(input.uv, float2(0.5, 0.5));
                    float edge = smoothstep(0.3, 0.5, dist);
                    float pulse = sin(_SelectedPulse * 6.28) * 0.5 + 0.5;
                    baseColor.rgb += _HighlightColor.rgb * edge * 0.8 * pulse;
                }

                // 溶解效果
                if (_Dissolve > 0.001)
                {
                    half mask = baseColor.a;
                    half edgeFactor = smoothstep(_Dissolve - _DissolveEdge, _Dissolve, mask);
                    half3 edgeColor = _DissolveColor.rgb * (1 - edgeFactor) * 3;
                    baseColor.rgb = lerp(edgeColor, baseColor.rgb, edgeFactor);
                    half alpha = step(_Dissolve - _DissolveEdge * 0.5, mask);
                    baseColor.a *= alpha;
                }

                return baseColor;
            }
            ENDHLSL
        }
    }
}
```

## 3. 棋子材质控制器

```csharp
using UnityEngine;

/// <summary>
/// 棋子材质控制器 - 控制Shader参数
/// </summary>
[RequireComponent(typeof(Renderer))]
public class GemMaterialController : MonoBehaviour
{
    [SerializeField] private Renderer gemRenderer;
    private MaterialPropertyBlock propertyBlock;

    [Header("Effect Settings")]
    [SerializeField] private float highlightTransitionSpeed = 5f;
    [SerializeField] private float dissolveSpeed = 2f;
    [SerializeField] private float selectedPulseSpeed = 3f;

    // 属性ID缓存
    private static readonly int HighlightID = Shader.PropertyToID("_Highlight");
    private static readonly int SelectedID = Shader.PropertyToID("_Selected");
    private static readonly int SelectedPulseID = Shader.PropertyToID("_SelectedPulse");
    private static readonly int DissolveID = Shader.PropertyToID("_Dissolve");

    private float currentHighlight = 0f;
    private float targetHighlight = 0f;
    private float currentDissolve = 0f;
    private float selectedPulse = 0f;
    private bool isSelected = false;
    private bool isDissolving = false;

    private void Awake()
    {
        if (gemRenderer == null) gemRenderer = GetComponent<Renderer>();
        propertyBlock = new MaterialPropertyBlock();
    }

    private void Update()
    {
        currentHighlight = Mathf.Lerp(currentHighlight, targetHighlight,
            Time.deltaTime * highlightTransitionSpeed);

        if (isSelected) selectedPulse += Time.deltaTime * selectedPulseSpeed;
        else selectedPulse = 0f;

        if (isDissolving)
        {
            currentDissolve += Time.deltaTime * dissolveSpeed;
            if (currentDissolve >= 1f) OnDissolveComplete();
        }

        ApplyProperties();
    }

    private void ApplyProperties()
    {
        propertyBlock.SetFloat(HighlightID, currentHighlight);
        propertyBlock.SetFloat(SelectedID, isSelected ? 1f : 0f);
        propertyBlock.SetFloat(SelectedPulseID, selectedPulse);
        propertyBlock.SetFloat(DissolveID, currentDissolve);
        gemRenderer.SetPropertyBlock(propertyBlock);
    }

    // ========== 公共API ==========

    public void SetHighlight(bool highlight)
    {
        targetHighlight = highlight ? 1f : 0f;
    }

    public void SetSelected(bool selected)
    {
        isSelected = selected;
    }

    public void StartDissolve()
    {
        isDissolving = true;
        currentDissolve = 0f;
    }

    public void ResetEffects()
    {
        currentHighlight = 0f;
        targetHighlight = 0f;
        currentDissolve = 0f;
        selectedPulse = 0f;
        isSelected = false;
        isDissolving = false;
        ApplyProperties();
    }

    public event System.Action OnDissolved;

    private void OnDissolveComplete()
    {
        OnDissolved?.Invoke();
    }
}
```

## 4. 消除特效系统

```csharp
using UnityEngine;
using UnityEngine.Pool;

public enum MatchEffectType
{
    Normal,
    Combo,
    Special,
    Skill
}

/// <summary>
/// 消除特效播放器
/// </summary>
public class MatchEffectPlayer : MonoBehaviour
{
    [Header("Effect Prefabs")]
    [SerializeField] private ParticleSystem normalMatchEffect;
    [SerializeField] private ParticleSystem comboMatchEffect;

    [Header("Pool Settings")]
    [SerializeField] private int poolSize = 20;

    private ObjectPool<ParticleSystem> normalPool;
    private ObjectPool<ParticleSystem> comboPool;

    private void Awake()
    {
        normalPool = new ObjectPool<ParticleSystem>(
            () => Instantiate(normalMatchEffect, transform),
            effect => effect.gameObject.SetActive(true),
            effect => effect.gameObject.SetActive(false),
            effect => Destroy(effect.gameObject),
            false, poolSize, poolSize * 2
        );

        comboPool = new ObjectPool<ParticleSystem>(
            () => Instantiate(comboMatchEffect, transform),
            effect => effect.gameObject.SetActive(true),
            effect => effect.gameObject.SetActive(false),
            effect => Destroy(effect.gameObject),
            false, poolSize / 2, poolSize
        );
    }

    public void PlayMatchEffect(Vector3 position, MatchEffectType type, int comboCount = 1)
    {
        var pool = type == MatchEffectType.Combo ? comboPool : normalPool;
        var effect = pool.Get();
        effect.transform.position = position;

        float scale = 1f + (comboCount - 1) * 0.1f;
        effect.transform.localScale = Vector3.one * scale;

        StartCoroutine(ReturnToPoolWhenComplete(effect, pool));
    }

    private System.Collections.IEnumerator ReturnToPoolWhenComplete(
        ParticleSystem effect, ObjectPool<ParticleSystem> pool)
    {
        yield return new WaitForSeconds(effect.main.duration);
        while (effect.isPlaying) yield return null;
        pool.Release(effect);
    }
}
```

## 5. UI流光Shader

```hlsl
Shader "Match3/UI/FlowLight"
{
    Properties
    {
        _BaseMap ("Base Map", 2D) = "white" {}
        _BaseColor ("Base Color", Color) = (1, 1, 1, 1)
        _FlowTexture ("Flow Texture", 2D) = "white" {}
        _FlowColor ("Flow Color", Color) = (1, 1, 1, 1)
        _FlowSpeed ("Flow Speed", Float) = 1
        _FlowIntensity ("Flow Intensity", Range(0, 2)) = 0.5
    }

    SubShader
    {
        Tags { "Queue" = "Transparent" "RenderPipeline" = "UniversalPipeline" }
        Blend SrcAlpha OneMinusSrcAlpha
        ZWrite Off

        Pass
        {
            HLSLPROGRAM
            #pragma vertex Vertex
            #pragma fragment Fragment

            #include "Packages/com.unity.render-pipelines.universal/ShaderLibrary/Core.hlsl"

            TEXTURE2D(_BaseMap);
            SAMPLER(sampler_BaseMap);
            TEXTURE2D(_FlowTexture);
            SAMPLER(sampler_FlowTexture);

            CBUFFER_START(UnityPerMaterial)
                float4 _BaseMap_ST;
                half4 _BaseColor;
                half4 _FlowColor;
                half _FlowSpeed;
                half _FlowIntensity;
            CBUFFER_END

            struct Attributes
            {
                float4 positionOS : POSITION;
                float2 uv : TEXCOORD0;
            };

            struct Varyings
            {
                float4 positionCS : SV_POSITION;
                float2 uv : TEXCOORD0;
            };

            Varyings Vertex(Attributes input)
            {
                Varyings output;
                output.positionCS = TransformObjectToHClip(input.positionOS.xyz);
                output.uv = TRANSFORM_TEX(input.uv, _BaseMap);
                return output;
            }

            half4 Fragment(Varyings input) : SV_Target
            {
                half4 baseColor = SAMPLE_TEXTURE2D(_BaseMap, sampler_BaseMap, input.uv);
                baseColor *= _BaseColor;

                // 流光UV动画
                float2 flowUV = input.uv;
                flowUV.x += _Time.y * _FlowSpeed;
                half4 flow = SAMPLE_TEXTURE2D(_FlowTexture, sampler_FlowTexture, flowUV);

                // 混合
                half3 finalColor = baseColor.rgb + flow.rgb * _FlowColor.rgb * _FlowIntensity;
                return half4(finalColor, baseColor.a);
            }
            ENDHLSL
        }
    }
}
```

## 6. 后处理控制器

```csharp
using UnityEngine;
using UnityEngine.Rendering;
using UnityEngine.Rendering.Universal;

public class Match3PostProcessingController : MonoBehaviour
{
    [SerializeField] private Volume globalVolume;

    private ColorAdjustments colorAdjustments;
    private Bloom bloom;
    private Vignette vignette;

    private float originalSaturation = 1f;

    private void Awake()
    {
        if (globalVolume == null) globalVolume = FindObjectOfType<Volume>();
        if (globalVolume.profile.TryGet(out colorAdjustments))
            originalSaturation = colorAdjustments.saturation.value;
    }

    public void OnMatchEffect(int comboCount)
    {
        StartCoroutine(FlashBloom(1.5f, 0.3f));
    }

    public void OnComboEffect(int comboCount)
    {
        float targetSaturation = originalSaturation + comboCount * 0.1f;
        StartCoroutine(LerpSaturation(targetSaturation, 0.3f));
    }

    public void ResetEffects()
    {
        StartCoroutine(LerpSaturation(originalSaturation, 0.5f));
    }

    private System.Collections.IEnumerator FlashBloom(float intensity, float duration)
    {
        if (bloom == null) yield break;

        float start = bloom.intensity.value;
        float elapsed = 0f;

        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;
            float t = elapsed / duration;
            bloom.intensity.Override(Mathf.Lerp(start, intensity, Mathf.Sin(t * Mathf.PI)));
            yield return null;
        }
    }

    private System.Collections.IEnumerator LerpSaturation(float target, float duration)
    {
        if (colorAdjustments == null) yield break;

        float start = colorAdjustments.saturation.value;
        float elapsed = 0f;

        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;
            colorAdjustments.saturation.Override(Mathf.Lerp(start, target, elapsed / duration));
            yield return null;
        }
    }
}
```

## 7. 效果整合管理器

```csharp
using UnityEngine;

public class Match3VisualManager : MonoSingleton<Match3VisualManager>
{
    [SerializeField] private MatchEffectPlayer effectPlayer;
    [SerializeField] private Match3PostProcessingController postProcessController;
    [SerializeField] private float effectDelay = 0.05f;

    public void PlayMatchEffect(Vector3 position, MatchEffectType type, int comboCount = 1)
    {
        effectPlayer.PlayMatchEffect(position, type, comboCount);
        postProcessController.OnMatchEffect(comboCount);
    }

    public void PlayMatchEffects(Vector3[] positions, MatchEffectType type, int comboCount = 1)
    {
        StartCoroutine(PlayEffectsSequential(positions, type, comboCount));
    }

    private System.Collections.IEnumerator PlayEffectsSequential(
        Vector3[] positions, MatchEffectType type, int comboCount)
    {
        foreach (var pos in positions)
        {
            effectPlayer.PlayMatchEffect(pos, type, comboCount);
            yield return new WaitForSeconds(effectDelay);
        }
        postProcessController.OnMatchEffect(comboCount);
    }

    public void PlayComboEffect(int comboCount, Vector3 centerPosition)
    {
        postProcessController.OnComboEffect(comboCount);
        effectPlayer.PlayMatchEffect(centerPosition, MatchEffectType.Combo, comboCount);
    }
}
```

## 本课小结

### 核心系统

| 系统 | 功能 | 关键组件 |
|------|------|----------|
| 棋子Shader | 棋子渲染效果 | GemShader |
| 材质控制 | 动态控制Shader参数 | GemMaterialController |
| 特效系统 | 粒子效果播放 | MatchEffectPlayer |
| 后处理 | 屏幕效果控制 | PostProcessingController |
| 效果管理 | 统一调度 | VisualManager |

### 效果流程

```
消除触发
    ↓
GemMaterialController.StartDissolve()
    ↓
MatchEffectPlayer.PlayMatchEffect()
    ↓
PostProcessingController.OnMatchEffect()
    ↓
视觉效果完成
```

### 性能优化建议

1. **粒子池化** - 使用ObjectPool复用粒子
2. **MaterialPropertyBlock** - 避免材质克隆
3. **效果分级** - 低端设备减少效果
4. **异步加载** - 特效资源异步加载
